59 void SShowAbout::configuring() throw(::fwTools::Failed)
60 {
61  this->::fwGui::IActionSrv::initialize();
62 
63  typedef SPTR (::fwRuntime::ConfigurationElement) ConfigurationElement;
64 
65  ConfigurationElement cfgFilename = m_configuration->findConfigurationElement("filename");
66  ConfigurationElement cfgTitle = m_configuration->findConfigurationElement("title");
67  ConfigurationElement cfgSize = m_configuration->findConfigurationElement("size");
68 
69  if(cfgFilename)
70  {
71  const std::string& filename = cfgFilename->getExistingAttributeValue("id");
72 
73  m_fsAboutPath = ::boost::filesystem::path(filename);
74  m_bServiceIsConfigured = ::boost::filesystem::exists(m_fsAboutPath);
75  SLM_WARN_IF("About file " + filename + " doesn't exist", !m_bServiceIsConfigured);
76  SLM_TRACE("Filename found '" + filename + "'");
77  }
78 
79  if(cfgTitle)

81  m_title = cfgTitle->getValue();
82  SLM_TRACE("Set title to '" + m_title + "'");
83  }
84 
85  if(cfgSize)
86  {
87  const std::string& w = cfgSize->getExistingAttributeValue("width");
88  const std::string& h = cfgSize->getExistingAttributeValue("height");
89 
90  m_size.setWidth(::boost::lexical_cast< int >(w));
91  m_size.setHeight(::boost::lexical_cast< int >(h));
92 
93  OSLM_TRACE("Set frame size to (" << m_size.width() << ", " << m_size.height() << ")");
94  }

99 void SShowAbout::updating( ) throw(::fwTools::Failed)
100 {
101  SLM_TRACE_FUNC();
102  SLM_ASSERT("The about service isn't configured properly.", m_bServiceIsConfigured );
103 
104  QDialog* dialog = new QDialog(qApp->activeWindow());
105  dialog->setWindowTitle(QString::fromStdString(m_title));
106  QUrl url(QString::fromStdString(m_fsAboutPath.string()));
107 #if defined(QT_WEBKIT)
108  QWebView* htmlView = new QWebView(dialog);
109  htmlView->page()->setLinkDelegationPolicy(QWebPage::DelegateAllLinks);
110  htmlView->load( url );
111  QObject::connect( htmlView, SIGNAL(linkClicked(const QUrl &)),this, SLOT(onUrlClicked(const QUrl &)));
112 #else
113  QTextBrowser * htmlView = new QTextBrowser(dialog);
114  htmlView->setSource(url);
115  htmlView->setOpenExternalLinks(true);
116  htmlView->setMinimumSize(m_size);
117 #endif
118  QPushButton* okButton = new QPushButton(QObject::tr("Ok"));
119  QHBoxLayout *hLayout = new QHBoxLayout();
120  hLayout->addStretch();
121  hLayout->addWidget(okButton);
122  hLayout->setContentsMargins(5, 5, 5, 5);
123 
124  QFrame* line = new QFrame(dialog);
125  line->setFrameShape(QFrame::HLine);
126  line->setFrameShadow(QFrame::Sunken);
127 
128  QVBoxLayout* layout = new QVBoxLayout();
129  layout->addWidget(htmlView, 0);
130  layout->addWidget(line, 0);
131  layout->addLayout(hLayout, 0);
132  layout->setContentsMargins(0, 0, 0, 0);
133  layout->setSpacing(0);
134  dialog->setLayout( layout );
135 
136  QObject::connect(okButton, SIGNAL(clicked()), dialog, SLOT(accept()));
137  QObject::connect(dialog, SIGNAL(accepted()), dialog, SLOT(deleteLater()));
138  dialog->setModal(true);
139  dialog->show();
140 }
