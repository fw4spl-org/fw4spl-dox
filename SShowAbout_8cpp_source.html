00058 void SShowAbout::configuring() throw(::fwTools::Failed)
00059 {
00060     this->::fwGui::IActionSrv::initialize();
00061 
00062     typedef SPTR(::fwRuntime::ConfigurationElement) ConfigurationElement;
00063 
00064     ConfigurationElement cfgFilename = m_configuration->findConfigurationElement("filename");
00065     ConfigurationElement cfgTitle = m_configuration->findConfigurationElement("title");
00066     ConfigurationElement cfgSize = m_configuration->findConfigurationElement("size");
00067 
00068     if(cfgFilename)
00069     {
00070         const std::string& filename = cfgFilename->getExistingAttributeValue("id");
00071 
00072         m_fsAboutPath = ::boost::filesystem::path(filename);
00073         m_bServiceIsConfigured = ::boost::filesystem::exists(m_fsAboutPath);
00074         SLM_WARN_IF("About file " + filename + " doesn't exist", !m_bServiceIsConfigured);
00075         SLM_TRACE("Filename found '" + filename + "'");
00076     }
00077 
00078     if(cfgTitle)
00079     {
00080         m_title = cfgTitle->getValue();
00081         SLM_TRACE("Set title to '" + m_title + "'");
00082     }
00083 
00084     if(cfgSize)
00085     {
00086         const std::string& w = cfgSize->getExistingAttributeValue("width");
00087         const std::string& h = cfgSize->getExistingAttributeValue("height");
00088 
00089         m_size.setWidth(::boost::lexical_cast< int >(w));
00090         m_size.setHeight(::boost::lexical_cast< int >(h));
00091 
00092         OSLM_TRACE("Set frame size to (" << m_size.width() << ", " << m_size.height() << ")");
00093     }
00094 }

00098 void SShowAbout::updating( ) throw(::fwTools::Failed)
00099 {
00100     SLM_TRACE_FUNC();
00101     SLM_ASSERT("The about service isn't configured properly.", m_bServiceIsConfigured );
00102 
00103     QDialog* dialog = new QDialog(qApp->activeWindow());
00104     dialog->setWindowTitle(QString::fromStdString(m_title));
00105     QUrl url(QString::fromStdString(m_fsAboutPath.string()));
00106 #if defined(QT_WEBKIT)
00107     QWebView* htmlView = new QWebView(dialog);
00108     htmlView->page()->setLinkDelegationPolicy(QWebPage::DelegateAllLinks);
00109     htmlView->load( url );
00110     QObject::connect( htmlView, SIGNAL(linkClicked(const QUrl&)),this, SLOT(onUrlClicked(const QUrl&)));
00111 #else
00112     QTextBrowser * htmlView = new QTextBrowser(dialog);
00113     htmlView->setSource(url);
00114     htmlView->setOpenExternalLinks(true);
00115     htmlView->setMinimumSize(m_size);
00116 #endif
00117     QPushButton* okButton = new QPushButton(QObject::tr("Ok"));
00118     QHBoxLayout *hLayout = new QHBoxLayout();
00119     hLayout->addStretch();
00120     hLayout->addWidget(okButton);
00121     hLayout->setContentsMargins(5, 5, 5, 5);
00122 
00123     QFrame* line = new QFrame(dialog);
00124     line->setFrameShape(QFrame::HLine);
00125     line->setFrameShadow(QFrame::Sunken);
00126 
00127     QVBoxLayout* layout = new QVBoxLayout();
00128     layout->addWidget(htmlView, 0);
00129     layout->addWidget(line, 0);
00130     layout->addLayout(hLayout, 0);
00131     layout->setContentsMargins(0, 0, 0, 0);
00132     layout->setSpacing(0);
00133     dialog->setLayout( layout );
00134 
00135     QObject::connect(okButton, SIGNAL(clicked()), dialog, SLOT(accept()));
00136     QObject::connect(dialog, SIGNAL(accepted()), dialog, SLOT(deleteLater()));
00137     dialog->setModal(true);
