24 bool setPreference(const std::string& key, const std::string& value)
25 {
26  bool isModified = false;
27  // Check preferences
28  auto preferencesServicesList = ::fwServices::OSR::getServices("::fwPreferences::IPreferences");
29  if(!preferencesServicesList.empty())
30  {
31  const ::fwServices::IService::sptr prefService = *preferencesServicesList.begin();
32  ::fwData::Composite::sptr prefs = prefService->getObject< ::fwData::Composite >();
33 
34  ::fwData::Composite::IteratorType iterPref = prefs->find(key);
35  if ( iterPref != prefs->end() )
36  {
37  ::fwData::String::sptr preferences = ::fwData::String::dynamicCast(iterPref->second);
38  preferences->value() = value;
39  }
40  else
41  {
42  (*prefs)[key] = ::fwData::String::New(value);
43  }
44  isModified = true;
45  }
46  return isModified;
47 }
