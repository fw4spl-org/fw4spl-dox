153 QStandardItem* SelectorModel::getInfo(T data, QString separator)
154 {
155  QString dataStr;
156  if(!data.empty())
157  {
158  typename T::iterator itr = data.begin();
159  std::ostringstream str;
160  str << *itr++;
161  dataStr = QString::fromStdString(str.str());
162 
163  for(; itr!= data.end(); ++itr)
164  {
165  str.str("");
166  str << *itr;
167  dataStr.append(separator + QString::fromStdString(str.str()));
168  }
169  }
170  QStandardItem *item = new QStandardItem(dataStr);
171  return item;
172 }
