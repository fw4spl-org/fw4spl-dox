00041 void SReconstructionFromModelSeriesUpdater::receiving( ::fwServices::ObjectMsg::csptr _msg ) throw ( ::fwTools::Failed )
00042 {
00043     ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
00044     for (   ManagedEvents::iterator it = m_managedEvents.begin();
00045             it != m_managedEvents.end();
00046             ++it )
00047     {
00048         // First element of tuple ( it->get<0>() ) is the event, test if message correspond to a defined event
00049         if( _msg->hasEvent( it->get<0>() ) )
00050         {
00051             ::fwData::Object::sptr obj = ::fwData::Object::dynamicCast( _msg->getSubject().lock() );
00052             SLM_ASSERT(obj,"Sorry, the subject of message is not a ::fwData::Object");
00053 
00054             // Test if we manage this event from this object message uid ( it->get<1>() )
00055             if( it->get<1>() == "*" || obj->getID() == it->get<1>() )
00056             {
00057                 ::fwData::Reconstruction::sptr reconst;
00058                 ActionType action = it->get<3>();
00059                 if( action != REMOVE && action != REMOVE_IF_PRESENT && action != DO_NOTHING)
00060                 {
00061                     reconst = this->getReconstruction(_msg);
00062                 }
00063                 // Udapte the composite object referenced by the composite key ( it->get<2>() )
00064                 this->updateComposite(composite, reconst, it->get<2>(), it->get<3>() );
00065             }
00066         }
00067     }

00075     ::fwComEd::ModelSeriesMsg::csptr pModelSeriesMsg = ::fwComEd::ModelSeriesMsg::dynamicConstCast( msg );
00076     SLM_FATAL_IF("Received message isn't a ModelSeriesMsg", !pModelSeriesMsg);

00078     SLM_ASSERT("Sorry, there's not NEW_RECONSTRUCTION_SELECTED in ModelSeriesMsg", pModelSeriesMsg->hasEvent( ::fwComEd::ModelSeriesMsg::NEW_RECONSTRUCTION_SELECTED ));
00079     ::fwData::Object::csptr dataInfo = pModelSeriesMsg->getDataInfo(::fwComEd::ModelSeriesMsg::NEW_RECONSTRUCTION_SELECTED);
00080     SLM_ASSERT("Sorry, there is no dataInfo", dataInfo);
00081     ::fwData::String::csptr reconstructionSelectedUID = ::fwData::String::dynamicConstCast( dataInfo ) ;
00082     SLM_ASSERT("Sorry,  there is no reconstruction selected", reconstructionSelectedUID);
00083     reconst =  ::fwData::Reconstruction::dynamicCast( ::fwTools::fwID::getObject(reconstructionSelectedUID->value() ));
