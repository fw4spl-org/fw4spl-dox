62 void TransferFunctionEditor::configuring() throw( ::fwTools::Failed )
63 {
64  SLM_TRACE_FUNC();
65  this->initialize();
66  ::fwRuntime::ConfigurationElement::sptr configuration = m_configuration->findConfigurationElement("config");
67 
68  SLM_ASSERT("Missing 'config' element.", configuration);
69 
70  if ( configuration->hasAttribute("selectedTFKey") )
71  {
72  m_selectedTFKey = configuration->getAttributeValue("selectedTFKey");
73  SLM_FATAL_IF("'selectedTFKey' must not be empty", m_selectedTFKey.empty());
74  }
75  if(!this->isVersion2())
76  {
77  if ( configuration->hasAttribute("tfSelectionFwID") )
78  {
79  m_tfSelectionFwID = configuration->getAttributeValue("tfSelectionFwID");
80  SLM_FATAL_IF("'tfSelectionFwID' must not be empty", m_tfSelectionFwID.empty());
81  }
82  }
83  else
84  {
85  m_tfSelectionFwID = this->getInOut< ::fwData::Composite>("TFSelections")->getID();
86  }
87 
88  std::vector< ::fwRuntime::ConfigurationElement::sptr > pathsCfg = configuration->find("path");
89  for(::fwRuntime::ConfigurationElement::sptr cfg : pathsCfg)
90  {
91  ::boost::filesystem::path path(cfg->getValue());
92  m_paths.push_back(path);
93  }
94 
95  bool useDefaultPath = true;
96  if ( configuration->hasAttribute("useDefaultPath") )
97  {
98  std::string value = configuration->getAttributeValue("useDefaultPath");
99  SLM_FATAL_IF("'useDefaultPath' value must be 'yes' or 'no'", value != "yes" && value != "no");
100 
101  useDefaultPath = (value == "yes");
102  }
103  if (useDefaultPath)
104  {
105  ::boost::filesystem::path pathRoot (std::string(BUNDLE_PREFIX) + "/uiTF_" + std::string(
106  UITF_VER) + "/tf");
107  m_paths.push_back(pathRoot);
108  }
109 }

278  newName = this->createTransferFunctionName(newName);

298  compositeHelper.add(newName, pNewTransferFunction);

306  else

308  ::fwGui::dialog::MessageDialog::showMessageDialog(
309  "Warning",
310  "This transfer function name already exists so you can not overwrite it.",

408 void TransferFunctionEditor::importTF()
409 {
410  ::fwData::Composite::sptr poolTF = this->getTFPool();
411  ::fwDataTools::helper::Composite compositeHelper(poolTF);
412 
413  ::fwData::TransferFunction::sptr tf = ::fwData::TransferFunction::New();

416 
417  ::fwServices::OSR::registerService(tf, srv);
418 
419  ::io::IReader::sptr reader = ::io::IReader::dynamicCast(srv);
420  reader->start();
421  reader->configureWithIHM();
422  reader->update();
423  reader->stop();
424  ::fwServices::OSR::unregisterService(srv);
425 
426  if (!tf->getName().empty())
427  {
428  if( this->hasTransferFunctionName( tf->getName() ) )
429  {
430  tf->setName( this->createTransferFunctionName( tf->getName() ) );
431  }
432 
433  compositeHelper.add(tf->getName(), tf);
434  m_pTransferFunctionPreset->addItem(QString(tf->getName().c_str()));
435  this->presetChoice(static_cast<int>((*poolTF).size()-1));
436 
437  compositeHelper.notify();
438  }

473 void TransferFunctionEditor::initTransferFunctions()
474 {
475  // Get transfer function composite (pool TF)
476  ::fwData::Composite::sptr poolTF = this->getTFPool();
477 
478  ::fwDataTools::helper::Composite compositeHelper(poolTF);
479 
480  const std::string defaultTFName = ::fwData::TransferFunction::s_DEFAULT_TF_NAME;
481  if(!this->hasTransferFunctionName(defaultTFName))
482  {
483  ::fwData::TransferFunction::sptr defaultTf = ::fwData::TransferFunction::createDefaultTF();
484  defaultTf->setWindow( 50. );
485  defaultTf->setLevel( 500. );
486  compositeHelper.add(defaultTFName, defaultTf);
487  }
488 
489  // Test if transfer function composite has few TF
490  if( poolTF->size() <= 1 )
491  {
492  // Parse all TF contained in uiTF Bundle's resources
493  std::vector< ::boost::filesystem::path > paths;
494  for(::boost::filesystem::path dirPath : m_paths)
495  {
496  SLM_ASSERT("Invalid directory path '"+dirPath.string()+"'", ::boost::filesystem::exists(dirPath));
497  for(::boost::filesystem::directory_iterator it(dirPath);
498  it != ::boost::filesystem::directory_iterator();
499  ++it )
500  {
501  if(!::boost::filesystem::is_directory(*it) &&
502  ::boost::filesystem::extension(*it) == ".json")
503  {
504  paths.push_back(*it);
505  }
506  }
507  }
508 
509  ::fwData::TransferFunction::sptr tf = ::fwData::TransferFunction::New();
510  ::fwServices::IService::sptr srv =
511  ::fwServices::registry::ServiceFactory::getDefault()->create("::ioAtoms::SReader");
512  ::fwServices::OSR::registerService(tf, srv);
513  ::io::IReader::sptr reader = ::io::IReader::dynamicCast(srv);
514 
515  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
516  ::fwRuntime::EConfigurationElement::sptr fileCfg = ::fwRuntime::EConfigurationElement::New("file");
517  srvCfg->addConfigurationElement(fileCfg);
518 
519  for( ::boost::filesystem::path file : paths )
520  {
521  fileCfg->setValue(file.string());
522  reader->setConfiguration(srvCfg);
523  reader->configure();
524  reader->start();
525  reader->update();
526  reader->stop();
527 
528  if (!tf->getName().empty())
529  {
530  ::fwData::TransferFunction::sptr newTF = ::fwData::Object::copy< ::fwData::TransferFunction >(tf);
531  if( this->hasTransferFunctionName( newTF->getName() ) )
532  {
533  newTF->setName( this->createTransferFunctionName( newTF->getName() ) );
534  }
535 
536  compositeHelper.add(newTF->getName(), newTF);
537  }
538  tf->initTF();
539  }
540  ::fwServices::OSR::unregisterService(srv);
541  }
542  compositeHelper.notify();
543 
544  this->updateTransferFunctionPreset();
545 }

550 void TransferFunctionEditor::updateTransferFunctionPreset()
551 {
552  ::fwData::Composite::sptr poolTF = this->getTFPool();
553  const std::string defaultTFName = ::fwData::TransferFunction::s_DEFAULT_TF_NAME;
554  // Manage TF preset
555  m_pTransferFunctionPreset->clear();
556  for(::fwData::Composite::value_type elt : *poolTF)
557  {
558  m_pTransferFunctionPreset->addItem( elt.first.c_str() );
559  }
560 
561  int index = m_pTransferFunctionPreset->findText( QString::fromStdString(defaultTFName) );
562  index = std::max(index, 0);
563  ::fwData::TransferFunction::sptr selectedTF = this->getSelectedTransferFunction();
564  if(selectedTF)
565  {
566  std::string tfName = selectedTF->getName();
567  int tmpIdx = m_pTransferFunctionPreset->findText(QString::fromStdString(tfName));
568  if(tmpIdx >= 0)
569  {
570  index = tmpIdx;
571  }
572  }
573  this->presetChoice(index);
574 }

586 std::string TransferFunctionEditor::createTransferFunctionName(const std::string& _sBasename)
587 {
588  bool bHasTransferFunctionName = true;
589  std::string newName = _sBasename;
590  int cpt = 1;
591  while (bHasTransferFunctionName)
592  {
593  std::stringstream tmpStr;
594  tmpStr << _sBasename << "_" << cpt;
595  newName = tmpStr.str();
596  bHasTransferFunctionName = this->hasTransferFunctionName(newName);
597  cpt++;
598  }
599 
600  return newName;
601 }

605 void TransferFunctionEditor::updateTransferFunction()
606 {
607  std::string newSelectedTFKey = m_pTransferFunctionPreset->currentText().toStdString();
608  OSLM_DEBUG("Transfer function selected : " << newSelectedTFKey);
609 
610  ::fwData::Composite::sptr tfSelection = this->getTFSelection();
611 
612  OSLM_ASSERT("TF "<< newSelectedTFKey <<" missing in pool", this->hasTransferFunctionName(newSelectedTFKey));
613  ::fwData::Composite::sptr poolTF = this->getTFPool();
614  ::fwData::Object::sptr newSelectedTF = (*poolTF)[newSelectedTFKey];
615  if(this->getSelectedTransferFunction() != newSelectedTF)
616  {
617  ::fwDataTools::helper::Composite compositeHelper(tfSelection);
618  if(tfSelection->getContainer().find(m_selectedTFKey) != tfSelection->getContainer().end())
619  {
620  compositeHelper.swap(m_selectedTFKey, newSelectedTF);
621  }
622  else
623  {
624  compositeHelper.add(m_selectedTFKey, newSelectedTF);
625  }
626  compositeHelper.notify();
627  }
628 }
