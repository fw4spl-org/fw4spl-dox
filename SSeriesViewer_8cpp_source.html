00085 void SSeriesViewer::updating() throw(::fwTools::Failed)
00086 {
00087     ::fwData::Vector::sptr vector = this->getObject< ::fwData::Vector >();
00088 
00089     if(m_configTemplateManager)
00090     {
00091         m_configTemplateManager->stopAndDestroy();
00092         m_configTemplateManager.reset();
00093     }
00094 
00095     if(vector->size() == 1)
00096     {
00097         ::fwData::Object::sptr obj = vector->front();
00098         std::string classname = obj->getClassname();
00099         SeriesConfigMapType::iterator itr = m_seriesConfigs.find(classname);
00100 
00101         if(itr != m_seriesConfigs.end())
00102         {
00103             SeriesConfigInfo info = itr->second;
00104             std::string configId =  info.configId;
00105 
00106             std::map< std::string, std::string > replaceMap;
00107             // Generate generic UID
00108             std::string genericUidAdaptor = ::fwServices::registry::AppConfig::getUniqueIdentifier( this->getID() );
00109             replaceMap["GENERIC_UID"] = genericUidAdaptor;
00110             replaceMap["WID_PARENT"] = m_parentView;
00111             replaceMap["objectID"] = obj->getID();
00112 
00113             BOOST_FOREACH(const ReplaceValuesMapType::value_type &elt, info.extractValues)
00114             {
00115                 ::fwData::Object::sptr object = ::fwDataCamp::getObject( obj, elt.second );
00116                 OSLM_ASSERT("Object from name "<< elt.second <<" not found", object);
00117                 replaceMap[elt.first] = object->getID();
00118             }
00119 
00120             BOOST_FOREACH(const ReplaceValuesMapType::value_type &elt, info.parameters)
00121             {
00122                 SLM_ASSERT("Value '" << elt.first << "' already used in extracted values.",
00123                            replaceMap.find(elt.first) == replaceMap.end());
00124                 replaceMap[elt.first] = elt.second;
00125             }
00126 
00127             // Init manager
00128             ::fwRuntime::ConfigurationElement::csptr config =
00129                    ::fwServices::registry::AppConfig::getDefault()->getAdaptedTemplateConfig( configId, replaceMap );
00130             m_configTemplateManager = ::fwServices::AppConfigManager::New();
00131             m_configTemplateManager->setConfig( config );
00132 
00133             // Launch config
00134             m_configTemplateManager->launch();
00135         }
00136     }

00141 void SSeriesViewer::configuring() throw(::fwTools::Failed)
00142 {
00143     std::vector < ::fwRuntime::ConfigurationElement::sptr > viewCfg = m_configuration->find("parentView");
00144     SLM_ASSERT("Missing tag 'parentView'", viewCfg.size() == 1);

00147 
00148     std::vector < ::fwRuntime::ConfigurationElement::sptr > configsCfg = m_configuration->find("configs");
00149     SLM_ASSERT("Missing tag 'configs'", configsCfg.size() == 1);
00150 
00151     std::vector < ::fwRuntime::ConfigurationElement::sptr > config = configsCfg[0]->find("config");
00152     SLM_ASSERT("Missing tag 'config'", !config.empty());
00153 
00154     BOOST_FOREACH(const ::fwRuntime::ConfigurationElement::sptr &elt, config)
00155     {
00156         SeriesConfigInfo info;
00157         info.configId = elt->getAttributeValue("id");
00158         SLM_ASSERT("'id' attribute must not be empty", !info.configId.empty());
00159         std::string seriesType = elt->getAttributeValue("type");
00160         SLM_ASSERT("'type' attribute must not be empty", !seriesType.empty());
00161         OSLM_ASSERT("Type " << seriesType << " is already defined.",
00162                     m_seriesConfigs.find(seriesType)== m_seriesConfigs.end() );
00163 
00164         BOOST_FOREACH(const ::fwRuntime::ConfigurationElement::sptr &extractElt, elt->find("extract"))
00165         {
00166             std::string path = extractElt->getAttributeValue("path");
00167             SLM_ASSERT("'path' attribute must not be empty", !path.empty());
00168             std::string pattern = extractElt->getAttributeValue("pattern");
00169             SLM_ASSERT("'pattern' attribute must not be empty", !pattern.empty());
00170             info.extractValues[pattern] = path;
00171         }
00172 
00173         BOOST_FOREACH(const ::fwRuntime::ConfigurationElement::sptr &param, elt->find("parameter"))
00174         {
00175             std::string replace = param->getAttributeValue("replace");
00176             SLM_ASSERT("'replace' attribute must not be empty", !replace.empty());
00177             std::string by = param->getAttributeValue("by");
00178             SLM_ASSERT("'by' attribute must not be empty", !by.empty());
00179             info.parameters[replace] = by;
00180         }
00181 
00182         m_seriesConfigs[seriesType] = info;
00183     }
