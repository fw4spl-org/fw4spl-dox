00080 void Axes::configuring() throw(fwTools::Failed)
00081 {
00082     assert( m_configuration->getName() == "config" );
00083 
00084     this->setRenderId( m_configuration->getAttributeValue("renderer") );
00085 
00086     if ( m_configuration->hasAttribute( "transform" ) )
00087     {
00088         this->setTransformId ( m_configuration->getAttributeValue ( "transform" ) );
00089     }
00090     if ( m_configuration->hasAttribute( "length" ) )
00091     {
00092         m_length = boost::lexical_cast<double>( m_configuration->getAttributeValue( "length" ) );
00093     }
00094     if ( m_configuration->hasAttribute( "label" ) )
00095     {
00096         std::string value = m_configuration->getAttributeValue( "label" );
00097         std::transform( value.begin(), value.end(), value.begin(), tolower );
00098         m_labelOn = ( value == "yes" );
