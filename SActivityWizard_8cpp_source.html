88 void SActivityWizard::configuring() throw(fwTools::Failed)
89 {
90  ::fwGui::IGuiContainerSrv::initialize();
91 
92  ::fwRuntime::ConfigurationElement::sptr config = m_configuration->findConfigurationElement("ioSelectorConfig");
93  if (config)
94  {
95  m_ioSelectorConfig = config->getValue();
96  }
97 
98  ::fwRuntime::ConfigurationElement::sptr confirmConfig = m_configuration->findConfigurationElement("confirm");
99  if (confirmConfig)
100  {
101  std::string confirmStr = confirmConfig->getValue();
102  SLM_ASSERT("'confirm' value must be 'true' or 'false', actual: '" + confirmStr + "'",
103  confirmStr == "true" || confirmStr == "false");
104  m_confirmUpdate = (confirmStr == "true");
105  }
106 
107  ::fwRuntime::ConfigurationElement::sptr cancelConfig = m_configuration->findConfigurationElement("cancel");
108  if (cancelConfig)
109  {
110  std::string cancelStr = cancelConfig->getValue();
111  SLM_ASSERT("'cancel' value must be 'true' or 'false', actual: '" + cancelStr + "'",
112  cancelStr == "true" || cancelStr == "false");
113  m_isCancelable = (cancelStr == "true");
114  }
115 
116  std::vector < ::fwRuntime::ConfigurationElement::sptr > iconsCfg = m_configuration->find("icons");
117  if (!iconsCfg.empty())
118  {
119  SLM_ASSERT("Only one 'config' tag is allowed for SSelector configuration", iconsCfg.size() == 1);
120 
121  std::vector < ::fwRuntime::ConfigurationElement::sptr > cfg = iconsCfg.front()->find("icon");
122 
123  for(::fwRuntime::ConfigurationElement::sptr elt : cfg)
124  {
125  std::string type = elt->getAttributeValue("type");
126  SLM_ASSERT("'series' attribute is missing", !type.empty());
127 
128  std::string icon = elt->getAttributeValue("icon");
129  SLM_ASSERT("'icon' attribute is missing", !icon.empty());
130 
131  m_objectIcons[type] = icon;
132  }
133  }

228 void SActivityWizard::createActivity(std::string activityID)
229 {
230  m_mode = Mode::CREATE;
231  ::fwActivities::registry::ActivityInfo info;
232  info = ::fwActivities::registry::Activities::getDefault()->getInfo(activityID);
233 
234  // load activity bundle
235  std::shared_ptr< ::fwRuntime::Bundle > bundle = ::fwRuntime::findBundle(info.bundleId,
236  info.bundleVersion);
237  if (!bundle->isStarted())
238  {
239  bundle->start();
240  }
241 
242  m_actSeries = ::fwMedData::ActivitySeries::New();
243 
244  m_actSeries->setModality("OT");
245  m_actSeries->setInstanceUID("fwActivities." + ::fwTools::UUID::generateUUID() );
246 
247  ::boost::posix_time::ptime now = ::boost::posix_time::second_clock::local_time();
248  m_actSeries->setDate(::fwTools::getDate(now));
249  m_actSeries->setTime(::fwTools::getTime(now));
250 
251  m_actSeries->setActivityConfigId(info.id);
252 
253  m_title->setText(QString("<h1>%1</h1>").arg(QString::fromStdString(info.title)));
254  m_description->setText(QString::fromStdString(info.description));
255 
256  bool needConfig = false;
257 
258  // If we have requirements but they are not needed to start (maxOccurs = 0), we can skip the config as well
259  for(const auto& req : info.requirements)
260  {
261  if(req.maxOccurs > 0)
262  {
263  needConfig = true;
264  break;
265  }
266  }
267 
268  if (needConfig)
269  {
270  m_activityDataView->fillInformation(info);
271  if (m_activityDataView->count() > 1)
272  {
273  m_okButton->setText("Next");
274  }
275 
276  this->slot(s_SHOW_SLOT)->asyncRun();
277  }
278  else
279  {
280  // Create data automatically if they are not provided by the user
281  for(const auto& req : info.requirements)
282  {
283  SLM_ASSERT("minOccurs and maxOccurs should be 0", req.minOccurs == 0 && req.maxOccurs == 0);
284  ::fwData::Composite::sptr data = m_actSeries->getData();
285  (*data)[req.name] = ::fwData::factory::New(req.type);
286  }
287 
288  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
289  ::fwMedDataTools::helper::SeriesDB helper(seriesDB);
290  helper.add(m_actSeries);
291  helper.notify();
292  m_sigActivityCreated->asyncEmit(m_actSeries);
293  }
294 }

299 {
300  ::fwActivities::registry::ActivityInfo info;
301  info = ::fwActivities::registry::Activities::getDefault()->getInfo(activitySeries->getActivityConfigId());
302 
303  // load activity bundle
304  std::shared_ptr< ::fwRuntime::Bundle > bundle = ::fwRuntime::findBundle(info.bundleId,
305  info.bundleVersion);
306  if (!bundle->isStarted())
307  {
308  bundle->start();
309  }
310 
311  m_title->setText(QString("<h1>%1</h1>").arg(QString::fromStdString(info.title)));
312  m_description->setText(QString::fromStdString(info.description));
313 
314  m_mode = Mode::UPDATE;
315  m_actSeries = activitySeries;
316 
317  bool needConfig = false;
318 
319  // If we have requirements but they are not needed to start (maxOccurs = 0), we can skip the config as well
320  for(const auto& req : info.requirements)
321  {
322  if(req.maxOccurs != 0)
323  {
324  needConfig = true;
325  break;
326  }
327  }
328 
329  if (needConfig)
330  {
331  m_activityDataView->fillInformation(m_actSeries);
332  if (m_activityDataView->count() > 1)
333  {
334  m_okButton->setText("Next");
335  }
336  }
337  else
338  {
339  // Start immediately without popping any configuration UI
340  ::fwData::Object::ModifiedSignalType::sptr sig;
341  sig = m_actSeries->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
342  sig->asyncEmit();
343  m_sigActivityUpdated->asyncEmit(m_actSeries);

401 void SActivityWizard::onBuildActivity()
402 {
403  int index = m_activityDataView->currentIndex();
404  int lastTab = m_activityDataView->count() -1;
405 
406  std::string errorMsg;
407  // Check current data
408  if (m_activityDataView->checkData(size_t(index), errorMsg))
409  {
410  if (index != lastTab)
411  {
412  // enable and select the next tab
413  m_activityDataView->setTabEnabled(index+1, true);
414  m_activityDataView->setCurrentIndex(index+1);
415  }
416 
417  else // index == lastTab
418  {
419  // Create/update activity
420  if (m_mode == Mode::UPDATE && m_confirmUpdate)
421  {
422  QMessageBox::StandardButton button = QMessageBox::question(
423  qApp->activeWindow(),
424  "Update activity",
425  "You will override your activity. You could loose some data.\n"
426  "Would you duplicate your activity ?",
427  QMessageBox::Yes | QMessageBox::No | QMessageBox::Cancel,
428  QMessageBox::No);
429 
430  if (button == QMessageBox::Cancel)
431  {
432  return;
433  }
434  else if (button == QMessageBox::Yes)
435  {
436  m_actSeries = ::fwData::Object::copy(m_actSeries);
437  m_mode = Mode::CREATE; // The new activity should be added in the seriesDB
438  }
439  }
440 
441  // check all data and create/update the activity
442  bool ok = m_activityDataView->checkAndComputeData(m_actSeries, errorMsg);
443  if (ok)
444  {
445  ::fwData::Composite::sptr data = m_actSeries->getData();
446 
447  // Copy the patient/study information of a series
448  ::fwMedData::Series::sptr series;
449  for(const auto& elt : (*data) )
450  {
451  series = ::fwMedData::Series::dynamicCast(elt.second);
452  if(series)
453  {
454  m_actSeries->setPatient( ::fwData::Object::copy(series->getPatient()) );
455  m_actSeries->setStudy( ::fwData::Object::copy(series->getStudy()) );
456  m_actSeries->setEquipment( ::fwData::Object::copy(series->getEquipment()) );
457  break;
458  }
459  }
460 
461  if (m_mode == Mode::CREATE)
462  {
463  // Add the new activity series in seriesDB
464  ::fwActivities::registry::ActivityInfo info;
465  info = ::fwActivities::registry::Activities::getDefault()->getInfo(
466  m_actSeries->getActivityConfigId());
467 
468  std::string description = ::fwGui::dialog::InputDialog::showInputDialog(
469  "Activity creation",
470  "Please, give a description of the activity.",
471  info.title);
472  if (description.empty())
473  {
474  return;
475  }
476  m_actSeries->setDescription(description);
477  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
478  ::fwMedDataTools::helper::SeriesDB helper(seriesDB);
479  helper.add(m_actSeries);
480  helper.notify();
481  m_sigActivityCreated->asyncEmit(m_actSeries);
482  }
483  else // m_mode == Mode::UPDATE
484  {
485  ::fwData::Object::ModifiedSignalType::sptr sig;
486  sig = m_actSeries->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
487  sig->asyncEmit();
488  m_sigActivityUpdated->asyncEmit(m_actSeries);
489  }
490  }
491  else
492  {
493  QString message = "This activity can not be created : \n";
494  message.append(QString::fromStdString(errorMsg));
495  QMessageBox::warning(qApp->activeWindow(), "Activity Creation", message);
496  SLM_ERROR(errorMsg);
497  }
498  }
499  }
500  else
501  {
502  QMessageBox::warning(qApp->activeWindow(), "Error", QString::fromStdString(errorMsg));
503  }
504 }
