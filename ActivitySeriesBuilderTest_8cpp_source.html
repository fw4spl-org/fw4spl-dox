00059 void ActivitySeriesBuilderTest::builDataTest()
00060 {
00061     ::fwData::Vector::sptr selection = ::fwData::Vector::New();
00062     ::fwMedData::ImageSeries::sptr imgSeriesSelected = ::fwMedData::ImageSeries::New();
00063     ::fwMedData::ModelSeries::sptr modelSeriesSelected = ::fwMedData::ModelSeries::New();
00064     selection->getContainer().push_back(imgSeriesSelected);
00065     selection->getContainer().push_back(modelSeriesSelected);
00066 
00067     ::fwActivities::registry::Activities::ActivitiesType activities;
00068     activities = m_activities->getInfos(selection);
00069     CPPUNIT_ASSERT_EQUAL(size_t(1), activities.size());
00070 
00071     ::fwActivities::registry::ActivityInfo activityInfo = activities[0];
00072     ::fwMedData::ActivitySeries::sptr actSeries;
00073     ::fwActivities::IBuilder::sptr builder;
00074     builder = ::fwActivities::builder::factory::New(activityInfo.builderImpl);
00075     CPPUNIT_ASSERT_MESSAGE("<" + activityInfo.builderImpl + "> instantiation failed", builder);
00076     actSeries = builder->buildData(activityInfo, selection);
00077 
00078     CPPUNIT_ASSERT_MESSAGE("ActivitySeries instantiation failed", actSeries);
00079 
00080     CPPUNIT_ASSERT_EQUAL(activityInfo.id, actSeries->getActivityConfigId());
00081 
00082     ::fwData::Composite::sptr dataActivity = actSeries->getData();
00083     CPPUNIT_ASSERT_EQUAL(size_t(2), dataActivity->size());
00084 
00085     const std::string imageKey = "imageSeries";
00086     const std::string modelKey = "modelSeries";
00087     CPPUNIT_ASSERT_MESSAGE(imageKey + " key is missing", dataActivity->find(imageKey) != dataActivity->end());
00088     CPPUNIT_ASSERT_MESSAGE(modelKey + " key is missing", dataActivity->find(modelKey) != dataActivity->end());
00089 
00090     //single param [1;1]
00091     ::fwData::Object::sptr obj = (*dataActivity)[imageKey];
00092     CPPUNIT_ASSERT(obj == imgSeriesSelected);
00093 
00094     //set of param [0;2]
00095     obj = (*dataActivity)[modelKey];
00096     ::fwData::Composite::sptr composite = ::fwData::Composite::dynamicCast(obj);
00097     CPPUNIT_ASSERT_MESSAGE(modelKey + " param dynamicCast to fwData::Composite failed", composite);
00098     CPPUNIT_ASSERT_EQUAL(size_t(1), composite->size());
00099     CPPUNIT_ASSERT(modelSeriesSelected == (*composite)["key1"]);
00100 }
