211 void WindowLevel::updating() throw(::fwTools::Failed)
212 {
213  SLM_TRACE_FUNC();
214  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
215 
216  bool imageIsValid = ::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity( image );
217  this->setEnabled(imageIsValid);
218 
219  this->updateTransferFunction(image);
220  if(imageIsValid)
221  {
222  this->updateImageInfos(image);
223 
224  // test if service must use image grey level tf ( when another tf pool is defined )
225  if( m_useImageGreyLevelTF &&
226  !this->getTFSelectionFwID().empty() )
227  {
228  ::fwData::TransferFunction::sptr newTF = this->getImageGreyLevelTF();
229  this->swapCurrentTFAndNotify( newTF );
230  m_toggleTFButton->setCheckable(true);
231  }
232 
233  if(m_autoWindowing)
234  {
235  double min, max;
236  ::fwComEd::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
237  this->updateImageWindowLevel(min, max);
238  }
239 
240  ::fwData::TransferFunction::sptr pTF = this->getTransferFunction();
241  SLM_ASSERT("TransferFunction null pointer", pTF);
242  ::fwData::TransferFunction::TFValuePairType minMax = pTF->getWLMinMax();
243  this->onImageWindowLevelChanged( minMax.first, minMax.second );
244  }

307 double WindowLevel::fromWindowLevel(double val)
308 {
309  double valMin = m_widgetDynamicRangeMin;
310  double valMax = valMin + m_widgetDynamicRangeWidth;
311 
312  valMin = std::min(val, valMin);

355 void WindowLevel::onDynamicRangeSelectionChanged(QAction *action)
356 {
357  WindowLevelMinMaxType wl = this->getImageWindowMinMax();
358  double min = m_widgetDynamicRangeMin;
359  double max = m_widgetDynamicRangeWidth + min;
360  int index = action->data().toInt();
361  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
362 
363  switch (index)
364  {
365  case 0:
366  break;
367  case 1: // -1024; 1023
368  min = -1024;
369  max = 1023;
370  break;
371  case 2: // -100; 300
372  min = -100;
373  max = 300;
374  break;
375  case 3: // Fit Window/Level
376  min = std::min(wl.first, wl.second);
377  max = std::max(wl.first, wl.second);
378  break;
379  case 4: // Fit Image Range
380  ::fwComEd::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
381  break;
382  case 5: // Custom : TODO
383  break;
384  default:
385  SLM_ASSERT("Unknown range selector index", 0);
386  }
387 
388  this->setWidgetDynamicRange(min, max);
389  this->updateWidgetMinMax(wl.first, wl.second);
390 }

410 void WindowLevel::onToggleTF(bool squareTF)
411 {
412  bool usedGreyLevelTF = false;
413 
414  ::fwData::TransferFunction::sptr oldTF = this->getTransferFunction();
415  ::fwData::TransferFunction::sptr newTF;
416 
417  if( squareTF )
418  {
419  newTF = ::fwData::TransferFunction::New();
420  ::fwData::TransferFunction::TFColor color(1.,1.,1.,1.);
421  newTF->setName("SquareTF");
422  newTF->addTFColor(0.0, color);
423  newTF->addTFColor(1.0, color);
424  newTF->setIsClamped(true);
425  }
426  else
427  {
428  // test if service must use image grey level tf ( when another tf pool is defined )
429  if( m_useImageGreyLevelTF &&
430  !this->getTFSelectionFwID().empty() )
431  {
432  newTF = this->getImageGreyLevelTF();
433  usedGreyLevelTF = true;
434  }
435  else
436  {
437  newTF = ::fwData::TransferFunction::createDefaultTF();
438  }
439  }
440 
441  newTF->setWindow( oldTF->getWindow() );
442  newTF->setLevel( oldTF->getLevel() );
443 
444  this->swapCurrentTFAndNotify( newTF );
445 
446  if ( usedGreyLevelTF )
447  {
448  // Send signal
449  auto sig = newTF->signal< ::fwData::TransferFunction::WindowingModifiedSignalType >(
450  ::fwData::TransferFunction::s_WINDOWING_MODIFIED_SIG);
451  {
452  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdateTFWindowing));
453  sig->asyncEmit( newTF->getWindow(), newTF->getLevel());
454  }
455  }
456 }
