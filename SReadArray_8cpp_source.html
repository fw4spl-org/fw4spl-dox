40 void SReadArray::updating() throw( ::fwTools::Failed )
41 {
42  ::fwData::Array::sptr array = this->getObject< ::fwData::Array >();
43  ::fwData::mt::ObjectWriteLock writeLock(array);
44  SLM_ASSERT("No array.", array);
45 
46  // Initialize the array size and type
47  const int arraySize = 10;
48  ::fwData::Array::SizeType size(1, arraySize);
49  array->resize("uint32", size, 1, true);
50 
51  // Fill the array values
52  ::fwComEd::helper::Array arrayHelper(array);
53  unsigned int *buffer = static_cast< unsigned int* >( arrayHelper.getBuffer() );
54  for (unsigned int i = 0; i < arraySize; i++)
55  {
56  buffer[i] = i;
57  }
58 
59  // Notify that the array is modified
60  ::fwData::Object::ModifiedSignalType::sptr sig
61  = array->signal< ::fwData::Object::ModifiedSignalType>( ::fwData::Object::s_MODIFIED_SIG );
62  sig->asyncEmit();
63 }
