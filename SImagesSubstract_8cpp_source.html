86 void SImagesSubstract::updating() throw ( ::fwTools::Failed )
87 {
88  ::fwTools::Type REQUESTED_TYPE = ::fwTools::Type::create("int16");
89 
90  const std::string image1Name("image1");
91  const std::string image2Name("image2");
92  const std::string imageResultName("imageResult");
93 
94  ::fwData::Composite::sptr compositeVisu = this->getObject< ::fwData::Composite >();
95  ::fwData::Image::sptr image1 = ::fwData::Image::dynamicCast(::fwTools::fwID::getObject(image1Name));
96  ::fwData::Image::sptr image2 = ::fwData::Image::dynamicCast(::fwTools::fwID::getObject(image2Name));
97  ::fwData::Image::sptr imageResult = ::fwData::Image::New();
98 
99  OSLM_ASSERT("The object " << image1Name << " is not an image", image1);
100  OSLM_ASSERT("The object " << image2Name << " is not an image", image2);
101 
102  // Test if the both images have the same type and it is signed short.
103  bool isSameType =
104  ( ((image1->getDataArray())->getType() == (image1->getDataArray())->getType())&&
105  ((image1->getDataArray())->getType() == REQUESTED_TYPE));
106 
107  if(isSameType)
108  {
109  // test if the both images have the same size.
110  bool isSameSize = (image1->getSize() == image2->getSize());
111  if(isSameSize)
112  {
113  typedef itk::Image< ::boost::int16_t, 3 > ImageType;
114 
115  ImageType::Pointer itkImage1 = ::fwItkIO::itkImageFactory< ImageType >( image1 );
116  SLM_ASSERT("Unable to convert fwData::Image to itkImage", itkImage1);
117 
118  ImageType::Pointer itkImage2 = ::fwItkIO::itkImageFactory< ImageType >( image2 );
119  SLM_ASSERT("Unable to convert fwData::Image to itkImage", itkImage2);
120 
121  ImageType::Pointer output;
122 
123  //Create filter
124  typedef ::itk::SubtractImageFilter< ImageType, ImageType, ImageType > SubtractImageFilterType;
125  SubtractImageFilterType::Pointer filter;
126  filter = SubtractImageFilterType::New();
127  assert(filter);
128 
129  filter->SetInput1( itkImage1 );
130  filter->SetInput2( itkImage2 );
131  filter->Update();
132  output = filter->GetOutput();
133  assert(output->GetSource());
134  ::fwItkIO::dataImageFactory< ImageType >( output, imageResult, true );
135 
136  ::fwDataTools::helper::Composite compositeHelper(compositeVisu);
137  if ( compositeVisu->find(imageResultName) != compositeVisu->end() )
138  {
139  compositeHelper.swap(imageResultName, imageResult);
140  }
141  else
142  {
143  compositeHelper.add(imageResultName, imageResult);
144  }
145  compositeHelper.notify();
146  }
147  else
148  {
149  ::fwGui::dialog::MessageDialog::showMessageDialog("Warning",
150  "Both images must have the same size.",
151  ::fwGui::dialog::IMessageDialog::WARNING);
152  }
153  }
154  else
155  {
156  ::fwGui::dialog::MessageDialog::showMessageDialog("Warning",
157  "Both Images must have signed short as type.",
158  ::fwGui::dialog::IMessageDialog::WARNING);
159  }
160 }
