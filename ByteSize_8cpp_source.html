00196     symbols<char, ByteSize::SizeType> unit;
00197 
00198     unit.add
00199         ( "b"    , ByteSize::Bytes ) ( "byte", ByteSize::Bytes ) ( "bytes", ByteSize::Bytes )
00200         ( "kb"   , ByteSize::KB  )
00201         ( "mb"   , ByteSize::MB  )
00202         ( "gb"   , ByteSize::GB  )
00203         ( "tb"   , ByteSize::TB  )
00204         ( "pb"   , ByteSize::PB  )
00205         ( "k"  , ByteSize::KiB ) ( "kib"  , ByteSize::KiB )
00206         ( "m"  , ByteSize::MiB ) ( "mib"  , ByteSize::MiB )
00207         ( "g"  , ByteSize::GiB ) ( "gib"  , ByteSize::GiB )
00208         ( "t"  , ByteSize::TiB ) ( "tib"  , ByteSize::TiB )
00209         ( "p"  , ByteSize::PiB ) ( "pib"  , ByteSize::PiB )
00210         ;
00211 
00212 
00213     bool r = false;
00214     r = phrase_parse(first, last,
00215                      //  Begin grammar
00216                      (
00217                       ( *space >>
00218                        ( &((double_) >> no_case[-unit] >> *space >> eoi)
00219                          >> double_[ ref(floatSize) = _1 ] >> no_case[-unit[ ref(multiplier) = _1 ]] )
00220                        | ( &((ulong_long) >> no_case[-unit]  >> *space >> eoi)
00221                            >> ulong_long[ ref(intSize) = _1 ] >> no_case[-unit[ ref(multiplier) = _1 ]] )
00222                       )
00223                       >> *space >> eoi
00224                      ),
00225                      //  End grammar
00226 
00227                      space);
00228 
00229 
00230     if (!r || first != last) // fail if we did not get a full match
00231     {
00232         return false;
00233     }
00234 
00235     if (intSize != 0)
00236     {
00237         size = intSize * multiplier;
00238     }
00239     else if (floatSize != 0)
00240     {
00241         if(floatSize < 0)
00242         {
00243             FW_RAISE_EXCEPTION_MSG( ::fwMemory::exception::BadCast , "Bad size : " << floatSize << " < 0");
00244         }
00245 
00246         size = static_cast< ByteSize::SizeType >(floatSize * multiplier);
00247     }
00248     else
00249     {
00250         return false;
00251     }
