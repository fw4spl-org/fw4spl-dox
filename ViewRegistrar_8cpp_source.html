00060 void ViewRegistrar::initialize( ::fwRuntime::ConfigurationElement::sptr configuration)
00061 {
00062     OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be viewRegistrar",
00063             configuration->getName() == "registry");
00064 
00065     // find parent container
00066     std::vector < ConfigurationType > vectParent = configuration->find("parent");
00067     if(!vectParent.empty())
00068     {
00069         ConfigurationType parent = vectParent.at(0);
00070         SLM_ASSERT("<parent> tag must have wid attribute", parent->hasAttribute("wid"));
00071         m_parentWid = parent->getAttributeValue("wid");
00072     }
00073 
00074     // index represents associated container with position in subViews vector
00075     unsigned int index = 0;
00076     // initialize m_sids and m_wids map with configuration
00077     std::vector < ConfigurationType > vectViews = configuration->find("view");
00078     BOOST_FOREACH( ConfigurationType view, vectViews)
00079     {
00080         SLM_ASSERT("<view> tag must have sid or wid attribute",
00081                 view->hasAttribute("sid") || view->hasAttribute("wid"));
00082         if(view->hasAttribute("sid"))
00083         {
00084             bool start = false;
00085             if(view->hasAttribute("start"))
00086             {
00087                 std::string startValue = view->getAttributeValue("start");
00088                 SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
00089                         startValue == "yes" || startValue == "no");
00090                 start = (startValue=="yes");
00091             }
00092             std::string sid = view->getAttributeValue("sid");
00093             m_sids[sid] = SIDContainerMapType::mapped_type(index, start);
00094         }
00095         else if(view->hasAttribute("wid"))
00096         {
00097             std::string wid = view->getAttributeValue("wid");
00098             m_wids[wid] = index;
00099         }
00100         index++;
00101     }
00102 
00103     // find menuBar
00104     std::vector < ConfigurationType > vectmenuBar = configuration->find("menuBar");
00105     if(!vectmenuBar.empty())
00106     {
00107         ConfigurationType menuBarCfg = vectmenuBar.at(0);
00108         if (menuBarCfg->hasAttribute("sid"))
00109         {
00110             bool start = false;
00111             if (menuBarCfg->hasAttribute("start"))
00112             {
00113                 std::string startValue = menuBarCfg->getAttributeValue("start");
00114                 SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
00115                            startValue == "yes" || startValue == "no");
00116                 start = (startValue=="yes");
00117             }
00118             std::string sid = menuBarCfg->getAttributeValue("sid");
00119             m_menuBarSid =  std::make_pair( sid, start);
00120         }
00121     }
00122

00146 void ViewRegistrar::manage(std::vector< ::fwGui::container::fwContainer::sptr > subViews )
00147 {
00148     ::fwGui::container::fwContainer::sptr container;
00149     BOOST_FOREACH( SIDContainerMapType::value_type sid, m_sids)
00150     {
00151         OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!", sid.second.first < subViews.size());
00152         container = subViews.at( sid.second.first );
00153         ::fwGui::GuiRegistry::registerSIDContainer(sid.first, container);
00154         if(sid.second.second) //service is auto started?
00155         {
00156             OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
00157             ::fwServices::IService::sptr service = ::fwServices::get( sid.first ) ;
00158             OSLM_ASSERT("Service "<<sid.first <<" must be stopped.", service->isStopped() );
00159             SLM_TRACE("Starting " + sid.first);
00160             service->start();
00161         }
00162     }
00163 
00164     BOOST_FOREACH( WIDContainerMapType::value_type wid, m_wids)
00165     {
00166         OSLM_ASSERT("Container index "<< wid.second <<" is bigger than subViews size!", wid.second < subViews.size());
00167         container = subViews.at( wid.second );
00168         ::fwGui::GuiRegistry::registerWIDContainer(wid.first, container);
00169     }
00170 }
