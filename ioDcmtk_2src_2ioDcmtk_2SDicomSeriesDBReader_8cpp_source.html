95 ::fwMedData::SeriesDB::sptr SDicomSeriesDBReader::createSeriesDB(const ::boost::filesystem::path& dicomDir)
96 {
97  SLM_TRACE_FUNC();
98  ::fwDcmtkIO::SeriesDBReader::sptr myLoader = ::fwDcmtkIO::SeriesDBReader::New();
99  ::fwMedData::SeriesDB::sptr dummy = ::fwMedData::SeriesDB::New();
100  myLoader->setObject(dummy);
101  myLoader->setFolder(dicomDir);
102 
103  if(myLoader->isDicomDirAvailable())
104  {
105  ::fwGui::dialog::MessageDialog messageBox;
106  messageBox.setTitle("Dicomdir file");
107  messageBox.setMessage( "There is a dicomdir file in the root folder. "
108  "Would you like to use it for the reading process ?" );
109  messageBox.setIcon(::fwGui::dialog::IMessageDialog::QUESTION);
110  messageBox.addButton(::fwGui::dialog::IMessageDialog::YES_NO);
111  ::fwGui::dialog::IMessageDialog::Buttons button = messageBox.show();
112 
113  myLoader->setDicomdirActivated(button == ::fwGui::dialog::IMessageDialog::YES);
114  }
115 
116  try
117  {
118  ::fwGui::dialog::ProgressDialog progressMeterGUI("Loading Dicom Image");
119  myLoader->addHandler( progressMeterGUI );
120  myLoader->readDicomSeries();
121  }
122  catch (const std::exception& e)
123  {
124  std::stringstream ss;
125  ss << "Warning during loading : " << e.what();
126  ::fwGui::dialog::MessageDialog::showMessageDialog(
127  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
128  }
129  catch( ... )
130  {
131  ::fwGui::dialog::MessageDialog::showMessageDialog(
132  "Warning", "Warning during loading", ::fwGui::dialog::IMessageDialog::WARNING);
133  }
134 
135  return myLoader->getConcreteObject();
136 }

140 void SDicomSeriesDBReader::updating() throw(::fwTools::Failed)
141 {
142  SLM_TRACE_FUNC();
143  if( this->hasLocationDefined() )
144  {
145  ::fwMedData::SeriesDB::sptr seriesDB = createSeriesDB( this->getFolder() );
146 
147  if( seriesDB->size() > 0 )
148  {
149  // Retrieve dataStruct associated with this service
150  ::fwMedData::SeriesDB::sptr associatedSeriesDB = this->getObject< ::fwMedData::SeriesDB >();
151  SLM_ASSERT("associated SeriesDB not instanced", associatedSeriesDB);
152  associatedSeriesDB->shallowCopy( seriesDB );
153 
154  ::fwGui::Cursor cursor;
155  cursor.setCursor(::fwGui::ICursor::BUSY);
156  this->notificationOfDBUpdate();
157  cursor.setDefaultCursor();
158  }
159  else
160  {
161  ::fwGui::dialog::MessageDialog::showMessageDialog(
162  "Image Reader","This file can not be read. Retry with another file reader.",
163  ::fwGui::dialog::IMessageDialog::WARNING);
164  }
165  }
166 }

170 void SDicomSeriesDBReader::notificationOfDBUpdate()
171 {
172  SLM_TRACE_FUNC();
173  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
174  SLM_ASSERT("Unable to get seriesDB", seriesDB);
175 
176  ::fwMedData::SeriesDB::ContainerType addedSeries;
177  for( ::fwMedData::Series::sptr s: seriesDB->getContainer() )
178  {
179  addedSeries.push_back(s);
180  }
181 
182  auto sig = seriesDB->signal< ::fwMedData::SeriesDB::AddedSeriesSignalType >(
183  ::fwMedData::SeriesDB::s_ADDED_SERIES_SIG);
184  sig->asyncEmit(addedSeries);
185 }
