69 void SIOSelector::configuring() throw( ::fwTools::Failed )
70 {
71  SLM_TRACE_FUNC();
72 
73  bool vectorIsAlreadyCleared = false;
74 
75  // Config Elem
76  // <selection mode="include" />
77  // <addSelection service="::ioAtoms::SWriter" />
78 
79  ::fwRuntime::ConfigurationElementContainer::Iterator iter = this->m_configuration->begin();
80  for(; iter != this->m_configuration->end(); ++iter )
81  {
82  OSLM_INFO( "SIOSelector " << (*iter)->getName());
83 
84  if( (*iter)->getName() == "selection" )
85  {
86  SLM_ASSERT( "The xml element <selection> must have the attribute 'mode'.", (*iter)->hasAttribute("mode"));
87  std::string mode = (*iter)->getExistingAttributeValue("mode");
88  m_servicesAreExcluded = ( mode == "exclude" );
89  SLM_ASSERT( "The xml attribute <mode> must be 'include' (to add the selection to selector list ) or "
90  "'exclude' (to exclude the selection of the selector list).",
91  mode == "exclude" || mode == "include" );
92  OSLM_DEBUG( "mode => " << mode );
93  }
94 
95  if( (*iter)->getName() == "addSelection" )
96  {
97  if( !vectorIsAlreadyCleared )
98  {
99  vectorIsAlreadyCleared = true;
100  m_selectedServices.clear();
101  }
102  SLM_ASSERT( "The xml elemenet <addSelection> must have the attribute 'service'.",
103  (*iter)->hasAttribute("service"));
104  m_selectedServices.push_back( (*iter)->getExistingAttributeValue("service") );
105  OSLM_DEBUG( "add selection => " << (*iter)->getExistingAttributeValue("service") );
106  }
107 
108  if( (*iter)->getName() == "type" )
109  {
110  SLM_ASSERT( "The xml elemenet <type> must have the attribute 'mode'.", (*iter)->hasAttribute("mode"));
111  std::string mode = (*iter)->getExistingAttributeValue("mode");
112  SLM_ASSERT("The xml attribute <mode> must be 'reader' (to open file) or 'writer' (to write a new file).",
113  mode == "writer" || mode == "reader" );
114  m_mode = ( mode == "writer" ) ? WRITER_MODE : READER_MODE;
115  OSLM_DEBUG( "mode => " << mode );
116  }
117 
118  if( (*iter)->getName() == "config" )
119  {
120  SLM_ASSERT( "The xml element <config> must have the attribute 'id'.", (*iter)->hasAttribute("id"));
121  SLM_ASSERT( "The xml element <config> must have the attribute 'service'.",
122  (*iter)->hasAttribute("service"));
123  std::string configId = (*iter)->getExistingAttributeValue("id");
124  std::string configSrv = (*iter)->getExistingAttributeValue("service");
125  m_serviceToConfig[ configSrv ] = configId;
126  }
127 
128  }
129 
130  typedef std::vector < SPTR(::fwRuntime::ConfigurationElement) > ConfigurationElementContainer;
131  ConfigurationElementContainer inject = m_configuration->find("inject");
132 
133  if(!inject.empty())
134  {
135  m_inject = inject.at(0)->getValue();
136  }
137 
138 }

156 void SIOSelector::updating() throw( ::fwTools::Failed )
157 {
158  SLM_TRACE_FUNC();
159 
160  // Retrieve implementation of type ::io::IReader for this object
161  std::vector< std::string > availableExtensionsId;
162  if ( m_mode == READER_MODE )
163  {
164  availableExtensionsId =
165  ::fwServices::registry::ServiceFactory::getDefault()->getImplementationIdFromObjectAndType(
166  this->getObject()->getClassname(),"::io::IReader");
167  }
168  else // m_mode == WRITER_MODE
169  {
170  availableExtensionsId =
171  ::fwServices::registry::ServiceFactory::getDefault()->getImplementationIdFromObjectAndType(
172  this->getObject()->getClassname(),"::io::IWriter");
173  }
174 
175  // Filter available extensions and replace id by service description
176  std::vector< std::pair < std::string, std::string > > availableExtensionsMap;
177  std::vector< std::string > availableExtensionsSelector;
178 
179  for( const std::string &serviceId : availableExtensionsId )
180  {
181  bool serviceIsSelectedByUser =
182  std::find( m_selectedServices.begin(), m_selectedServices.end(),
183  serviceId ) != m_selectedServices.end();
184 
185  // Test if the service is considered here as available by users, if yes push in availableExtensionsSelector
186  // excluded mode => add services that are not selected by users
187  // included mode => add services selected by users
188  if( (m_servicesAreExcluded && !serviceIsSelectedByUser) ||
189  (!m_servicesAreExcluded && serviceIsSelectedByUser) )
190  {
191  // Add this service
192  std::string infoUser =
193  ::fwServices::registry::ServiceFactory::getDefault()->getServiceDescription(serviceId);
194 
195  std::map< std::string, std::string >::const_iterator iter = m_serviceToConfig.find( serviceId );
196  if ( iter != m_serviceToConfig.end() )
197  {
198  infoUser = ::fwServices::registry::ServiceConfig::getDefault()->getConfigDesc(iter->second);
199  }
200 
201  if (infoUser != "")
202  {
203  availableExtensionsMap.push_back( std::pair < std::string, std::string > (serviceId, infoUser) );
204  availableExtensionsSelector.push_back( infoUser );
205  }
206  else
207  {
208  availableExtensionsMap.push_back( std::pair < std::string, std::string > (serviceId, serviceId) );
209  availableExtensionsSelector.push_back( serviceId );
210  }
211  }
212  }
213 
214  // Sort available services (lexical string sort)
215  std::sort( availableExtensionsSelector.begin(), availableExtensionsSelector.end() );
216 
217  // Test if we have an extension
218  if ( !availableExtensionsMap.empty() )
219  {
220  std::string extensionId = availableExtensionsMap[0].first;
221  bool extensionSelectionIsCanceled = false;
222 
223  // Selection of extension when availableExtensions.size() > 1
224  if ( availableExtensionsSelector.size() > 1 )
225  {
226  ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
227 
228  if ( m_mode != READER_MODE )
229  {
230  selector->setTitle("Writer to use");
231  }
232  else
233  {
234  selector->setTitle("Reader to use");
235  }
236  selector->setSelections(availableExtensionsSelector);
237  std::string selection = selector->show();
238  if( !selection.empty() )
239  {
240  bool extensionIdFound = false;
241 
242  typedef std::pair < std::string, std::string > PairType;
243  for(PairType pair : availableExtensionsMap)
244  {
245  if (pair.second == selection )
246  {
247  extensionId = pair.first;
248  extensionIdFound = true;
249  }
250  }
251  OSLM_ASSERT("Problem to find the selected string.", extensionIdFound );
252  }
253  else
254  {
255  extensionSelectionIsCanceled = true;
256  }
257  }
258 
259  if ( !extensionSelectionIsCanceled )
260  {
261 
262  // Get Config
263  bool hasConfigForService = false;
264  ::fwRuntime::ConfigurationElement::csptr srvCfg;
265  if ( m_serviceToConfig.find( extensionId ) != m_serviceToConfig.end() )
266  {
267  hasConfigForService = true;
268  srvCfg = ::fwServices::registry::ServiceConfig::getDefault()->getServiceConfig(
269  m_serviceToConfig[extensionId], extensionId );
270  SLM_ASSERT(
271  "No service configuration of type ::fwServices::registry::ServiceConfig was found",
272  srvCfg );
273  }
274 
275  // Configure and start service
276  if ( m_mode == READER_MODE )
277  {
278  ::fwData::Object::sptr object;
279  if(m_inject.empty() || this->getObject()->getClassname().compare("::fwData::Composite") )
280  {
281  object = this->getObject< ::fwData::Object >();
282  }
283  else
284  {
285  ::fwServices::registry::ServiceFactory::sptr services =
286  ::fwServices::registry::ServiceFactory::getDefault();
287  std::string objType = services->getObjectImplementation(extensionId);
288  if(!objType.compare("::fwData::Object"))
289  {
290  object = this->getObject< ::fwData::Composite>();
291  }
292  else
293  {
294  object = ::fwData::factory::New(objType);
295  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite>();
296  ::fwComEd::helper::Composite helper(composite);
297  helper.add(m_inject, object);
298 
299  helper.notify();
300  }
301  }
302 
303  ::io::IReader::sptr reader = ::fwServices::add< ::io::IReader >( object, extensionId );
304  reader->setWorker(m_associatedWorker);
305 
306  if ( hasConfigForService )
307  {
308  reader->setConfiguration( ::fwRuntime::ConfigurationElement::constCast(srvCfg) );
309  reader->configure();
310  }
311 
312  auto jobCreatedSignal = reader->signal("jobCreated");
313  if(jobCreatedSignal)
314  {
315  jobCreatedSignal->connect(m_slotForwardJob);
316  }
317 
318  reader->start();
319  reader->configureWithIHM();
320 
321  ::fwGui::Cursor cursor;
322  cursor.setCursor(::fwGui::ICursor::BUSY);
323  reader->update();
324  cursor.setDefaultCursor();
325 
326  reader->stop();
327  ::fwServices::OSR::unregisterService(reader);
328  }
329  else
330  {
331  ::io::IWriter::sptr writer = ::fwServices::add< ::io::IWriter >( this->getObject(), extensionId );
332  writer->setWorker(m_associatedWorker);
333 
334  if ( hasConfigForService )
335  {
336  writer->setConfiguration( ::fwRuntime::ConfigurationElement::constCast(srvCfg) );
337  writer->configure();
338  }
339 
340  auto jobCreatedSignal = writer->signal("jobCreated");
341  if(jobCreatedSignal)
342  {
343  jobCreatedSignal->connect(m_slotForwardJob);
344  }
345 
346  writer->start();
347  writer->configureWithIHM();
348 
349  ::fwGui::Cursor cursor;
350  cursor.setCursor(::fwGui::ICursor::BUSY);
351  writer->update();
352  cursor.setDefaultCursor();
353 
354  writer->stop();
355  ::fwServices::OSR::unregisterService(writer);
356  }
357  }
358  }
359  else
360  {
361  SLM_WARN("SIOSelector::load : availableExtensions is empty.");
362  if ( m_mode == READER_MODE )
363  {
364  ::fwGui::dialog::MessageDialog messageBox;
365  messageBox.setTitle("Reader not found");
366  messageBox.setMessage( "There are not available readers for this data type." );
367  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
368  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
369  messageBox.show();
370  }
371  else // m_mode == WRITER_MODE
372  {
373  ::fwGui::dialog::MessageDialog messageBox;
374  messageBox.setTitle("Writer not found");
375  messageBox.setMessage( "There are not available writers for this data type." );
376  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
377  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
378  messageBox.show();
379  }
380  }
381 }
