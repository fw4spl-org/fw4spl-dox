103  void fillPickList()
104  {

118  virtual void Execute( vtkObject* caller, unsigned long eventId, void*)
119  {
120  int pos[2];
121  m_service->getInteractor()->GetLastEventPosition(pos);
122  OSLM_TRACE( "EventId: " << eventId);
123 
124  if ( eventId == vtkCommand::RightButtonPressEvent )
125  {
126  std::copy(pos, pos+1, m_lastPos);
127  m_display[0] = pos[0];
128  m_display[1] = pos[1];
129 
130  this->fillPickList();
131  if (m_picker->Pick( m_display, m_service->getRenderer() ) )
132  {
133  if(getSelectedPoint())
134  {
135  SetAbortFlag(1);
136  }
137  else
138  {
139  m_pickedPoint.reset();
140  m_pickedPointList.reset();
141  }
142  }
143  }
144  else if ( (eventId == vtkCommand::RightButtonReleaseEvent ) && !m_pickedPoint.expired() &&
145  !m_pickedPointList.expired() && std::equal(pos, pos+1, m_lastPos) )
146  {
147  ::fwData::Image::sptr image = m_service->getObject< ::fwData::Image >();
148  ::fwData::PointList::PointListContainer::iterator itr = std::find(
149  m_pickedPointList.lock()->getRefPoints().begin(),
150  m_pickedPointList.lock()->getRefPoints().end(), m_pickedPoint.lock() );
151  if(itr != m_pickedPointList.lock()->getRefPoints().end())
152  {
153  ::fwData::Point::sptr point = *itr;
154  m_pickedPointList.lock()->getRefPoints().erase(itr);
155  notifyRemoveLandMark(image, m_service, point);
156  }
157  }
158  }
159  bool getSelectedPoint()
160  {
161  bool isFind = false;
162  vtkPropCollection* propc = m_picker->GetActors();
163  vtkProp* prop;
164 
165  propc->InitTraversal();
166  while ( (prop = propc->GetNextProp()) )
167  {
168  m_pickedPoint = ::fwData::Point::dynamicCast(m_service->getAssociatedObject(prop,2));
169  m_pickedPointList = ::fwData::PointList::dynamicCast(m_service->getAssociatedObject(prop,1));
170 
171  if( !m_pickedPoint.expired() && !m_pickedPointList.expired() )
172  {
173  ::fwData::PointList::PointListContainer::iterator itr = std::find(
174  m_pickedPointList.lock()->getRefPoints().begin(),
175  m_pickedPointList.lock()->getRefPoints().end(), m_pickedPoint.lock());
176  if(itr != m_pickedPointList.lock()->getRefPoints().end() )
177  {
178  isFind = true;
179  break;
180  }
181  }
182  }
183  return isFind;
184  }

243 void ImageLandmarks::doUpdate() throw(fwTools::Failed)
244 {
245  // get PointList in image Field then install distance service if required
246  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
247 
248  ::fwData::PointList::sptr landmarks;
249  bool isShown;
250  landmarks = image->getField< ::fwData::PointList >( ::fwDataTools::fieldHelper::Image::m_imageLandmarksId );
251  isShown = image->getField("ShowLandmarks", ::fwData::Boolean::New(true))->value();
252 
253  if (!isShown || !landmarks || m_needSubservicesDeletion)
254  {
255  this->unregisterServices();
256  m_needSubservicesDeletion = false;
257  }
258 
259  if( isShown && landmarks )
260  {
261  if ( !landmarks->getPoints().empty() )
262  {
263  ::fwRenderVTK::IVtkAdaptorService::sptr servicePointList;
264  servicePointList = ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService >( landmarks,
265  "::visuVTKAdaptor::PointList");
266  SLM_ASSERT("servicePointList not instanced", servicePointList);
267 
268  servicePointList->setPickerId( this->getPickerId() );
269  servicePointList->setRenderService( this->getRenderService() );
270  servicePointList->setAutoRender( this->getAutoRender() );
271  servicePointList->start();
272 
273  this->registerService( servicePointList );
274 
275 
276  for( ::fwData::Point::sptr point : landmarks->getRefPoints() )
277  {
278  ::fwRenderVTK::IVtkAdaptorService::sptr serviceLabel;
279  serviceLabel = ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService >(point,
280  "::visuVTKAdaptor::PointLabel");
281  SLM_ASSERT("serviceLabel not instanced", serviceLabel);
282  serviceLabel->setRenderService( this->getRenderService() );
283  serviceLabel->setAutoRender( this->getAutoRender() );
284  serviceLabel->start();
285  this->registerService( serviceLabel );
286  }
287  }
288  }
289  this->setVtkPipelineModified();
290 }
