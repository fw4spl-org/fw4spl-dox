92 
93  ::fwData::location::Folder::sptr result;

95  if (result)
96  {
97  _sDefaultPath = result->getFolder();
98  this->setFolder( _sDefaultPath );
99  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
100  }
101  else
102  {
103  this->clearLocations();
104  }
105 
106  // Select filter
107  if(!m_filterSelectorSrvConfig.empty())
108  {
109  // Get the config
110  ::fwRuntime::ConfigurationElement::csptr filterSelectorConfig;
111  filterSelectorConfig = ::fwServices::registry::ServiceConfig::getDefault()->getServiceConfig(
112  m_filterSelectorSrvConfig, "::ioDicom::SFilterSelectorDialog");
113 
114  SLM_ASSERT("Sorry, there is no service configuration "
115  << m_filterSelectorSrvConfig
116  << " for ::ioDicom::SFilterSelectorDialog", filterSelectorConfig);
117 
118  // Init and execute the service
119  ::fwServices::IService::sptr filterSelectorSrv;
120  ::fwData::String::sptr key = ::fwData::String::New();
121  filterSelectorSrv = ::fwServices::add(key,
122  "::gui::editor::IDialogEditor",
123  "::ioDicom::SFilterSelectorDialog");
124  filterSelectorSrv->setConfiguration( ::fwRuntime::ConfigurationElement::constCast(filterSelectorConfig) );
125  filterSelectorSrv->configure();
126  filterSelectorSrv->start();
127  filterSelectorSrv->update();
128  filterSelectorSrv->stop();
129  ::fwServices::OSR::unregisterService( filterSelectorSrv );
130 
131  m_filterType = key->getValue();
132 
133  }

138 void SSeriesDBReader::configuring() throw (fwTools::Failed)
139 {
140  ::io::IReader::configuring();
141 
142  // Use filter selector
143  ::fwRuntime::ConfigurationElement::sptr selectorConfig =
144  m_configuration->findConfigurationElement("FilterSelectorSrvConfig");
145  if(selectorConfig)
146  {
147  SLM_ASSERT("Missing 'name' attribute", selectorConfig->hasAttribute("name"));
148  m_filterSelectorSrvConfig = selectorConfig->getAttributeValue("name");
149  }
150 
151  // Get SOP Class selection
152  const ::fwRuntime::ConfigurationElement::sptr sopClassSelection =
153  m_configuration->findConfigurationElement("SOPClassSelection");
154  if(sopClassSelection)
155  {
156  ::fwRuntime::ConfigurationElementContainer sopClassElements =
157  sopClassSelection->findAllConfigurationElement("SOPClass");
158 
159  for(::fwRuntime::ConfigurationElementContainer::Iterator it = sopClassElements.begin();
160  it != sopClassElements.end(); ++it)
161  {
162  const ::fwRuntime::ConfigurationElement::AttributePair attributePair = (*it)->getSafeAttributeValue("uid");
163  if(attributePair.first)
164  {
165  SLM_TRACE("New SOP class supported : " + attributePair.second);
166  m_supportedSOPClassSelection.push_back(attributePair.second);
167  }
168  }
169  }
170 
171  // Set filter
172  ::fwRuntime::ConfigurationElement::sptr config = m_configuration->findConfigurationElement("config");
173  if(config)
174  {
175  m_filterType = config->getAttributeValue("filterType");
176  }
177 
178  // Show log dialog
179  ::fwRuntime::ConfigurationElement::sptr logDialog = m_configuration->findConfigurationElement("showLogDialog");
180  if(logDialog)
181  {
182  std::string logDialogStr = logDialog->getValue();
183  SLM_ASSERT("<showLogDialog> value must be 'yes' or 'no'", logDialogStr == "yes" || logDialogStr == "no");
184  m_showLogDialog = (logDialogStr == "yes");
185  }
186 
187  // Enable buffer rotation
188  ::fwRuntime::ConfigurationElement::sptr bufRot = m_configuration->findConfigurationElement("enableBufferRotation");
189  if(bufRot)
190  {
191  std::string bufRotStr = bufRot->getValue();
192  SLM_ASSERT("<enableBufferRotation> value must be 'yes' or 'no'", bufRotStr == "yes" || bufRotStr == "no");
193  m_enableBufferRotation = (bufRotStr == "yes");
194  }
195 
196  // Enable dicomdir
197  ::fwRuntime::ConfigurationElement::sptr dicomDir = m_configuration->findConfigurationElement("dicomdirSupport");
198  if(dicomDir)
199  {
200  std::string dicomDirStr = dicomDir->getValue();
201  SLM_ASSERT("<dicomdirSupport> value must be 'always' or 'never' or 'user_selection'",
202  dicomDirStr == "always" || dicomDirStr == "never" || dicomDirStr == "user_selection");
203 
204  if(dicomDirStr == "always")
205  {
206  m_dicomDirSupport = ALWAYS;
207  }
208  else if(dicomDirStr == "never")
209  {
210  m_dicomDirSupport = NEVER;
211  }
212  else if(dicomDirStr == "user_selection")
213  {
214  m_dicomDirSupport = USER_SELECTION;
215  }
216  }
217 }

249 ::fwMedData::SeriesDB::sptr SSeriesDBReader::createSeriesDB( const ::boost::filesystem::path& dicomDir)
250 {
251  SLM_TRACE_FUNC();
252  ::fwGdcmIO::reader::SeriesDB::sptr reader = ::fwGdcmIO::reader::SeriesDB::New();
253  ::fwMedData::SeriesDB::sptr dummy = ::fwMedData::SeriesDB::New();
254  reader->setObject(dummy);
255  reader->setFolder(dicomDir);
256  reader->setDicomFilterType(m_filterType);
257  reader->setBufferRotationEnabled(m_enableBufferRotation);
258  reader->setSupportedSOPClassContainer(m_supportedSOPClassSelection);
259  auto job = reader->getJob();
260  m_sigJobCreated->emit(job);
261 
262  if(m_dicomDirSupport == USER_SELECTION && reader->isDicomDirAvailable())
263  {
264  ::fwGui::dialog::MessageDialog messageBox;
265  messageBox.setTitle("Dicomdir file");
266  messageBox.setMessage( "There is a dicomdir file in the root folder. "
267  "Would you like to use it for the reading process ?" );
268  messageBox.setIcon(::fwGui::dialog::IMessageDialog::QUESTION);
269  messageBox.addButton(::fwGui::dialog::IMessageDialog::YES_NO);
270  ::fwGui::dialog::IMessageDialog::Buttons button = messageBox.show();
271 
272  reader->setDicomdirActivated(button == ::fwGui::dialog::IMessageDialog::YES);
273  }
274  else if(m_dicomDirSupport == ALWAYS)
275  {
276  reader->setDicomdirActivated(true);
277  }
278  else //m_dicomDirSupport == NEVER
279  {
280  reader->setDicomdirActivated(false);
281  }
282 
283  try
284  {
285  reader->read();
286 
287  // Retrieve logger
288  ::fwLog::Logger::sptr logger = reader->getLogger();
289  logger->sort();
290 
291  // Set default cursor
292  ::fwGui::Cursor cursor;
293  cursor.setDefaultCursor();
294 
295  // Display logger dialog if enabled
296  if(m_showLogDialog && !logger->empty())
297  {
298 
299  std::stringstream ss;
300  if(dummy->size() > 1)
301  {
302  ss << "The reading process is over : <b>" << dummy->size() << " series</b> have been found. "
303  "<br>Please verify the log report to be informed of the potential errors.";
304  }
305  else
306  {
307  ss << "The reading process is over : <b>" << dummy->size() << " series</b> has been found. "
308  "<br>Please verify the log report to be informed of the potential errors.";
309  }
310 
311  bool result = false;
312  if(!job->cancelRequested())
313  {
314  result = ::fwGui::dialog::LoggerDialog::showLoggerDialog("Reading process over", ss.str(), logger);
315  }
316 
317  // If the user cancel the reading process we delete the loaded series
318  if(!result || job->cancelRequested())
319  {
320  ::fwMedDataTools::helper::SeriesDB sDBhelper(dummy);
321  sDBhelper.clear();
322  }
323  }
324  }
325  catch (const std::exception& e)
326  {
327  std::stringstream ss;
328  ss << "Warning during loading : " << e.what();
329  ::fwGui::dialog::MessageDialog::showMessageDialog(
330  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
331  }
332  catch( ... )
333  {
334  ::fwGui::dialog::MessageDialog::showMessageDialog(
335  "Warning", "Warning during loading", ::fwGui::dialog::IMessageDialog::WARNING);
336  }
337 
338  return dummy;
339 }

343 void SSeriesDBReader::updating() throw(::fwTools::Failed)
344 {
345  SLM_TRACE_FUNC();
346  if( this->hasLocationDefined() )
347  {
348  ::fwMedData::SeriesDB::sptr localSeriesDB = this->createSeriesDB(this->getFolder());
349 
350  if( !localSeriesDB->empty() )
351  {
352  // Retrieve dataStruct associated with this service
353  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
354  // seriesDB->shallowCopy( localSeriesDB ) ;
355 
356  ::fwMedDataTools::helper::SeriesDB sDBhelper(seriesDB);
357 
358  ::fwData::mt::ObjectWriteLock lock(seriesDB);
359  sDBhelper.merge(localSeriesDB);
360  sDBhelper.notify();
361  }
362  }
