44 void MenuLayoutManager::createLayout( ::fwGui::container::fwMenu::sptr parent )
45 {
46  SLM_TRACE_FUNC();
47 
48  m_parent = ::fwGuiQt::container::QtMenuContainer::dynamicCast(parent);
49  SLM_ASSERT("dynamicCast fwMenu to QtMenuContainer failed", m_parent);
50 
51  QMenu* menu = m_parent->getQtMenu();
52 
53  QActionGroup * actionGroup = 0;
54  unsigned int menuItemIndex = 0;
55  for ( ::fwGui::layoutManager::IMenuLayoutManager::ActionInfo actionInfo : m_actionInfo)
56  {
57  ::fwGuiQt::container::QtMenuItemContainer::sptr menuItem = ::fwGuiQt::container::QtMenuItemContainer::New();
58 
59  QAction *action = menu->addAction( QString::fromStdString(actionInfo.m_name) );
60 
61  action->setSeparator(actionInfo.m_isSeparator);
62 
63  if (!actionInfo.m_icon.empty())
64  {
65  QIcon icon(QString::fromStdString(actionInfo.m_icon));
66  action->setIcon(icon);
67  }
68  if (actionInfo.m_type == ::fwGui::layoutManager::IMenuLayoutManager::QUIT)
69  {
70  action->setMenuRole(QAction::QuitRole);
71  }
72  else if (actionInfo.m_type == ::fwGui::layoutManager::IMenuLayoutManager::ABOUT)
73  {
74  action->setMenuRole(QAction::AboutRole);
75  }
76  else
77  {
78  action->setMenuRole(QAction::NoRole);
79  }
80 
81  action->setCheckable(actionInfo.m_isCheckable || actionInfo.m_isRadio);
82 
83  if (actionInfo.m_isRadio)
84  {
85  if (!actionGroup)
86  {
87  actionGroup = new QActionGroup(menu);
88  }
89  actionGroup->addAction(action);
90  }
91 
92  // create shortcut
93  if( !actionInfo.m_shortcut.empty() )
94  {
95  action->setShortcut(QKeySequence(QString::fromStdString(actionInfo.m_shortcut)));
96  }
97 
98  if (actionInfo.m_isMenu)
99  {
100  ::fwGuiQt::container::QtMenuContainer::sptr menu = ::fwGuiQt::container::QtMenuContainer::New();
101  QMenu* qtMenu = new QMenu();
102  menu->setQtMenu(qtMenu);
103  action->setMenu(qtMenu);
104  m_menus.push_back(menu);
105  }
106 
107  menuItem->setQtMenuItem(action);
108 
109  if(!actionInfo.m_isSeparator && !actionInfo.m_isMenu )
110  {
111  m_menuItems.push_back(menuItem);
112  OSLM_ASSERT("No callback found for menu" << actionInfo.m_name, menuItemIndex < m_callbacks.size());
113  ::fwGui::IMenuItemCallback::sptr callback = m_callbacks.at(menuItemIndex);
114 
115  ::fwGuiQt::ActionCallback::sptr qtCallback = ::fwGuiQt::ActionCallback::dynamicCast(callback);
116  SLM_ASSERT("dynamicCast IMenuItemCallback to ActionCallback failed", qtCallback);
117 
118  QObject::connect( action, SIGNAL(triggered(bool)), qtCallback.get(), SLOT(executeQt(bool)));
119  QObject::connect( action, SIGNAL(toggled(bool)), qtCallback.get(), SLOT(checkQt(bool)));
120  menuItemIndex++;
121  }
122  else
123  {
124  actionGroup = 0;
125  }
126  }
127 }
