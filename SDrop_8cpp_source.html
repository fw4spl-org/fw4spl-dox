51 void SDrop::addObject( std::string uid )
52 {
53  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
54 
55  ::fwData::Object::sptr object = ::fwData::Object::dynamicCast(::fwTools::UUID::get(uid));
56  if(object)
57  {
58  ::fwComEd::helper::Composite helper( this->getObject< ::fwData::Composite >() );
59  helper.clear();
60  if(object->isA("::fwData::Image"))
61  {
62  helper.add("image", object);
63  }
64  else if(object->isA("::fwData::Mesh"))
65  {
66  helper.add("mesh", object);
67  }
68  else if(object->isA("::fwData::Reconstruction"))
69  {
70  helper.add("reconstruction", object);
71  }
72  else if(object->isA("::fwData::Resection"))
73  {
74  helper.add("resection", object);
75  }
76  else if(object->isA("::fwData::ResectionDB"))
77  {
78  helper.add("resectionDB", object);
79  }
80  else if(object->isA("::fwData::Plane"))
81  {
82  helper.add("plane", object);
83  }
84  else if(object->isA("::fwData::PlaneList"))
85  {
86  helper.add("planeList", object);
87  }
88  helper.notify();
89  }
90 }
