66 void Image::readImagePlaneModule()
67 {
68  // Retrieve GDCM image
69  SPTR(::gdcm::ImageReader) imageReader = std::static_pointer_cast< ::gdcm::ImageReader >(m_reader);
70  const ::gdcm::Image& gdcmImage = imageReader->GetImage();
71 
72  // Retrieve dataset
73  const ::gdcm::DataSet& dataset = m_reader->GetFile().GetDataSet();
74 
75  // Image Position (Patient) - Type 1
76  const double* gdcmOrigin = gdcmImage.GetOrigin();
77  ::fwData::Image::OriginType origin(3,0);
78  if ( gdcmOrigin != 0 )
79  {
80  std::copy( gdcmOrigin, gdcmOrigin+3, origin.begin() );
81  }
82  m_object->setOrigin(origin);
83 
84  // Pixel Spacing - Type 1
85  // Image dimension
86  const unsigned int dimension = gdcmImage.GetNumberOfDimensions();
87  OSLM_TRACE("Image's dimension : " << dimension);
88 
89  // Image's spacing
90  const double* gdcmSpacing = gdcmImage.GetSpacing();
91  ::fwData::Image::SpacingType spacing (3,1);
92  if ( gdcmSpacing != 0 )
93  {
94  std::copy( gdcmSpacing, gdcmSpacing+dimension, spacing.begin() );
95  }
96 
97  // Compute Z image spacing when extra information is required
98  std::string sliceThickness = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue<0x0018,0x0050>(dataset);
99  std::string spacingBetweenSlices = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue<0x0018,0x0088>(dataset);
100 
101  if(m_dicomSeries->hasComputedValues("SliceThickness"))
102  {
103  spacing[2] = ::boost::lexical_cast< double >(m_dicomSeries->getComputedTagValues().at("SliceThickness"));
104  std::stringstream ss;
105  ss << "Voxel depth has been computed using the image slice positions. The computed value is " << spacing[2] <<
106  " instead of " << sliceThickness << " for the SliceThickness tag.";
107  m_logger->warning(ss.str());
108  }
109  else if(!sliceThickness.empty())
110  {
111  spacing[2] = ::boost::lexical_cast< double >(sliceThickness);
112  std::stringstream ss;
113  ss << "Voxel depth has been retrieved using the SliceThickness tag value. The retrieved value is " <<
114  spacing[2] << ".";
115  m_logger->information(ss.str());
116  }
117  else if(!spacingBetweenSlices.empty())
118  {
119  spacing[2] = ::boost::lexical_cast< double >(spacingBetweenSlices);
120  std::stringstream ss;
121  ss << "Voxel depth has been retrieved using the SpacingBetweenSlices tag value. The retrieved value is " <<
122  spacing[2] << ".";
123  m_logger->information(ss.str());
124  }
125 
126  OSLM_TRACE("Image's spacing : "<<spacing[0]<<"x"<<spacing[1]<<"x"<<spacing[2]);
127  m_object->setSpacing( spacing );
128 
129 }

184 void Image::readImagePixelModule()
185 {
186  // Retrieve GDCM image
187  SPTR(::gdcm::ImageReader) imageReader = std::static_pointer_cast< ::gdcm::ImageReader >(m_reader);
188  const ::gdcm::Image& gdcmImage = imageReader->GetImage();
189 
190  // Retrieve dataset
191  const ::gdcm::DataSet& dataset = m_reader->GetFile().GetDataSet();
192 
193  // Retrieve image information before processing the rescaling
194  ::gdcm::PixelFormat pixelFormat = ::gdcm::ImageHelper::GetPixelFormatValue(imageReader->GetFile());
195 
196  // Retrieve rescale intercept/slope values
197  std::vector<double> rescale = getRescaleInterceptSlopeValue(imageReader.get());
198  double rescaleIntercept = rescale[0];
199  double rescaleSlope = rescale[1];
200 
201  unsigned short samplesPerPixel = pixelFormat.GetSamplesPerPixel();
202  unsigned short bitsAllocated = pixelFormat.GetBitsAllocated();
203  unsigned short bitsStored = pixelFormat.GetBitsStored();
204  unsigned short highBit = pixelFormat.GetHighBit();
205  unsigned short pixelRepresentation = pixelFormat.GetPixelRepresentation();
206 
207  // Compute final image type
208  ::fwDicomTools::Image imageHelper(
209  samplesPerPixel,bitsAllocated,bitsStored, highBit, pixelRepresentation, rescaleSlope, rescaleIntercept);
210  ::fwTools::Type imageType = imageHelper.findImageTypeFromMinMaxValues();
211  m_object->setType(imageType);
212  ::gdcm::PixelFormat targetPixelFormat = ::fwGdcmIO::helper::DicomData::getPixelType(m_object);
213 
214  // Compute number of components
215  std::string photometricInterpretation = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue<0x0028,0x0004>(dataset);
216  std::string pixelPresentation = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue<0x0008,0x9205>(dataset);
217 
218  if(photometricInterpretation == "MONOCHROME2")
219  {
220  m_object->setNumberOfComponents(1);
221  }
222  else if(photometricInterpretation == "RGB" || photometricInterpretation == "YBR")
223  {
224  m_object->setNumberOfComponents(3);
225  }
226  else if(photometricInterpretation == "ARGB" || photometricInterpretation == "CMYK")
227  {
228  m_object->setNumberOfComponents(4);
229  }
230  else if(photometricInterpretation == "PALETTE COLOR" || pixelPresentation == "COLOR")
231  {
232  m_object->setNumberOfComponents(3);
233  }
234  else
235  {
236  std::stringstream ss;
237  ss << "The photometric interpretation \"" + photometricInterpretation + "\" is not supported.";
238  throw ::fwGdcmIO::exception::Failed(ss.str());
239  }
240 
241  // Retrieve image dimensions
242  std::vector<unsigned int> dimensions = ::gdcm::ImageHelper::GetDimensionsValue(imageReader->GetFile());
243 
244  // Compute real image size (we assume every instance has the same number of
245  // slices (1 for CT and MR, may be more for enhanced CT and MR)
246  const unsigned long frameBufferSize = gdcmImage.GetBufferLength();
247  const unsigned long depth = frameBufferSize / (dimensions[0] * dimensions[1] * (bitsAllocated/8));
248  dimensions[2] = static_cast<unsigned int>(m_dicomSeries->getLocalDicomPaths().size() * depth);
249  m_object->setSize(::boost::assign::list_of(dimensions[0])(dimensions[1])(dimensions[2]));
250 
251  OSLM_TRACE("Image dimensions : [" << dimensions[0] << "," <<dimensions[1] << "," << dimensions[2] << "]");
252 
253  const unsigned long imageBufferSize = dimensions[0] * dimensions[1] * dimensions[2] * (bitsAllocated/8);
254  const unsigned long newImageBufferSize = dimensions[0] * dimensions[1] * dimensions[2] *
255  (targetPixelFormat.GetBitsAllocated()/8);
256 
257  OSLM_TRACE("Image buffer size : " << imageBufferSize);
258  OSLM_TRACE("New image buffer size : " << newImageBufferSize);
259 
260  // Let's read the image buffer
261  bool performRescale = (photometricInterpretation != "PALETTE COLOR" && pixelPresentation != "COLOR");
262  char* imageBuffer = this->readImageBuffer(dimensions, bitsAllocated, targetPixelFormat.GetBitsAllocated(),
263  performRescale);
264 
265 
266  // Correct image buffer according to the image orientation
267  if(!m_cancelled && m_enableBufferRotation)
268  {
269  imageBuffer = this->correctImageOrientation(imageBuffer, dimensions,
270  (performRescale) ? targetPixelFormat.GetBitsAllocated() : bitsAllocated);
271  }
272 
273  // Apply lookup table if required
274  if(!m_cancelled && (photometricInterpretation == "PALETTE COLOR" || pixelPresentation == "COLOR"))
275  {
276  try
277  {
278  // Create new buffer
279  char* coloredBuffer = 0;
280  coloredBuffer = new char[newImageBufferSize*3];
281 
282  // Apply lookup
283  gdcmImage.GetLUT().Decode(coloredBuffer, newImageBufferSize*3, imageBuffer, imageBufferSize);
284 
285  // Swap buffers
286  delete[] imageBuffer;
287  imageBuffer = coloredBuffer;
288  }
289  catch (...)
290  {
291  throw ::fwGdcmIO::exception::Failed("There is not enough memory available to open this image.");
292  }
293  }
294 
295  // Set image buffer
296  ::fwData::Array::sptr array = m_object->getDataArray();
297  ::fwDataTools::helper::Array helper(array);
298  helper.setBuffer(imageBuffer, true, m_object->getType(), m_object->getSize(), m_object->getNumberOfComponents());
299 
300 }

304 char* Image::readImageBuffer(const std::vector<unsigned int>& dimensions, unsigned short bitsAllocated,
305  unsigned short newBitsAllocated, bool performRescale) throw(::fwGdcmIO::exception::Failed)
306 {
307  // Retrieve GDCM image
308  SPTR(::gdcm::ImageReader) imageReader = std::static_pointer_cast< ::gdcm::ImageReader >(m_reader);
309  const ::gdcm::Image& gdcmFirstImage = imageReader->GetImage();
310 
311  // Path container
312  ::fwMedData::DicomSeries::DicomPathContainerType pathContainer = m_dicomSeries->getLocalDicomPaths();
313 
314  // Raw buffer for all frames
315  char* frameBuffer;
316  char* imageBuffer;
317  const unsigned long frameBufferSize = gdcmFirstImage.GetBufferLength();
318  const unsigned long newFrameBufferSize = frameBufferSize * (newBitsAllocated/bitsAllocated);
319  const unsigned long imageBufferSize = dimensions.at(0) * dimensions.at(1) * dimensions.at(2) *
320  ((performRescale ? newBitsAllocated : bitsAllocated)/8);
321 
322  // Allocate raw buffer
323  try
324  {
325  frameBuffer = new char[frameBufferSize];
326  imageBuffer = new char[imageBufferSize];
327  }
328  catch (...)
329  {
330  throw ::fwGdcmIO::exception::Failed("There is not enough memory available to open this image.");
331  }
332 
333  // Read every frames
334  unsigned int frameNumber = 0;
335  for(::fwMedData::DicomSeries::DicomPathContainerType::value_type v: pathContainer)
336  {
337  // Get filename
338  const std::string& filename = v.second.string();
339 
340  // Read a frame
341  ::gdcm::ImageReader frameReader;
342  frameReader.SetFileName( filename.c_str() );
343  if ( frameReader.Read() )
344  {
345  const ::gdcm::Image& gdcmImage = frameReader.GetImage();
346 
347  // Check frame buffer size
348  if(frameBufferSize != gdcmImage.GetBufferLength())
349  {
350  throw ::fwGdcmIO::exception::Failed("The frame buffer does not have the expected size : " + filename);
351  }
352 
353  // Get raw buffer and set it in the image buffer
354  if ( !gdcmImage.GetBuffer( frameBuffer ) )
355  {
356  throw ::fwGdcmIO::exception::Failed("Failed to get a frame buffer");
357  }
358  }
359  else
360  {
361  std::stringstream ss;
362  ss << "Reading error on frame : " << frameNumber;
363  throw ::fwGdcmIO::exception::Failed(ss.str());
364  }
365 
366 
367  // Rescale
368  if(performRescale)
369  {
370  // Retrieve rescale intercept/slope values
371  std::vector<double> rescale = getRescaleInterceptSlopeValue(&frameReader);
372  double rescaleIntercept = rescale[0];
373  double rescaleSlope = rescale[1];
374 
375  // Retrieve image information before processing the rescaling
376  ::gdcm::PixelFormat pixelFormat =
377  ::gdcm::ImageHelper::GetPixelFormatValue(frameReader.GetFile());
378  ::gdcm::PixelFormat::ScalarType scalarType = pixelFormat.GetScalarType();
379  ::gdcm::PixelFormat targetPixelFormat = ::fwGdcmIO::helper::DicomData::getPixelType(m_object);
380 
381  // Create rescaler
382  ::gdcm::Rescaler rescaler;
383  rescaler.SetIntercept(rescaleIntercept);
384  rescaler.SetSlope(rescaleSlope);
385  rescaler.SetPixelFormat(scalarType);
386  rescaler.SetTargetPixelType(targetPixelFormat.GetScalarType());
387  rescaler.SetUseTargetPixelType(true);
388 
389  // Rescale the image
390  rescaler.Rescale(imageBuffer + (frameNumber * newFrameBufferSize), frameBuffer, frameBufferSize);
391  }
392  else
393  {
394  // Copy bytes
395  memcpy(imageBuffer + frameNumber * frameBufferSize, frameBuffer, frameBufferSize);
396  }
397 
398  // Reference SOP Instance UID in dicomInstance for SR reading
399  const ::gdcm::DataSet& gdcmDatasetRoot = frameReader.GetFile().GetDataSet();
400  const std::string sopInstanceUID = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue<0x0008,0x0018>(
401  gdcmDatasetRoot);
402  if(!sopInstanceUID.empty())
403  {
404  m_instance->getRefSOPInstanceUIDContainer().push_back(sopInstanceUID);
405  }
406  else
407  {
408  m_logger->warning("A frame with an undefined SOP instance UID has been detected.");
409  }
410 
411  // Next frame
412  ++frameNumber;
413 
414  unsigned int progress =
415  static_cast<unsigned int>(18 + (frameNumber*100/static_cast<double>(pathContainer.size())) * 0.6);
416  m_progressCallback(progress);
417 
418  if(m_cancelled)
419  {
420  break;
421  }
422  }
423 
424  // Delete frame buffer
425  delete[] frameBuffer;
426 
427  return imageBuffer;
428 }

432 char* Image::correctImageOrientation(char* buffer, const std::vector<unsigned int>& dimensions,
433  unsigned short bitsAllocated)
434 {
435  char* result = buffer;
436 
437  // Retrieve GDCM image
438  SPTR(::gdcm::ImageReader) imageReader = std::static_pointer_cast< ::gdcm::ImageReader >(m_reader);
439  const ::gdcm::Image& gdcmImage = imageReader->GetImage();
440 
441  // Retrieve image orientation
442  const double* directionCosines = gdcmImage.GetDirectionCosines();
443 
444  // Compute U vector
445  fwVec3d imageOrientationU =
446  {::boost::math::round(directionCosines[0]), ::boost::math::round(directionCosines[1]),
447  ::boost::math::round(directionCosines[2])};
448  // Try to find the closest axe
449  if((fabs(imageOrientationU[0]) + fabs(imageOrientationU[1]) + fabs(imageOrientationU[2])) > 1)
450  {
451  if(fabs(directionCosines[0]) < fabs(directionCosines[1]) ||
452  fabs(directionCosines[0]) < fabs(directionCosines[2]))
453  {
454  imageOrientationU[0] = 0;
455  }
456  if(fabs(directionCosines[1]) < fabs(directionCosines[0]) ||
457  fabs(directionCosines[1]) < fabs(directionCosines[2]))
458  {
459  imageOrientationU[1] = 0;
460  }
461  if(fabs(directionCosines[2]) < fabs(directionCosines[0]) ||
462  fabs(directionCosines[2]) < fabs(directionCosines[1]))
463  {
464  imageOrientationU[2] = 0;
465  }
466  m_logger->warning(
467  "Unable to determine clearly the orientation of the image. The software may display the image in the wrong direction.");
468  }
469 
470  // Compute V vector
471  fwVec3d imageOrientationV =
472  {::boost::math::round(directionCosines[3]), ::boost::math::round(directionCosines[4]),
473  ::boost::math::round(directionCosines[5])};
474  // Try to find the closest axe
475  if((fabs(imageOrientationV[0]) + fabs(imageOrientationV[1]) + fabs(imageOrientationV[2])) > 1)
476  {
477  if(fabs(directionCosines[3]) < fabs(directionCosines[4]) ||
478  fabs(directionCosines[3]) < fabs(directionCosines[5]))
479  {
480  imageOrientationV[0] = 0;
481  }
482  if(fabs(directionCosines[4]) < fabs(directionCosines[3]) ||
483  fabs(directionCosines[4]) < fabs(directionCosines[5]))
484  {
485  imageOrientationV[1] = 0;
486  }
487  if(fabs(directionCosines[5]) < fabs(directionCosines[3]) ||
488  fabs(directionCosines[5]) < fabs(directionCosines[4]))
489  {
490  imageOrientationV[2] = 0;
491  }
492  m_logger->warning(
493  "Unable to determine clearly the orientation of the image. The software may display the image in the wrong direction.");
494  }
495 
496  // Compute W vector
497  const fwVec3d imageOrientationW = ::fwMath::cross(imageOrientationU, imageOrientationV);
498 
499  // Create orientation matrix
500  Image::MatrixType matrix(4,4);
501  matrix(0,0) = imageOrientationU[0];
502  matrix(1,0) = imageOrientationU[1];
503  matrix(2,0) = imageOrientationU[2];
504  matrix(3,0) = 0;
505  matrix(0,1) = imageOrientationV[0];
506  matrix(1,1) = imageOrientationV[1];
507  matrix(2,1) = imageOrientationV[2];
508  matrix(3,1) = 0;
509  matrix(0,2) = imageOrientationW[0];
510  matrix(1,2) = imageOrientationW[1];
511  matrix(2,2) = imageOrientationW[2];
512  matrix(3,2) = 0;
513  matrix(0,3) = 0;
514  matrix(1,3) = 0;
515  matrix(2,3) = 0;
516  matrix(3,3) = 1;
517 
518  // Compute inverse matrix in order to rotate the buffer
519  Image::MatrixType inverseMatrix = this->computeInverseMatrix(matrix);
520  Image::MatrixType identityMatrix = ::boost::numeric::ublas::identity_matrix< double >(inverseMatrix.size1());
521 
522  // Check whether the image must be rotated or not
523  if(!::boost::numeric::ublas::detail::expression_type_check(inverseMatrix,identityMatrix))
524  {
525  // Compute new image size
526  VectorType sizeVector(4);
527  sizeVector(0) = dimensions.at(0);
528  sizeVector(1) = dimensions.at(1);
529  sizeVector(2) = dimensions.at(2);
530  VectorType newSizeVector = ::boost::numeric::ublas::prod(sizeVector, inverseMatrix);
531  unsigned short newSizeX = static_cast<unsigned short>(fabs(newSizeVector[0]));
532  unsigned short newSizeY = static_cast<unsigned short>(fabs(newSizeVector[1]));
533  unsigned short newSizeZ = static_cast<unsigned short>(fabs(newSizeVector[2]));
534  newSizeVector(0) = newSizeX;
535  newSizeVector(1) = newSizeY;
536  newSizeVector(2) = newSizeZ;
537 
538  // Compute old size from the new absolute size in order to retrieve pixel flips
539  VectorType oldSizeVector = ::boost::numeric::ublas::prod(newSizeVector, matrix);
540 
541  // Create new buffer to store rotated image
542  const unsigned long size = dimensions.at(0) * dimensions.at(1) * dimensions.at(2) * (bitsAllocated/8);
543  char* newBuffer = new char[size];
544 
545  // Rotate image
546  unsigned short x, y, z, oldx, oldy, oldz;
547  for(z = 0; z < newSizeZ && !m_cancelled; ++z)
548  {
549  for(y = 0; y < newSizeY; ++y)
550  {
551  for(x = 0; x < newSizeX; ++x)
552  {
553  // Create new position
554  VectorType newPosition(4);
555  newPosition(0) = x;
556  newPosition(1) = y;
557  newPosition(2) = z;
558 
559  // Compute old position
560  VectorType oldPosition = ::boost::numeric::ublas::prod(newPosition, matrix);
561  oldx =
562  (oldSizeVector[0]>
563  0) ? static_cast<unsigned short>(oldPosition[0]) : static_cast<unsigned short>((
564  dimensions.
565  at(0)-1)+
566  oldPosition[0]);
567  oldy =
568  (oldSizeVector[1]>
569  0) ? static_cast<unsigned short>(oldPosition[1]) : static_cast<unsigned short>((
570  dimensions.
571  at(1)-1)+
572  oldPosition[1]);
573  oldz =
574  (oldSizeVector[2]>
575  0) ? static_cast<unsigned short>(oldPosition[2]) : static_cast<unsigned short>((
576  dimensions.
577  at(2)-1)+
578  oldPosition[2]);
579 
580  // Compute indices
581  unsigned int positionIndex = (x + (y*newSizeX) + z*(newSizeX*newSizeY)) * (bitsAllocated/8);
582  unsigned int oldPositionIndex =
583  (oldx + (oldy*dimensions.at(0)) + oldz*(dimensions.at(0)*dimensions.at(1))) *
584  (bitsAllocated/8);
585 
586  // Copy bytes
587  memcpy(&newBuffer[positionIndex], &buffer[oldPositionIndex], (bitsAllocated/8));
588  }
589  }
590  unsigned int progress = static_cast<unsigned int>(78 + (z*100./newSizeZ) * 0.2);
591  m_progressCallback(progress);
592  }
593 
594  result = newBuffer;
595  delete[] buffer;
596 
597  // Update image size
598  m_object->setSize(::boost::assign::list_of(newSizeX)(newSizeY)(newSizeZ));
599 
600  // Update image spacing
601  ::fwData::Image::SpacingType spacing = m_object->getSpacing();

635 Image::MatrixType Image::computeInverseMatrix(MatrixType matrix)
636 {
637  // Create output matrix (identity)
638  Image::MatrixType output(matrix.size1(),matrix.size2());
639  output.assign(::boost::numeric::ublas::identity_matrix< double >(output.size1()));
640 
641  // Create a permutation matrix for the LU-factorization
642  boost::numeric::ublas::permutation_matrix< std::size_t > perm(matrix.size1());
643 
644  // Perform LU-factorization
645  long unsigned int res = boost::numeric::ublas::lu_factorize(matrix, perm);
646  if (res != 0)
647  {
648  SLM_WARN("Cannot compute matrix.");
649  }
650  else
651  {
652  // Backsubstitute to get the inverse
653  boost::numeric::ublas::lu_substitute(matrix, perm, output);
654  }
655 
656  // Return inverse matrix or identity
657  return output;
