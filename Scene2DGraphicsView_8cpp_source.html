21 Scene2DGraphicsView::Scene2DGraphicsView(QGraphicsScene* scene, QWidget* widget) : QGraphicsView(scene,widget)
22 {
23  if( !this->hasMouseTracking() )
24  {
25  this->setMouseTracking( true );
26  }
27 
28  /* Change the method usedn by the scene to index its items:
29  * 'NoIndex' is better than the default method 'BspTreeIndex' (Binary Space Partioning)
30  * which consists in dividing the scene into multiple parts and store them into a tree.
31  * With 'NoIndex', searching an item is faster, especially for scenes with a lot of items
32  * (Histogram for instance) and for scenes which manage moves (ViewportRangeSelector and
33  * ViewportUpdater for instance).
34  */
35  scene->setItemIndexMethod( QGraphicsScene::NoIndex );
36 
37  this->setVerticalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
38  this->setHorizontalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
39 
40  this->setViewportUpdateMode( QGraphicsView::BoundingRectViewportUpdate );
41  this->setOptimizationFlags( QGraphicsView::DontSavePainterState //
42  | QGraphicsView::DontAdjustForAntialiasing ); // Prevent from rendering artifacts
43  this->setCacheMode( QGraphicsView::CacheBackground ); // Activates background cache
44  this->setAttribute( Qt::WA_TranslucentBackground, false );
45  this->setFrameStyle( 0 );
46 }
