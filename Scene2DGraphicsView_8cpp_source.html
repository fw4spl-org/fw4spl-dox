00020 Scene2DGraphicsView::Scene2DGraphicsView(QGraphicsScene* scene, QWidget* widget) : QGraphicsView(scene,widget)
00021 {
00022     if( !this->hasMouseTracking() )
00023     {
00024         this->setMouseTracking( true );
00025     }
00026 
00027     /* Change the method usedn by the scene to index its items:
00028      * 'NoIndex' is better than the default method 'BspTreeIndex' (Binary Space Partioning)
00029      * which consists in dividing the scene into multiple parts and store them into a tree.
00030      * With 'NoIndex', searching an item is faster, especially for scenes with a lot of items
00031      * (Histogram for instance) and for scenes which manage moves (ViewportRangeSelector and
00032      * ViewportUpdater for instance).
00033      */
00034     scene->setItemIndexMethod( QGraphicsScene::NoIndex );
00035 
00036     this->setVerticalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
00037     this->setHorizontalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
00038 
00039     this->setViewportUpdateMode( QGraphicsView::BoundingRectViewportUpdate );
00040     this->setOptimizationFlags( QGraphicsView::DontSavePainterState          //
00041                               | QGraphicsView::DontAdjustForAntialiasing );  // Prevent from rendering artifacts
00042     this->setCacheMode( QGraphicsView::CacheBackground );   // Activates background cache
00043     this->setAttribute( Qt::WA_TranslucentBackground, false );
00044     this->setFrameStyle( 0 );
00045 }
