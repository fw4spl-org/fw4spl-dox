82 Vector<float> &computeCellNormal( const PointsMultiArrayType &points, const ::fwData::Mesh::CellValueType *cell,
83  size_t cellSize, Vector<float> &n)
84 {
85  n = Vector<float>();
86  Vector<float> v;
87 
88  for (size_t i = 0; i< cellSize; ++i)
89  {
90  const Point &p1 = points[cell[i ]];
91  const Point &p2 = points[cell[(i+1)% cellSize]];
92  const Point &p3 = points[cell[(i+2)% cellSize]];
93 
94  computeTriangleNormal(p1, p2, p3, n);
95 
96  n += v;
97  }
98 
99  n /= ::fwTools::numericRoundCast<float>(cellSize);
100  n.normalize();
101  return n;

115  ::boost::extents[mesh->getNumberOfPoints()]

118 
119  ::fwData::Mesh::CellTypesMultiArrayType cellTypes = meshHelper->getCellTypes();
120  ::fwData::Mesh::CellDataMultiArrayType cellData = meshHelper->getCellData();
121  ::fwData::Mesh::CellDataOffsetsMultiArrayType cellDataOffsets = meshHelper->getCellDataOffsets();
122 
123  const Vector<float> vZero;
124  ::fwData::Mesh::CellTypes type;
125  ::fwData::Mesh::CellDataOffsetType offset;
126  ::fwData::Mesh::CellValueType *cell;
127  ::fwData::Mesh::Id cellLen = 0;

130  const ::fwData::Mesh::Id cellDataSize = mesh->getCellDataSize();

134

136  {
137  Vector<float> &n = normals[i];
138 
139  type = cellTypes[i];
140  offset = cellDataOffsets[i];
141  cell = &cellData[offset];

143  {
144  case 0:
145  case 1:
146  case 2:
147  n = vZero;
148  break;
149  case 3:
150  {
151  computeTriangleNormal(point, cell, n);
152  }
153  break;
154  case 4:
155  case 5:

158  cellLen = (( i1 < numberOfCells ) ? cellDataOffsets[i1] : cellDataSize) - cellDataOffsets[i];

213 void generateRegionCellNormalsByPoints(FloatVectors &normalsData, CharVectors &normalCounts, size_t dataId,
214  ::fwComEd::helper::Mesh::sptr meshHelper, const ::fwData::Mesh::Id regionMin,

222  normalsResults.resize(3*nbOfPoints, 0.f);
223  normalCount.resize(nbOfPoints, 0);

230  ::fwData::Mesh::CellTypes type;

239  Vector< ::fwData::Mesh::NormalValueType > *normals =
240  arrayHelper.begin< Vector< ::fwData::Mesh::NormalValueType > >();

242  reinterpret_cast< Vector< ::fwData::Mesh::NormalValueType > * >( &(*normalsResults.begin()));
243 
244 
245  ::fwData::Mesh::CellValueType *pointId;
246  ::fwData::Mesh::CellValueType *cellEnd;

248  for(::fwData::Mesh::Id i = regionMin; i<regionMax; ++i)

251  offset = cellDataOffsets[i];
252  cell = &cellData[offset];
253  cellLen = type;
254 
255  switch (type)
256  {
257  case 0:
258  case 1:
259  case 2:
260  case 3:

480 void Mesh::transform( ::fwData::Mesh::sptr mesh, ::fwData::TransformationMatrix3D::sptr t )
481 {
482  size_t nbPts = mesh->getNumberOfPoints();
483  ::fwComEd::helper::Mesh meshHelper(mesh);
484  ::fwData::Mesh::PointsMultiArrayType points = meshHelper.getPoints();
485  ::fwData::Mesh::PointValueType x, y, z, xp, yp, zp, factor;
486  for(size_t i = 0; i<nbPts; ++i )
487  {
488  x = points[i][0];
489  y = points[i][1];
490  z = points[i][2];
491  xp = ::fwTools::numericRoundCast< ::fwData::Mesh::PointValueType >(t->getCoefficient(0,0) * x
492  + t->getCoefficient(0,1) * y
493  + t->getCoefficient(0,2) * z
494  + t->getCoefficient(0,3));
495  yp = ::fwTools::numericRoundCast< ::fwData::Mesh::PointValueType >(t->getCoefficient(1,0) * x
496  + t->getCoefficient(1,1) * y
497  + t->getCoefficient(1,2) * z
498  + t->getCoefficient(1,3));
499  zp = ::fwTools::numericRoundCast< ::fwData::Mesh::PointValueType >(t->getCoefficient(2,0) * x
500  + t->getCoefficient(2,1) * y
501  + t->getCoefficient(2,2) * z
502  + t->getCoefficient(2,3));
503  factor = ::fwTools::numericRoundCast< ::fwData::Mesh::PointValueType >(t->getCoefficient(3,0) * x
504  + t->getCoefficient(3,1) * y
505  + t->getCoefficient(3,2) * z
506  + t->getCoefficient(3,3));
507  points[i][0] = xp/factor;
508  points[i][1] = yp/factor;
509  points[i][2] = zp/factor;
510  }
511 }
