177 //-----------------------------------------------------------------------------
178 void RendererService::createAndAddActorToRender()
179 {
180  //Check there is indeed a Composite object in this:
181  assert(this->getObject< ::fwData::Composite >());
182 
183  //Create a pointer on this object:
184  ::fwData::Composite::sptr myComposite = this->getObject< ::fwData::Composite >();
185 
186  OSLM_INFO( "VTK Pipeline ready TO UPDATE" << '\n' << "Object received:" << myComposite->getLeafClassname());
187 
188  // elementnumber increases for each mesh found:
189  unsigned int elementNumber = 0;
190 
191  //Loop through the composite objects, if it's a mesh, then render it:
192  for(::fwData::Composite::ContainerType::const_iterator it = myComposite->begin(); it != myComposite->end(); ++it)
193  {
194  OSLM_INFO("ObjectName: " << it->first);
195  OSLM_INFO("ObjectPointer: " << it->second);
196  OSLM_INFO("ObjectType: " << it->second->getClassname () << '\n');
197 
198  ::fwData::Mesh::sptr myMesh = ::fwData::Mesh::dynamicCast (it->second);
199 
200  // If it's a mesh, then put it in the pipeline:
201  if( myMesh )
202  {
203  vtkSmartPointer<vtkPolyData> vtk_polyData = vtkSmartPointer<vtkPolyData>::New();
204  ::fwVtkIO::helper::Mesh::toVTKMesh( myMesh, vtk_polyData);
205  OSLM_INFO("Loaded: " << it->first);
206  vtkPolyDataMapper* mapper = vtkPolyDataMapper::New();
207 
208  m_normals = vtkPolyDataNormals::New();
209  m_normals->SetInputData(vtk_polyData);
210  mapper->SetInputConnection(m_normals->GetOutputPort());
211 
212  vtkActor* actor = vtkActor::New();
213  actor->SetMapper( mapper);
214 
215  // Add the actors
216  m_render->AddActor( actor);
217 
218 
219  OSLM_INFO("Mesh found: " << it->first);
220  if (!myMesh->getField( "MaterialMesh" ))
221  {
222  // No Material data then default
223  if(elementNumber == 0)
224  {
225  actor->GetProperty()->EdgeVisibilityOn();
226  actor->GetProperty()->SetInterpolationToFlat();
227  actor->GetProperty()->SetColor (1.0, 0.0, 0.0);
228  actor->GetProperty()->SetEdgeColor (1.0, 0.0, 0.0);
229  }
230  else
231  {
232  actor->GetProperty()->SetRepresentationToWireframe ();
233  actor->GetProperty()->SetColor (1.0, 1.0, 1.0);
234  }
235  }
236  else
237  {
238  // Material exists
239  ::fwData::Material::sptr matObjPtr = myMesh->getField< ::fwData::Material >( "MaterialMesh" );
240  actor->GetProperty()->SetColor (matObjPtr->diffuse()->red(),
241  matObjPtr->diffuse()->green(), matObjPtr->diffuse()->blue());
242  }
243  mapper->Delete();
244  elementNumber++;
245  OSLM_INFO("displayed: " << it->first);
246  }
247  }
248  m_interactorManager->getInteractor()->SetInteractorStyle(vtkInteractorStyleTrackballCamera::New());
249  m_loc = new vtkLocalCommand(this);
250  m_interactorManager->getInteractor()->AddObserver(vtkCommand::AnyEvent, m_loc);
251 
252  // Repaint and resize window
253  m_render->ResetCamera();
254 }

284 void RendererService::notifyCamPositionUpdated()
285 {
286  vtkCamera* camera = m_render->GetActiveCamera();
287 
288  SharedArray position = SharedArray(new double[3]);
289  SharedArray focal = SharedArray(new double[3]);
290  SharedArray viewUp = SharedArray(new double[3]);
291 
292  std::copy(camera->GetPosition(), camera->GetPosition()+3, position.get());
293  std::copy(camera->GetFocalPoint(), camera->GetFocalPoint()+3, focal.get());
294  std::copy(camera->GetViewUp(), camera->GetViewUp()+3, viewUp.get());
295 
296  {
297  ::fwCom::Connection::Blocker block(m_sigCamUpdated->getConnection(m_slotUpdateCamPosition));
298  m_sigCamUpdated->asyncEmit(position, focal, viewUp);
299  }
300 }
