64 SPTR(T) read(const ::fwRuntime::EConfigurationElement::sptr &srvCfg, const std::string &reader)
65 {
66 
67  typename T::sptr readObj = T::New();
68  ::fwServices::IService::sptr readerSrv = ::fwServices::registry::ServiceFactory::getDefault()->create( reader );
69  CPPUNIT_ASSERT(readerSrv);
70 
71  ::fwServices::OSR::registerService( readObj, readerSrv );
72  readerSrv->setConfiguration(srvCfg);
73  readerSrv->configure();
74  readerSrv->start();
75  readerSrv->update();
76  readerSrv->stop();
77  ::fwServices::OSR::unregisterService( readerSrv );
78 
79  return readObj;
80 }

115 void PatchTest::patchMedicalDataTest()
116 {
117  const ::boost::filesystem::path file = ::fwTest::Data::dir() /"fw4spl/patch/md_1.jsonz";
118 
119  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
120 
121  ::fwRuntime::EConfigurationElement::sptr fileCfg = ::fwRuntime::EConfigurationElement::New("file");
122  fileCfg->setValue(file.string());
123  srvCfg->addConfigurationElement(fileCfg);
124 
125  //<patcher context="..." version="..." />
126  ::fwRuntime::EConfigurationElement::sptr patcherCfg = ::fwRuntime::EConfigurationElement::New("patcher");
127  patcherCfg->setAttributeValue("context","MedicalData");
128  patcherCfg->setAttributeValue("version","V09ALA");
129  srvCfg->addConfigurationElement(patcherCfg);
130 
131  ::fwMedData::SeriesDB::sptr sdb = read< ::fwMedData::SeriesDB >(srvCfg, "::ioAtoms::SReader" );
132 
133  // check data
134  CPPUNIT_ASSERT_EQUAL( (size_t) 3, sdb->size() );
135 
136  // Check ACH Data
137  ::fwMedData::Series::sptr series = getACHSeries( sdb );
138  CPPUNIT_ASSERT( series );
139  CPPUNIT_ASSERT( ::fwMedData::ImageSeries::dynamicCast(series) );
140  CPPUNIT_ASSERT_EQUAL(std::string("1.2.392.200036.9116.2.6.1.48.1211418863.1225184516.765855"),
141  series->getInstanceUID());
142  CPPUNIT_ASSERT_EQUAL(std::string("20081028"), series->getDate());
143  CPPUNIT_ASSERT_EQUAL(std::string("174446"), series->getTime());
144  CPPUNIT_ASSERT_EQUAL(std::string("Original image"),series->getDescription());
145  CPPUNIT_ASSERT_EQUAL(std::string("CT"), series->getModality());
146 
147 
148  ::fwMedData::Patient::sptr patient = series->getPatient();
149  CPPUNIT_ASSERT( patient );
150  CPPUNIT_ASSERT_EQUAL(std::string("12592 ARTHRO GENOU G"), patient->getPatientId());
151  CPPUNIT_ASSERT_EQUAL(std::string("19790618"), patient->getBirthdate());
152  CPPUNIT_ASSERT_EQUAL(std::string("M"), patient->getSex());
153 
154 
155  ::fwMedData::Study::sptr study = series->getStudy();
156  CPPUNIT_ASSERT( study );
157  CPPUNIT_ASSERT_EQUAL(std::string("1.2.392.200036.9116.2.6.1.48.1211418863.1225183167.375775"),
158  study->getInstanceUID());
159  CPPUNIT_ASSERT_EQUAL(std::string("20081028"), study->getDate());
160  CPPUNIT_ASSERT_EQUAL(std::string("174446"), study->getTime());
161  CPPUNIT_ASSERT_EQUAL(std::string(""), study->getReferringPhysicianName());
162  CPPUNIT_ASSERT_EQUAL(std::string(""), study->getDescription());
163  CPPUNIT_ASSERT_EQUAL(std::string(""), study->getPatientAge());
164 
165  ::fwMedData::Equipment::sptr equipment = series->getEquipment();
166  CPPUNIT_ASSERT( equipment );
167  CPPUNIT_ASSERT_EQUAL(std::string("SCANNER DE LA MODER"), equipment->getInstitutionName());
168 
169  // Test split between meshes and image
170  std::vector< ::fwMedData::Series::sptr > otherSeries = getOtherSeries( sdb );
171  CPPUNIT_ASSERT_EQUAL( (size_t) 2, otherSeries.size() );
172  CPPUNIT_ASSERT( otherSeries[0]->getStudy() != otherSeries[1]->getStudy() );
173  CPPUNIT_ASSERT( otherSeries[0]->getPatient() != otherSeries[1]->getPatient() );
174  CPPUNIT_ASSERT( otherSeries[0]->getEquipment() != otherSeries[1]->getEquipment() );
175 
176  ::fwMedData::Patient::sptr p1 = otherSeries[0]->getPatient();
177  ::fwMedData::Patient::sptr p2 = otherSeries[1]->getPatient();
178  CPPUNIT_ASSERT_EQUAL( p1->getName(), p2->getName() );
179  CPPUNIT_ASSERT_EQUAL( p1->getPatientId(), p2->getPatientId() );
180  CPPUNIT_ASSERT_EQUAL( p1->getBirthdate(), p2->getBirthdate() );
181  CPPUNIT_ASSERT_EQUAL( p1->getSex(), p2->getSex() );
182 
183  ::fwMedData::Study::sptr s1 = otherSeries[0]->getStudy();
184  ::fwMedData::Study::sptr s2 = otherSeries[1]->getStudy();
185  CPPUNIT_ASSERT_EQUAL( s1->getInstanceUID(), s2->getInstanceUID() );
186  CPPUNIT_ASSERT_EQUAL( s1->getDate(), s2->getDate() );
187  CPPUNIT_ASSERT_EQUAL( s1->getTime(), s2->getTime() );
188  CPPUNIT_ASSERT_EQUAL( s1->getDescription(), s2->getDescription() );
189  CPPUNIT_ASSERT_EQUAL( s1->getReferringPhysicianName(), s2->getReferringPhysicianName() );
190  CPPUNIT_ASSERT_EQUAL( s1->getPatientAge(), s2->getPatientAge() );
191 
192  ::fwMedData::Equipment::sptr e1 = otherSeries[0]->getEquipment();
193  ::fwMedData::Equipment::sptr e2 = otherSeries[1]->getEquipment();
194  CPPUNIT_ASSERT_EQUAL( e1->getInstitutionName(), e2->getInstitutionName() );
195 
196  ::fwMedData::Series::sptr se1 = otherSeries[0];
197  ::fwMedData::Series::sptr se2 = otherSeries[1];
198  CPPUNIT_ASSERT( se1->getInstanceUID() != se2->getInstanceUID() );
199  CPPUNIT_ASSERT( se1->getModality() != se2->getModality() );
200  CPPUNIT_ASSERT_EQUAL( se1->getDate(), se2->getDate() );
201  CPPUNIT_ASSERT_EQUAL( se1->getTime(), se2->getTime() );
202  CPPUNIT_ASSERT_EQUAL( se1->getDescription(), se2->getDescription() );
203 }
