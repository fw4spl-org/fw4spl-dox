95 void TransferFunctionTest::classicGetSetTest()
96 {
97  TransferFunction::TFColor expectedColor1(0.0, 0.0, 0.0, 0.0);
98  TransferFunction::TFColor expectedColor2(1.0, 1.0, 1.0, 1.0);
99  TransferFunction::TFValueType expectedKey1(0.0);
100  TransferFunction::TFValueType expectedKey2(1.0);
101 
102  ::fwData::TransferFunction::sptr tf = ::fwData::TransferFunction::createDefaultTF();
103 
104  // Test getTFData()
105  const TransferFunction::TFDataType& data = tf->getTFData();
106 
107  TransferFunction::TFDataType::const_iterator itr = data.begin();
108  TransferFunction::TFValueType key1 = itr->first;
109  TransferFunction::TFValueType key2 = (++itr)->first;
110 
111  TransferFunction::TFDataType::const_iterator itrColor = data.begin();
112  TransferFunction::TFColor color1 = itrColor->second;
113  TransferFunction::TFColor color2 = (++itrColor)->second;
114 
115  CPPUNIT_ASSERT(expectedKey1 == key1);
116  CPPUNIT_ASSERT(expectedKey2 == key2);
117 
118  CPPUNIT_ASSERT(expectedColor1 == color1);
119  CPPUNIT_ASSERT(expectedColor2 == color2);
120 
121  // Test getTFColor()
122  CPPUNIT_ASSERT(expectedColor1 == tf->getTFColor(key1));
123 
124  // Test getTFValues()
125  TransferFunction::TFValueVectorType values = tf->getTFValues();
126 
127  CPPUNIT_ASSERT_DOUBLES_EQUAL_MESSAGE("Wrong key value", expectedKey1, values[0], 0.0);
128  CPPUNIT_ASSERT_DOUBLES_EQUAL_MESSAGE("Wrong key value", expectedKey2, values[1], 0.0);
129 
130  // Test getTFColors()
131  TransferFunction::TFColorVectorType colors = tf->getTFColors();
132 
133  CPPUNIT_ASSERT(expectedColor1 == colors[0]);
134  CPPUNIT_ASSERT(expectedColor2 == colors[1]);
135 
136  // Test erase
137  tf->eraseTFValue( 1.0 );
138  CPPUNIT_ASSERT_EQUAL( (size_t)1, tf->getTFData().size() );
139  CPPUNIT_ASSERT_DOUBLES_EQUAL_MESSAGE("Wrong key value", expectedKey1, tf->getTFValues()[0], 0.0);
140 
141  // Test clear()
142  tf->clear();
143  size_t expectedClearedSize = 0;
144  CPPUNIT_ASSERT_EQUAL(expectedClearedSize, tf->getTFData().size());
145 
146 }

192  tf->setBackgroundColor( ::fwData::TransferFunction::TFColor( 1.0, 0.3, 0.6, 0.1) );

201  tf->addTFColor( -0.2, ::fwData::TransferFunction::TFColor( 0.1, 0.9, 0.3, 0.4) );
202  tf->addTFColor( 150, ::fwData::TransferFunction::TFColor( 0.1, 0.2, 0.3, 0.9) );
203 
204  ::fwData::String::sptr myString = ::fwData::String::New("fieldStringValue");
205  tf->setField( "fieldStringKey", myString );
206 
207  return tf;

230  CPPUNIT_ASSERT_EQUAL( values[3], 150.0 );

234  CPPUNIT_ASSERT( ::fwData::TransferFunction::TFColor( 0.1, 0.2, 0.9, 0.4) == tf->getTFColor( 3 ) );
