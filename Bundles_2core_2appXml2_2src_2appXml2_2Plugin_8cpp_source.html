44 void Plugin::initialize() throw( ::fwRuntime::RuntimeException )
45 {
46  SLM_ASSERT("The OSR is already initialized.", !m_appConfigMng );
47  SLM_ASSERT("The configuration name parameter is not initialized.", !m_configurationName.empty());
48 
49  m_appConfigMng = ::fwServices::AppConfigManager2::New();
50 
51  if( m_parametersName.empty() )
52  {
53  const ::fwServices::registry::FieldAdaptorType fields;
54  m_appConfigMng->setConfig( m_configurationName, fields );
55  }
56  else
57  {
58  const ::fwServices::registry::FieldAdaptorType & fields =
59  ::fwServices::registry::AppConfigParameters::getDefault()->getParameters( m_parametersName );
60  m_appConfigMng->setConfig( m_configurationName, fields );
61  }
62
