121 {
122  ::gdcm::PhotometricInterpretation pi;
123  size_t components = image.getNumberOfComponents();
124 
125  // Attempt a guess (VTK do the same choice)
126  switch (components)
127  {
128  case 1: // It could well be MONOCHROME1
129  pi = ::gdcm::PhotometricInterpretation::MONOCHROME2;
130  break;
131  case 3: // It could well be YBR
132  pi = ::gdcm::PhotometricInterpretation::RGB;
133  break;
134  case 4: // It could well be CMYK
135  pi = ::gdcm::PhotometricInterpretation::ARGB;
136  break;
137  default:
138  SLM_ERROR("Photometric interpretation not found");
139  pi = ::gdcm::PhotometricInterpretation::UNKNOWN;
140  break;
141  }
142 
143  return pi;

213 int DicomData::convertPointToFrameNumber(const ::fwData::Image::csptr& image, const ::fwData::Point::sptr& point)
214 throw(::fwGdcmIO::exception::Failed)
215 {
216  // Retrieve Z spacing
217  double zSpacing = (image->getNumberOfDimensions() > 2) ? (image->getSpacing()[2]) : 1;
218 
219  // Retrieve Z coordinate of image origin
220  double zOrigin = (image->getNumberOfDimensions() > 2) ? (image->getOrigin()[2]) : 0;
221 
222  // Retrieve Z coordinate
223  double zCoordinate = static_cast<double>(point->getCoord()[2]);
224 
225  // Compute frame number
226  unsigned int frameNumber = static_cast<int>(floor((zCoordinate - zOrigin) / zSpacing + 0.5)) + 1;
227  FW_RAISE_EXCEPTION_IF(::fwGdcmIO::exception::Failed("Coordinates out of image bounds."),
228  frameNumber < 1 || frameNumber > image->getSize()[2]);
229 
230  return frameNumber;
231 }

235 double DicomData::convertFrameNumberToZCoordinate(const ::fwData::Image::csptr& image, int frameNumber)
236 throw(::fwGdcmIO::exception::Failed)
237 {
238  // Retrieve Z spacing
239  double zSpacing = (image->getNumberOfDimensions() > 2) ? (image->getSpacing()[2]) : 1;
240 
241  // Retrieve Z coordinate of image origin
242  double zOrigin = (image->getNumberOfDimensions() > 2) ? (image->getOrigin()[2]) : 0;
243 
244  // Compute coordinate
245  unsigned int frameIndex = (frameNumber-1);
246  FW_RAISE_EXCEPTION_IF(::fwGdcmIO::exception::Failed("Coordinates out of image bounds."),
247  frameIndex >= image->getSize()[2]);
248  double zCoordinate = zOrigin + frameIndex * zSpacing;
249 
250  return zCoordinate;
251 }
