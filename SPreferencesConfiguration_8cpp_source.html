56 void SPreferencesConfiguration::starting() throw(::fwTools::Failed)
57 {
58  this->actionServiceStarting();
59 
60  // Check preferences
61  auto preferencesServicesList = ::fwServices::OSR::getServices("::fwPreferences::IPreferences");
62  if(!preferencesServicesList.empty())
63  {
64  const ::fwServices::IService::sptr prefService = *preferencesServicesList.begin();
65  ::fwData::Composite::sptr prefs = prefService->getObject< ::fwData::Composite >();
66 
67  for(PreferenceElt& pref : m_preferences)
68  {
69  pref.m_dataPreference = ::fwData::String::New(pref.m_defaultValue);
70  ::fwData::Composite::IteratorType iterPref = prefs->find( pref.m_preferenceKey );
71  if ( iterPref != prefs->end() )
72  {
73  pref.m_dataPreference = ::fwData::String::dynamicCast(iterPref->second);
74  }
75  else
76  {
77  (*prefs)[pref.m_preferenceKey] = pref.m_dataPreference;
78  }
79  }
80  }
81 }

92 void SPreferencesConfiguration::configuring() throw(fwTools::Failed)
93 {
94  this->initialize();
95 
96  ::fwRuntime::ConfigurationElementContainer config = m_configuration->findAllConfigurationElement("preference");
97  for(const auto& elt : config.getElements())
98  {
99  PreferenceElt pref;
100 
101  ConfigurationType typeCfg = elt->findConfigurationElement("type");
102  SLM_ASSERT("element 'type' is missing.", typeCfg);
103  if(typeCfg->getValue() == "checkbox")
104  {
105  pref.m_type = PreferenceType::CHECKBOX;
106  }
107  else if (typeCfg->getValue() == "text" )
108  {
109  pref.m_type = PreferenceType::TEXT;
110  }
111  else if (typeCfg->getValue() == "number")
112  {
113  pref.m_type = PreferenceType::U_INT;
114  }
115  else if(typeCfg->getValue() == "path")
116  {
117  pref.m_type = PreferenceType::PATH;
118  }
119  else
120  {
121  OSLM_ERROR("Preference type "<<typeCfg->getValue()<<" is not implemented");
122  }
123 
124  ConfigurationType nameCfg = elt->findConfigurationElement("name");
125  SLM_ASSERT("element 'name' is missing.", nameCfg);
126  pref.m_name = nameCfg->getValue();
127 
128  ConfigurationType keyCfg = elt->findConfigurationElement("key");
129  SLM_ASSERT("element 'key' is missing.", keyCfg);
130  pref.m_preferenceKey = keyCfg->getValue();
131 
132  ConfigurationType defaultValueCfg = elt->findConfigurationElement("default_value");
133  SLM_ASSERT("element 'default_value' is missing.", defaultValueCfg);
134  pref.m_defaultValue = defaultValueCfg->getValue();
135 
136  if(pref.m_type == PreferenceType::TEXT || pref.m_type == PreferenceType::PATH)
137  {
138  pref.m_lineEdit = new QLineEdit(QString::fromStdString(pref.m_defaultValue));
139  }
140  else if(pref.m_type == PreferenceType::CHECKBOX)
141  {
142  pref.m_checkBox = new QCheckBox();
143  pref.m_checkBox->setChecked(pref.m_defaultValue == "true");
144  }
145  else if(pref.m_type == PreferenceType::U_INT)
146  {
147  pref.m_lineEdit = new QLineEdit(QString::fromStdString(pref.m_defaultValue));
148  pref.m_lineEdit->setValidator( new QIntValidator( 0, 999999));
149  }
150  m_preferences.push_back(pref);
151  }
152 }

156 void SPreferencesConfiguration::updating() throw(::fwTools::Failed)
157 {
158  QPointer<QDialog> dialog = new QDialog();
159  QPointer<QGridLayout> layout = new QGridLayout();
160 
161  int index = 0;
162  for(PreferenceElt& pref : m_preferences)
163  {
164  QPointer<QLabel> label = new QLabel(QString::fromStdString(pref.m_name));
165  layout->addWidget(label, index, 0);
166 
167  if(pref.m_type == PreferenceType::TEXT)
168  {
169  pref.m_lineEdit->setText(QString::fromStdString(pref.m_dataPreference->value()));
170  layout->addWidget(pref.m_lineEdit, index, 1);
171  }
172  else if(pref.m_type == PreferenceType::CHECKBOX)
173  {
174  pref.m_checkBox->setChecked(pref.m_dataPreference->value() == "true");
175  layout->addWidget(pref.m_checkBox, index, 1);
176  }
177  else if(pref.m_type == PreferenceType::U_INT)
178  {
179  pref.m_lineEdit->setText(QString::fromStdString(pref.m_dataPreference->value()));
180  layout->addWidget(pref.m_lineEdit, index, 1);
181  }
182  else if(pref.m_type == PreferenceType::PATH)
183  {
184  pref.m_lineEdit->setText(QString::fromStdString(pref.m_dataPreference->value()));
185  layout->addWidget(pref.m_lineEdit, index, 1);
186  QPointer<QPushButton> directorySelector = new QPushButton("...");
187  layout->addWidget(directorySelector, index, 2);
188  QObject::connect(directorySelector.data(), &QPushButton::clicked, [this,pref]()
189  {
190  this->onSelectDir(pref.m_lineEdit);
191  });
192  }
193 
194  ++index;
195  }
196 
197  QPointer<QPushButton> cancelButton = new QPushButton("Cancel");
198  QPointer<QPushButton> okButton = new QPushButton("OK");
199  okButton->setDefault(true);
200 
201  QPointer<QHBoxLayout> buttonLayout = new QHBoxLayout();
202  buttonLayout->addWidget(cancelButton);
203  buttonLayout->addWidget(okButton);
204 
205  layout->addLayout(buttonLayout, index, 1, 4, 2 );
206 
207  QObject::connect(cancelButton.data(), &QPushButton::clicked, dialog.data(), &QDialog::reject);
208  QObject::connect(okButton.data(), &QPushButton::clicked, dialog.data(), &QDialog::accept);
209 
210  dialog->setLayout(layout);
211 
212 
213  if (dialog->exec() == QDialog::Accepted)
214  {
215  for(PreferenceElt& pref : m_preferences)
216  {
217  if((pref.m_type == PreferenceType::TEXT || pref.m_type == PreferenceType::PATH) &&
218  !pref.m_lineEdit->text().isEmpty())
219  {
220  pref.m_dataPreference->value() = pref.m_lineEdit->text().toStdString();
221  }
222  else if(pref.m_type == PreferenceType::CHECKBOX)
223  {
224  pref.m_dataPreference->value() = pref.m_checkBox->isChecked() ? "true" : "false";
225  }
226  else if(pref.m_type == PreferenceType::U_INT)
227  {
228  pref.m_dataPreference->value() = pref.m_lineEdit->text().toStdString();
229  }
230  }
231  m_sigParametersModified->asyncEmit();
232  }
233 }

237 void SPreferencesConfiguration::onSelectDir(QPointer<QLineEdit> lineEdit)
238 {
239  static ::boost::filesystem::path _sDefaultPath;
240 
241  ::fwGui::dialog::LocationDialog dialogFile;
242  dialogFile.setTitle("Select Storage directory");
243  dialogFile.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
244  dialogFile.setOption(::fwGui::dialog::ILocationDialog::WRITE);
245  dialogFile.setType(::fwGui::dialog::ILocationDialog::FOLDER);
246 
247  ::fwData::location::Folder::sptr result;
248  result = ::fwData::location::Folder::dynamicCast( dialogFile.show() );
249  if (result)
250  {
251  _sDefaultPath = result->getFolder();
252  lineEdit->setText( QString::fromStdString(result->getFolder().string()) );
253  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
254  }
255 }
