48 void ViewportRangeSelector::configuring() throw( ::fwTools::Failed)
49 {
50  SLM_TRACE_FUNC();
51 
52  ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
53 
54  this->IAdaptor::configuring();
55 
56  const double viewportWidth = viewport->getWidth();
57  const double defaultWidth = 2 * viewportWidth / 4;
58 
59  if (!m_configuration->getAttributeValue("initialWidth").empty())
60  {
61  m_initialWidth = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("initialWidth") );
62 
63  if( m_initialWidth > viewportWidth || m_initialWidth < m_clickCatchRange )
64  {
65  OSLM_WARN("Set viewport width to a default value instead of the given one because it can't be accepted.");
66  m_initialWidth = defaultWidth;
67  }
68  }
69  else
70  {
71  m_initialWidth = defaultWidth;
72  }
73 
74  const double defaultPos = (viewportWidth - m_initialWidth) / 2;
75 
76  if (!m_configuration->getAttributeValue("initialPos").empty())
77  {
78  m_initialX = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("initialPos") );
79 
80  if( m_initialX < viewport->getX() || (m_initialX + m_initialWidth) > viewportWidth)
81  {
82  OSLM_WARN("Set viewport position to a default value since the given one is not correct.");
83  m_initialX = defaultPos;
84  }
85  }
86  else
87  {
88  m_initialX = defaultPos;
89  }
90 }

94 void ViewportRangeSelector::doStart() throw( ::fwTools::Failed)
95 {
96  ::scene2D::data::Viewport::sptr viewport = this->getObject< ::scene2D::data::Viewport>();
97 
98  QRectF sceneRect = this->getScene2DRender()->getScene()->sceneRect();
99 
100  std::pair< double, double > pair = this->mapAdaptorToScene(
101  std::pair< double, double >( m_initialX, viewport->getHeight() ), m_xAxis, m_yAxis );
102  m_shutter = new QGraphicsRectItem(
103  pair.first, 0, m_initialWidth * m_xAxis->getScale(), pair.second );
104  m_shutter->setBrush( QBrush(QColor(127, 127, 127, 127)) );
105  m_shutter->setPen( Qt::NoPen );
106 
107  m_layer = new QGraphicsItemGroup();
108  m_layer->addToGroup( m_shutter );
109 
110  // Adjust the layer's position and zValue depending on the associated axis
111  m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
112  m_layer->setZValue(m_zValue);
113 
114  this->getScene2DRender()->getScene()->addItem( m_layer );
115 
116  QRectF rect = m_shutter->rect();
117  updateViewportFromShutter( rect.x(), rect.y(), rect.width(), rect.height() );
118 
119  ::fwData::Object::ModifiedSignalType::sptr sig;
120  sig = viewport->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
121  {
122  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
123  sig->asyncEmit();
124  }
125 }

150 void ViewportRangeSelector::processInteraction( ::scene2D::data::Event::sptr _event )
151 {
152  // Event coordinates in scene
153  ::scene2D::data::Coord coord;
154  coord = this->getScene2DRender()->mapToScene( _event->getCoord() );
155 
156  // Shutter coordinates in scene
157  std::pair< double, double > shutterCoordPair;
158  shutterCoordPair = this->mapAdaptorToScene(
159  std::pair< double, double >( m_shutter->rect().x(), m_shutter->rect().y() ), m_xAxis, m_yAxis);
160  double shutterWidth = m_shutter->rect().width() * m_xAxis->getScale();
161 
162  QRectF sceneRect = this->getScene2DRender()->getScene()->sceneRect();
163 
164  bool onShutterLeft = mouseOnShutterLeft( coord );
165  bool onShutterRight = mouseOnShutterRight( coord );
166  bool onShutterMiddle = mouseOnShutterMiddle( coord );
167 
168  QRectF rect = m_shutter->rect();
169 
170  if( _event->getType() == ::scene2D::data::Event::MouseButtonPress )
171  {
172  if( onShutterLeft )
173  {
174  m_isLeftInteracting = true;
175  }
176  else if( onShutterRight )
177  {
178  m_isRightInteracting = true;
179  }
180  else if( onShutterMiddle )
181  {
182  this->getScene2DRender()->getView()->setCursor( Qt::ClosedHandCursor );
183 
184  // Interaction when clicking on the center of the shutter
185  m_isInteracting = true;
186  m_dragStartPoint = coord;
187  m_dragStartShutterPos.setX( shutterCoordPair.first );
188  m_dragStartShutterPos.setY( shutterCoordPair.second );
189  }
190  }
191  else if( _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
192  {
193  m_isInteracting = false;
194  m_isLeftInteracting = false;
195  m_isRightInteracting = false;
196 
197  // Reset cursor
198  if( onShutterMiddle )
199  {
200  this->getScene2DRender()->getView()->setCursor( Qt::OpenHandCursor );
201  }
202  else
203  {
204  this->getScene2DRender()->getView()->setCursor( Qt::ArrowCursor );
205  }
206  }
207  else if( _event->getType() == ::scene2D::data::Event::MouseMove )
208  {
209  // If the mouse is moving onto the shutter, without interactions, the cursor is changed to an other cursor
210  // that symbolize the available interactions
211  if( !m_isLeftInteracting && !m_isRightInteracting && !m_isInteracting )
212  {
213  if( onShutterLeft || onShutterRight )
214  {
215  this->getScene2DRender()->getView()->setCursor( Qt::SizeHorCursor ); // horizontal double arrow
216  }
217  else if( onShutterMiddle )
218  {
219  this->getScene2DRender()->getView()->setCursor( Qt::OpenHandCursor ); // open hand, for moving the whole shutter
220  }
221  else
222  {
223  this->getScene2DRender()->getView()->setCursor( Qt::ArrowCursor ); // reset the cursor to the default cursor
224  }
225  }
226 
227  bool update = false; // if a viewport update will be requested
228 
229  if( m_isLeftInteracting )
230  {
231  // Shutter right side position
232  double rightSide = rect.x() + rect.width();
233 
234  if( coord.getX() < rightSide - m_clickCatchRange)
235  {
236  if( coord.getX() >= sceneRect.x() )
237  {
238  rect.setX( coord.getX() );
239  }
240  else
241  {
242  rect.setX( sceneRect.x() );
243  }
244  }
245  else
246  {
247  rect.setX( rightSide - m_clickCatchRange );
248  }
249 
250  update = true;
251  }
252  else if( m_isRightInteracting )
253  {
254  double newWidth = coord.getX() - shutterCoordPair.first;
255  double shutterRightPos = abs(sceneRect.x()) + shutterCoordPair.first + newWidth;
256 
257  if( newWidth > m_clickCatchRange ) // Shutter's width must be greater than the allowed picking range
258  {
259  if( shutterRightPos < sceneRect.width() )
260  {
261  rect.setWidth( newWidth );
262  }
263  else
264  {
265  rect.setWidth( sceneRect.width() - shutterCoordPair.first - abs(sceneRect.x()) );
266  }
267  }
268  else
269  {
270  rect.setWidth( m_clickCatchRange );
271  }
272 
273  update = true;
274  }
275  else if( m_isInteracting )
276  {
277  double offset = coord.getX() - m_dragStartPoint.getX();
278  double newX = m_dragStartShutterPos.getX() + offset;
279  double shutterRightPos = abs(sceneRect.x()) + newX + shutterWidth;
280 
281  if( newX >= sceneRect.x() && shutterRightPos < sceneRect.width() )
282  {
283  rect.setX( newX );
284  }
285  else if( newX < sceneRect.x() )
286  {
287  rect.setX( sceneRect.x() );
288  }
289  else if( shutterRightPos >= sceneRect.width() )
290  {
291  rect.setX( sceneRect.width() - shutterWidth - abs(sceneRect.x()) );
292  }
293 
294  rect.setWidth( shutterWidth );
295  update = true;
296  }
297 
298  if( update )
299  {
300  // Update graphical shutter
301  m_shutter->setRect( rect );
302  m_layer->removeFromGroup( m_shutter );
303  m_layer->addToGroup( m_shutter );
304 
305  // Update object
306  updateViewportFromShutter( rect.x(), rect.y(), rect.width(), rect.height() );
307 
308  ::fwData::Object::ModifiedSignalType::sptr sig;
309  sig = this->getObject< ::scene2D::data::Viewport>()->signal< ::fwData::Object::ModifiedSignalType >(
310  ::fwData::Object::s_MODIFIED_SIG);
311  {
312  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
313  sig->asyncEmit();
314  }
315  }
316  }
317 }

321 void ViewportRangeSelector::updateViewportFromShutter( double _x, double _y, double _width, double _height )
322 {
323  SLM_TRACE_FUNC();
324 
325  ::scene2D::data::Viewport::sptr viewport = this->getObject< ::scene2D::data::Viewport>();
326 
327  Point2DType fromSceneCoord = this->mapSceneToAdaptor(Point2DType( _x, _y ), m_xAxis, m_yAxis );
328  viewport->setX( fromSceneCoord.first );
329  viewport->setY( fromSceneCoord.second );
330 
331  Point2DType pair = this->mapSceneToAdaptor(Point2DType(_width, _height), m_xAxis, m_yAxis);
332  viewport->setWidth( pair.first );
333  viewport->setHeight( this->getScene2DRender()->getViewport()->getHeight() );
334 }

338 bool ViewportRangeSelector::mouseOnShutterMiddle( ::scene2D::data::Coord _coord)
339 {
340  Point2DType shutterCoordPair;
341  shutterCoordPair = this->mapAdaptorToScene( Point2DType( m_shutter->rect().x(),
342  m_shutter->rect().y() ), m_xAxis, m_yAxis );
343 
344  return ( _coord.getX() > m_shutter->rect().x() + m_clickCatchRange )
345  && ( _coord.getX() < m_shutter->rect().x() + m_shutter->rect().width() - m_clickCatchRange );
346 }

350 bool ViewportRangeSelector::mouseOnShutterLeft( ::scene2D::data::Coord _coord)
351 {
352  std::pair< double, double > shutterCoordPair;
353  shutterCoordPair =
354  this->mapAdaptorToScene( std::pair< double, double >( m_shutter->rect().x(),
355  m_shutter->rect().y() ), m_xAxis, m_yAxis );
356 
357  return ( _coord.getX() >= shutterCoordPair.first - m_clickCatchRange )
358  && ( _coord.getX() <= shutterCoordPair.first + m_clickCatchRange );
359 }

363 bool ViewportRangeSelector::mouseOnShutterRight( ::scene2D::data::Coord _coord)
364 {
365  std::pair< double, double > shutterCoordPair;
366  shutterCoordPair =
367  this->mapAdaptorToScene( std::pair< double, double >( m_shutter->rect().x(),
368  m_shutter->rect().y() ), m_xAxis, m_yAxis );

371 
372  return ( _coord.getX() >= shutterRightPos - m_clickCatchRange )
373  && ( _coord.getX() <= shutterRightPos + m_clickCatchRange );
