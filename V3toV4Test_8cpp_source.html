52 void V3ToV4Test::applyPatchTest()
53 {
54  ::fwAtoms::Object::sptr matObjV3 = ::fwAtoms::Object::New();
55  ::fwAtoms::Object::sptr matObjV4;
56 
57  ::fwAtomsPatch::helper::setClassname(matObjV3, "::fwData::Material");
58  ::fwAtomsPatch::helper::setVersion(matObjV3, "3");
59 
60  ::fwAtomsPatch::helper::Object helper(matObjV3);
61 
62  ::fwAtoms::Sequence::sptr ambientSeq = ::fwAtoms::Sequence::New();
63  ambientSeq->push_back(::fwAtoms::Numeric::New(1.f));
64  ambientSeq->push_back(::fwAtoms::Numeric::New(0.5f));
65  ambientSeq->push_back(::fwAtoms::Numeric::New(0.6f));
66  ambientSeq->push_back(::fwAtoms::Numeric::New(0.12f));
67  ::fwAtoms::Object::sptr ambient = ::fwAtoms::Object::New();
68  ambient->setAttribute("rgba", ambientSeq);
69 
70  ::fwAtoms::Sequence::sptr diffuseSeq = ::fwAtoms::Sequence::New();
71  diffuseSeq->push_back(::fwAtoms::Numeric::New(0.33f));
72  diffuseSeq->push_back(::fwAtoms::Numeric::New(0.32f));
73  diffuseSeq->push_back(::fwAtoms::Numeric::New(0.2f));
74  diffuseSeq->push_back(::fwAtoms::Numeric::New(0.125f));
75  ::fwAtoms::Object::sptr diffuse = ::fwAtoms::Object::New();
76  diffuse->setAttribute("rgba", diffuseSeq);
77 
78  helper.addAttribute("shading_mode", ::fwAtoms::String::New("PHONG"));
79  helper.addAttribute("representation_mode", ::fwAtoms::String::New("SURFACE"));
80  helper.addAttribute("options_mode", ::fwAtoms::String::New("STANDARD"));
81  helper.addAttribute("ambient", ambient);
82  helper.addAttribute("diffuse", diffuse);
83  helper.addAttribute("diffuse_texture", ::fwAtoms::Object::sptr());
84  helper.addAttribute("diffuse_texture_filtering", ::fwAtoms::String::New("NEAREST"));
85  helper.addAttribute("diffuse_texture_wrapping", ::fwAtoms::String::New("CLAMP"));
86  helper.addAttribute("lighting", ::fwAtoms::Boolean::New(false));
87 
88  // 1. Check that if lighting is false, then shading_mode is set to AMBIENT
89  // 2. Check that diffuse is replaced by previous ambient and ambient is 0.05f
90  {
91  matObjV4 = ::fwAtoms::Object::dynamicCast(matObjV3->clone());
92 
93  ::fwAtomsPatch::IPatch::NewVersionsType newVersions;
94  newVersions[matObjV3] = matObjV4;
95 
96  ::fwStructuralPatch::fwData::Material::V3ToV4::sptr mat2ToMat3Patch;
97  mat2ToMat3Patch = ::fwStructuralPatch::fwData::Material::V3ToV4::New();
98  CPPUNIT_ASSERT_NO_THROW(mat2ToMat3Patch->apply(matObjV3, matObjV4, newVersions));
99 
100  CPPUNIT_ASSERT(matObjV4);
101  CPPUNIT_ASSERT(!matObjV4->getAttribute("lighting"));
102  ::fwAtoms::String::sptr shading = ::fwAtoms::String::dynamicCast(matObjV4->getAttribute("shading_mode"));
103  CPPUNIT_ASSERT(shading->getValue() == "AMBIENT");
104 
105  ::fwAtoms::Object::sptr ambientV4 = ::fwAtoms::Object::dynamicCast(matObjV4->getAttribute("ambient"));
106  CPPUNIT_ASSERT(ambientV4);
107  ::fwAtoms::Sequence::sptr ambientSeqV4 = ::fwAtoms::Sequence::dynamicCast(ambientV4->getAttribute("rgba"));
108  CPPUNIT_ASSERT(ambientSeqV4);
109 
110  ::fwAtoms::Object::sptr diffuseV4 = ::fwAtoms::Object::dynamicCast(matObjV4->getAttribute("diffuse"));
111  CPPUNIT_ASSERT(diffuseV4);
112  ::fwAtoms::Sequence::sptr diffuseSeqV4 = ::fwAtoms::Sequence::dynamicCast(diffuseV4->getAttribute("rgba"));
113  CPPUNIT_ASSERT(diffuseSeqV4);
114 
115  CPPUNIT_ASSERT_EQUAL(std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeq)[0])->getValue<float>(),
116  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*diffuseSeqV4)[0])->getValue<float>());
117  CPPUNIT_ASSERT_EQUAL(std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeq)[1])->getValue<float>(),
118  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*diffuseSeqV4)[1])->getValue<float>());
119  CPPUNIT_ASSERT_EQUAL(std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeq)[2])->getValue<float>(),
120  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*diffuseSeqV4)[2])->getValue<float>());
121  CPPUNIT_ASSERT_EQUAL(std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeq)[3])->getValue<float>(),
122  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*diffuseSeqV4)[3])->getValue<float>());
123 
124  CPPUNIT_ASSERT_EQUAL(0.05f,
125  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeqV4)[0])->getValue<float>());
126  CPPUNIT_ASSERT_EQUAL(0.05f,
127  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeqV4)[1])->getValue<float>());
128  CPPUNIT_ASSERT_EQUAL(0.05f,
129  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeqV4)[2])->getValue<float>());
130  CPPUNIT_ASSERT_EQUAL(1.f,
131  std::dynamic_pointer_cast< ::fwAtoms::Numeric >((*ambientSeqV4)[3])->getValue<float>());
132  }
133 
134  // Check that if lighting is true, then shading_mode is not modifed
135  {
136  helper.replaceAttribute("lighting", ::fwAtoms::Boolean::New(true));
137 
138  matObjV4 = ::fwAtoms::Object::dynamicCast(matObjV3->clone());
139 
140  ::fwAtomsPatch::IPatch::NewVersionsType newVersions;
141  newVersions[matObjV3] = matObjV4;
142 
143  ::fwStructuralPatch::fwData::Material::V3ToV4::sptr mat2ToMat3Patch;
144  mat2ToMat3Patch = ::fwStructuralPatch::fwData::Material::V3ToV4::New();
145  CPPUNIT_ASSERT_NO_THROW(mat2ToMat3Patch->apply(matObjV3, matObjV4, newVersions));
146 
147  CPPUNIT_ASSERT(matObjV4);
148  CPPUNIT_ASSERT(!matObjV4->getAttribute("lighting"));
149  ::fwAtoms::String::sptr shading = ::fwAtoms::String::dynamicCast(matObjV4->getAttribute("shading_mode"));
150  CPPUNIT_ASSERT(shading->getValue() == "PHONG");
151  }
