44 void GzBufferImageWriter::write()
45 {
46  assert( getFile().empty() == false );
47 
48  ::fwData::Image::sptr image = getConcreteObject();
49  OSLM_TRACE( "GzBufferImageWriter::write()" << image.get() << " " << image->className());
50 
52  gzFile rawFile = gzopen( getFile().string().c_str(), "wb1");
53  SLM_ASSERT("rawFile not instanced", rawFile);
54  if ( rawFile == 0 )
55  {
56  std::string str = "GzBufferImageWriter::write unable to open ";
57  str += getFile().string();
58  gzclose(rawFile);
59  throw std::ios_base::failure(str);
60  }
61 
62  ::fwComEd::helper::Image imageHelper(image);
63 
64  // file is OK : process now
65  size_t imageSizeInBytes = image->getSizeInBytes();
66 
67  char *ptr = static_cast<char*>(imageHelper.getBuffer());
68  size_t writtenBytes = 0;
69 
70  int uncompressedbyteswrited;
71 
72  while ( writtenBytes < imageSizeInBytes
73  && (uncompressedbyteswrited = gzwrite(rawFile, ptr+writtenBytes, imageSizeInBytes-writtenBytes)) > 0 )
74  {
75  writtenBytes += uncompressedbyteswrited;
76  }
77 
78  gzclose(rawFile);
79 
80  assert( uncompressedbyteswrited != 0 && writtenBytes==imageSizeInBytes);
81 
82  if ( uncompressedbyteswrited != 0 && writtenBytes==imageSizeInBytes)
83  {
84  std::string str = "GzBufferImageWriter::write unable to write ";
85  str += getFile().string();
86  throw std::ios_base::failure(str);
87  }
88 }
