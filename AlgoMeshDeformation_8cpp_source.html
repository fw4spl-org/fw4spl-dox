76 void AlgoMeshDeformation::initSimu()
77 {
78  SLM_TRACE_FUNC();
79  m_originPoints = ::fwData::Object::copy( m_mesh.lock()->getPointsArray() );
80  m_step = 0;
81 
82  if ( !m_mesh.lock()->getPointColorsArray() )
83  {
84  ::fwDataTools::Mesh::colorizeMeshPoints( m_mesh.lock() );
85  }
86 
87  m_meshHelper = ::fwComEd::helper::Mesh::New(m_mesh.lock());
88 
89  float max = std::numeric_limits<float>::min();
90  float min = std::numeric_limits<float>::max();
91 
92  ::fwData::Mesh::PointsMultiArrayType points = m_meshHelper->getPoints();
93  float coord;
94  for(unsigned int i = 0; i < m_nbPoints; ++i)
95  {
96  coord = points[i][1];
97  if ( coord < min )
98  {
99  min = coord;
100  }
101  if ( coord > max )
102  {
103  max = coord;
104  }
105  }
106 
107  m_yCenter = (max - min) / 2 + min;
108 }

112 void AlgoMeshDeformation::computeSimu()
113 {
114  SLM_TRACE_FUNC();
115  m_step += m_direction;
116  if ( m_step == m_nbStep )
117  {
118  m_direction = -1;
119  }
120  else if ( m_step == 0 )
121  {
122  m_direction = 1;
123  }
124 
125  const float scale = m_step / (float) m_nbStep;
126 
127  ::fwComEd::helper::Array originPointsHelper(m_originPoints);
128 
129  ::fwData::Mesh::PointsMultiArrayType points = m_meshHelper->getPoints();
130  ::fwData::Mesh::PointColorsMultiArrayType colors = m_meshHelper->getPointColors();
131 
132  ::fwData::Mesh::PointsMultiArrayType opoints =
133  ::fwData::Mesh::PointsMultiArrayType(
134  static_cast< ::fwData::Mesh::PointsMultiArrayType::element* >(originPointsHelper.getBuffer()),
135  ::boost::extents[m_nbPoints][3] );
136 
137  for(unsigned int i = 0; i < m_nbPoints; ++i)
138  {
139  points[i][0] = opoints[i][0];
140  OSLM_TRACE("opoints[i][1] - m_yCenter = " << opoints[i][1] - m_yCenter);
141  if( opoints[i][1] - m_yCenter > 0 )
142  {
143  points[i][1] = opoints[i][1] + (opoints[i][1] - m_yCenter) * scale;
144  colors[i][0] = ::fwTools::numericRoundCast< ::fwData::Mesh::ColorValueType >(255 * scale);
145  }
146  else
147  {
148  colors[i][0] = 0;
149  }
150  points[i][2] = opoints[i][2];
151  }
152 
153  ::fwDataTools::Mesh::generatePointNormals( m_mesh.lock() );
154 }
