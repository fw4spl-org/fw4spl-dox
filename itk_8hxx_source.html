00095     // update spacing information ; workaround due to GetSpacing const

00097     for (boost::uint8_t  d=0; d<ITKIMAGE::ImageDimension; ++d)
00098     {
00099         spacing[d] = imageData->getSpacing()[d];

00101     itkImage->SetSpacing(spacing);
00102 
00103     // update origin information ; workaround due to GetOrigin const

00105                  imageData->getOrigin().end(),
00106                  const_cast< typename ITKIMAGE::PointType * >( &itkImage->GetOrigin())->Begin()
00107              );
00108 
00109     itk::ImageRegion< ITKIMAGE::ImageDimension > itkRegion;
00110 
00111     unsigned long nbpixels=1;
00112     for (::boost::uint8_t  d=0; d<ITKIMAGE::ImageDimension; ++d)
00113     {
00114         // itkRegion.SetIndex( d,  static_cast<int>(imageData->getCRefOrigin()[d]) );
00115         itkRegion.SetSize( d,   static_cast<unsigned long>(imageData->getSize()[d]) );
00116         nbpixels *= itkRegion.GetSize()[d];;
00117     }
00118 
00119     itkImage->SetRegions(itkRegion);
00120 
00121     if( bufferManagerIsDataImage )
00122     {
00123         itkImage->GetPixelContainer()->SetImportPointer(static_cast< typename ITKIMAGE::PixelType *>( imageHelper.getBuffer() ) , nbpixels , false );
00124     }
00125     else
00126     {
00127         SLM_ASSERT("Sorry, this method requires that imageData manages its buffer.",   imageData->getDataArray()->getIsBufferOwner() );
00128         itkImage->GetPixelContainer()->SetImportPointer(static_cast< typename ITKIMAGE::PixelType *>( imageHelper.getBuffer() ) , nbpixels , true );
00129         imageData->getDataArray()->setIsBufferOwner( false );
00130     }
00131 
00132     return itkImage;
