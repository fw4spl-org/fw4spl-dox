92 template< class ITKIMAGE>
93 typename ITKIMAGE::Pointer fwDataImageToItkImage( ::fwData::Image::sptr imageData, bool bufferManagerIsDataImage )
94 {
95  // Pre Condition
96  SLM_ASSERT("Sorry, itk image dimension not correspond to fwData image",
97  imageData->getNumberOfDimensions() == ITKIMAGE::ImageDimension );
98 
99  ::fwComEd::helper::Image imageHelper(imageData);
100 
101  typename ITKIMAGE::Pointer itkImage = ITKIMAGE::New();
102 
103  // update spacing information ; workaround due to GetSpacing const
104  typename ITKIMAGE::SpacingType spacing = itkImage->GetSpacing();
105  for (boost::uint8_t d = 0; d<ITKIMAGE::ImageDimension; ++d)
106  {
107  spacing[d] = imageData->getSpacing()[d];
108  }
109  itkImage->SetSpacing(spacing);
110 
111  // update origin information ; workaround due to GetOrigin const
112  std::copy( imageData->getOrigin().begin(),
113  imageData->getOrigin().end(),
114  const_cast< typename ITKIMAGE::PointType * >( &itkImage->GetOrigin())->Begin()
115  );
116 
117  itk::ImageRegion< ITKIMAGE::ImageDimension > itkRegion;
118 
119  unsigned long nbpixels = 1;
120  for (::boost::uint8_t d = 0; d<ITKIMAGE::ImageDimension; ++d)
121  {
122  // itkRegion.SetIndex( d, static_cast<int>(imageData->getCRefOrigin()[d]) );
123  itkRegion.SetSize( d, static_cast<unsigned long>(imageData->getSize()[d]) );
124  nbpixels *= itkRegion.GetSize()[d];
125  }
126 
127  itkImage->SetRegions(itkRegion);
128 
129  if( bufferManagerIsDataImage )
130  {
131  itkImage->GetPixelContainer()->SetImportPointer(static_cast< typename ITKIMAGE::PixelType *>( imageHelper.
132  getBuffer() ), nbpixels,
133  false );
134  }
135  else
136  {
137  SLM_ASSERT("Sorry, this method requires that imageData manages its buffer.",
138  imageData->getDataArray()->getIsBufferOwner() );
139  itkImage->GetPixelContainer()->SetImportPointer(static_cast< typename ITKIMAGE::PixelType *>( imageHelper.
140  getBuffer() ), nbpixels,
141  true );
142  imageData->getDataArray()->setIsBufferOwner( false );
143  }
144 
145  return itkImage;
146 }
