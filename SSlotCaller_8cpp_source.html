00111     BOOST_FOREACH(ConfigurationType elem,  slotCfgs.getElements())

00113         src = elem->getValue();
00114         if( ::boost::regex_match(src, match, re) )
00115         {
00116             OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
00117             uid.assign(match[1].first, match[1].second);
00118             key.assign(match[2].first, match[2].second);
00119 
00120             SLM_ASSERT("Missing hasSlotsId attribute", !uid.empty());
00121             SLM_ASSERT("Missing slotKey attribute", !key.empty());
00122 
00123             m_slotInfos.push_back( std::make_pair(uid, key) );
