72 void SSlotCaller::updating() throw( ::fwTools::Failed )
73 {
74  SLM_TRACE_FUNC();
75 
76  for(SlotInfoType info : m_slotInfos)
77  {
78  HasSlotIDType HasSlotId = info.first;
79  ::fwCom::Slots::SlotKeyType slotKey = info.second;
80 
81  if (::fwTools::fwID::exist(HasSlotId))
82  {
83  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(HasSlotId);
84  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
85  SLM_ASSERT("Object with id " << HasSlotId << " is not a HasSlots", hasSlots);
86 
87  ::fwCom::SlotBase::sptr slot = hasSlots->slot(slotKey);
88 
89  slot->asyncRun();
90  }
91  }
92 }

96 void SSlotCaller::configuring() throw( ::fwTools::Failed )
97 {
98  SLM_TRACE_FUNC();
99  this->initialize();
100 
101  OSLM_ASSERT("Missing slots configuration element in " << this->getID(),
102  m_configuration->hasConfigurationElement("slots"));
103  ConfigurationType cfg = m_configuration->findConfigurationElement("slots");
104 
105  ::fwRuntime::ConfigurationElementContainer slotCfgs = cfg->findAllConfigurationElement("slot");
106 
107  ::boost::regex re("(.*)/(.*)");
108  ::boost::smatch match;
109  std::string src, uid, key;
110 
111  for(ConfigurationType elem : slotCfgs.getElements())
112  {
113  src = elem->getValue();
114  if( ::boost::regex_match(src, match, re) )
115  {
116  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
117  uid.assign(match[1].first, match[1].second);
118  key.assign(match[2].first, match[2].second);
119 
120  SLM_ASSERT("Missing hasSlotsId attribute", !uid.empty());
121  SLM_ASSERT("Missing slotKey attribute", !key.empty());
122 
123  m_slotInfos.push_back( std::make_pair(uid, key) );
124  }
125  }
126 }
