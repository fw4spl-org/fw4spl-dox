73 void SPointListRegistration::updating() throw ( ::fwTools::Failed )
74 {
75  ::fwData::PointList::sptr registeredPL;
76  ::fwData::PointList::sptr referencePL;
77  ::fwData::TransformationMatrix3D::sptr matrix;
78 
79  // Get fw4spl data
80  if(this->isVersion2())
81  {
82  registeredPL = this->getInOut< ::fwData::PointList >("registeredPL");
83  referencePL = this->getInOut< ::fwData::PointList >("referencePL");
84  matrix = this->getInOut< ::fwData::TransformationMatrix3D >("output");
85  }
86  else
87  {
88  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
89  registeredPL = ::fwData::PointList::dynamicCast( (*composite)[ m_registeredPointsKey ] );
90  referencePL = ::fwData::PointList::dynamicCast( (*composite)[ m_referencePointsKey ] );
91  matrix =
92  ::fwData::TransformationMatrix3D::dynamicCast( (*composite)[ m_matrixKey ] );
93  }
94 
95 
96  if( registeredPL->getPoints().size() >= 3 &&
97  registeredPL->getPoints().size() == referencePL->getPoints().size() )
98  {
99  vtkSmartPointer<vtkLandmarkTransform> landmarkTransform = vtkSmartPointer<vtkLandmarkTransform>::New();
100 
101  vtkPoints* sourcePts = vtkPoints::New();
102  vtkPoints* targetPts = vtkPoints::New();
103 
104  // Match each point in both list according to the label
105  for( ::fwData::Point::sptr pointRef : referencePL->getPoints() )
106  {
107  const std::string& labelRef =
108  pointRef->getField< ::fwData::String >(::fwDataTools::fieldHelper::Image::m_labelId )->value();
109 
110  for( ::fwData::Point::sptr pointReg : registeredPL->getPoints() )
111  {
112  const std::string& labelReg =
113  pointReg->getField< ::fwData::String >(::fwDataTools::fieldHelper::Image::m_labelId )->value();
114 
115  if(labelRef == labelReg)
116  {
117  auto coord = pointRef->getRefCoord();
118  sourcePts->InsertNextPoint(coord[0], coord[1], coord[2]);
119 
120  OSLM_ERROR("referencePL : " << pointRef->getField< ::fwData::String >(
121  ::fwDataTools::fieldHelper::Image::m_labelId )->value() );
122  OSLM_ERROR(
123  "referencePL : " << pointRef->getCoord()[0] << " " << pointRef->getCoord()[1] << " " <<
124  pointRef->getCoord()[2] );
125 
126  coord = pointReg->getRefCoord();
127  targetPts->InsertNextPoint(coord[0], coord[1], coord[2]);
128  OSLM_ERROR("registeredPL : " << pointReg->getField< ::fwData::String >(
129  ::fwDataTools::fieldHelper::Image::m_labelId )->value() );
130  OSLM_ERROR(
131  "registeredPL : " << pointReg->getCoord()[0] << " " << pointReg->getCoord()[1] << " " <<
132  pointReg->getCoord()[2] );
133  }
134  }
135  }
136 
137  landmarkTransform->SetSourceLandmarks(sourcePts);
138  landmarkTransform->SetTargetLandmarks(targetPts);
139  landmarkTransform->SetModeToRigidBody();
140  landmarkTransform->Update();
141 
142  // Get the resulting transformation matrix (this matrix takes the source points to the target points)
143  vtkSmartPointer<vtkMatrix4x4> m = landmarkTransform->GetMatrix();
144  m->Invert();
145  for(size_t l = 0; l < 4; ++l)
146  {
147  for(size_t c = 0; c < 4; ++c)
148  {
149  matrix->setCoefficient(l, c, m->GetElement(l,c));
150  }
151  }
152 
153  // Notify Matrix modified
154  auto sig = matrix->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
155  {
156  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
157  sig->asyncEmit();
158  }
159  }
160  else
161  {
162  ::fwGui::dialog::MessageDialog::showMessageDialog("Error",
163  "You must enter 2 or more points for the registration to work.",
164  ::fwGui::dialog::IMessageDialog::WARNING);
165  }
166 }
