00137 void ToolBarRegistrar::manage(std::vector< ::fwGui::container::fwMenuItem::sptr > menuItems )
00138 {
00139     ::fwGui::container::fwMenuItem::sptr menuItem;
00140     BOOST_FOREACH( SIDToolBarMapType::value_type sid, m_actionSids)
00141     {
00142         OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!", sid.second.first < menuItems.size());
00143         menuItem = menuItems.at( sid.second.first );
00144         ::fwGui::GuiRegistry::registerActionSIDToParentSID(sid.first, m_sid);
00145         if(sid.second.second) //service is auto started?
00146         {
00147             OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
00148             ::fwServices::IService::sptr service = ::fwServices::get( sid.first ) ;
00149             OSLM_ASSERT("Service "<<sid.first <<" must be stopped.", service->isStopped() );
00150             service->start();
00151         }
00152         else
00153         {
00154             bool service_exists = ::fwTools::fwID::exist(sid.first );
00155             if (!service_exists || ::fwServices::get( sid.first )->isStopped())
00156             {
00157                 ::fwGui::GuiRegistry::actionServiceStopping(sid.first);
00158             }
00159             else
00160             {
00161                 ::fwGui::GuiRegistry::actionServiceStarting(sid.first);
00162             }
00163         }
00164     }
