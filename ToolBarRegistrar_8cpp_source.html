55 void ToolBarRegistrar::initialize( ::fwRuntime::ConfigurationElement::sptr configuration)
56 {
57  OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be registry",
58  configuration->getName() == "registry");
59 
60  // index represents associated toolBar with position in toolBars vector
61  unsigned int index = 0;
62  m_callbacks.clear();
63  // initialize m_actionSids map with configuration
64  std::vector < ConfigurationType > vectMenuItems = configuration->find("menuItem");
65  for( ConfigurationType menuItem : vectMenuItems)
66  {
67  SLM_ASSERT("<menuItem> tag must have sid attribute", menuItem->hasAttribute("sid"));
68  if(menuItem->hasAttribute("sid"))
69  {
70  bool start = false;
71  if(menuItem->hasAttribute("start"))
72  {
73  std::string startValue = menuItem->getAttributeValue("start");
74  SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
75  startValue == "yes" || startValue == "no");
76  start = (startValue=="yes");
77  }
78  std::string sid = menuItem->getAttributeValue("sid");
79  OSLM_ASSERT("Action " << sid << " already exists for this toolBar", m_actionSids.find(
80  sid) == m_actionSids.end());
81  m_actionSids[sid] = SIDToolBarMapType::mapped_type(index, start);
82 
83  ::fwGui::ActionCallbackBase::sptr callback;
84  ::fwGui::GuiBaseObject::sptr guiObj = ::fwGui::factory::New(ActionCallbackBase::REGISTRY_KEY);
85  callback = ::fwGui::ActionCallbackBase::dynamicCast(guiObj);
86  OSLM_ASSERT("ClassFactoryRegistry failed for class "<< ::fwGui::ActionCallbackBase::REGISTRY_KEY, callback);
87 
88  callback->setSID(sid);
89  m_callbacks.push_back(callback);
90  }
91  index++;
92  }
93  index = 0;
94  // initialize m_menuSids map with configuration
95  std::vector < ConfigurationType > vectMenus = configuration->find("menu");
96  for( ConfigurationType menu : vectMenus)
97  {
98  SLM_ASSERT("<menu> tag must have sid attribute", menu->hasAttribute("sid"));
99  if(menu->hasAttribute("sid"))
100  {
101  bool start = false;
102  if(menu->hasAttribute("start"))
103  {
104  std::string startValue = menu->getAttributeValue("start");
105  SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
106  startValue == "yes" || startValue == "no");
107  start = (startValue=="yes");
108  }
109  std::string sid = menu->getAttributeValue("sid");
110  OSLM_ASSERT("Action " << sid << " already exists for this toolBar",
111  m_menuSids.find(sid) == m_menuSids.end());
112  m_menuSids[sid] = SIDToolBarMapType::mapped_type(index, start);
113  }
114  index++;
115  }
116 
117  index = 0;
118  // initialize m_menuSids map with configuration
119  std::vector < ConfigurationType > vectEditors = configuration->find("editor");
120  for( ConfigurationType editor : vectEditors)
121  {
122  SLM_ASSERT("<editor> tag must have sid attribute", editor->hasAttribute("sid") || editor->hasAttribute("wid"));
123  if(editor->hasAttribute("sid"))
124  {
125  bool start = false;
126  if(editor->hasAttribute("start"))
127  {
128  std::string startValue = editor->getAttributeValue("start");
129  SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
130  startValue == "yes" || startValue == "no");
131  start = (startValue=="yes");
132  }
133  std::string sid = editor->getAttributeValue("sid");
134  OSLM_ASSERT("Action " << sid << " already exists for this toolBar", m_editorSids.find(
135  sid) == m_editorSids.end());
136  m_editorSids[sid] = SIDToolBarMapType::mapped_type(index, start);
137  }
138  else if(editor->hasAttribute("wid"))
139  {
140  std::string wid = editor->getAttributeValue("wid");
141  m_editorWids[wid] = index;
142  }
143  index++;
144  }
145 }

149 void ToolBarRegistrar::manage(std::vector< ::fwGui::container::fwMenuItem::sptr > menuItems )
150 {
151  ::fwGui::container::fwMenuItem::sptr menuItem;
152  for( SIDToolBarMapType::value_type sid : m_actionSids)
153  {
154  OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!",
155  sid.second.first < menuItems.size());
156  menuItem = menuItems.at( sid.second.first );
157  ::fwGui::GuiRegistry::registerActionSIDToParentSID(sid.first, m_sid);
158  if(sid.second.second) //service is auto started?
159  {
160  OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
161  ::fwServices::IService::sptr service = ::fwServices::get( sid.first );
162  OSLM_ASSERT("Service "<<sid.first <<" must be stopped.", service->isStopped() );
163  service->start();
164  }
165  else
166  {
167  bool service_exists = ::fwTools::fwID::exist(sid.first );
168  if (!service_exists || ::fwServices::get( sid.first )->isStopped())
169  {
170  ::fwGui::GuiRegistry::actionServiceStopping(sid.first);
171  }
172  else
173  {
174  ::fwGui::GuiRegistry::actionServiceStarting(sid.first);
175  }
176  }
177  }
