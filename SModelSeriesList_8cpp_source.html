59 class ValueView
60 {
61 public:
62  virtual std::string apply(::fwData::Object::sptr obj)
63  {
64  if(obj->isA("::fwData::String"))
65  {
66  ::fwData::String::sptr fwValue = ::fwData::String::dynamicCast(obj);
67  return fwValue->getValue();
68  }
69  else if(obj->isA("::fwData::Integer"))
70  {
71  ::fwData::Integer::sptr fwValue = ::fwData::Integer::dynamicCast(obj);
72  return ::boost::lexical_cast<std::string>(fwValue->getValue());
73  }
74  else if(obj->isA("::fwData::Float"))
75  {
76  ::fwData::Float::sptr fwValue = ::fwData::Float::dynamicCast(obj);
77  return ::boost::lexical_cast<std::string>(fwValue->getValue());
78  }
79  else
80  {
81  SLM_WARN(obj->getClassname() + " is not a printable object : ");
82  return "";
83  }
84  }

222 void SModelSeriesList::configuring() throw(fwTools::Failed)
223 {
224  SLM_TRACE_FUNC();
225  this->initialize();
226 
227  if( m_configuration->findConfigurationElement( "enable_hide_all" ) )
228  {
229  const std::string& hide = m_configuration->findConfigurationElement("enable_hide_all")->getValue();
230  SLM_ASSERT("'enable_hide_all' attribute value must be 'true' or 'false' (found '" + hide + "')",
231  hide == "true" || hide == "false");
232  m_enableHideAll = (hide == "true");
233  }
234 
235  const ::fwRuntime::ConfigurationElement::sptr& columns = m_configuration->findConfigurationElement( "columns" );
236  if(columns)
237  {
238  ::fwRuntime::ConfigurationElement::Container::const_iterator cIt = columns->begin();
239  m_tree->setColumnCount(static_cast<int>(columns->size()));
240  QStringList header;
241  for(; cIt != columns->end(); cIt++)
242  {
243  ValueView* view;
244  ::fwRuntime::ConfigurationElement::AttributePair configView = (*cIt)->getSafeAttributeValue("view");
245  if(!configView.first)
246  {
247  view = new ValueView();
248  }
249  else if (configView.second ==("positive"))
250  {
251  view = new PositiveView();
252  }
253 
254  m_displayedInfo.insert(DisplayedInformation::value_type((*cIt)->getValue(), view));
255  header << QString::fromStdString((*cIt)->getName());
256  }
257  m_tree->setHeaderLabels(header);
258  }
259 }

314 void SModelSeriesList::fillTree()
315 {
316  ::fwMedData::ModelSeries::sptr modelSeries = this->getObject< ::fwMedData::ModelSeries >();
317  auto& reconstructions = modelSeries->getReconstructionDB();
318 
319  if(!m_tree->selectedItems().empty())
320  {
321  m_sigEmptiedSelection->asyncEmit();
322  }
323 
324  m_tree->clear();
325 
326  // Create items
327  for(auto const& reconstruction : reconstructions)
328  {
329  QStringList info;
330  for(auto const& cIt : m_displayedInfo)
331  {
332  ::fwData::Object::sptr obj = ::fwDataCamp::getObject(reconstruction, cIt.first);
333  OSLM_ASSERT("Invalid seshat path : '"<< cIt.first <<"'", obj);
334  info << QString::fromStdString(cIt.second->apply(obj));
335  }
336  QTreeWidgetItem* item = new QTreeWidgetItem(info);
337  item->setCheckState(0, Qt::Unchecked);
338  m_tree->addTopLevelItem(item);
339  item->setData(0, Qt::UserRole, QString::fromStdString(reconstruction->getID()));
340  }
341 
342 
343  for(int i = 0; i< m_tree->topLevelItemCount(); i++)
344  {
345  m_tree->resizeColumnToContents(i);
346  }
347 }
