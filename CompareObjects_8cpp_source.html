70  PropType operator()(const camp::UserObject& value)
71  {
72  const camp::Class& metaclass = value.getClass();
73  PropType prop;
74 
75  if (value.pointer())
76  {
77  std::string classname = value.call("classname").to<std::string>();
78 
79  if (metaclass.name() != classname)
80  {
81  const camp::Class& newMetaclass = ::camp::classByName(classname);
82  CompareObjects visitor(value, m_prefix, m_props);
83  newMetaclass.visit(visitor);
84  }
85  else if(classname == "::fwMemory::BufferObject")
86  {
87  ::fwMemory::BufferObject* bo = value.get< ::fwMemory::BufferObject* >();
88  if(bo)
89  {
90  ::fwMemory::BufferObject::Lock lock = bo->lock();
91  if(lock.getBuffer())
92  {
93  char* buffer = static_cast< char* >(lock.getBuffer());
94  std::size_t seed = 0;
95  const std::size_t buffsize = bo->getSize();
96  for(size_t i = 0; i < buffsize; ++i)
97  {
98  ::boost::hash_combine(seed, buffer[i]);
99  }
100  return std::make_pair(m_prefix, ::boost::lexical_cast< std::string >(seed));
101  }
102  }
103  }
104  else
105  {
106  CompareObjects visitor(value, m_prefix, m_props);
107  metaclass.visit(visitor);
108  }
109  }
110  else
111  {
112  OSLM_INFO("try visiting class= '" << metaclass.name() << " but a null pointer was found");
113  }
114  return prop;
115  }

252 void CompareObjects::compare(SPTR(::fwData::Object)objRef, SPTR(::fwData::Object)objComp)
253 throw (::fwCore::Exception)
254 {
255  if(objRef->getClassname() != objComp->getClassname())
256  {
257  std::stringstream ss;
258  ss << "Classnames mismatch : '" << objRef->getClassname() << "' (reference object) vs. '"
259  << objComp->getClassname() << "' (compared object)";
260  throw ::fwCore::Exception(ss.str());
261  }
262 
263  m_objRef = objRef;
264  m_objComp = objComp;
265 
266  SLM_ASSERT("Reference object not defined", m_objRef);
267  m_campObj = ::camp::UserObject(m_objRef.get());
268  const ::camp::Class & classRef = ::camp::classByName(m_objRef->getClassname());
269  classRef.visit(*this);
270  m_propsRef = ::boost::move(*m_props);
271  m_props->clear();
272 
273  SLM_ASSERT("Reference object not defined", m_objComp);
274  m_campObj = ::camp::UserObject(m_objComp.get());
275  const ::camp::Class & classComp = ::camp::classByName(m_objComp->getClassname());
276  classComp.visit(*this);
277  m_propsComp = ::boost::move(*m_props);
278  m_props->clear();
279 
280  for(PropsMapType::value_type prop : m_propsComp)
281  {
282  if(m_propsRef.find(prop.first) != m_propsRef.end())
283  {
284  if(m_propsRef[prop.first] != prop.second)
285  {
286  (*m_props)[prop.first] = prop.second;
287  }
288  }
289  else
290  {
291  (*m_props)[prop.first] = s_MISSING_PROPERTY;
292  }
293  }
294 
295  for(PropsMapType::value_type prop : m_propsRef)
296  {
297  if(m_propsComp.find(prop.first) == m_propsComp.end() && m_props->find(prop.first) == m_props->end())
298  {
299  (*m_props)[prop.first] = s_MISSING_PROPERTY;
300  }
301  }
302 }
