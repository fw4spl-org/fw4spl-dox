37 void LineLayoutManager::createLayout( ::fwGui::container::fwContainer::sptr parent )
38 {
39  SLM_TRACE_FUNC();
40  m_parentContainer = ::fwGuiQt::container::QtContainer::dynamicCast(parent);
41  SLM_ASSERT("dynamicCast fwContainer to QtContainer failed", m_parentContainer);
42 
43  QBoxLayout *layout = new QBoxLayout(QBoxLayout::LeftToRight);
44  layout->setContentsMargins(0, 0, 0, 0);
45 
46  QWidget *qtContainer = m_parentContainer->getQtContainer();
47  Orientation orientation = this->getOrientation();
48 
49  if( orientation == VERTICAL )
50  {
51  layout->setDirection(QBoxLayout::TopToBottom);
52  }
53 
54  if (qtContainer->layout())
55  {
56  QWidget().setLayout(qtContainer->layout());
57  }
58  qtContainer->setLayout(layout);
59 
60  const std::list< ViewInfo> &views = this->getViewsInfo();
61  for ( ViewInfo viewInfo : views)
62  {
63  if(viewInfo.m_isSpacer)
64  {
65  layout->addStretch();
66  }
67  else
68  {
69  QWidget *panel;
70  int border = viewInfo.m_border;
71  if(viewInfo.m_caption.first)
72  {
73  QGroupBox *groupbox = new QGroupBox(qtContainer);
74  groupbox->setTitle(QString::fromStdString(viewInfo.m_caption.second));
75  panel = groupbox;
76  border += groupbox->style()->pixelMetric(QStyle::PM_LayoutTopMargin);
77  }
78  else
79  {
80  panel = new QWidget(qtContainer);
81  }
82  panel->setMinimumSize(std::max(viewInfo.m_minSize.first,0), std::max(viewInfo.m_minSize.second,0));
83  panel->setContentsMargins(border, border,border, border);
84 
85  ::fwGuiQt::container::QtContainer::sptr subContainer = ::fwGuiQt::container::QtContainer::New();
86  subContainer->setQtContainer(panel);
87  m_subViews.push_back(subContainer);
88 
89  if(viewInfo.m_useScrollBar)
90  {
91  QScrollArea *scrollArea = new QScrollArea(qtContainer);
92  scrollArea->setWidget(panel);
93  scrollArea->setWidgetResizable ( true );
94 
95  layout->addWidget( scrollArea );
96  layout->setStretchFactor(scrollArea, viewInfo.m_proportion);
97  }
98  else
99  {
100  layout->addWidget( panel );
101  layout->setStretchFactor(panel, viewInfo.m_proportion);
102  }
103 
104  subContainer->setVisible( viewInfo.m_visible );
105  }
106  }
107 }
