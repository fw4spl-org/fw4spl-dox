151 # Check the header guard consistency
152 def fix_header_guard(path, enableReformat):
153 
154  ret = FormatReturn()
155 
156  file = open(path, 'r')
157  content = file.read()
158  file.close()
159 
160  # Assume the first ifndef is the first header guard
161  match = re.search('#ifndef +(_*(?:[0-9A-Z]+_*)*)$', content, re.MULTILINE)
162  if match == None:
163  common.error("Can't find header guard, maybe check the naming convention ?")
164  common.error(FILEWARN(path))
165  return FormatReturn.Error
166 
167  guard = match.group(1)
168 
169  # Now check that the guard name follows the file name
170  fileupper = path.upper()
171 
172  match = re.search('(?:(?:INCLUDE\/)|(?:SRC\/)).*', fileupper)
173  if match == None:
174  common.error("warning: can't find 'src' or 'include' in the file path." )
175  common.error(" Header guard naming check skipped." )
176  common.error(FILEWARN(path))
177 
178  else:
179 
180  # look if this is a unit test
181  testMatch = re.search('([^\/]*)(?:\/TEST\/TU\/INCLUDE)(.*)', fileupper)
182 
183  if testMatch:
184  expectedGuard = '__' + testMatch.group(1) + '_' + 'UT' + re.sub('/|\.', '_', testMatch.group(2)) + '__'
185  else:
186 
187  # find the last interesting part of the path (we can have SRC or INCLUDE multiple times)
188  splitBySrc = string.split(fileupper, 'SRC/')[-1]
189  splitByInc = string.split(fileupper, 'INCLUDE/')[-1]
190  if len(splitBySrc) < len(splitByInc):
191  fileupper = splitBySrc
192  else:
193  fileupper = splitByInc
194  expectedGuard = '__' + re.sub('/|\.', '_', fileupper) + '__'
195 
196  if guard != expectedGuard:
197  common.note(FILEWARN(path))
198 
199  strNewFile = re.sub(r"([^_])" + guard, r"\1" + expectedGuard, content)
200 
201  if strNewFile != content:
202  if enableReformat:
203  open(path, 'wb').write(strNewFile)
204  content = strNewFile
205  common.note("The header guard does not follow our naming convention." )
206  common.note(" Replacing " + guard + " by " + expectedGuard + " ." )
207  ret.add(FormatReturn.Modified)
208  else:
209  common.error("The header guard does not follow our naming convention." )
210  return FormatReturn.Error
211 
212 
213  match = re.search('#define +' + expectedGuard, content, re.MULTILINE)
214  if match == None:
215  common.error("Can't find #define header guard matching : " + expectedGuard)
216  common.error(FILEWARN(path))
217  return FormatReturn.Error
218 
219  match = re.search('#endif */[/\*] *' + expectedGuard + '(?:.*\*/)?', content, re.MULTILINE)
220  if match == None:
221  common.error("Can't find #endif with a comment matching the header guard : " + expectedGuard + "\n")
222  common.error(FILEWARN(path))
223  return FormatReturn.Error
