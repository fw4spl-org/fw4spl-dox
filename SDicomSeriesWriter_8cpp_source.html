90 void SDicomSeriesWriter::updating() throw(::fwTools::Failed)
91 {
92  if( this->hasLocationDefined() )
93  {
94  // Retrieve dataStruct associated with this service
95  ::fwMedData::DicomSeries::sptr series = this->getObject< ::fwMedData::DicomSeries >();
96  const ::boost::filesystem::path& folder = this->getFolder();
97  if(!::boost::filesystem::is_empty(folder))
98  {
99  ::fwGui::dialog::MessageDialog dialog;
100  dialog.setMessage("Folder '"+folder.string()+"' isn't empty, files can be overwritten."
101  "\nDo you want to continue ?");
102  dialog.setTitle("Folder not empty.");
103  dialog.setIcon(::fwGui::dialog::MessageDialog::QUESTION);
104  dialog.addButton( ::fwGui::dialog::MessageDialog::YES_NO );
105  ::fwGui::dialog::MessageDialog::Buttons button = dialog.show();
106 
107  if(button == ::fwGui::dialog::MessageDialog::NO)
108  {
109  return;
110  }
111  }
112 
113  if (series->getModality() == "OT")
114  {
115  ::fwGui::dialog::MessageDialog dialog;
116  dialog.setMessage("Series modality is '" + series->getModality() + "' some information can be lost."
117  "\nDo you want to continue ?");
118  dialog.setTitle("Series modality.");
119  dialog.setIcon(::fwGui::dialog::MessageDialog::QUESTION);
120  dialog.addButton( ::fwGui::dialog::MessageDialog::YES_NO );
121  ::fwGui::dialog::MessageDialog::Buttons button = dialog.show();
122 
123  if(button == ::fwGui::dialog::MessageDialog::NO)
124  {
125  return;
126  }
127  }
128  ::fwGui::Cursor cursor;
129  cursor.setCursor(::fwGui::ICursor::BUSY);
130  this->saveDicomSeries(folder, series);
131  cursor.setDefaultCursor();
132  }
133 }

138 void SDicomSeriesWriter::saveDicomSeries( const ::boost::filesystem::path folder,
139  ::fwMedData::DicomSeries::sptr series ) const
140 {
141  ::fwGdcmIO::helper::DicomSeriesWriter::sptr writer = ::fwGdcmIO::helper::DicomSeriesWriter::New();
142 
143  writer->setObject(series);
144  ::fwData::location::Folder::sptr loc = ::fwData::location::Folder::New();
145  loc->setFolder(folder);
146  writer->setLocation(loc);
147  m_sigJobCreated->emit(writer->getJob());
148 
149  try
150  {
151  ::fwGui::dialog::ProgressDialog progressMeterGUI("Saving series ");
152  writer->write();
153  }
154  catch (const std::exception& e)
155  {
156  std::stringstream ss;
157  ss << "Warning during saving : " << e.what();
158  ::fwGui::dialog::MessageDialog::showMessageDialog(
159  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
160  }
161  catch( ... )
162  {
163  ::fwGui::dialog::MessageDialog::showMessageDialog(
164  "Warning", "Warning during saving", ::fwGui::dialog::IMessageDialog::WARNING);
165  }
166 }
