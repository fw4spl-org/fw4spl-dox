00051 void MenuRegistrar::initialize( ::fwRuntime::ConfigurationElement::sptr configuration)
00052 {
00053     OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be registry",
00054             configuration->getName() == "registry");
00055 
00056     // index represents associated menu with position in menus vector
00057     unsigned int index = 0;
00058     m_callbacks.clear();
00059     // initialize m_actionSids map with configuration
00060     std::vector < ConfigurationType > vectMenuItems = configuration->find("menuItem");
00061     BOOST_FOREACH( ConfigurationType menuItem, vectMenuItems)
00062     {
00063         SLM_ASSERT("<menuItem> tag must have sid attribute", menuItem->hasAttribute("sid"));
00064         if(menuItem->hasAttribute("sid"))
00065         {
00066             bool start = false;
00067             if(menuItem->hasAttribute("start"))
00068             {
00069                 std::string startValue = menuItem->getAttributeValue("start");
00070                 SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
00071                         startValue == "yes" || startValue == "no");
00072                 start = (startValue=="yes");
00073             }
00074             std::string sid = menuItem->getAttributeValue("sid");
00075             OSLM_ASSERT("Action " << sid << " already exists for this menu", m_actionSids.find(sid) == m_actionSids.end());
00076             m_actionSids[sid] = SIDMenuMapType::mapped_type(index, start);
00077 
00078             ::fwGui::ActionCallbackBase::sptr callback ;
00079             ::fwGui::GuiBaseObject::sptr guiObj = ::fwGui::factory::New(ActionCallbackBase::REGISTRY_KEY);
00080             callback = ::fwGui::ActionCallbackBase::dynamicCast(guiObj);
00081             OSLM_ASSERT("ClassFactoryRegistry failed for class "<< ::fwGui::ActionCallbackBase::REGISTRY_KEY, callback);
00082 
00083             callback->setSID(sid);
00084             m_callbacks.push_back(callback);
00085         }
00086         index++;
00087     }
00088 
00089     index = 0;
00090     // initialize m_actionSids map with configuration
00091     std::vector < ConfigurationType > vectMenus = configuration->find("menu");
00092     BOOST_FOREACH( ConfigurationType menu, vectMenus)
00093     {
00094         SLM_ASSERT("<menu> tag must have sid attribute", menu->hasAttribute("sid"));
00095         if(menu->hasAttribute("sid"))
00096         {
00097             bool start = false;
00098             if(menu->hasAttribute("start"))
00099             {
00100                 std::string startValue = menu->getAttributeValue("start");
00101                 SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
00102                         startValue == "yes" || startValue == "no");
00103                 start = (startValue=="yes");
00104             }
00105             std::string sid = menu->getAttributeValue("sid");
00106             std::pair<int, bool> indexStart =  std::make_pair( index, start);
00107             OSLM_ASSERT("Action " << sid << " already exists for this toolBar", m_actionSids.find(sid) == m_actionSids.end());
00108             m_menuSids[sid] = indexStart;
00109         }
00110         index++;
00111     }

00116 void MenuRegistrar::manage(std::vector< ::fwGui::container::fwMenuItem::sptr > menuItems )
00117 {
00118     ::fwGui::container::fwMenuItem::sptr menuItem;
00119     BOOST_FOREACH( SIDMenuMapType::value_type sid, m_actionSids)
00120     {
00121         OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!", sid.second.first < menuItems.size());
00122         menuItem = menuItems.at( sid.second.first );
00123         ::fwGui::GuiRegistry::registerActionSIDToParentSID(sid.first, m_sid);
00124         if(sid.second.second) //service is auto started?
00125         {
00126             OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
00127             ::fwServices::IService::sptr service = ::fwServices::get( sid.first ) ;
00128             OSLM_ASSERT("Service "<<sid.first <<" must be stopped.", service->isStopped() );
00129             service->start();
00130         }
00131         else
00132         {
00133             bool service_exists = ::fwTools::fwID::exist(sid.first );
00134             if (!service_exists || ::fwServices::get( sid.first )->isStopped())
00135             {
00136                 ::fwGui::GuiRegistry::actionServiceStopping(sid.first);
00137             }
00138             else
00139             {
00140                 ::fwGui::GuiRegistry::actionServiceStarting(sid.first);
00141             }
00142         }
00143     }
