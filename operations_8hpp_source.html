91 template<typename T>
92 T * createExecutableInstance(
93  const std::shared_ptr<ConfigurationElement> element,
94  const std::string & attribute = "class" ) throw(RuntimeException)
95 {
96 
97  // Retrieves the executable type.
98  if( element->hasAttribute( attribute ) == false )
99  {
100  throw RuntimeException( "Configuration element has no attribute '" + attribute + "'." );
101  }
102  const std::string type( element->getExistingAttributeValue(attribute) );
103 
104  // Creates the executable instance.
105  Runtime * rntm ( Runtime::getDefault() );
106  std::auto_ptr< IExecutable > executable ( rntm->createExecutableInstance(type, element) );
107 
108  // Converts the executable instance to the right type.
109  T * result = dynamic_cast<T*>( executable.get() );
110  if( result == 0 )
111  {
112  throw RuntimeException( "Executable creation failed. Bad cast" );
113  }
114  executable.release();
115 
116  // That's all folks !
117  return result;
118 }

223 template< typename Container, typename T >
224 const Container getAllExecutableForPoint( const std::string & identifier,
225  const std::string & attribute = "class" ) throw(RuntimeException)
226 {
227  // Defines the element container
228  typedef std::vector< std::shared_ptr< ConfigurationElement > > ConfigurationElementContainer;
229 
230  // Retrieves all configuration elements.
231  ConfigurationElementContainer elements( getAllConfigurationElementsForPoint< ConfigurationElementContainer >(
232  identifier) );
233 
234  // Defines an insert iterator type for the executable container.
235  typedef std::back_insert_iterator< Container > Inserter;
236 
237  // Walks through collected configuration elements and create desired executable instances
238  Container result;
239  ConfigurationElementContainer::iterator iElement;
240  Inserter iInserter( result );
241  for( iElement = elements.begin(); iElement != elements.end(); ++iElement, ++iInserter )
242  {
243  std::shared_ptr< ConfigurationElement > element( *iElement );
244  std::shared_ptr< T > executable( createExecutableInstance< T >(element, attribute) );
245 
246  iInserter = executable;
247  }
248  return result;
249 }
