261 void TimerAsio::rearmNoLock(TimeDurationType duration)
262 {
263  this->cancelNoLock();
264  ::boost::posix_time::time_duration d =
265  ::boost::posix_time::microseconds(boost::chrono::duration_cast<boost::chrono::microseconds>(duration).count());
266  m_timer.expires_from_now( d );
267  m_timer.async_wait( ::boost::bind(&TimerAsio::call, this, _1));

270 void TimerAsio::call(const ::boost::system::error_code & error)
271 {
272  if(!error)
273  {
274  TimeDurationType duration;
275  bool oneShot;
276  {
277  ::fwCore::mt::ScopedLock lock(m_mutex);
278  oneShot = m_oneShot;
279  duration = m_duration;
280  }
281 
282  if (!oneShot)
283  {
284  this->rearmNoLock(duration);
285  m_function();
286  }
287  else

291  m_running = false;
292  }
