00049 
00050     std::fstream file;

00053

00056         OSLM_ERROR( "Trian file writing error for " << getFile().string());

00061 
00062     ::fwComEd::helper::Mesh meshHelper(mesh);
00063

00068     for( i=0 ; i<nbPts ; ++i )
00069     {
00070         file << points[i][0] << " " << points[i][1] << " " << points[i][2] << std::endl;
00071     }
00072 
00073     nbCells = mesh->getNumberOfCells();
00074     ::fwData::Array::sptr cells = mesh->getCellDataArray();
00075 
00076     ::fwComEd::helper::Array cellsArrayHelper(cells);
00077 
00078     FW_RAISE_IF("Not able to write " << cells->getType().string() << " cell type in trian file.",
00079             cells->getType() != ::fwTools::Type::create< ::boost::uint64_t >());
00080 
00081     ::boost::uint64_t *cellBuf = cellsArrayHelper.begin< ::boost::uint64_t >();
00082     ::boost::uint64_t *cellBufEnd = cellBuf + 3*nbCells;
00083 
00084     SLM_ASSERT("Wrong CellDataMultiArray size", cells->getNumberOfElements() >= nbCells*3);
00085     file << nbCells << std::endl;
00086 
00087     ::fwData::Array::sptr normals = mesh->getCellNormalsArray();
00088 
00089     if(normals
00090             && !normals->empty()
00091             && normals->getType() == ::fwTools::Type::create<float>()
00092             && normals->getNumberOfComponents() == 3
00093             && normals->getNumberOfDimensions() == 1
00094             && nbCells == normals->getSize().at(0)
00095             )
00096     {
00097         ::fwComEd::helper::Array normalsArrayHelper(normals);
00098         float *normalBuf = normalsArrayHelper.begin< float >();
00099 
00100         while (cellBuf != cellBufEnd)
00101         {
00102             file << (*cellBuf++) << " " ;
00103             file << (*cellBuf++) << " " ;
00104             file << (*cellBuf++) << " " ;
00105             file << (*normalBuf++) << " " ;
00106             file << (*normalBuf++) << " " ;
00107             file << (*normalBuf++) << std::endl;
00108         }
00109     }
00110     else
00111     {
00112         while (cellBuf != cellBufEnd)
00113         {
00114             file << (*cellBuf++) << " ";
00115             file << (*cellBuf++) << " ";
00116             file << (*cellBuf++) << " -1 -1 -1" << std::endl;
00117         }
00118     }
00119     file.close();
00120 }
