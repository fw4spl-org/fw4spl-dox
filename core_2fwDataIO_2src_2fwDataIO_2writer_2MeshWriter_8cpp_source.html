44 void MeshWriter::write()
45 {
46  OSLM_INFO( "[MeshReader::read] Trian file: " << getFile());
47  assert( getFile().empty() == false );
48 
49  ::fwData::Mesh::sptr mesh = this->getConcreteObject();
50  FW_RAISE_IF("Can't convert this Mesh to Trian file",
51  !::fwDataTools::Mesh::hasUniqueCellType(mesh, ::fwData::Mesh::TRIANGLE));
52 
53  std::fstream file;
54  file.open(getFile().string().c_str(), std::fstream::out);
55  file.precision(std::numeric_limits< float >::digits10);
56 
57  if (!file.is_open())
58  {
59  OSLM_ERROR( "Trian file writing error for " << getFile().string());
60  std::string str = "Unable to write ";
61  str += getFile().string();
62  throw std::ios_base::failure(str);
63  }
64 
65  ::fwDataTools::helper::Mesh meshHelper(mesh);
66 
67  size_t i, nbPts, nbCells;
68  nbPts = mesh->getNumberOfPoints();
69  ::fwData::Mesh::PointsMultiArrayType points = meshHelper.getPoints();
70  file<<nbPts<<std::endl;
71  for( i = 0; i<nbPts; ++i )
72  {
73  file << points[i][0] << " " << points[i][1] << " " << points[i][2] << std::endl;
74  }
75 
76  nbCells = mesh->getNumberOfCells();
77  ::fwData::Array::sptr cells = mesh->getCellDataArray();
78 
79  ::fwDataTools::helper::Array cellsArrayHelper(cells);
80 
81  FW_RAISE_IF("Not able to write " << cells->getType().string() << " cell type in trian file.",
82  cells->getType() != ::fwTools::Type::create< ::boost::uint64_t >());
83 
84  ::boost::uint64_t* cellBuf = cellsArrayHelper.begin< ::boost::uint64_t >();
85  ::boost::uint64_t* cellBufEnd = cellBuf + 3*nbCells;
86 
87  SLM_ASSERT("Wrong CellDataMultiArray size", cells->getNumberOfElements() >= nbCells*3);
88  file << nbCells << std::endl;
89 
90  ::fwData::Array::sptr normals = mesh->getCellNormalsArray();
91 
92  if(normals
93  && !normals->empty()
94  && normals->getType() == ::fwTools::Type::create<float>()
95  && normals->getNumberOfComponents() == 3
96  && normals->getNumberOfDimensions() == 1
97  && nbCells == normals->getSize().at(0)
98  )
99  {
100  ::fwDataTools::helper::Array normalsArrayHelper(normals);
101  float* normalBuf = normalsArrayHelper.begin< float >();
102 
103  while (cellBuf != cellBufEnd)
104  {
105  file << (*cellBuf++) << " ";
106  file << (*cellBuf++) << " ";
107  file << (*cellBuf++) << " ";
108  file << (*normalBuf++) << " ";
109  file << (*normalBuf++) << " ";
110  file << (*normalBuf++) << std::endl;
111  }
112  }
113  else
114  {
115  while (cellBuf != cellBufEnd)
116  {
117  file << (*cellBuf++) << " ";
118  file << (*cellBuf++) << " ";
119  file << (*cellBuf++) << " -1 -1 -1" << std::endl;
120  }
121  }
122  file.close();
123 }
