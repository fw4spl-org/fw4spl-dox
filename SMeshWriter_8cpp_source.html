103 void SMeshWriter::updating() throw(::fwTools::Failed)
104 {
105  SLM_TRACE_FUNC();
106 
107  if( this->hasLocationDefined() )
108  {
109  // Retrieve dataStruct associated with this service
110  ::fwData::Mesh::sptr pMesh = this->getObject< ::fwData::Mesh >();
111  SLM_ASSERT("pMesh not instanced", pMesh);
112 
113  ::fwGui::Cursor cursor;
114  cursor.setCursor(::fwGui::ICursor::BUSY);
115 
116  ::fwVtkIO::MeshWriter::sptr writer = ::fwVtkIO::MeshWriter::New();
117 
118  m_sigJobCreated->emit(writer->getJob());
119 
120  writer->setObject(pMesh);
121  writer->setFile(this->getFile());
122 
123  try
124  {
125  writer->write();
126  }
127  catch (const std::exception & e)
128  {
129  std::stringstream ss;
130  ss << "Warning during saving : " << e.what();
131 
132  ::fwGui::dialog::MessageDialog::showMessageDialog(
133  "Warning",
134  ss.str(),
135  ::fwGui::dialog::IMessageDialog::WARNING);
136  }
137  catch( ... )
138  {
139  ::fwGui::dialog::MessageDialog::showMessageDialog(
140  "Warning",
141  "Warning during saving",
142  ::fwGui::dialog::IMessageDialog::WARNING);
143  }
144 
145  cursor.setDefaultCursor();
146  }
147 }
