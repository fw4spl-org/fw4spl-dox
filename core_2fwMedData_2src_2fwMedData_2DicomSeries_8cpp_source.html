95 bool DicomSeries::isInstanceAvailable(std::size_t instanceIndex) const
96 {
97  DicomPathContainerType::const_iterator localPathIter;
98 
99  bool available = false;
100 
101  switch(m_dicomAvailability)
102  {
103  case NONE:
104  case PATHS:
105  case BLOB:
106  localPathIter = m_localDicomPaths.find(instanceIndex);
107  available = localPathIter != m_localDicomPaths.end() && ::boost::filesystem::exists(
108  localPathIter->second);
109  break;
110  case BINARIES:
111  available = instanceIndex < m_dicomBinaries.size();
112  break;
113  default:
114  SLM_ASSERT("You shall not pass.",0);
115  }
116 
117  return available;
118 }
