88 IJob::SharedFuture IJob::cancel()
89 {
90  ::fwCore::mt::ReadToWriteLock lock(m_mutex);
91  if( m_cancelable && (m_state == WAITING || m_state == RUNNING) )
92  {
93  State nextState = (m_state == WAITING) ? CANCELED : CANCELING;
94 
95  {
96  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
97  m_cancelRequested = true;
98  this->setStateNoLock(CANCELING);
99  }
100 
101  // unlock mutex for cancel callbacks sanity
102  lock.unlock();
103  // m_cancelHooks can not be changed when m_state != WAITING or
104  // RUNNING, no need to lock m_mutex
105  for(auto cancel : m_cancelHooks)
106  {
107  (cancel)(*this);
108  }
109  lock.lock();
110 
111  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
112 
113  SLM_ASSERT("State shall be only CANCELING or CANCELED, not " << m_state,
114  m_state == CANCELED || m_state == CANCELING);
115 
116  if(m_state == CANCELING)
117  {
118  this->setStateNoLock(nextState);
119 
120  if(nextState == CANCELED)
121  {
122  m_runFuture = ::boost::make_shared_future();
123  }
124  }
125  }
126  // else if (m_state == CANCELING)
127  // NOP
128  return m_runFuture;
129 }

198 IJob::SharedFuture IJob::run()
199 {
200  ::fwCore::mt::ReadToWriteLock lock(m_mutex);
201 
202  if( m_state == WAITING )
203  {
204  {
205  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
206  this->setStateNoLock(RUNNING);
207  }
208 
209  lock.unlock();
210  auto future = this->runImpl();
211  lock.lock();
212 
213  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
214  m_runFuture = future;
215  }
216 
217  return m_runFuture;
218 }

230 void IJob::setStateNoLock(IJob::State state)
231 {
232  m_state = state;
233  switch(state)
234  {
235  case WAITING:
236  break;
237  case RUNNING:
238  m_sigStarted->asyncEmit();
239  break;
240  case CANCELING:
241  m_sigCancelRequested->asyncEmit();
242  break;
243  case CANCELED:
244  m_sigCanceled->asyncEmit();
245  break;
246  case FINISHED:
247  m_sigFinished->asyncEmit();
248  break;
249  default:
250  SLM_ASSERT("You shall not pass !", 0);
251  }
252 
253  std::for_each(m_stateHooks.begin(), m_stateHooks.end(),
254  [&]( const StateHookSeq::value_type& f )
255  {
256  f(m_state);
257  } );
258 }

385 
386 void IJob::doneWork( std::uint64_t units, ::fwCore::mt::ReadToWriteLock &lock )
387 {
388  auto oldDoneWork = m_doneWorkUnits;
389  decltype(m_doneWorkHooks)doneWorkHooks;
390 
391  if(m_doneWorkUnits == units)
392  {
393  return;
394  }
395 
396  doneWorkHooks = m_doneWorkHooks;
397 
398  {
399  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
400  m_doneWorkUnits = units;
401  }
402 
403  m_sigDoneWork->asyncEmit(m_doneWorkUnits, m_totalWorkUnits);
404 
405  lock.unlock();
406 
407  std::for_each(doneWorkHooks.begin(), doneWorkHooks.end(),
408  [&]( const DoneWorkHookSeq::value_type& f )
409  {
410  f(*this, oldDoneWork);
411  } );
412 }

425 {

427  decltype(m_totalWorkUnitsHooks)totalWorkUnitsHook;

430 
431  if(m_doneWorkUnits > units)
432  {

434  }

436  {
437  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
438  m_totalWorkUnits = units;
