63 const ::boost::filesystem::path ILocationDialog::getDefaultLocation()
64 {
65  ::boost::filesystem::path defaultPath;
66  ::fwData::Composite::sptr prefUI = this->getPreferenceUI();
67  ::fwData::location::ILocation::sptr location;
68  if(prefUI)
69  {
70  if ( prefUI->find( ILocationDialog::DLG_DEFAULT_LOCATION ) != prefUI->end() )
71  {
72  location = ::fwData::location::ILocation::dynamicCast( (*prefUI)[ ILocationDialog::DLG_DEFAULT_LOCATION ] );
73  SLM_ASSERT("LOCATION not correct", location);
74  }
75  }
76 
77  if(!location)
78  {
79  location = m_defaultLocaction;
80  }
81 
82  ::fwData::location::SingleFile::csptr singleFile = ::fwData::location::SingleFile::dynamicConstCast(location);
83  ::fwData::location::Folder::csptr folder = ::fwData::location::Folder::dynamicConstCast(location);
84  if (singleFile)
85  {
86  defaultPath = singleFile->getPath();
87  }
88  else if (folder)
89  {
90  defaultPath = folder->getFolder();
91  }
92 
93  return defaultPath;
94 }

109 ::fwData::Composite::sptr ILocationDialog::getPreferenceUI()
110 {
111  ::fwData::Composite::sptr prefUI;
112 
113  // Get preferences
114  std::vector< ::fwServices::IService::sptr > preferencesServicesList = ::fwServices::OSR::getServices(
115  "::preferences::IPreferencesService");
116  if(!preferencesServicesList.empty())
117  {
118  ::fwServices::IService::sptr prefService = preferencesServicesList[0];
119  ::fwData::Composite::sptr prefs = prefService->getObject< ::fwData::Composite >();
120 
121  ::fwData::Composite::sptr framesUI;
122  // Retrieves software UI pref
123  if ( prefs->find( ILocationDialog::SOFTWARE_UI ) != prefs->end() )
124  {
125  framesUI = ::fwData::Composite::dynamicCast( (*prefs)[ ILocationDialog::SOFTWARE_UI ]);
126  }
127  else
128  {
129  framesUI = ::fwData::Composite::New();
130  (*prefs)[ ILocationDialog::SOFTWARE_UI ] = framesUI;
131  }
132  // Retrieves associated dialog UI pref
133  if ( framesUI->find( this->getTitle() ) != framesUI->end() )
134  {
135  prefUI = ::fwData::Composite::dynamicCast( (*framesUI)[ this->getTitle() ] );
136  }
137  else
138  {
139  prefUI = ::fwData::Composite::New();
140  (*framesUI)[ this->getTitle() ] = prefUI;
141  }
142  }
143  return prefUI;
144 }
