60 void SFilterSelectorDialog::configuring() throw( ::fwTools::Failed )
61 {
62  SLM_TRACE_FUNC();
63 
64  m_selectedFilters.clear();
65 
66  // Config Elem
67  // <selection mode="exclude">
68  // <addSelection filter="::fwDicomIOFilter::composite::CTImageStorageDefaultComposite" />
69  // <addSelection filter="::fwDicomIOFilter::composite::CTImageStorageDefaultComposite" />
70 
71  ::fwRuntime::ConfigurationElementContainer::Iterator iter = this->m_configuration->begin();
72  for(; iter != this->m_configuration->end(); ++iter )
73  {
74  SLM_INFO( "SFilterSelectorDialog " + (*iter)->getName());
75 
76  if( (*iter)->getName() == "selection" )
77  {
78  SLM_ASSERT( "The xml element <selection> must have the attribute 'mode'.", (*iter)->hasAttribute("mode"));
79  const std::string mode = (*iter)->getExistingAttributeValue("mode");
80  m_filtersAreExcluded = ( mode == "exclude" );
81  SLM_ASSERT( "The xml attribute <mode> must be either 'exclude' or 'include'.", mode == "exclude" ||
82  mode == "include" );
83  SLM_DEBUG( "mode => " + mode );
84  }
85 
86  if( (*iter)->getName() == "addSelection" )
87  {
88  SLM_ASSERT( "The xml element <addSelection> must have the attribute 'filter'.",
89  (*iter)->hasAttribute("filter"));
90  m_selectedFilters.push_back( (*iter)->getExistingAttributeValue("filter") );
91  SLM_DEBUG( "add selection => " + (*iter)->getExistingAttributeValue("filter") );
92  }
93 
94  }
95 
96 }

114 void SFilterSelectorDialog::updating() throw( ::fwTools::Failed )
115 {
116  SLM_TRACE_FUNC();
117 
118  // Retrieve available filters
119  std::vector< ::fwDicomIOFilter::IFilter::sptr > registredFilters;
120  for(std::string key: ::fwDicomIOFilter::registry::get()->getFactoryKeys())
121  {
122  ::fwDicomIOFilter::IFilter::sptr filter = ::fwDicomIOFilter::factory::New(key);
123  registredFilters.push_back(filter);
124  }
125 
126  // Filter available extensions and replace id by service description
127  std::map< std::string, ::fwDicomIOFilter::IFilter::sptr > availableFiltersMap;
128  std::vector< std::string > availableFilterNames;
129 
130  for( ::fwDicomIOFilter::IFilter::sptr filter: registredFilters )
131  {
132  const bool filterIsSelectedByUser = std::find( m_selectedFilters.begin(), m_selectedFilters.end(),
133  filter->getClassname() ) != m_selectedFilters.end();
134 
135  // Test if the filter is considered here as available by users
136  // excluded mode => add filters that are not selected by users
137  // included mode => add filters selected by users
138  if( (m_filtersAreExcluded && !filterIsSelectedByUser) ||
139  (!m_filtersAreExcluded && filterIsSelectedByUser) )
140  {
141  // Add this filter
142  std::string filterName = filter->getName();
143  filterName = (filterName.empty()) ? filter->getClassname() : filterName;
144  availableFiltersMap[filterName] = filter;
145  availableFilterNames.push_back( filterName );
146  }
147  }
148 
149  // Sort available services (lexical string sort)
150  std::sort( availableFilterNames.begin(), availableFilterNames.end() );
151 
152  // Test if we have an extension
153  if ( !availableFilterNames.empty() )
154  {
155  std::string filterName = *availableFilterNames.begin();
156  bool filterSelectionIsCanceled = false;
157 
158  // Selection of extension when availableFilterNames.size() > 1
159  if ( availableFilterNames.size() > 1 )
160  {
161  ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
162 
163  selector->setTitle("Filter to use");
164  selector->setSelections(availableFilterNames);
165  filterName = selector->show();
166  filterSelectionIsCanceled = filterName.empty();
167 
168  SLM_ASSERT("Unable to find the selected filter name in the filter map.",
169  filterSelectionIsCanceled || availableFiltersMap.find(filterName) != availableFiltersMap.end() );
170  }
171 
172  if ( !filterSelectionIsCanceled )
173  {
174 
175  ::fwDicomIOFilter::IFilter::sptr filter = availableFiltersMap[filterName];
176 
177  ::fwData::String::sptr obj = this->getObject< ::fwData::String >();
178  SLM_ASSERT("The filter selector service must work on a ::fwData::String object.", obj);
179  obj->setValue(filter->getClassname());
180 
181  auto sig
182  = obj->signal< ::fwData::Object::ModifiedSignalType>( ::fwData::Object::s_MODIFIED_SIG );
183  {
184  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
185  sig->asyncEmit();
186  }
187  }
188  }
189  else
190  {
191  SLM_WARN("SFilterSelectorDialog::load : availableFilters is empty.");
192  ::fwGui::dialog::MessageDialog messageBox;
193  messageBox.setTitle("Filter not found");
194  messageBox.setMessage( "There is no available filter for this reader." );
195  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
196  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
197  messageBox.show();
198  }
199 }
