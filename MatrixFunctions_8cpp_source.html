00037 
00038 //     Rotation Matrix
00039 //     [     cos(z)           sin(z)         0     0 ]
00040 //     [                                             ]
00041 //     [ - cos(x) sin(z)   cos(x) cos(z)   sin(x)  0 ]

00044 //     [                                             ]

00046 
00047     R[0][0] =  cos(RZ);
00048     R[0][1] =  sin(RZ);
00049     R[0][2] =  0;
00050     R[0][3] =  0.;
00051 
00052     R[1][0] = -cos(RX) * sin(RZ);
00053     R[1][1] =  cos(RX) * cos(RZ);
00054     R[1][2] =  sin(RX);
00055     R[1][3] =  0.;
00056 
00057     R[2][0] =  sin(RX) * sin(RZ);
00058     R[2][1] = -sin(RX) * cos(RZ);
00059     R[2][2] =  cos(RX);
00060     R[2][3] =  0.;

00062     R[3][2] =  0.;
