59 void SConfigController::configuring() throw(fwTools::Failed)
60 {
61  if(this->isVersion2())
62  {
63  typedef ::fwRuntime::ConfigurationElement::sptr ConfigType;
64  typedef ::fwRuntime::EConfigurationElement::sptr EditableConfigType;
65 
66  ConfigType cfg = this->getConfiguration();
67 
68  ConfigType appCfg = cfg->findConfigurationElement("appConfig");
69  SLM_ASSERT("Missing 'appConfig' tag.", appCfg);
70  std::string appCfgId = appCfg->getAttributeValue("id");
71 
72  EditableConfigType srvCfg = ::fwRuntime::EConfigurationElement::New( "service" );
73  EditableConfigType newCfg = srvCfg->addConfigurationElement("config");
74  EditableConfigType newAppCfg = newCfg->addConfigurationElement("appConfig");
75  newAppCfg->setAttributeValue("id", appCfgId);
76 
77  EditableConfigType newParamsCfg = newAppCfg->addConfigurationElement("parameters");
78 
79  const std::vector< ConfigType > inoutsCfg = cfg->find("inout");
80  for (const auto& inoutCfg : inoutsCfg)
81  {
82  std::string key = inoutCfg->getAttributeValue("key");
83  SLM_ASSERT("[" + appCfgId + "] Missing 'key' tag.", !key.empty());
84 
85  std::string uid = inoutCfg->getAttributeValue("uid");
86  SLM_ASSERT("[" + appCfgId + "] Missing 'uid' tag.", !uid.empty());
87 
88  EditableConfigType newParamCfg = newParamsCfg->addConfigurationElement("parameter");
89  newParamCfg->setAttributeValue("replace", key);
90 
91  auto obj = this->getInOut< ::fwData::Object>(key);
92  newParamCfg->setAttributeValue("uid", obj->getID());
93  }
94 
95  const std::vector< ConfigType > paramsCfg = cfg->find("parameter");
96  for (const auto& paramCfg : paramsCfg)
97  {
98  std::string replace = paramCfg->getAttributeValue("replace");
99  SLM_ASSERT("[" + appCfgId + "] Missing 'replace' tag.", !replace.empty());
100 
101  EditableConfigType newParamCfg = newParamsCfg->addConfigurationElement("parameter");
102  newParamCfg->setAttributeValue("replace", replace);
103 
104  std::string uid = paramCfg->getAttributeValue("uid");
105  if(!uid.empty())
106  {
107  newParamCfg->setAttributeValue("uid", uid);
108  }
109  else
110  {
111  SLM_ASSERT("[" + appCfgId + "] Missing 'uid' or 'by' tag.", paramCfg->hasAttribute("by"));
112  std::string by = paramCfg->getAttributeValue("by");
113  // 'by' can be empty
114  newParamCfg->setAttributeValue("by", by);
115  }
116  }
117 
118  m_configLauncher->parseConfig(::fwRuntime::Convert::toPropertyTree(srvCfg));
119  }
120  else
121  {
122  m_configLauncher->parseConfig(this->getConfigTree());
123  }
124 }
