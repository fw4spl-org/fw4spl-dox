48 void TagReader::read()
49 {
50  assert( ::std::dynamic_pointer_cast< ::fwData::location::SingleFile >(m_location) );
51  ::boost::filesystem::path path =
52  ::std::dynamic_pointer_cast< ::fwData::location::SingleFile >(m_location)->getPath();
53 
54 
55  OSLM_INFO( "[TagReader::read] Tag file: " << path);
56  assert( path.empty() == false );
57 
58  ::std::shared_ptr< ::fwData::Tag > tag = getConcreteObject();
59 
60  std::fstream file;
61  file.open(path.string().c_str(), std::fstream::in);
62  if (!file.is_open())
63  {
64  OSLM_ERROR( "Tag file loading error for " << path.string());
65  std::string str = "Unable to open ";
66  str += path.string();
67  throw std::ios_base::failure(str);
68  }
69 
71  std::string name, type;
72  int n;
73  file>>name;
74  file>>n;
75  if(n>=1)
76  {
77  file>>type;
78 
79  if(type=="ARTAG" || type=="CHESSBOARD" || type=="ARToolKitPlus_MARKER_ID_BCH")
80  {
81  int nbPts;
82  double x,y,z;
83  double radius = 0.0;
84  file>>x>>y>>z;
85  file>>nbPts;
86  tag->setType(type);
87  for(int i = 0; i < nbPts; i++)
88  {
89  ::fwData::Point::sptr p;
90  fwVec3d vPoint;
91  file>>vPoint[0]>>vPoint[1]>>vPoint[2]>>radius;
92  p->setCoord(vPoint);
93  tag->getRefPointList()->getRefPoints().push_back(p);
94  }
95  }
96  else
97  {
98  SLM_ERROR( "Tag file loading error for " + path.string() + " with type " + type);
99  std::string str = "Unable to open ";
100  str += path.string();
101  throw std::ios_base::failure(str);
102  }
103  }
104 
105  file.close();
106 
107 }
