147 void SlicesCursor::buildPolyData()
148 {
149  int nbPoints = 8;
150  vtkPoints* points = vtkPoints::New(VTK_DOUBLE);
151  points->SetNumberOfPoints(nbPoints);
152  int i;
153  for (i = 0; i < nbPoints; i++)
154  {
155  //points->SetPoint(i, 300*i +0.0, 50*i*i + 0.0 , 0.0);
156  points->SetPoint(i, 0.0, 0.0, 0.0);
157  }
158 
159  vtkCellArray *cells = vtkCellArray::New();
160  cells->Allocate(cells->EstimateSize(nbPoints,2));
161 
162  vtkIdType pts[2];
163  for ( int line = 0; line<4; ++line)
164  {
165  pts[0] = line; pts[1] = line+4;
166  //cells->InsertNextCell(2,pts);
167  vtkLine *lineCell = vtkLine::New();
168  lineCell->GetPointIds()->SetId(0, line );
169  lineCell->GetPointIds()->SetId(1, line+ 4 );
170  cells->InsertNextCell(lineCell);
171  lineCell->Delete();
172  }
173 
174  m_cursorPolyData->SetPoints(points);
175  points->Delete();
176  m_cursorPolyData->SetLines(cells);
177  cells->Delete();
178  this->setVtkPipelineModified();
179 }

193 void SlicesCursor::computeCrossPoints( double _ptA[3], double _ptB[3], double _ptP[3], double _scale,
194  double _ptAprime[3], double _ptBprime[3] )
195 {
196  double ptPBprime[3];
197  double norm2PBprime = 0.0;
198  double norm2BP = 0;
199  double norm2AP = 0;
200 
201  // Compute AB x scale / 2 = |A'P| x AB = |PB'| x AB
202  for (int i = 0; i<3; ++i )
203  {
204  ptPBprime[i] = ( _ptB[i] - _ptA[i] ) * (1-_scale)/2.0;
205  norm2PBprime += ptPBprime[i]*ptPBprime[i];
206  norm2AP += (_ptP[i] - _ptA[i])*(_ptP[i] - _ptA[i]);
207  norm2BP += (_ptP[i] - _ptB[i])*(_ptP[i] - _ptB[i]);
208  }
209 
210  // Compute APrime = P + PA' = P - |A'P| x AB
211  // BPrime = P + pB' = P + |PB'| x AB
212  if ( norm2PBprime > norm2BP )
213  {
214  for (int i = 0; i<3; ++i )
215  {
216  _ptAprime[i] = _ptP[i] - ptPBprime[i];
217  _ptBprime[i] = _ptB[i];
218  }
219  }
220  else if ( norm2PBprime > norm2AP )
221  {
222  for (int i = 0; i<3; ++i )
223  {
224  _ptAprime[i] = _ptA[i];
225  _ptBprime[i] = _ptP[i] + ptPBprime[i];
226  }
227  }
228  else
229  {
230  for (int i = 0; i<3; ++i )
231  {
232  _ptAprime[i] = _ptP[i] - ptPBprime[i];
233  _ptBprime[i] = _ptP[i] + ptPBprime[i];
234  }
235  }
236 }

240 void SlicesCursor::buildColorAttribute()
241 {
242  unsigned char red[3] = {255, 0, 0};
243  unsigned char green[3] = {0, 255, 0};
244  unsigned char blue[3] = {0, 0, 255};
245 
246  typedef unsigned char* RGBColor;
247  typedef std::map< std::string, std::pair< RGBColor, RGBColor> > DicoType;
248  DicoType dict;
249  dict["colorXAxis"] = std::make_pair(green,red);
250  dict["colorYAxis"] = std::make_pair(red, blue);
251  dict["colorZAxis"] = std::make_pair(blue, green );
252 
253 
254  for ( DicoType::iterator i = dict.begin(); i!= dict.end(); ++i )
255  {
256  vtkUnsignedCharArray* colors = vtkUnsignedCharArray::New();
257  colors->SetNumberOfComponents(3);
258  colors->SetName( i->first.c_str() );
259  colors->InsertNextTupleValue( i->second.first );
260  colors->InsertNextTupleValue( i->second.second );
261  colors->InsertNextTupleValue( i->second.first );
262  colors->InsertNextTupleValue( i->second.second );
263  m_cursorPolyData->GetCellData()->AddArray(colors);
264  }
265  m_cursorMapper->SetScalarModeToUseCellFieldData();
266  this->setVtkPipelineModified();
267 }

308 void SlicesCursor::updateImageSliceIndex( ::fwData::Image::sptr image )
309 {
310  float scale = m_isSelected ? 1.0 : m_scale;
311  if (scale <= 0)
312  {
313  m_cursorActor->VisibilityOff();
314  }
315  else
316  {
317  m_cursorActor->VisibilityOn();
318 
319  unsigned int pos[3];
320 
321  pos[2] = m_axialIndex->value();
322  pos[1] = m_frontalIndex->value();
323  pos[0] = m_sagittalIndex->value();
324 
325  double sliceWorld[3];
326  for (int dim = 0; dim<3; ++dim )
327  {
328  sliceWorld[dim] = pos[dim]*image->getSpacing()[dim] + image->getOrigin().at(dim);
329  }
330 
331  double cursorPoints[8][3]; // point AB,BC,CD,AD,ABM,BCM,CDM,ADM
332 
333  for ( int p = 0; p<2; ++p )
334  {
335  for (int dim = 0; dim<3; ++dim )
336  {
337  //cursorPoints[p][dim] = ((m_orientation==dim || ( (dim != m_orientation) && p ) )?sliceWorld[dim] : 0 );
338  //cursorPoints[p+2][dim] = ((m_orientation==dim || ( (dim != m_orientation) && p ) )?sliceWorld[dim] : (image->getSize()[dim]-1)*image->getSpacing()[dim] );
339  cursorPoints[p][dim] = sliceWorld[dim];
340  cursorPoints[p+2][dim] = sliceWorld[dim];
341  if ( (dim + p + 1)%3 == m_orientation )
342  {
343  cursorPoints[p][dim] = image->getOrigin().at(dim);
344  cursorPoints[p+2][dim] = (image->getSize()[dim]-1)*image->getSpacing()[dim] + image->getOrigin().at(
345  dim);
346  }
347  }
348  }
349 
350  // Compute ABM & CDM
351  computeCrossPoints( cursorPoints[0], cursorPoints[2], sliceWorld, scale, cursorPoints[4], cursorPoints[6] );
352  // Compute BCM & ADM
353  computeCrossPoints( cursorPoints[1], cursorPoints[3], sliceWorld, scale, cursorPoints[5], cursorPoints[7] );
354 
355  vtkPoints* points = m_cursorPolyData->GetPoints();
356 
357  for ( int i = 0; i < 8; ++i)
358  {
359  points->SetPoint(i,cursorPoints[i]);
360  }
361 
362  }
363  m_cursorPolyData->Modified();
364  this->setVtkPipelineModified();
365 }
