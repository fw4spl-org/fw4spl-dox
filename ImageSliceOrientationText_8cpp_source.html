00240 
00241         if( toSliceType == static_cast<int>(m_orientation) )
00242         {
00243             this->setOrientation( static_cast< Orientation >( fromSliceType ));
00244         }
00245         else if(fromSliceType == static_cast<int>(m_orientation))
00246         {
00247             this->setOrientation( static_cast< Orientation >( toSliceType ));
00248         }
00249     }
00250     else if (msg->hasEvent(::fwComEd::ImageMsg::BUFFER)
00251               || msg->hasEvent(::fwComEd::ImageMsg::NEW_IMAGE)
00252               ||msg->hasEvent(::fwComEd::ImageMsg::MODIFIED))
