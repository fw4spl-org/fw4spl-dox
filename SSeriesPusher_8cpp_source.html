98 void SSeriesPusher::updating() throw(::fwTools::Failed)
99 {
100  ::fwData::Vector::csptr selectedSeries = this->getInput< ::fwData::Vector >("selectedSeries");
101 
102  if(m_isPushing)
103  {
104  // Display a message to inform the user that the service is already pushing data.
105  ::fwGui::dialog::MessageDialog messageBox;
106  messageBox.setTitle("Pushing Series");
107  messageBox.setMessage( "The service is already pushing data. Please wait until the pushing is done "
108  "before sending a new push request." );
109  messageBox.setIcon(::fwGui::dialog::IMessageDialog::INFO);
110  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
111  messageBox.show();
112  }
113  else if(selectedSeries->empty())
114  {
115  // Display a message to inform the user that there is no series selected.
116  ::fwGui::dialog::MessageDialog messageBox;
117  messageBox.setTitle("Pushing Series");
118  messageBox.setMessage( "Unable to push series, there is no series selected." );
119  messageBox.setIcon(::fwGui::dialog::IMessageDialog::INFO);
120  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
121  messageBox.show();
122  }
123  else
124  {
125  // Initialize enquirer
126  m_seriesEnquirer->initialize(
127  m_pacsConfiguration->getLocalApplicationTitle(),
128  m_pacsConfiguration->getPacsHostName(),
129  m_pacsConfiguration->getPacsApplicationPort(),
130  m_pacsConfiguration->getPacsApplicationTitle(),
131  m_pacsConfiguration->getMoveApplicationTitle(),
132  m_slotProgressCallback);
133 
134 
135  // Set pushing boolean to true
136  m_isPushing = true;
137 
138  // Check whether some selected series are already on the PACS or not
139  bool pushOK = this->checkSeriesOnPACS();
140  if(pushOK)
141  {
142  // Push series to the PACS
143  m_pushSeriesWorker->post(::boost::bind(&::ioPacs::SSeriesPusher::pushSeries, this));
144  }
145 
146  }

150 //------------------------------------------------------------------------------
151 bool SSeriesPusher::checkSeriesOnPACS()
152 {
153  // Return true if the push operation must be performed
154  bool result = true;
155 
156  ::fwData::Vector::csptr seriesVector = this->getInput< ::fwData::Vector >("selectedSeries");
157 
158  // Catch any errors
159  try
160  {
161  // Find which selected series must be pushed
162  DicomSeriesContainerType duplicateSeriesVector;
163 
164  // Connect to PACS
165  m_seriesEnquirer->connect();
166 
167  ::fwData::Vector::ConstIteratorType it = seriesVector->begin();
168  for(; it != seriesVector->end(); ++it)
169  {
170  ::fwMedData::DicomSeries::csptr series = ::fwMedData::DicomSeries::dynamicCast(*it);
171  SLM_ASSERT("The SeriesDB should contain only DicomSeries.", series);
172 
173  // Try to find series on PACS
174  OFList< QRResponse* > responses;
175  responses = m_seriesEnquirer->findSeriesByUID(series->getInstanceUID());
176 
177  // If the series has been found on the PACS
178  if(responses.size() > 1)
179  {
180  duplicateSeriesVector.push_back(series);
181  }
182 
183  ::fwPacsIO::helper::Series::releaseResponses(responses);
184  }
185 
186  // Disconnect from PACS
187  m_seriesEnquirer->disconnect();
188 
189  // Inform the user that some series are already on the PACS
190  if(!duplicateSeriesVector.empty())
191  {
192  ::std::stringstream ss;
193  ss << "Those series are already on the PACS: \n";
194 
195  // Display duplicated Series
196  for(const ::fwMedData::Series::csptr& series: duplicateSeriesVector)
197  {
198  std::string description = series->getDescription();
199  description = (description.empty()) ? "[No description]" : description;
200  ss << "- " << description << std::endl;
201  }
202 
203  ss << std::endl << "Would you like to perform the operation anyway ?" << std::endl
204  << "(This will result in a merge operation)";
205 
206  ::fwGui::dialog::MessageDialog messageBox;
207  messageBox.setTitle("Duplicate series");
208  messageBox.setMessage( ss.str() );
209  messageBox.setIcon(::fwGui::dialog::IMessageDialog::INFO);
210  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
211  messageBox.addButton(::fwGui::dialog::IMessageDialog::CANCEL);
212  ::fwGui::dialog::IMessageDialog::Buttons answer = messageBox.show();
213 
214  result = (answer == ::fwGui::dialog::IMessageDialog::OK);
215 
216  }
217  }
218  catch (::fwPacsIO::exceptions::Base& exception)
219  {
220  ::std::stringstream ss;
221  ss << "Unable to connect to the pacs. Please check your configuration: \n"
222  << "Pacs host name: " << m_pacsConfiguration->getPacsHostName() << "\n"
223  << "Pacs application title: " << m_pacsConfiguration->getPacsApplicationTitle() << "\n"
224  << "Pacs port: " << m_pacsConfiguration->getPacsApplicationPort() << "\n";
225  m_slotDisplayMessage->asyncRun(ss.str(), true);
226  SLM_WARN(exception.what());
227  result = false;
228 
229  // Set pushing boolean to false
230  m_isPushing = false;
231 
232  }
233 
234  return result;
235 }

239 void SSeriesPusher::pushSeries()
240 {
241  ::fwData::Vector::csptr seriesVector = this->getInput< ::fwData::Vector >("selectedSeries");
242 
243  // Catch any errors
244  try
245  {
246  // List of files that must be pushed
247  DicomFileContainer dicomFilesContainer;
248 
249  // Connect to PACS
250  ::fwData::Vector::ConstIteratorType it = seriesVector->begin();
251  for(; it != seriesVector->end(); ++it)
252  {
253  ::fwMedData::DicomSeries::csptr series = ::fwMedData::DicomSeries::dynamicCast(*it);
254  SLM_ASSERT("The SeriesDB should contain only DicomSeries.", series);
255 
256  BOOST_FOREACH(::fwMedData::DicomSeries::DicomPathContainerType::value_type value,
257  series->getLocalDicomPaths())
258  {
259  dicomFilesContainer.push_back(value.second);
260  }
261  }
262 
263  // Number of instances that must be uploaded
264  m_instanceCount = dicomFilesContainer.size();
265 
266  // Connect from PACS
267  m_seriesEnquirer->connect();
268  m_sigStartedProgress->asyncEmit(m_progressbarId);
269 
270  // Push series
271  m_seriesEnquirer->pushSeries(dicomFilesContainer);
272 
273  // Disconnect from PACS
274  m_seriesEnquirer->disconnect();
275 
276  }
277  catch (::fwPacsIO::exceptions::Base& exception)
278  {
279  ::std::stringstream ss;
280  ss << "Unable to connect to the pacs. Please check your configuration: \n"
281  << "Pacs host name: " << m_pacsConfiguration->getPacsHostName() << "\n"
282  << "Pacs application title: " << m_pacsConfiguration->getPacsApplicationTitle() << "\n"
283  << "Pacs port: " << m_pacsConfiguration->getPacsApplicationPort() << "\n";
284  m_slotDisplayMessage->asyncRun(ss.str(), true);
285  SLM_WARN(exception.what());
286  }
287 
288  // Set pushing boolean to false
289  m_isPushing = false;
290 
291 }

296  const std::string& filePath)
