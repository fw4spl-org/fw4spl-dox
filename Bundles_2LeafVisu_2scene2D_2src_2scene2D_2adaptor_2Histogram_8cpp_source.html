82 void Histogram::doUpdate() throw( ::fwTools::Failed)
83 {
84  SLM_TRACE_FUNC();
85 
86  this->doStop();
87 
88  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
89  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
90 
91  if (!values.empty())
92  {
93  m_layer = new QGraphicsItemGroup();
94  // Update color with opacity
95  QColor color = m_color.color();
96  color.setAlphaF( m_opacity );
97  m_color.setColor( color );
98 
99  const float min = histogram->getMinValue();
100  const float binsWidth = histogram->getBinsWidth();
101 
102  // Initialize the path with a start point:
103  // The value preceding the current value that we'll use to build the arcs of the path
104  std::pair< double, double > startPoint = this->mapAdaptorToScene(
105  std::pair<double, double>(min, values[0]), m_xAxis, m_yAxis);
106 
107  std::pair<double, double> pair;
108 
109  QBrush brush = QBrush(m_color.color());
110 
111  // Build the graphic items:
112  const int nbValues = (int)values.size();
113  for(int i = 1; i < nbValues; ++i)
114  {
115  pair = this->mapAdaptorToScene(
116  std::pair<double, double>(min + i * binsWidth, values[i]), m_xAxis, m_yAxis);
117 
118  QPainterPath painter( QPointF(startPoint.first, 0) );
119  painter.lineTo( startPoint.first, startPoint.second );
120  painter.lineTo( pair.first, pair.second );
121  painter.lineTo( pair.first, 0 );
122 
123  QGraphicsPathItem* item = new QGraphicsPathItem( painter );
124  item->setPath( painter );
125  item->setBrush( brush );
126  item->setPen( Qt::NoPen );
127  item->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
128 
129  m_layer->addToGroup( item );
130 
131  startPoint = pair;
132  }
133 
134  // Adjust the layer's position and zValue depending on the associated axis
135  m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
136  m_layer->setZValue(m_zValue);
137 
138  // Add to the scene the unique item which gather the whole set of rectangle graphic items:
139  this->getScene2DRender()->getScene()->addItem( m_layer );
140  }
141 }

145 void Histogram::updateCurrentPoint( ::scene2D::data::Event::sptr _event )
146 {
147  SLM_TRACE_FUNC();
148 
149  SLM_ASSERT("m_histogramPointUID must be defined in order to update the related ::fwData::Point data.",
150  !m_histogramPointUID.empty());
151 
152  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
153  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
154  const float histogramMinValue = histogram->getMinValue();
155  const float histogramBinsWidth = histogram->getBinsWidth();
156 
157  // Event coordinates in scene
158  ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( _event->getCoord() );
159 
160  const int histIndex = (int) sceneCoord.getX();
161  const int index = histIndex - histogramMinValue;
162  const int nbValues = (int)values.size() * histogramBinsWidth;
163 
164  if(index >= 0 && index < nbValues)
165  {
166  ::fwData::Point::sptr point = this->getSafeInOut< ::fwData::Point>( m_histogramPointUID );
167  SLM_ASSERT("m_histogramPointUID can't be null here.", point);
168 
169  point->getRefCoord()[0] = sceneCoord.getX();
170  point->getRefCoord()[1] = values.at( index / histogramBinsWidth ) * m_scale;
171  }
172 }

195 void Histogram::processInteraction( ::scene2D::data::Event::sptr _event)
196 {
197  SLM_TRACE_FUNC();
198 
199  bool updatePointedPos = false;
200 
201  // Vertical scaling
202  if( _event->getType() == ::scene2D::data::Event::MouseWheelUp )
203  {
204  m_scale *= SCALE;
205  m_layer->setTransform(QTransform::fromScale(1, SCALE), true);
206 
207  //_event->setAccepted( true );
208  m_yAxis->setScale( m_scale );
209 
210  updatePointedPos = true;
211  }
212  else if( _event->getType() == ::scene2D::data::Event::MouseWheelDown )
213  {
214  m_scale /= SCALE;
215  m_layer->setTransform(QTransform::fromScale(1, 1 / SCALE), true);
216 
217  //_event->setAccepted( true );
218  m_yAxis->setScale( m_scale );
219 
220  updatePointedPos = true;
221  }
222  else if( _event->getType() == ::scene2D::data::Event::MouseMove )
223  {
224  updatePointedPos = true;
225  }
226 
227  if( !m_histogramPointUID.empty() && updatePointedPos )
228  {
229  updateCurrentPoint( _event );
230  }
231 }
