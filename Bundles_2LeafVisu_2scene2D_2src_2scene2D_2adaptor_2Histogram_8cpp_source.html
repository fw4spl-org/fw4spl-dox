00083 void Histogram::doUpdate() throw( ::fwTools::Failed)
00084 {
00085     SLM_TRACE_FUNC();
00086 
00087     this->doStop();
00088 
00089     ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
00090     ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
00091 
00092     if (!values.empty())
00093     {
00094         m_layer = new QGraphicsItemGroup();
00095         // Update color with opacity
00096         QColor color = m_color.color();
00097         color.setAlphaF( m_opacity );
00098         m_color.setColor( color );
00099 
00100         const float min = histogram->getMinValue();
00101         const float binsWidth = histogram->getBinsWidth();
00102 
00103         // Initialize the path with a start point:
00104         // The value preceding the current value that we'll use to build the arcs of the path
00105         std::pair< double, double > startPoint = this->mapAdaptorToScene(
00106                 std::pair<double, double>(min, values[0]), m_xAxis, m_yAxis);
00107 
00108         std::pair<double, double> pair;
00109 
00110         QBrush brush = QBrush(m_color.color());
00111 
00112         // Build the graphic items:
00113         const int nbValues = (int)values.size();
00114         for(int i = 1; i < nbValues; ++i)
00115         {
00116             pair = this->mapAdaptorToScene(
00117                     std::pair<double, double>(min + i * binsWidth, values[i]), m_xAxis, m_yAxis);
00118 
00119             QPainterPath painter( QPointF(startPoint.first, 0) );
00120             painter.lineTo( startPoint.first, startPoint.second );
00121             painter.lineTo( pair.first, pair.second );
00122             painter.lineTo( pair.first, 0 );
00123 
00124             QGraphicsPathItem* item = new QGraphicsPathItem( painter );
00125             item->setPath( painter );
00126             item->setBrush( brush );
00127             item->setPen( Qt::NoPen );
00128             item->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
00129 
00130             m_layer->addToGroup( item );
00131 
00132             startPoint = pair;
00133         }
00134 
00135         // Adjust the layer's position and zValue depending on the associated axis
00136         m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
00137         m_layer->setZValue(m_zValue);
00138 
00139         // Add to the scene the unique item which gather the whole set of rectangle graphic items:
00140         this->getScene2DRender()->getScene()->addItem( m_layer );
00141     }
00142 }

00146 void Histogram::updateCurrentPoint( ::scene2D::data::Event::sptr _event )
00147 {
00148     SLM_TRACE_FUNC();
00149 
00150     SLM_ASSERT("m_histogramPointUID must be defined in order to update the related ::fwData::Point data.",
00151             !m_histogramPointUID.empty());
00152 
00153     ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
00154     ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
00155     const float histogramMinValue = histogram->getMinValue();
00156     const float histogramBinsWidth = histogram->getBinsWidth();
00157 
00158     // Event coordinates in scene
00159     ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( _event->getCoord() );
00160 
00161     const int histIndex = (int) sceneCoord.getX();
00162     const int index = histIndex - histogramMinValue;
00163     const int nbValues = (int)values.size() * histogramBinsWidth;
00164 
00165     if(index >= 0 && index < nbValues)
00166     {
00167         ::fwData::Point::sptr point =
00168             ::fwData::Point::dynamicCast( ::fwTools::fwID::getObject( m_histogramPointUID ) );
00169 
00170         SLM_ASSERT("m_histogramPointUID can't be null here.", point);
00171 
00172         point->getRefCoord()[0] = sceneCoord.getX();
00173         point->getRefCoord()[1] = values.at( index / histogramBinsWidth ) * m_scale;
00174     }
00175 }

00209 void Histogram::processInteraction( ::scene2D::data::Event::sptr _event)
00210 {
00211     SLM_TRACE_FUNC();
00212 
00213     bool updatePointedPos = false;
00214 
00215     // Vertical scaling
00216     if( _event->getType() == ::scene2D::data::Event::MouseWheelUp )
00217     {
00218         m_scale *= SCALE;
00219         m_layer->setTransform(QTransform::fromScale(1, SCALE), true);
00220 
00221         //_event->setAccepted( true );
00222         m_yAxis->setScale( m_scale );
00223 
00224         updatePointedPos = true;
00225     }
00226     else if( _event->getType() == ::scene2D::data::Event::MouseWheelDown )
00227     {
00228         m_scale /= SCALE;
00229         m_layer->setTransform(QTransform::fromScale(1, 1 / SCALE), true);
00230 
00231         //_event->setAccepted( true );
00232         m_yAxis->setScale( m_scale );
00233 
00234         updatePointedPos = true;
00235     }
00236     else if( _event->getType() == ::scene2D::data::Event::MouseMove )
00237     {
00238         updatePointedPos = true;
00239     }
00240 
00241     if( !m_histogramPointUID.empty() && updatePointedPos )
00242     {
00243         updateCurrentPoint( _event );
00244     }
