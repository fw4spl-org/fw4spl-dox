82 void Reconstruction::createMeshService()
83 {
84  ::fwData::Reconstruction::sptr reconstruction = this->getObject < ::fwData::Reconstruction >();
85 
86  ::fwData::Mesh::sptr mesh = reconstruction->getMesh();
87 
88  SLM_TRACE_IF("Mesh is null", !mesh);
89  if (mesh)
90  {
91  ::fwRenderVTK::IVtkAdaptorService::sptr meshService;
92  meshService = ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService > (
93  mesh,
94  "::visuVTKAdaptor::Mesh" );
95  SLM_ASSERT("meshService not instanced", meshService);
96  ::visuVTKAdaptor::Mesh::sptr meshAdaptor = Mesh::dynamicCast(meshService);
97 
98  meshService->setRenderService( this->getRenderService() );
99  meshService->setRenderId ( this->getRenderId() );
100  meshService->setPickerId ( this->getPickerId() );
101  meshService->setTransformId ( this->getTransformId() );
102  meshService->setAutoRender ( this->getAutoRender() );
103 
104  meshAdaptor->setClippingPlanesId( m_clippingPlanesId );
105  //meshAdaptor->setSharpEdgeAngle ( m_sharpEdgeAngle );
106  meshAdaptor->setShowClippedPart ( true );
107  meshAdaptor->setMaterial ( reconstruction->getMaterial() );
108  meshAdaptor->setAutoResetCamera ( m_autoResetCamera );
109  meshService->start();
110  meshAdaptor->updateVisibility ( reconstruction->getIsVisible() );
111  meshAdaptor->update();
112 
113  m_meshService = meshService;
114  this->registerService(meshService);
115  OSLM_TRACE("Mesh is visible : "<< reconstruction->getIsVisible());
116  OSLM_TRACE("Mesh nb points : "<< mesh->getNumberOfPoints());
117  }
118 }

130 void Reconstruction::doUpdate() throw(fwTools::Failed)
131 {
132  SLM_TRACE_FUNC();
133 
134  if (!m_meshService.expired())
135  {
136  ::fwRenderVTK::IVtkAdaptorService::sptr meshService = m_meshService.lock();
137 
138  ::fwData::Reconstruction::sptr reconstruction = this->getObject < ::fwData::Reconstruction >();
139  ::visuVTKAdaptor::Mesh::sptr meshAdaptor = Mesh::dynamicCast(meshService);
140  //meshAdaptor->setSharpEdgeAngle( m_sharpEdgeAngle );
141 
142  meshAdaptor->setMaterial ( reconstruction->getMaterial());
143  meshAdaptor->swap ( reconstruction->getMesh() );
144  meshAdaptor->updateVisibility( reconstruction->getIsVisible());
145 
146  }
147  else
148  {
149  this->createMeshService();
150  }
151 }
