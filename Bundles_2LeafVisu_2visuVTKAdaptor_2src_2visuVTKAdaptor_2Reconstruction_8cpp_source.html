00100         ::fwRenderVTK::IVtkAdaptorService::sptr meshService;
00101         meshService = ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService > (

00104         SLM_ASSERT("meshService not instanced", meshService);
00105         ::visuVTKAdaptor::Mesh::sptr meshAdaptor
00106             = Mesh::dynamicCast(meshService);
00107 
00108         meshService->setRenderService( this->getRenderService() );
00109         meshService->setRenderId     ( this->getRenderId()      );
00110         meshService->setPickerId     ( this->getPickerId()      );
00111         meshService->setTransformId  ( this->getTransformId() );
00112         meshService->setAutoRender   ( this->getAutoRender() );
00113 
00114         meshAdaptor->setClippingPlanesId( m_clippingPlanesId             );
00115         //meshAdaptor->setSharpEdgeAngle  ( m_sharpEdgeAngle               );
00116         meshAdaptor->setShowClippedPart ( true );
00117         meshAdaptor->setMaterial        ( reconstruction->getMaterial()  );
00118         meshAdaptor->setAutoResetCamera ( m_autoResetCamera );
00119         meshService->start();
00120         meshAdaptor->updateVisibility  ( reconstruction->getIsVisible() );
00121         meshAdaptor->update();
00122 
00123         m_meshService = meshService;
00124         this->registerService(meshService);
00125         OSLM_TRACE("Mesh is visible : "<< reconstruction->getIsVisible());
00126         OSLM_TRACE("Mesh nb points : "<< mesh->getNumberOfPoints());
