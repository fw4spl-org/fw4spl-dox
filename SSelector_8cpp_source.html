158 void SSelector::configuring() throw(::fwTools::Failed)
159 {
160  this->::fwGui::IGuiContainerSrv::initialize();
161 
162  if(!this->isVersion2())
163  {
164  std::vector < ::fwRuntime::ConfigurationElement::sptr > selectionCfg = m_configuration->find("selectionId");
165  SLM_ASSERT("Missing tag 'selectionId'", !selectionCfg.empty());
166  m_selectionId = selectionCfg.front()->getValue();
167  SLM_ASSERT("selectionId must not be empty", !m_selectionId.empty());
168  }
169 
170  std::vector < ::fwRuntime::ConfigurationElement::sptr > selectionModeCfg = m_configuration->find("selectionMode");
171  if(!selectionModeCfg.empty())
172  {
173  const std::string& selectionMode = selectionModeCfg.front()->getValue();
174 
175  if(!selectionMode.empty())
176  {
177  if(selectionMode == "single")
178  {
179  m_selectionMode = QAbstractItemView::SingleSelection;
180  }
181  else if(selectionMode == "extended")
182  {
183  m_selectionMode = QAbstractItemView::ExtendedSelection;
184  }
185  else
186  {
187  SLM_WARN("value " + selectionMode + " is not managed for <selectionMode>");
188  }
189  }
190  }
191 
192  std::vector < ::fwRuntime::ConfigurationElement::sptr > allowedRemoveCfg = m_configuration->find("allowedRemove");
193  if(!allowedRemoveCfg.empty())
194  {
195  const std::string& allowedRemove = allowedRemoveCfg.front()->getValue();
196 
197  if(allowedRemove == "yes")
198  {
199  m_allowedRemove = true;
200  }
201  else if(allowedRemove == "no")
202  {
203  m_allowedRemove = false;
204  }
205  else
206  {
207  SLM_WARN("value " + allowedRemove + " is not managed for <allowedRemove>");
208  }
209  }
210 
211  std::vector < ::fwRuntime::ConfigurationElement::sptr > insertCfg = m_configuration->find("insertMode");
212  if(!insertCfg.empty())
213  {
214  const std::string& insert = insertCfg.front()->getValue();
215 
216  if(insert == "yes")
217  {
218  m_insertMode = true;
219  }
220  else if(insert == "no")
221  {
222  m_insertMode = false;
223  }
224  else
225  {
226  SLM_WARN("value " + insert + " is not managed for <insertMode>");
227  }
228  }
229 
230  std::vector < ::fwRuntime::ConfigurationElement::sptr > iconsCfg = m_configuration->find("icons");
231  if (!iconsCfg.empty())
232  {
233  SLM_ASSERT("Only one 'config' tag is allowed for SSelector configuration", iconsCfg.size() == 1);
234 
235  std::vector < ::fwRuntime::ConfigurationElement::sptr > cfg = iconsCfg.front()->find("icon");
236 
237  for(::fwRuntime::ConfigurationElement::sptr elt : cfg)
238  {
239  std::string series = elt->getAttributeValue("series");
240  SLM_ASSERT("'series' attribute is missing", !series.empty());
241 
242  std::string icon = elt->getAttributeValue("icon");
243  SLM_ASSERT("'icon' attribute is missing", !icon.empty());
244 
245  m_seriesIcons[series] = icon;
246  }
247  }
248 }

273 void SSelector::onDoubleClick(const QModelIndex& index)
274 {
275  m_selectorWidget->clearSelection();
276  m_selectorWidget->setCurrentIndex(index);
277 
278  ::fwData::Vector::sptr selectionVector = this->getSelection();
279 
280  if (m_selectorWidget->getItemType(index) == ::uiMedData::widget::SelectorModel::STUDY)
281  {
282  std::stringstream str;
283  str << "Selected study. TODO";
284 
285  ::fwGui::dialog::MessageDialog::showMessageDialog("Double click",
286  str.str());
287  }
288  else if (m_selectorWidget->getItemType(index) == ::uiMedData::widget::SelectorModel::SERIES)
289  {
290  SLM_ASSERT("There must be only one object selected", selectionVector->size() == 1);
291  ::fwData::Object::sptr obj = selectionVector->front();
292  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(obj);
293  SLM_ASSERT("Object must be a '::fwMedData::Series'", series);
294 
295  m_sigSeriesDoubleClicked->asyncEmit (series);
296  }

301 void SSelector::onRemoveSeries(QVector< ::fwMedData::Series::sptr > selection)
302 {
303  ::fwMedData::SeriesDB::sptr seriesDB;
304  if (this->isVersion2())
305  {
306  seriesDB = this->getInOut< ::fwMedData::SeriesDB >("seriesDB");
307  }
308  else
309  {
310  seriesDB = this->getObject< ::fwMedData::SeriesDB >();
311  }
312  ::fwMedDataTools::helper::SeriesDB seriesDBHelper(seriesDB);
313 
314  // Remove duplicated series
315  std::set< ::fwMedData::Series::sptr > seriesSet;
316  std::copy(selection.begin(), selection.end(), std::inserter(seriesSet, seriesSet.begin()));
317 
318  for( ::fwMedData::Series::sptr series : seriesSet)
319  {
320  seriesDBHelper.remove(series);
321  }
322 
323  {
324  auto sig = seriesDB->signal< ::fwMedData::SeriesDB::RemovedSeriesSignalType >(
325  ::fwMedData::SeriesDB::s_REMOVED_SERIES_SIG );
326  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotRemoveSeries));
327  seriesDBHelper.notify();
328  }
329 }
