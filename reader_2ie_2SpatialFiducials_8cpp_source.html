56 void SpatialFiducials::readLandmark(const ::gdcm::DataSet& fiducialDataset)
57 {
58  ::fwData::PointList::sptr pointList =
59  m_object->getField< ::fwData::PointList >(::fwDataTools::fieldHelper::Image::m_imageLandmarksId);
60  if(!pointList)
61  {
62  pointList = ::fwData::PointList::New();
63  m_object->setField(::fwDataTools::fieldHelper::Image::m_imageLandmarksId, pointList);
64  }
65 
66  const ::gdcm::DataElement& graphicCoordinatesDataElement =
67  fiducialDataset.GetDataElement( ::gdcm::Tag(0x0070, 0x0318) );
68  const ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > graphicCoordinatesDataSequence =
69  graphicCoordinatesDataElement.GetValueAsSQ();
70 
71  const std::string label = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue< 0x0070, 0x030F >(fiducialDataset);
72 
73  for(unsigned int i = 1; i <= graphicCoordinatesDataSequence->GetNumberOfItems(); ++i)
74  {
75  ::gdcm::Item graphicCoordinatesItem = graphicCoordinatesDataSequence->GetItem(i);
76  const ::gdcm::DataSet& graphicCoordinatesDataset = graphicCoordinatesItem.GetNestedDataSet();
77 
78  // 2D Points
79  ::gdcm::Attribute< 0x0070, 0x0022 > coordinatesAttribute;
80  coordinatesAttribute.SetFromDataElement(graphicCoordinatesDataset.GetDataElement(::gdcm::Tag(0x0070, 0x0022)));
81  const float* pointValues = coordinatesAttribute.GetValues();
82 
83  // Frame number
84  const ::gdcm::DataElement& referencedImageDataElement =
85  graphicCoordinatesDataset.GetDataElement( ::gdcm::Tag(0x0008, 0x1140) );
86  const ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > referencedImageDataSequence =
87  referencedImageDataElement.GetValueAsSQ();
88  ::gdcm::Item referencedImageItem = referencedImageDataSequence->GetItem(1);
89  ::gdcm::DataSet& referencedImageDataset = referencedImageItem.GetNestedDataSet();
90 
91  int frameNumber = ::fwGdcmIO::helper::DicomData::getTagValue< 0x0008, 0x1160, int >(referencedImageDataset);
92  double zCoordinate = ::fwGdcmIO::helper::DicomData::convertFrameNumberToZCoordinate(m_object,frameNumber);
93 
94  ::fwData::Point::sptr point = ::fwData::Point::New(static_cast<double>(pointValues[0]),
95  static_cast<double>(pointValues[1]), zCoordinate);
96  point->setField(::fwDataTools::fieldHelper::Image::m_labelId, ::fwData::String::New(label));
97  pointList->getRefPoints().push_back(point);
98 
99  }
100 }
