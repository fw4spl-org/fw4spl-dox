65 void SPreferences::load()
66 {
67  if (::boost::filesystem::is_regular_file(m_prefFile))
68  {
69  const ::boost::filesystem::path folderPath = m_prefFile.parent_path();
70  const ::boost::filesystem::path filename = m_prefFile.filename();
71 
72  ::fwData::Object::sptr data = this->getObject< ::fwData::Object >();
73 
74  // Read atom
75  ::fwZip::IReadArchive::sptr readArchive = ::fwZip::ReadDirArchive::New(folderPath.string());
76  ::fwAtomsBoostIO::Reader reader;
77  try
78  {
79  ::fwAtoms::Object::sptr atom = ::fwAtoms::Object::dynamicCast( reader.read( readArchive, filename ) );
80 
81  ::fwData::Object::sptr newData = ::fwAtomConversion::convert(atom,
82  ::fwAtomConversion::AtomVisitor::ChangePolicy());
83  data->shallowCopy(newData);
84  }
85  catch(...)
86  {
87  SLM_ERROR("Failed to load preference file '"+m_prefFile.string()+"'.");
88  }
89  }
90 }

94 void SPreferences::save()
95 {
96  const ::boost::filesystem::path folderPath = m_prefFile.parent_path();
97  const ::boost::filesystem::path filename = m_prefFile.filename();
98 
99  ::fwData::Object::sptr obj = this->getObject();
100 
101  // Mutex data lock
102  ::fwDataCamp::visitor::RecursiveLock recursiveLock (obj);
103 
104  // Convert data to atom
105  ::fwAtoms::Object::sptr atom = ::fwAtomConversion::convert(obj);
106  // Write atom
107  ::fwZip::IWriteArchive::sptr writeArchive = ::fwZip::WriteDirArchive::New(folderPath.string());
108  ::fwAtomsBoostIO::FormatType format = ::fwAtomsBoostIO::JSON;
109 
110  if(::boost::filesystem::exists(m_prefFile) && ::boost::filesystem::is_regular_file(m_prefFile)
111  && !(::boost::filesystem::status(m_prefFile).permissions() & ::boost::filesystem::owner_write) )
112  {
113  SLM_ERROR("SPreference need write access to the file '"+m_prefFile.string()+"'."
114  "Please, change file permission.");
115  }
116  else
117  {
118  ::fwAtomsBoostIO::Writer(atom).write( writeArchive, filename, format );
119  }
120 }
