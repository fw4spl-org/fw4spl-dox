65 bool Image::createTransferFunctionPool()
66 {
67  bool fieldIsCreated = false;
68  const std::string poolFieldName = ::fwComEd::Dictionary::m_transferFunctionCompositeId;
69  ::fwData::Composite::sptr tfPool;
70 
71  tfPool = m_image->getField< ::fwData::Composite >(poolFieldName);
72  // Transfer functions
73  if ( !tfPool )
74  {
75  tfPool = ::fwData::Composite::New();
76 
77  // Set in selected image
78  ::fwComEd::helper::Field fieldHelper(m_image);
79  fieldHelper.setField(poolFieldName, tfPool);
80  fieldHelper.notify();
81 
82  // TF pool is modified
83  fieldIsCreated = true;
84  }
85 
86  const std::string defaultTFName = ::fwData::TransferFunction::s_DEFAULT_TF_NAME;
87  if(tfPool->find(defaultTFName) == tfPool->end())
88  {
89  ::fwData::TransferFunction::sptr tf = ::fwData::TransferFunction::createDefaultTF();
90  if (m_image->getWindowWidth() != 0 )
91  {
92  tf->setWindow( m_image->getWindowWidth() );
93  tf->setLevel( m_image->getWindowCenter() );
94  }
95  else if(::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(m_image))
96  {
97  double min, max;
98  ::fwComEd::fieldHelper::MedicalImageHelpers::getMinMax(m_image, min, max);
99  ::fwData::TransferFunction::TFValuePairType wlMinMax(min, max);
100  tf->setWLMinMax(wlMinMax);
101  }
102  // Set in TFPool
103  ::fwComEd::helper::Composite compositeHelper(tfPool);
104  compositeHelper.add(defaultTFName, tf);
105  compositeHelper.notify();
106  }
107 
108  return fieldIsCreated;
109 }

114 bool Image::createImageSliceIndex()
115 {
116  bool fieldIsCreated = false;
117 
118  const ::fwData::Image::SizeType &imageSize = m_image->getSize();
119 
120  ::fwData::Integer::sptr axialIdx = m_image->getField< ::fwData::Integer >(
121  ::fwComEd::Dictionary::m_axialSliceIndexId );
122  ::fwData::Integer::sptr frontalIdx = m_image->getField< ::fwData::Integer >(
123  ::fwComEd::Dictionary::m_frontalSliceIndexId);
124  ::fwData::Integer::sptr sagittalIdx = m_image->getField< ::fwData::Integer >(
125  ::fwComEd::Dictionary::m_sagittalSliceIndexId );
126 
127  // Manage image slice index
128  if ( !(axialIdx && frontalIdx && sagittalIdx) )
129  {
130  // Set value
131  axialIdx = ::fwData::Integer::New(-1);
132  m_image->setField( ::fwComEd::Dictionary::m_axialSliceIndexId, axialIdx );
133 
134  frontalIdx = ::fwData::Integer::New(-1);
135  m_image->setField( ::fwComEd::Dictionary::m_frontalSliceIndexId, frontalIdx );
136 
137  sagittalIdx = ::fwData::Integer::New(-1);
138  m_image->setField( ::fwComEd::Dictionary::m_sagittalSliceIndexId, sagittalIdx );
139 
140  fieldIsCreated = true;
141  }
142 
143 
144  SLM_ASSERT (
145  "Information on image slice index is not correct, miss one of these fields : "
146  "m_axialSliceIndexId, m_frontalSliceIndexId, m_sagittalSliceIndexId.",
147  axialIdx && frontalIdx && sagittalIdx
148  );
149 
150  // Get value
151  if( axialIdx->value() < 0 || imageSize[2] < axialIdx->value() )
152  {
153  axialIdx->value() = static_cast< ::fwData::Integer::ValueType >(imageSize[2] / 2);
154  fieldIsCreated = true;
155  }
156 
157  if( frontalIdx->value() < 0 || imageSize[1] < frontalIdx->value() )
158  {
159  frontalIdx->value() = static_cast< ::fwData::Integer::ValueType >(imageSize[1] / 2);
160  fieldIsCreated = true;
161  }
162 
163  if( sagittalIdx->value() < 0 || imageSize[0] < sagittalIdx->value() )
164  {
165  sagittalIdx->value() = static_cast< ::fwData::Integer::ValueType >(imageSize[0] / 2);
166  fieldIsCreated = true;
167  }
168 
169  return fieldIsCreated;
170 }
