00149 
00150         fieldIsCreated = true;
00151     }
00152 
00153 
00154     SLM_ASSERT (
00155             "Information on image slice index is not correct, miss one of these fields : "
00156             "m_axialSliceIndexId, m_frontalSliceIndexId, m_sagittalSliceIndexId.",

00159 
00160     // Get value
00161     if( axialIdx->value() < 0 ||  imageSize[2] < axialIdx->value() )
