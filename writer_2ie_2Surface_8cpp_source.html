80 void Surface::writeSurfaceSegmentationModule(unsigned short segmentationNumber)
81 {
82  // Retrieve Surface Writer
83  SPTR(::gdcm::SurfaceWriter) surfaceWriter = std::static_pointer_cast< ::gdcm::SurfaceWriter >(m_writer);
84 
85  // Retrieve dataset
86  ::gdcm::DataSet& dataset = m_writer->GetFile().GetDataSet();
87 
88  // Retrieve reconstruction
89  ::fwData::Reconstruction::sptr reconstruction = m_object->getReconstructionDB()[segmentationNumber];
90 
91  // Write Content Identification Macro
92  this->writeContentIdentificationMacro(segmentationNumber);
93 
94  ::boost::posix_time::ptime ptime = boost::posix_time::second_clock::local_time();
95 
96  // Content Date - Type 1 - FIXME: Keep series date ?
97  std::string date = m_object->getDate().empty() ? ::fwTools::getDate(ptime) : m_object->getDate();
98  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x0023 >(date, dataset);
99 
100  // Content Time - Type 1 - FIXME: Keep series time ?
101  std::string time = m_object->getTime().empty() ? ::fwTools::getTime(ptime) : m_object->getTime();
102  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x0033 >(time, dataset);
103 
104  // Identification of the segment
105  const std::string& segmentDescription = reconstruction->getOrganName();
106 
107  // Get the info of the struture type and format it
108  std::string segmentLabel = reconstruction->getStructureType();
109 
110  if(segmentLabel.empty())
111  {
112  SLM_WARN("Trying to retrieve structure type from organ name : \"" + segmentDescription + "\".");
113  segmentLabel = segmentDescription;
114  }
115 
116  // Format segment label
117  ::boost::algorithm::trim(segmentLabel);
118  if(segmentLabel.size() > 1)
119  {
120  segmentLabel = ::boost::algorithm::to_upper_copy(segmentLabel.substr(0,1)) + segmentLabel.substr(1);
121  }
122 
123  ::fwData::StructureTraitsDictionary::StructureTypeNameContainer segmentLabelContainer =
124  m_structureDictionary->getStructureTypeNames();
125 
126  ::fwData::StructureTraits::sptr structure;
127  if (std::find(segmentLabelContainer.begin(), segmentLabelContainer.end(), segmentLabel) !=
128  segmentLabelContainer.end())
129  {
130  structure = m_structureDictionary->getStructure(segmentLabel);
131  }
132  else
133  {
134  SLM_WARN("No structure traits found in dictionary for value \"" + segmentLabel + "\".");
135  structure = ::fwData::StructureTraits::New();
136  }
137 
138  // Identify the segmentation from its name
139  const ::fwGdcmIO::container::DicomCodedAttribute* anatomicRegion =
140  ::fwGdcmIO::helper::DictionarySegment::guessAnatRegionFromLabel(structure->getAnatomicRegion());
141 
142  const ::fwGdcmIO::container::DicomCodedAttribute* propertyCategory =
143  ::fwGdcmIO::helper::DictionarySegment::guessPropCategoryFromLabel(structure->getPropertyCategory());
144 
145  const ::fwGdcmIO::container::DicomCodedAttribute* propertyType =
146  ::fwGdcmIO::helper::DictionarySegment::guessPropTypeFromLabel(structure->getPropertyType());
147 
148 
149  // Add segmentation to GDCM Surface Writer
150  ::gdcm::SmartPointer< ::gdcm::Segment > segment = new ::gdcm::Segment();
151  surfaceWriter->AddSegment(segment);
152 
153  // Create Segment Sequence
154  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > surfaceSequence = new ::gdcm::SequenceOfItems();
155  surfaceSequence->SetLengthToUndefined();
156 
157  // Create Surface item
158  ::gdcm::Item surfaceItem;
159  surfaceItem.SetVLToUndefined();
160  ::gdcm::DataSet& segmentationDataset = surfaceItem.GetNestedDataSet();
161 
162  // Segment Number (0x0062,0x0004) - Type 1 (start at 1)
163  unsigned short realSegmentationNumber = static_cast<unsigned short>(segmentationNumber + 1);
164  segment->SetSegmentNumber(realSegmentationNumber);
165  OSLM_TRACE("Segment Number : " << realSegmentationNumber);
166 
167  // Segment Label (0x0062,0x0005) - Type 1
168  segment->SetSegmentLabel(segmentLabel.c_str());
169  OSLM_TRACE("Segment Label : " << segmentLabel);
170 
171  // Segment Description (0x0062,0x0006) - Type 3
172  segment->SetSegmentDescription(segmentDescription.c_str());
173  OSLM_TRACE("Segment Description : " << segmentDescription);
174 
175  // Segment Algorithm Type (0x0062,0x0008) - Type 1
176  segment->SetSegmentAlgorithmType(::gdcm::Segment::MANUAL);
177 
178  // Create Anatomic Region Sequence - See General Anatomy Macros PS 3.3 C.10.5
179  if (anatomicRegion != 0)
180  {
181  // Anatomic Region Sequence (0x0008,0x2218) - Type 1
182  ::gdcm::SegmentHelper::BasicCodedEntry anatomicRegionEntry(anatomicRegion->getCodeValue().c_str(),
183  anatomicRegion->getCodingSchemeDesignator().c_str(),
184  anatomicRegion->getCodeMeaning().c_str());
185  segment->SetAnatomicRegion(anatomicRegionEntry);
186  }
187 
188  // Create Segmented Property Category Code Sequence
189  if (propertyCategory != 0)
190  {
191  // Segmented Property Category Code Sequence (0x0062,0x0003) - Type 1
192  // See: PS.3.3 Table 8.8-1 and Context ID is 7150
193  ::gdcm::SegmentHelper::BasicCodedEntry propertyCategoryEntry(propertyCategory->getCodeValue().c_str(),
194  propertyCategory->getCodingSchemeDesignator().c_str(),
195  propertyCategory->getCodeMeaning().c_str());
196  segment->SetPropertyCategory(propertyCategoryEntry);
197  }
198 
199  // Create Segmented Property Type Code Sequence
200  if (propertyType != 0)
201  {
202  // Segmented Property Type Code Sequence (0x0062,0x000F) - Type 1
203  // See: PS.3.3 Table 8.8-1 and PS 3.16 Context ID 7151
204  ::gdcm::SegmentHelper::BasicCodedEntry propertyTypeEntry(propertyType->getCodeValue().c_str(),
205  propertyType->getCodingSchemeDesignator().c_str(),
206  propertyType->getCodeMeaning().c_str());
207  segment->SetPropertyType(propertyTypeEntry);
208  }
209 
210  // Surface Count (0x0066,0x002A) - Type 1
211  // WARNING : Hard coded because currently, reconstruction just contains one surface.
212  segment->SetSurfaceCount(1);
213 
214  // Create Referenced Surface Sequence
215  // Each item of this sequence references a surface for the current segment.
216  // A segment can have several surfaces (see tag(0x0066,0x002A), Surface Count).
217  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > referencedSurfaceSequence = new ::gdcm::SequenceOfItems();
218  referencedSurfaceSequence->SetLengthToUndefined();
219  {
220  // Create its item
221  ::gdcm::Item referencedSurfaceItem;
222  referencedSurfaceItem.SetVLToUndefined();
223  ::gdcm::DataSet& referencedSurfaceDataset = referencedSurfaceItem.GetNestedDataSet();
224 
225  // WARNING : index+1 == segmentNumber == surfaceNumber because currently, reconstruction just contains one surface.*
226  // Referenced Surface Number - Type 1
227  ::fwGdcmIO::helper::DicomData::setTagValue< uint16_t, 0x0066, 0x002C >(realSegmentationNumber,
228  referencedSurfaceDataset);
229 
230  // Create Segment Surface Source Instance Sequence
231  // Reference the image of the fwMedData::ImageSeries
232  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > surfaceSourceInstanceSequence = new ::gdcm::SequenceOfItems();
233  surfaceSourceInstanceSequence->SetLengthToUndefined();
234 
235  // Include ‘Image SOP Instance Reference Macro’ Table C.10-3
236  const std::string& referencedSOPClassUID = m_imageInstance->getSOPClassUID();
237  const std::vector< std::string >& referencedSOPInstanceUIDContainer =
238  m_imageInstance->getSOPInstanceUIDContainer();
239 
240  for(const std::string& sopInstanceUID: referencedSOPInstanceUIDContainer)
241  {
242  // Create one item
243  ::gdcm::Item imageSOPItem;
244  imageSOPItem.SetVLToUndefined();
245  ::gdcm::DataSet& imageSOPDataset = imageSOPItem.GetNestedDataSet();
246 
247  // SOP Class UID - Type 1
248  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x1150 >(referencedSOPClassUID, imageSOPDataset);
249 
250  // SOP Instance UID - Type 1
251  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x1155 >(sopInstanceUID, imageSOPDataset);
252 
253  surfaceSourceInstanceSequence->AddItem(imageSOPItem);
254  }
255 
256  // Segment Surface Source Instance Sequence - Type 1C
257  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x002E >(surfaceSourceInstanceSequence,
258  referencedSurfaceDataset);
259 
260  referencedSurfaceSequence->AddItem(referencedSurfaceItem);
261 
262  // Referenced Surface Sequence
263  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x002B >(referencedSurfaceSequence, segmentationDataset);
264  }
265  surfaceSequence->AddItem(surfaceItem);
266 
267  // Segment Sequence
268  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0062, 0x0002 >(surfaceSequence, dataset);
269 
270 }

274 void Surface::writeSurfaceMeshModule(unsigned short segmentationNumber)
275 {
276  // Retrieve Surface Writer
277  SPTR(::gdcm::SurfaceWriter) surfaceWriter = std::static_pointer_cast< ::gdcm::SurfaceWriter >(m_writer);
278 
279  // Retrieve dataset
280  ::gdcm::DataSet& dataset = m_writer->GetFile().GetDataSet();
281 
282  // Get the last segment added in SegmentWriter::writeSurfaceSegmentation()
283  ::gdcm::SmartPointer< ::gdcm::Segment > segment = surfaceWriter->GetSegments().back();
284 
285  // Add it a surface
286  ::gdcm::SmartPointer< ::gdcm::Surface > surface = new ::gdcm::Surface();
287  segment->AddSurface(surface);
288 
289  // Retrieve reconstruction
290  ::fwData::Reconstruction::sptr reconstruction = m_object->getReconstructionDB()[segmentationNumber];
291 
292  // Retrieve material
293  ::fwData::Material::csptr material = reconstruction->getMaterial();
294 
295  // Set DicomSurface data - NOTE: must be called before points and primitives writing
296  SPTR(::fwGdcmIO::container::DicomSurface) surfaceContainer =
297  std::make_shared< ::fwGdcmIO::container::DicomSurface >();
298  surfaceContainer->setFromData(reconstruction);
299 
300  // Create Surface Sequence
301  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > surfaceSequence = new ::gdcm::SequenceOfItems();
302  surfaceSequence->SetLengthToUndefined();
303  {
304  // Create one item for the surface
305  ::gdcm::Item surfaceItem;
306  surfaceItem.SetVLToUndefined();
307  ::gdcm::DataSet& surfaceDataset = surfaceItem.GetNestedDataSet();
308 
309  // Surface Number (0x0066,0x0003) - Type 1
310  surface->SetSurfaceNumber(segmentationNumber + 1); // (start at 1)
311  OSLM_TRACE("Surface Number : " << surface->GetSurfaceNumber());
312 
313  // Surface Processing (0x0066,0x0009) - Type 2
314  surface->SetSurfaceProcessing(false);
315  OSLM_TRACE("Surface Processing : NO");
316 
317  // Get reconstruction's color
318  std::array< float, 4 > rgba = material->diffuse()->getRGBA();
319  std::vector< float > rgb(3);
320  rgb[0] = rgba[0];
321  rgb[1] = rgba[1];
322  rgb[2] = rgba[2];
323 
324  // Recommended Display Grayscale Value (0x0062,0x000C) - Type 1
325  const unsigned short grayscale = ::gdcm::SurfaceHelper::RGBToRecommendedDisplayGrayscale(rgb, 1.);
326  surface->SetRecommendedDisplayGrayscaleValue(grayscale);
327  OSLM_TRACE("Recommended Display Grayscale Value : " << grayscale);
328 
329  // Recommended Display CIELab Value (0x0062,0x000D) - Type 1
330  const std::vector< unsigned short > CIELab = ::gdcm::SurfaceHelper::RGBToRecommendedDisplayCIELab(rgb, 1.);
331  surface->SetRecommendedDisplayCIELabValue(CIELab);
332  OSLM_TRACE("Recommended Display CIELab Value : " << CIELab[0] << " " << CIELab[1] << " " << CIELab[2]);
333 
334  // Recommended Presentation Opacity (0x0066,0x000C) - Type 1
335  surface->SetRecommendedPresentationOpacity(rgba[3]);
336  OSLM_TRACE("Recommended Presentation Opacity : " << rgba[3]);
337 
338  // Recommended Presentation Type (0x0066,0x000D)
339  surface->SetRecommendedPresentationType(
340  ::fwGdcmIO::helper::DicomData::convertToPresentationType(material->getRepresentationMode()));
341  OSLM_TRACE( "Recommended Presentation Type : " <<
342  ::fwGdcmIO::helper::DicomData::convertToPresentationTypeString(material->getRepresentationMode()));
343 
344  // Finite Volume (0x0066,0x000E) - Type 1
345  ::fwDataTools::helper::Mesh helperMesh(reconstruction->getMesh());
346  surface->SetFiniteVolume(helperMesh.isClosed() ? (::gdcm::Surface::YES): (::gdcm::Surface::NO));
347 
348  // Manifold (0x0066,0x0010) - Type 1
349  surface->SetManifold(::gdcm::Surface::NO);
350  SLM_TRACE("Manifold : NO");
351 
352  // Create Surface Points Sequence
353  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > pointsSequence = new ::gdcm::SequenceOfItems();
354  pointsSequence->SetLengthToUndefined();
355  {
356  // Create its item
357  ::gdcm::Item pointsItem;
358  pointsItem.SetVLToUndefined();
359 
360  // Number Of Points
361  surface->SetNumberOfSurfacePoints(surfaceContainer->getPointCoordSize());
362  OSLM_TRACE("Number Of Points : " << surfaceContainer->getPointCoordSize());
363 
364  // Point Coordinates Data
365  ::gdcm::DataElement& pointCoordData = surface->GetPointCoordinatesData();
366  const float* pointCoordinatesData = surfaceContainer->getPointCoordData().get();
367  pointCoordData.SetByteValue((char*)(pointCoordinatesData),
368  static_cast<unsigned int>(3 * surfaceContainer->getPointCoordSize() *
369  sizeof(*pointCoordinatesData)));
370  OSLM_TRACE("Point Coordinates Data buffer size : " << 3 * surfaceContainer->getPointCoordSize()
371  * sizeof(*pointCoordinatesData));
372 
373  pointsSequence->AddItem(pointsItem);
374 
375  // Surface Points Sequence
376  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x0011 >(pointsSequence, surfaceDataset);
377  }
378 
379  // Create Surface Points Normals Sequence (0066,0012) - Type 2
380  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > normalSequence = new ::gdcm::SequenceOfItems();
381  normalSequence->SetLengthToUndefined();
382  if (surfaceContainer->getNormalCoordSize() > 0)
383  {
384  // Create its item
385  ::gdcm::Item normalsItem;
386  normalsItem.SetVLToUndefined();
387 
388  // Number Of Points
389  surface->SetNumberOfVectors(surfaceContainer->getNormalCoordSize());
390  OSLM_TRACE("Number Of Normals : " << surfaceContainer->getNormalCoordSize());
391 
392  surface->SetVectorDimensionality(1);
393 
394  // Normal Coordinates Data
395  ::gdcm::DataElement& normalCoordData = surface->GetVectorCoordinateData();
396  const float* normalCoordinatesData = surfaceContainer->getNormalCoordData().get();
397  normalCoordData.SetByteValue((char*)(normalCoordinatesData),
398  static_cast<unsigned int>(3 * surfaceContainer->getNormalCoordSize() *
399  sizeof(*normalCoordinatesData)));
400  OSLM_TRACE("Point Coordinates Data buffer size : " << 3 * surfaceContainer->getNormalCoordSize() *
401  sizeof(*normalCoordinatesData));
402 
403  normalSequence->AddItem(normalsItem);
404  }
405  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x0012 >(normalSequence, surfaceDataset);
406 
407  // Create Surface Mesh Primitives Sequence - PS 3.3 C.27.4
408  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > primitivesSequence = new ::gdcm::SequenceOfItems();
409  primitivesSequence->SetLengthToUndefined();
410 
411  {
412  // Mesh primitive type (fixed to TRIANGLE by FXW4SPL)
413  ::gdcm::SmartPointer< ::gdcm::MeshPrimitive > primitive = surface->GetMeshPrimitive();
414  primitive->SetPrimitiveType(::gdcm::MeshPrimitive::TRIANGLE);
415 
416  // Triangle Point Index List (0066,0023) - Type 2
417  ::gdcm::DataElement& pointIndexData = primitive->GetPrimitiveData();
418  pointIndexData.SetVL(sizeof(uint32_t));
419  const uint32_t* pointIndexListData = surfaceContainer->getPointIndexList().get();
420  unsigned long indexCount = 3 * surfaceContainer->getPointIndexSize();
421  pointIndexData.SetByteValue((char*)(pointIndexListData),
422  static_cast<unsigned int>(indexCount * sizeof(uint32_t)));
423  OSLM_TRACE("Point Index List buffer size : " << indexCount * sizeof(uint32_t));
424 
425  }
426 
427  // Surface Mesh Primitives Sequence
428  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x0013 >(primitivesSequence, surfaceDataset);
429 
430  surfaceSequence->AddItem(surfaceItem);
431  }
432 
433  // Surface Sequence
434  ::fwGdcmIO::helper::DicomData::insertSQ< 0x0066, 0x0002 >(surfaceSequence, dataset);
435 
436  // Complete Segment Sequence
437  // Create Algorithm Family Code Sequence
438  // See: Supplement 132 Table 10-X
439  {
440  //See: PS.3.3 Table 8.8-1 and PS 3.16 Context ID 7162
441  ::gdcm::SegmentHelper::BasicCodedEntry algoFamily;
442  algoFamily.CV = "1231009";
443  algoFamily.CSD = "DCM";
444  algoFamily.CM = "Manual Processing";
445  // Algorithm Family Code Sequence (0x0066,0x002F) - Type 1
446  surface->SetAlgorithmFamily(algoFamily);
447  }
448 
449  // Retrieve runtime profile
450  ::fwRuntime::profile::Profile::sptr profile = ::fwRuntime::profile::getCurrentProfile();
451 
452  // Algorithm Version (0x0066,0x0031) - Type 1
453  std::string algorithVersion = (profile) ? (profile->getName()+" "+profile->getVersion()) : "Unknown";
454  surface->SetAlgorithmVersion(algorithVersion.c_str());
455 
456  // Algorithm Name (0x0066,0x0036) - Type 1
457  std::string algorithName = (profile) ? (profile->getName()) : "Unknown";
458  surface->SetAlgorithmName(algorithName.c_str());
459 
460 }
