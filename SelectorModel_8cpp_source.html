00369     BOOST_FOREACH(QStandardItem *item, seriesItems)
00370     {
00371         QStandardItem * studyItem = item->parent();
00372 
00373         // Remove series item if it is not included in a study which will be remove.
00374         if (std::find(studyItems.begin(), studyItems.end(), studyItem) == studyItems.end())
00375         {
00376             this->removeSeriesItem(item);
00377         }
00378     }
00379 
00380     // Remove study items from selector
00381     BOOST_FOREACH(QStandardItem *item, studyItems)
00382     {
00383         this->removeStudyItem(item);
