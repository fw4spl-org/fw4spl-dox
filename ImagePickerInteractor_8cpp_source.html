102  bool pickSomething()
103  {
104  int x,y;
105  double display[3];
106 
107  m_adaptor->getInteractor()->GetEventPosition(x, y);
108  display[0] = x;
109  display[1] = y;
110  display[2] = 0;
111 
112  return (m_picker->Pick( display, m_adaptor->getRenderer() ) != 0);
113  }

117  void process(vtkRenderWindowInteractor *caller, unsigned long eventId) // from
118  {
119  if( m_eventId->find( static_cast< ImagePickerInteractor::EventID>(eventId) ) != m_eventId->end() )
120  {
121 #ifdef __linux
122  if(eventId == ImagePickerInteractor::MOUSE_MOVE)
125  {
126  m_skipMove++;
127  if( m_skipMove % 10 )
128  {
129  return;
130  }
131  }
132 #endif
133  SLM_ASSERT("bad vtk caller", caller);
134 
135  if ( this->pickSomething() )
136  {
137  double world[3] = {-1,0,0};
138  ::fwRenderVTK::vtk::getNearestPickedPosition(m_picker, m_adaptor->getRenderer(), world);
139  OSLM_TRACE("PICK" << world[0] << " ," << world[1] << " ," << world[2] );
140 
141  ::fwComEd::PickingInfo info;
142 
143  m_adaptor->worldToImageSliceIndex(world, info.m_worldPos);
144 
145  const auto iter = s_vtkEventIDConversion.find(eventId);
146  SLM_ASSERT("Unknown eventId", iter != s_vtkEventIDConversion.end());
147  info.m_eventId = iter->second;
148 
149  info.m_keyPressed = caller->GetKeyCode();
150 
151  info.m_modifierMask =
152  caller->GetControlKey() ? ::fwComEd::PickingInfo::CTRL : ::fwComEd::PickingInfo::NONE;
153  info.m_modifierMask |=
154  caller->GetShiftKey() ? ::fwComEd::PickingInfo::SHIFT : ::fwComEd::PickingInfo::NONE;
155 
156  auto sig = m_adaptor->signal<ImagePickerInteractor::PickedSignalType>(
157  ImagePickerInteractor::s_PICKED_SIGNAL);
158  sig->asyncEmit(info);
159  }
160  }
161

215 void ImagePickerInteractor::doConfigure() throw(fwTools::Failed)
216 {
217  SLM_ASSERT("Required element 'config' is missing.", m_configuration->getName() == "config");
218 
219  if (m_configuration->hasAttribute("event"))
220  {
221  const std::string eventTxt = m_configuration->getAttributeValue("event");
222 
223  ::boost::char_separator<char> sep(", ;");
224  ::boost::tokenizer< ::boost::char_separator<char> > tok(eventTxt, sep);
225  for( const auto it : tok)
226  {
227  const auto iter = m_eventIdConversion.find(it);
228  SLM_ASSERT("Unknown eventId '"+ it+"'.", iter != m_eventIdConversion.end());
229  m_eventId.insert(iter->second);
230  }
231  }
232  else
233  {
234  for(auto elt : m_eventIdConversion)
235  {
236  m_eventId.insert(elt.second);
237  }
238  }
239 }

244 {
245  ImagePickerInteractorCallback *observer = ImagePickerInteractorCallback::New();
246  observer->setAdaptor( ImagePickerInteractor::dynamicCast(this->getSptr()) );
247  observer->setPicker(this->getPicker());

252  vtkRenderWindowInteractor *interactor = this->getInteractor();

254  interactor->AddObserver(vtkCommand::LeftButtonPressEvent, m_interactionCommand, priority);
255  interactor->AddObserver(vtkCommand::LeftButtonReleaseEvent, m_interactionCommand, priority);
256  interactor->AddObserver(vtkCommand::MiddleButtonPressEvent, m_interactionCommand, priority);
257  interactor->AddObserver(vtkCommand::MiddleButtonReleaseEvent, m_interactionCommand, priority);
258  interactor->AddObserver(vtkCommand::RightButtonPressEvent, m_interactionCommand, priority);
259  interactor->AddObserver(vtkCommand::RightButtonReleaseEvent, m_interactionCommand, priority);
260  interactor->AddObserver(vtkCommand::MouseMoveEvent, m_interactionCommand, priority);
261  interactor->AddObserver(vtkCommand::MouseWheelForwardEvent, m_interactionCommand, priority);
