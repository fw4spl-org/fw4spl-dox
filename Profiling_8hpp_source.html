00048 class fwProfileFrameTimer
00049 {
00050 public:
00051     fwProfileFrameTimer(double interval) :
00052         m_interval(interval),
00053         m_average(0),
00054         m_count(0)
00055     {
00056         m_timer.start();
00057     }
00058 
00059     bool tick()
00060     {
00061         m_average = m_timer.getElapsedTimeInMilliSec() / ++m_count;
00062         return m_timer.getElapsedTimeInMilliSec() >= m_interval * 1000;
00063     }
00064 
00065     bool tick(::fwCore::HiResClock::HiResClockType time)
00066     {
00067         m_average = (m_average * m_count + time) / (m_count + 1);
00068         ++m_count;
00069         return m_timer.getElapsedTimeInMilliSec() >= m_interval * 1000;
00070     }
00071 
00072     void reset()
00073     {
00074         m_count   = 0;
00075         m_average = 0;
00076         m_timer.reset();
00077     }
