77 void PlaneSelectionNotifier::doStart() throw(fwTools::Failed)
78 {
79  SLM_TRACE_FUNC();
80 
81  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
82  SLM_ASSERT("m_currentPlaneList should be expired", m_currentPlaneList.expired());
83 
84  if (composite->find(m_planeListId) != composite->end())
85  {
86  m_currentPlaneList = ::fwData::PlaneList::dynamicCast(composite->getContainer()[m_planeListId]);
87  }
88 
89  ::fwData::PlaneList::sptr planeList = m_currentPlaneList.lock();
90  if(planeList)
91  {
92  m_plConnection->connect(planeList, ::fwData::PlaneList::s_PLANE_REMOVED_SIG,
93  this->getSptr(), s_REMOVE_PLANE_SLOT);
94  m_plConnection->connect(planeList, ::fwData::PlaneList::s_VISIBILITY_MODIFIED_SIG,
95  this->getSptr(), s_SHOW_PLANES_SLOT);
96 
97  for( ::fwData::Plane::sptr plane : planeList->getPlanes() )
98  {
99  m_planeConnections[plane->getID()] = plane->signal(::fwData::Plane::s_SELECTED_SIG)->connect(
100  this->slot(s_UPDATE_SELECTION_SLOT));
101  }
102  }
103 }

182 void PlaneSelectionNotifier::selectPlane( ::fwData::Object::sptr plane )
183 {
184  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
185 
186  ::fwComEd::helper::Composite helper(composite);
187  helper.swap(m_planeSelectionId, plane);
188 
189  auto sig = composite->signal< ::fwData::Composite::ChangedObjectsSignalType >(
190  ::fwData::Composite::s_CHANGED_OBJECTS_SIG);
191  {
192  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdatePlaneList));
193  helper.notify();
194  }
195 }

199 void PlaneSelectionNotifier::deselectPlane()
200 {
201  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
202  if ( composite->find(m_planeSelectionId) != composite->end() )
203  {
204  ::fwComEd::helper::Composite helper(composite);
205  helper.remove(m_planeSelectionId);
206  auto sig = composite->signal< ::fwData::Composite::RemovedObjectsSignalType >(
207  ::fwData::Composite::s_REMOVED_OBJECTS_SIG);
208  {
209  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdatePlaneList));
210  helper.notify();
211  }
212  }
