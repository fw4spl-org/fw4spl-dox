76 void compare(::fwData::Object::sptr objRef, ::fwData::Object::sptr objComp)
77 {
78  ::fwDataCamp::visitor::CompareObjects visitor;
79  visitor.compare(objRef, objComp);
80  SPTR(::fwDataCamp::visitor::CompareObjects::PropsMapType) props = visitor.getDifferences();
81  for( ::fwDataCamp::visitor::CompareObjects::PropsMapType::value_type prop : (*props) )
82  {
83  OSLM_ERROR( "new object difference found : " << prop.first << " '" << prop.second << "'" );
84  }
85  CPPUNIT_ASSERT_MESSAGE("Object Not equal", props->size() == 0 );
86 }

103  ::fwData::Reconstruction::New(),
104  ::fwData::Composite::New(),
105  ::fwData::Point::New(),

111 
112  ::fwAtoms::Object::sptr atom;
113 
114  for ( fwData::Object::sptr object : VALUES )
115  {
116  atom = ::fwAtomConversion::convert(object);
117 
118  const ::camp::Class& metaClass = ::camp::classByName(object->getClassname());
119 
120  //Type test
121  CPPUNIT_ASSERT_EQUAL( object->getClassname(),
122  atom->getMetaInfo( ::fwAtomConversion::DataVisitor::CLASSNAME_METAINFO ) );
123 
124  //Test attribute type
125  for( ::fwAtoms::Object::AttributesType::value_type elem : atom->getAttributes() )
126  {
127  std::string classname = atom->getMetaInfo( ::fwAtomConversion::DataVisitor::CLASSNAME_METAINFO );
128  if ( !( classname == "::fwData::Graph" &&
129  elem.first == "connections" ) )
130  {
131  int type = metaClass.property(elem.first).type();
132  std::string attribute = metaClass.property(elem.first).name();
133  switch(type)
134  {
135  case camp::stringType:
136  CPPUNIT_ASSERT(elem.second->isString());
137  CPPUNIT_ASSERT(elem.second->isValue());
138  break;
139  case camp::realType:
140  case camp::intType:
141  CPPUNIT_ASSERT(elem.second->isNumeric());
142  CPPUNIT_ASSERT(elem.second->isValue());
143  break;
144  case camp::boolType:
145  CPPUNIT_ASSERT(elem.second->isBoolean());
146  CPPUNIT_ASSERT(elem.second->isValue());
147  break;
148  case camp::userType:
149  if( ( ( classname == "::fwData::Mesh" ) && ( attribute == "cell_colors" ) ) ||
150  ( ( classname == "::fwData::Mesh" ) && ( attribute == "cell_normals" ) ) ||
151  ( ( classname == "::fwData::Mesh" ) && ( attribute == "point_colors" ) ) ||
152  ( ( classname == "::fwData::Mesh" ) && ( attribute == "point_normals" ) ) ||
153  ( ( classname == "::fwData::Mesh" ) && ( attribute == "cell_tex_coords" ) ) ||
154  ( ( classname == "::fwData::Mesh" ) && ( attribute == "point_tex_coords" ) ) ||
155  ( ( classname == "::fwData::Material" ) && ( attribute == "diffuse_texture" ) ) ||
156  ( ( classname == "::fwData::Reconstruction" ) && ( attribute == "image" ) ) ||
157  ( ( classname == "::fwData::Reconstruction" ) && ( attribute == "mesh" ) ) )
158  {
159  CPPUNIT_ASSERT(!elem.second);
160  }
161  else
162  {
163  CPPUNIT_ASSERT(elem.second->isObject() || elem.second->isBlob());
164  }
165  break;
166  case camp::mappingType:
167  CPPUNIT_ASSERT(elem.second->isMapping());
168  break;
169  case camp::enumType:
170  CPPUNIT_ASSERT(elem.second->isString());
171  CPPUNIT_ASSERT(elem.second->isValue());
172  break;
173  case camp::arrayType:
174  CPPUNIT_ASSERT(elem.second->isSequence());
175  break;
176  }
177  }
178  }
179  }

205 void ConversionTest::graphConversionTest()
206 {
207  ::fwAtoms::Object::sptr atom;
208  ::fwTools::UUID::UUIDType gID,n1ID,n2ID,n3ID,e12ID,e23ID;
209  {
210  ::fwData::Graph::sptr g = ::fwData::Graph::New();
211  gID = ::fwTools::UUID::get(g);
212  ::fwData::Node::sptr n1 = ::fwData::Node::New();
213  n1ID = ::fwTools::UUID::get(n1);
214  ::fwData::Node::sptr n2 = ::fwData::Node::New();
215  n2ID = ::fwTools::UUID::get(n2);
216  ::fwData::Node::sptr n3 = ::fwData::Node::New();
217  n3ID = ::fwTools::UUID::get(n3);
218  ::fwData::Edge::sptr e12 = ::fwData::Edge::New();
219  e12ID = ::fwTools::UUID::get(e12);
220  ::fwData::Edge::sptr e23 = ::fwData::Edge::New();
221  e23ID = ::fwTools::UUID::get(e23);
222 
223  // build graph
224  g->addNode(n1);
225  g->addNode(n2);
226  g->addNode(n3);
227 
228  n1->addOutputPort( ::fwData::Port::New() );
229  n2->addInputPort( ::fwData::Port::New() );
230  n2->addOutputPort( ::fwData::Port::New() );
231  n3->addInputPort( ::fwData::Port::New() );
232 
233  e12->setIdentifiers("IDNOTdefined","IDNOTdefined");
234  e23->setIdentifiers("IDNOTdefined","IDNOTdefined");
235 
236  g->addEdge(e12,n1,n2);
237  g->addEdge(e23,n2,n3);
238 
239  // Test field on edge
240  e12->setField("infoTest",::fwData::String::New("valueInfoTest"));
241 
242  // Create Atom
243  atom = ::fwAtomConversion::convert( g );
244  }
245 
246  // Create Data from Atom
247  ::fwData::Graph::sptr newGraph = ::fwData::Graph::dynamicCast( ::fwAtomConversion::convert(atom) );
248 
249  // nodes
250  ::fwData::Node::sptr n1, n2, n3;
251 
252  // Test nodes
253  const ::fwData::Graph::NodeContainer & nodes = newGraph->getCRefNodes();
254  CPPUNIT_ASSERT_EQUAL_MESSAGE("Graph nodes size", (size_t)3, nodes.size() );
255  for( ::fwData::Node::sptr node : nodes )
256  {
257  ::fwTools::UUID::UUIDType nodeID = ::fwTools::UUID::get(node);
258  CPPUNIT_ASSERT_MESSAGE("Test node uuid", nodeID == n1ID || nodeID == n2ID || nodeID == n3ID );
259  if ( nodeID == n1ID )
260  {
261  n1 = node;
262  }
263  else if ( nodeID == n2ID )
264  {
265  n2 = node;
266  }
267  else if ( nodeID == n3ID )
268  {
269  n3 = node;
270  }
271 
272  }
273  CPPUNIT_ASSERT_MESSAGE("Test node n1", n1 );
274  CPPUNIT_ASSERT_MESSAGE("Test node n2", n2 );
275  CPPUNIT_ASSERT_MESSAGE("Test node n3", n3 );
276 
277  // Test edges
278  const ::fwData::Graph::ConnectionContainer & connections = newGraph->getCRefConnections();
279  CPPUNIT_ASSERT_EQUAL_MESSAGE("Graph connections size", (size_t)2, connections.size() );
280  for( ::fwData::Graph::ConnectionContainer::value_type elem : connections )
281  {
282  ::fwTools::UUID::UUIDType edgeID = ::fwTools::UUID::get(elem.first);
283  CPPUNIT_ASSERT_MESSAGE("Test edge uuid", edgeID == e12ID || edgeID == e23ID );
284  if ( edgeID == e12ID )
285  {
286  CPPUNIT_ASSERT( newGraph->getSourceNode( elem.first ) == n1 );
287  CPPUNIT_ASSERT( newGraph->getDestinationNode( elem.first ) == n2 );
288 
289  // test field
290  CPPUNIT_ASSERT( elem.first->getField("infoTest") );
291  CPPUNIT_ASSERT( elem.first->getField< ::fwData::String >("infoTest")->value() == "valueInfoTest" );
292  }
293  else
294  {
295  CPPUNIT_ASSERT( newGraph->getSourceNode( elem.first ) == n2 );
296  CPPUNIT_ASSERT( newGraph->getDestinationNode( elem.first ) == n3 );
297  }
298  }
299 }
