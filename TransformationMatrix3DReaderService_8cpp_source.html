103 void TransformationMatrix3DReaderService::updating() throw(::fwTools::Failed)
104 {
105  SLM_TRACE_FUNC();
106  if(this->hasLocationDefined())
107  {
108  // Retrieve object
109  ::fwData::TransformationMatrix3D::sptr matrix = this->getObject< ::fwData::TransformationMatrix3D >( );
110  SLM_ASSERT("matrix not instanced", matrix);
111 
112  ::fwDataIO::reader::TransformationMatrix3DReader::sptr reader =
113  ::fwDataIO::reader::TransformationMatrix3DReader::New();
114  reader->setObject( matrix );
115  reader->setFile(this->getFile());
116  reader->read();
117 
118  // Notify reading
119  auto sig = this->getObject()->signal< ::fwData::Object::ModifiedSignalType >(
120  ::fwData::Object::s_MODIFIED_SIG);
121  {
122  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
123  sig->asyncEmit();
124  }
125  }
126 }
