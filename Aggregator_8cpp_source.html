40 IJob::SharedFuture Aggregator::runImpl()
41 {
42  decltype(m_jobs)jobs;
43 
44  {
45  ::fwCore::mt::ReadLock lock(m_mutex);
46  jobs = m_jobs;
47  }
48 
49  std::vector< SharedFuture > futures;
50  for(const ::fwJobs::IJob::sptr &iJob : jobs)
51  {
52  futures.push_back(iJob->run());
53  }
54 
55  auto future = boost::async(
56  [ = ]() mutable
57  {
58  boost::wait_for_all(futures.begin(), futures.end());
59 
60  this->finish();
61 
62  // forwards exceptions that might have been thrown
63  for( SharedFuture f : futures )
64  {
65  f.get();
66  }
67  }
68  );
69 
70  return ::boost::move(future);
71 }

75 void Aggregator::add(const ::fwJobs::IJob::sptr &iJob, double weight)
76 {
77  SLM_ASSERT("iJob shall not be null", iJob);
78 
79  SLM_ASSERT("iJob shall not be added to itself", this!=iJob.get());
80 
81  if(!iJob)
82  {
83  return;
84  }
85 
86  ::fwCore::mt::ReadToWriteLock lock(m_mutex);
87 
88  SLM_ASSERT("Jobs can't be added when Aggregator is running", m_state == WAITING || m_state == RUNNING);
89 
90  const auto normValue = std::uint64_t(weight*100);
91 
92  if( m_state == WAITING || m_state == RUNNING )
93  {
94  m_jobInfo[iJob.get()] = JobInfo( *iJob );
95  auto &jobInfo = m_jobInfo[iJob.get()];
96 
97  this->setTotalWorkUnitsUpgradeLock( m_totalWorkUnits + (jobInfo.totalWork ? normValue : 0), lock);
98  lock.lock();
99  // doneWork call after setTotalWorkUnitsUpgradeLock, because
100  // doneWork value can be thresholded by setTotalWorkUnitsUpgradeLock
101  // call
102  jobInfo.lastValue = std::uint64_t(jobInfo.progress() * normValue);
103  {
104  ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
105  m_jobs.push_back(iJob);
106  }
107  // take care : doneWork unlocks 'lock'
108  this->doneWork(jobInfo.lastValue, lock);
109 
110  // TODO : add a way to disconnect on aggregator destruction
111  iJob->addDoneWorkHook(
112  [ =, &jobInfo](IJob& subJob, std::uint64_t oldDoneWork)
113  {
114  ::fwCore::mt::ReadToWriteLock sublock(m_mutex);
115 
116  auto oldInfo = jobInfo;
117  jobInfo = Aggregator::JobInfo( subJob );
118 
119  jobInfo.lastValue = std::uint64_t(normValue * jobInfo.progress());
120 
121 
122  auto doneWork = m_doneWorkUnits + jobInfo.lastValue;
123  // minimize numerical uncertainty by substracting in a second time :
124  doneWork -= oldInfo.lastValue;
125 
126 
127  this->doneWork( static_cast<std::uint64_t>(doneWork), sublock );
128  }
129  );
130 
131 
132  iJob->addTotalWorkUnitsHook(
133  [ = ](IJob& subJob, std::uint64_t oldTotalWorkUnits)
134  {
135  ::fwCore::mt::ReadToWriteLock sublock(m_mutex);
136 
137  auto workUnits = m_totalWorkUnits;
138  auto newTotalWorkUnits = subJob.getTotalWorkUnits();
139 
140  if( oldTotalWorkUnits != newTotalWorkUnits)
141  {
142  if( oldTotalWorkUnits && 0 == newTotalWorkUnits)
143  {
144  workUnits -= normValue;
145  }
146  else if( 0 == oldTotalWorkUnits && newTotalWorkUnits)
147  {
148  workUnits += normValue;
149  }
150  }
151 
152  this->setTotalWorkUnitsUpgradeLock( workUnits, sublock );
153  }
154  );
155 
156 
157  this->addCancelHookNoLock( [iJob]( IJob& /* cancelingJob */ )
158  {
159  iJob->cancel();
160  } );
161 
162  auto iJobName = iJob->getName();
163  iJobName = iJobName.empty() ? "" : "[" + iJobName + "] ";
164  iJob->addLogHook( [ = ]( IJob& /* job */, const std::string &message)
165  {
166  this->log( iJobName + message);
167  } );
168 
169  auto iJobLogs = iJob->getLogs();
170  std::for_each(iJobLogs.begin(), iJobLogs.end(),
171  [&]( const Logs::value_type& message )
172  {
173  this->logNoLock(iJobName + message);
174  }
175  );
176  }
