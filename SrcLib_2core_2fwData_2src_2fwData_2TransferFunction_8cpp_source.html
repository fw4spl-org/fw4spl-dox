89 TransferFunction::TFValueVectorType TransferFunction::getScaledValues() const
90 {
91  TFValueVectorType values;
92  values.reserve(m_tfData.size());
93  TFValuePairType minMax = this->getMinMaxTFValues();
94  TFValuePairType windowMinMax = this->getWLMinMax();
95 
96  const double scale = m_window / (minMax.second - minMax.first);
97 
98  for(const TFDataType::value_type &data : m_tfData)
99  {
100  const double value = (data.first - minMax.first) * scale + windowMinMax.first;
101  values.push_back(value);
102  }
103 
104  return values;
105 }

142 TransferFunction::TFValueType TransferFunction::getNearestValue( TFValueType value ) const
143 {
144  OSLM_ASSERT("It must have at least one value.", m_tfData.size()>= 1);
145  std::pair<double, double> minMax = ::fwTools::Type::s_DOUBLE.minMax<double>();
146  double previousValue = minMax.first;
147  double nextValue = minMax.second;
148 
149  TFValueType val;
150  for(const TFDataType::value_type &data : m_tfData)
151  {
152  if(value < data.first )
153  {
154  nextValue = data.first;
155  break;
156  }
157  else
158  {
159  previousValue = data.first;
160  }
161  }
162  if(previousValue == minMax.first)
163  {
164  val = nextValue;
165  }
166  else if(nextValue == minMax.second)
167  {
168  val = previousValue;
169  }
170  else
171  {
172  if((value - previousValue) < (nextValue - value))
173  {
174  val = previousValue;
175  }
176  else
177  {
178  val = nextValue;
179  }
180  }
181  return val;
182 }

229 TransferFunction::TFColor TransferFunction::getNearestColor( TFValueType value ) const
230 {
231  OSLM_ASSERT("It must have at least one value.", m_tfData.size()>= 1);
232 
233  double min = std::numeric_limits<double>::min();
234  double max = std::numeric_limits<double>::max();
235 
236  double previousValue = min;
237  double nextValue = max;
238 
239  TFColor blackColor(0.0, 0.0, 0.0, 0.0);
240  TFColor color;
241  TFColor previousColor = blackColor;
242  TFColor nextColor = blackColor;
243 
244  for(const TFDataType::value_type &data : m_tfData)
245  {
246  if(value < data.first )
247  {
248  nextValue = data.first;
249  nextColor = data.second;
250  break;
251  }
252  else
253  {
254  previousValue = data.first;
255  previousColor = data.second;
256  }
257  }
258  if(previousValue == min)
259  {
260  if(m_isClamped)
261  {
262  color = blackColor;
263  }
264  else
265  {
266  color = nextColor;
267  }
268  }
269  else if(nextValue == max)
270  {
271  if(m_isClamped && (value != previousValue))
272  {
273  color = blackColor;
274  }
275  else
276  {
277  color = previousColor;
278  }
279  }
280  else
281  {
282  if((value - previousValue) < (nextValue - value))
283  {
284  color = previousColor;
285  }
286  else
287  {
288  color = nextColor;
289  }
290  }
291 
292  return color;
293 }

297 TransferFunction::TFColor TransferFunction::getLinearColor( TFValueType value ) const
298 {
299  OSLM_ASSERT("It must have at least one value.", m_tfData.size()>= 1);
300 
301  double min = std::numeric_limits<double>::min();
302  double max = std::numeric_limits<double>::max();
303  double previousValue = min;
304  double nextValue = max;
305 
306  TFColor blackColor(0.0, 0.0, 0.0, 0.0);
307  TFColor color;
308  TFColor previousColor = blackColor;
309  TFColor nextColor = blackColor;
310 
311  for(const TFDataType::value_type &data : m_tfData)
312  {
313  if(value < data.first )
314  {
315  nextValue = data.first;
316  nextColor = data.second;
317  break;
318  }
319  else
320  {
321  previousValue = data.first;
322  previousColor = data.second;
323  }
324  }
325  if(previousValue == min)
326  {
327  if(m_isClamped)
328  {
329  color = blackColor;
330  }
331  else
332  {
333  color = nextColor;
334  }
335  }
336  else if(nextValue == max)
337  {
338  if(m_isClamped && (value != previousValue))
339  {
340  color = blackColor;
341  }
342  else
343  {
344  color = previousColor;
345  }
346  }
347  else
348  {
349  // Interpolate the color.
350  const double distanceToNextValue = nextValue - value;
351  const double distanceToPreviousValue = value - previousValue;
352  const double distance = 1.0 / (nextValue - previousValue);
353  const double coefPrevious = 1.0 - (distanceToPreviousValue * distance);
354  const double coefNext = 1.0 - (distanceToNextValue * distance);
355 
356  color.r = coefPrevious*previousColor.r + coefNext*nextColor.r;
357  color.g = coefPrevious*previousColor.g + coefNext*nextColor.g;
358  color.b = coefPrevious*previousColor.b + coefNext*nextColor.b;
359  color.a = coefPrevious*previousColor.a + coefNext*nextColor.a;
360  }
361  return color;
362 }
