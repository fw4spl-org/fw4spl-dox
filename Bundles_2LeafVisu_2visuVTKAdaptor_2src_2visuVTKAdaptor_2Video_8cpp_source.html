00090 void Video::doUpdate() throw(fwTools::Failed)
00091 {
00092     ::fwData::Video::sptr video = this->getObject< ::fwData::Video >();
00093     if (!video->dataAvailable())
00094         return;
00095 
00096     const ::boost::uint8_t * _pImageBuffer = video->getImageBuffer();
00097 
00098     int size[3];
00099     size[0] = video->getXSize() ;
00100     size[1] = video->getYSize();
00101     size[2] = 1;
00102 
00103     m_array->SetNumberOfComponents(video->m_ui8BPP );
00104     m_array->SetVoidArray( (unsigned char*)_pImageBuffer, video->getXSize() * video->getYSize() ,1);
00105     m_imageData->AllocateScalars(VTK_UNSIGNED_CHAR, video->m_ui8BPP);
00106     m_imageData->GetPointData()->SetScalars(m_array) ;
00107     m_imageData->SetDimensions(size) ;
00108     m_imageData->SetSpacing(1.0, 1.0, 1.0 );
00109     m_imageData->SetOrigin(0.0, 0.0, 0.0 );
00110 
00111     if(!bText_init)
00112     {
00113         m_texture->SetInputData(m_imageData ) ;
00114         this->getActor()->SetScale(size[0], size[1], 1.0);
00115         this->getRenderer()->InteractiveOff();
00116         this->getRenderer()->GetActiveCamera()->ParallelProjectionOn();
00117         this->getRenderer()->ResetCamera();
00118 
00119          //fit the image to the render window
00120         this->getRenderer()->GetActiveCamera()->SetParallelScale(size[1] / 2.0);
00121 
00122         this->addToRenderer(this->getActor());
00123 
00124         bText_init  = true;
00125 
00126         ::boost::filesystem::path pathShaderVP ( "Bundles/visuVTKAdaptor_" + std::string(VISUVTKADAPTOR_VER) + "/vp.glsl" );
00127         ::boost::filesystem::path pathShaderFP ( "Bundles/visuVTKAdaptor_" + std::string(VISUVTKADAPTOR_VER) + "/fp.glsl" );
00128 
00129         vtkSmartPointer<vtkShaderProgram2> pProgram;
00130         pProgram = ::fwRenderVTK::vtk::buildShaderFromFile(this->getRenderer()->GetRenderWindow(),
00131                                                          pathShaderVP.string().c_str(),
00132                                                          pathShaderFP.string().c_str());
00133         SLM_ERROR_IF("Failed to build shader program", !pProgram);
00134 
00135         vtkProperty *property = this->getActor()->GetProperty();
00136         vtkOpenGLProperty *pOpenGLProperty = vtkOpenGLProperty::SafeDownCast(property);
00137         SLM_ASSERT("Shader program not supported", pOpenGLProperty);
00138         pOpenGLProperty->SetPropProgram(pProgram);
00139     }
00140     m_imageData->Modified();
00141     this->setVtkPipelineModified();
00142 }
