53 void SeriesDBTest::testImportSeriesDB()
54 {
55  ::fwMedData::SeriesDB::sptr seriesDB = ::fwMedData::SeriesDB::New();
56 
57  const ::boost::filesystem::path imagePath( ::fwTest::Data::dir() / "fw4spl/image/vtk/img.vtk" );
58  const ::boost::filesystem::path meshPath( ::fwTest::Data::dir() / "fw4spl/mesh/vtk/sphere.vtk" );
59 
60  CPPUNIT_ASSERT_MESSAGE(std::string("Missing file: ") + imagePath.string(), ::boost::filesystem::exists(imagePath));
61  CPPUNIT_ASSERT_MESSAGE(std::string("Missing file: ") + meshPath.string(), ::boost::filesystem::exists(meshPath));
62 
63  ::fwData::location::ILocation::VectPathType paths;
64  paths.push_back(imagePath);
65  paths.push_back(meshPath);
66  paths.push_back(meshPath);
67 
68  ::fwVtkIO::SeriesDBReader::sptr reader = ::fwVtkIO::SeriesDBReader::New();
69  reader->setObject(seriesDB);
70  reader->setFiles(paths);
71  reader->read();
72 
73  CPPUNIT_ASSERT_EQUAL(size_t(2), seriesDB->getContainer().size());
74 
75  ::fwMedData::ImageSeries::sptr imgSeries = ::fwMedData::ImageSeries::dynamicCast(seriesDB->at(0));
76  CPPUNIT_ASSERT_MESSAGE("ImageSeries dynamicCast failed", imgSeries);
77 
78  ::fwMedData::ModelSeries::sptr modelSeries = ::fwMedData::ModelSeries::dynamicCast(seriesDB->at(1));
79  CPPUNIT_ASSERT_MESSAGE("ModelSeries dynamicCast failed", modelSeries);
80 
81  ::fwMedData::ModelSeries::ReconstructionVectorType recVect = modelSeries->getReconstructionDB();
82  CPPUNIT_ASSERT_EQUAL(size_t(2), recVect.size());
83 
84  ::fwData::Reconstruction::sptr rec1 = recVect.at(0);
85  ::fwData::Reconstruction::sptr rec2 = recVect.at(0);
86 
87  CPPUNIT_ASSERT_EQUAL(std::string("sphere"), rec1->getOrganName());
88  CPPUNIT_ASSERT_EQUAL(std::string("sphere"), rec2->getOrganName());
89 
90  ::fwData::Mesh::sptr mesh1 = rec1->getMesh();
91  ::fwData::Mesh::sptr mesh2 = rec1->getMesh();
92 
93  CPPUNIT_ASSERT_EQUAL(mesh1->getNumberOfCells(), (::fwData::Mesh::Id)720);
94  CPPUNIT_ASSERT_EQUAL(mesh1->getNumberOfPoints(), (::fwData::Mesh::Id)362);
95 
96  ::fwDataCamp::visitor::CompareObjects visitor;
97  visitor.compare(mesh1, mesh2);
98  SPTR(::fwDataCamp::visitor::CompareObjects::PropsMapType) props = visitor.getDifferences();
99  for( ::fwDataCamp::visitor::CompareObjects::PropsMapType::value_type prop : (*props) )
100  {
101  OSLM_ERROR( "new object difference found : " << prop.first << " '" << prop.second << "'" );
102  }
103  CPPUNIT_ASSERT_MESSAGE("Object Not equal", props->size() == 0 );
104 }

159

162  CPPUNIT_ASSERT_MESSAGE("ImageSeries is still lazy-loaded", isLoaded(bo));
163  }
164

168  CPPUNIT_ASSERT_MESSAGE("ModelSeries dynamicCast failed", modelSeries);

172 
173  ::fwData::Mesh::sptr mesh = recVect[0]->getMesh();
174 
175  ::fwMemory::BufferObject::sptr points = mesh->getPointsArray()->getBufferObject();
176  CPPUNIT_ASSERT_MESSAGE("ModelSeries points are lazy-loaded", isLoaded(points));
177 
178  ::fwMemory::BufferObject::sptr cell = mesh->getCellDataArray()->getBufferObject();
179  CPPUNIT_ASSERT_MESSAGE("ModelSeries cells are lazy-loaded", isLoaded(cell));
