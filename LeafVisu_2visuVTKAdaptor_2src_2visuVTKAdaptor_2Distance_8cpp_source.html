96 void Distance::doStart()
97 {
98  ::fwData::PointList::sptr ptList = this->getObject< ::fwData::PointList >();
99 
100  m_point1 = ptList->getPoints().front();
101  m_point2 = ptList->getPoints().back();
102 
103  m_point1Connection = m_point1.lock()->signal(::fwData::Object::s_MODIFIED_SIG)->connect(
104  this->slot(::fwServices::IService::s_UPDATE_SLOT));
105  m_point2Connection = m_point2.lock()->signal(::fwData::Object::s_MODIFIED_SIG)->connect(
106  this->slot(::fwServices::IService::s_UPDATE_SLOT));
107 
108  // set color to distance if Point List have Color Field
109  if ( ptList->getField( ::fwDataTools::fieldHelper::Image::m_colorId ) )
110  {
111  ::fwData::Color::sptr color;
112  color = ptList->getField< ::fwData::Color >( ::fwDataTools::fieldHelper::Image::m_colorId );
113  this->setAxisColor( color );
114  }
115 
116  this->addToRenderer(m_lineActor);
117  this->addToRenderer(m_distanceRepresentation);
118 
119  this->doUpdate();
120 }

124 void Distance::doUpdate() throw(fwTools::Failed)
125 {
126  ::fwData::Point::sptr p1 = m_point1.lock();
127  ::fwData::Point::sptr p2 = m_point2.lock();
128 
129  double ps1[3];
130  std::copy(p1->getCRefCoord().begin(),(p1)->getCRefCoord().end(), ps1 );
131 
132  double ps2[3];
133  std::copy(p2->getCRefCoord().begin(),(p2)->getCRefCoord().end(), ps2 );
134 
135  m_distanceRepresentation->GetPoint1Representation()->SetWorldPosition(ps1);
136  m_distanceRepresentation->GetPoint2Representation()->SetWorldPosition(ps2);
137 
138  m_distanceRepresentation->SetPoint1DisplayPosition(ps1);
139  m_distanceRepresentation->SetPoint2DisplayPosition(ps2);
140 
141  m_distanceRepresentation->BuildRepresentation();
142 
143  m_lineSource->SetPoint1(ps1);
144  m_lineSource->SetPoint2(ps2);
145  this->setVtkPipelineModified();
146 }
