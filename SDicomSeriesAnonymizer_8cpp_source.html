123 void SDicomSeriesAnonymizer::anonymize()
124 {
125  ::fwData::Vector::sptr vector = this->getInOut< ::fwData::Vector >("selectedSeries");
126  ::fwMedDataTools::helper::SeriesDB sDBhelper(m_seriesDB);
127 
128  ::fwGdcmIO::helper::DicomSeriesAnonymizer::sptr anonymizer =
129  ::fwGdcmIO::helper::DicomSeriesAnonymizer::New();
130  m_sigJobCreated->emit(anonymizer->getJob());
131 
132  std::vector< ::fwMedData::DicomSeries::sptr > anonymizedDicomSeriesVector;
133 
134  for(::fwData::Vector::ContainerType::value_type value: vector->getContainer())
135  {
136  ::fwMedData::DicomSeries::sptr dicomSeries = ::fwMedData::DicomSeries::dynamicCast(value);
137  ::fwMedData::DicomSeries::sptr anonymizedDicomSeries = ::fwMedData::DicomSeries::New();
138  anonymizer->anonymize(dicomSeries, anonymizedDicomSeries);
139  anonymizedDicomSeriesVector.push_back(anonymizedDicomSeries);
140 
141  m_cancelled = anonymizer->getJob()->cancelRequested();
142  if(m_cancelled)
143  {
144  break;
145  }
146  }
147 
148  if(!m_cancelled)
149  {
150  for(::fwData::Vector::ContainerType::value_type value: vector->getContainer())
151  {
152  ::fwMedData::DicomSeries::sptr dicomSeries = ::fwMedData::DicomSeries::dynamicCast(value);
153  sDBhelper.remove(dicomSeries);
154  }
155 
156  for(auto anonymizedDicomSeries: anonymizedDicomSeriesVector)
157  {
158  sDBhelper.add(anonymizedDicomSeries);
159  }
160  }
161 
162  // Notify SeriesDB
163  sDBhelper.notify();
164 }
