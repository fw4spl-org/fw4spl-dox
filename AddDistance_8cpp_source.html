61 void AddDistance::updating() throw(::fwTools::Failed)
62 {
63  SLM_TRACE("AddDistance::updating");
64  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
65  if (!image)
66  {
67  return;
68  }
69 
70  ::fwData::Point::sptr pt1 = ::fwData::Point::New();
71  std::copy( image->getOrigin().begin(), image->getOrigin().begin() +3, pt1->getRefCoord().begin() );
72 
73  ::fwData::Point::sptr pt2 = ::fwData::Point::New();
74  std::copy( image->getSize().begin(), image->getSize().begin() +3, pt2->getRefCoord().begin() );
75 
76  std::transform( pt2->getRefCoord().begin(),pt2->getRefCoord().end(),
77  image->getSpacing().begin(),
78  pt2->getRefCoord().begin(),
79  std::multiplies<double>() );
80  std::transform( pt2->getRefCoord().begin(),pt2->getRefCoord().end(),
81  image->getOrigin().begin(),
82  pt2->getRefCoord().begin(),
83  std::plus<double>() );
84 
85  ::fwData::PointList::sptr pl = ::fwData::PointList::New();
86 
87  pl->getRefPoints().push_back( pt1 );
88  pl->getRefPoints().push_back( pt2 );
89 
90  ::fwData::Vector::sptr vectDist;
91  vectDist = image->setDefaultField(::fwComEd::Dictionary::m_imageDistancesId, ::fwData::Vector::New());
92 
93  vectDist->getContainer().push_back(pl);
94 
95  // force distance to be shown
96  image->setField("ShowDistances", ::fwData::Boolean::New(true));
97 
98  auto sig = image->signal< ::fwData::Image::DistanceAddedSignalType >(::fwData::Image::s_DISTANCE_ADDED_SIG);
99  sig->asyncEmit(pl);
100 }
