80 struct ThresholdFilter
81 {
82  struct Parameter
83  {
84  double thresholdValue;
85  ::fwData::Image::csptr imageIn;
86  ::fwData::Image::sptr imageOut;
87  };
88 
93  template<class PIXELTYPE>
94  void operator()(Parameter& param)
95  {
96  const PIXELTYPE thresholdValue = static_cast<PIXELTYPE>(param.thresholdValue);
97  ::fwData::Image::csptr imageIn = param.imageIn;
98  ::fwData::Image::sptr imageOut = param.imageOut;
99  SLM_ASSERT("Sorry, image must be 3D", imageIn->getNumberOfDimensions() == 3 );
100  imageOut->copyInformation(imageIn); // Copy image size, type... without copying the buffer
101  imageOut->allocate(); // Allocate the image buffer
102 
103  ::fwDataTools::helper::ImageGetter imageInHelper(imageIn); // helper used to access the image source buffer
104  ::fwDataTools::helper::Image imageOutHelper(imageOut); // helper used to access the image target buffer
105 
106  // Get image buffers
107  const PIXELTYPE* buffer1 = (PIXELTYPE*)imageInHelper.getBuffer();
108  PIXELTYPE* buffer2 = (PIXELTYPE*)imageOutHelper.getBuffer();
109 
110  // Get number of pixels
111  const size_t NbPixels = imageIn->getSize()[0] * imageIn->getSize()[1] * imageIn->getSize()[2];
112 
113  // Fill the target buffer considering the thresholding
114  for( size_t i = 0; i<NbPixels; ++i, ++buffer1, ++buffer2 )
115  {
116  * buffer2 = ( *buffer1 < thresholdValue ) ? 0 : std::numeric_limits<PIXELTYPE>::max();
117  }
118  }
119 };

123 void SThreshold::updating() throw ( ::fwTools::Failed )
124 {
125  SLM_TRACE_FUNC();
126 
127  // threshold value: the pixel with the value less than 50 will be set to 0, else the value is set to the maximum
128  // value of the image pixel type.
129  const double threshold = 50.0;
130 
131  ThresholdFilter::Parameter param; // filter parameters: threshold value, image source, image target
132 
133  // Get source image
134  param.imageIn = this->getInput< ::fwData::Image >("source");
135  SLM_ASSERT("'source' key not found", param.imageIn);
136 
137  // Get target image
138  param.imageOut = this->getInOut< ::fwData::Image >("target");
139  SLM_ASSERT("'target' key not found", param.imageOut);
140 
141  param.thresholdValue = threshold;
142 
143  /*
144  * The dispatcher allows to apply the filter on any type of image.
145  * It invokes the template functor ThresholdFilter using the image type.
146  */
147  ::fwTools::DynamicType type = param.imageIn->getPixelType(); // image type
148 
149  // Invoke filter functor
150  ::fwTools::Dispatcher< ::fwTools::IntrinsicTypes, ThresholdFilter >::invoke( type, param );
151 
152  // Notify that the image target is modified
153  auto sig = param.imageOut->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
154  {
155  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
156  sig->asyncEmit();
157  }
158 }
