85 struct ThresholdFilter
86 {
87  struct Parameter
88  {
89  double thresholdValue;
90  ::fwData::Image::sptr imageIn;
91  ::fwData::Image::sptr imageOut;
92  };
93 
98  template<class PIXELTYPE>
99  void operator()(Parameter &param)
100  {
101  const PIXELTYPE thresholdValue = static_cast<PIXELTYPE>(param.thresholdValue);
102  ::fwData::Image::sptr imageIn = param.imageIn;
103  ::fwData::Image::sptr imageOut = param.imageOut;
104  SLM_ASSERT("Sorry, image must be 3D", imageIn->getNumberOfDimensions() == 3 );
105  imageOut->copyInformation(imageIn); // Copy image size, type... without copying the buffer
106  imageOut->allocate(); // Allocate the image buffer
107 
108  ::fwComEd::helper::Image imageInHelper(imageIn); // helper used to access the image source buffer
109  ::fwComEd::helper::Image imageOutHelper(imageOut); // helper used to access the image target buffer
110 
111  // Get image buffers
112  PIXELTYPE *buffer1 = (PIXELTYPE *)imageInHelper.getBuffer();
113  PIXELTYPE *buffer2 = (PIXELTYPE *)imageOutHelper.getBuffer();
114 
115  // Get number of pixels
116  const size_t NbPixels = imageIn->getSize()[0] * imageIn->getSize()[1] * imageIn->getSize()[2];
117 
118  // Fill the target buffer considering the thresholding
119  for( size_t i = 0; i<NbPixels; ++i, ++buffer1, ++buffer2 )
120  {
121  *buffer2 = ( *buffer1 < thresholdValue ) ? 0 : std::numeric_limits<PIXELTYPE>::max();
122  }
123  }
124 };

128 void SThreshold::updating() throw ( ::fwTools::Failed )
129 {
130  SLM_TRACE_FUNC();
131 
132  // threshold value: the pixel with the value less than 50 will be set to 0, else the value is set to the maximum
133  // value of the image pixel type.
134  const double threshold = 50.0;
135 
136  ThresholdFilter::Parameter param; // filter parameters: threshold value, image source, image target
137 
138  // Get source image
139  OSLM_ASSERT("Image 1 not found. UID : " << m_imageSrcUID, ::fwTools::fwID::exist(m_imageSrcUID));
140  param.imageIn = ::fwData::Image::dynamicCast( ::fwTools::fwID::getObject(m_imageSrcUID) );
141 
142  // Get target image
143  OSLM_ASSERT("Image 2 not found. UID : " << m_imageTgtUID, ::fwTools::fwID::exist(m_imageTgtUID));
144  param.imageOut = ::fwData::Image::dynamicCast( ::fwTools::fwID::getObject(m_imageTgtUID) );
145 
146  param.thresholdValue = threshold;
147 
148  /*
149  * The dispatcher allows to apply the filter on any type of image.
150  * It invokes the template functor ThresholdFilter using the image type.
151  */
152  ::fwTools::DynamicType type = param.imageIn->getPixelType(); // image type
153 
154  // Invoke filter functor
155  ::fwTools::Dispatcher< ::fwTools::IntrinsicTypes, ThresholdFilter >::invoke( type, param );
156 
157  // Notify that the image target is modified
158  auto sig = param.imageOut->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
159  {
160  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
161  sig->asyncEmit();
162  }
163 }
