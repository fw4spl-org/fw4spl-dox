00079 Extension::Validity Extension::validate()
00080 {
00081     // Skips the validation if already done.
00082     if( m_validity != UnknownValidity )
00083     {
00084         return m_validity;
00085     }
00086 
00087     // Retrieves the extension point.
00088     Runtime * rntm( Runtime::getDefault() );
00089      ::boost::shared_ptr< ExtensionPoint >  point( rntm->findExtensionPoint(m_point) );
00090 
00091     // Checks that the point exists.
00092     if( !point )
00093     {
00094         throw RuntimeException(m_point + " : invalid point reference.");
00095     }
00096 
00097     // Validates the extension.
00098     ::boost::shared_ptr< io::Validator >   validator( point->getExtensionValidator() );
00099     OSLM_ASSERT("Sorry, validator creation failed for point "<<point->getIdentifier(), validator );
00100 
00101     // Check extension XML Node <extension id="xxx" implements="yyy" >...</extension>
00102     validator->clearErrorLog();
00103     if( validator->validate( m_xmlNode ) == true )
00104     {
00105         m_validity = Valid;
00106     }
00107     else
00108     {
00109         m_validity = Invalid;
00110         const std::string   identifier = m_id.empty() ? "anonymous" : m_id;
00111         OSLM_ERROR("In bundle " << getBundle()->getIdentifier() << ". " << identifier
00112                 << ": invalid extension XML element node does not respect schema. Verification error log is : "
00113                 << std::endl << validator->getErrorLog() );
00114     }
00115 
00116     return m_validity;
00117 }
