107 IAdaptor::Point2DType IAdaptor::mapAdaptorToScene(Point2DType _xy, ::scene2D::data::Axis::sptr _xAxis,
108  ::scene2D::data::Axis::sptr _yAxis )
109 {
110  double x, y;
111 
112  if (_xAxis->getScaleType() == "LOG")
113  {
114  // Logarithme 10 cannot get negative values
115  if (_xy.first <= 0)
116  {
117  x = 0;
118  }
119  else
120  {
121  // Apply the x scale and the log to the x value
122  x = _xAxis->getScale() * log10( _xy.first );
123  }
124  }
125  else
126  {
127  // Apply just the x scale to the x value
128  x = _xAxis->getScale() * _xy.first;
129  }
130 
131  if (_yAxis->getScaleType() == "LOG")
132  {
133  // Logarithm 10 cannot get negative values
134  if (_xy.second <= 0)
135  {
136  y = 0;
137  }
138  else
139  {
140  // Apply the y scale and the log to the y value
141  y = _yAxis->getScale() * log10( _xy.second );
142  }
143  }
144  else
145  {
146  // Apply just the y scale to the y value
147  y = _yAxis->getScale() * _xy.second;
148  }
149 
150  return Point2DType( x, y );
151 }

155 IAdaptor::Point2DType IAdaptor::mapSceneToAdaptor(Point2DType _xy, ::scene2D::data::Axis::sptr _xAxis,
156  ::scene2D::data::Axis::sptr _yAxis )
157 {
158  // Do the reverse operation of the mapAdaptorToScene function
159  double x, y;
160 
161  if (_xAxis->getScaleType() == "LOG")
162  {
163  x = 10 * exp( _xy.first ) / _xAxis->getScale();
164  }
165  else
166  {
167  x = ( _xy.first ) / _xAxis->getScale();
168  }
169 
170  if (_yAxis->getScaleType() == "LOG")
171  {
172  y = 10 * ( _xy.second ) / _yAxis->getScale();
173  }
174  else
175  {
176  y = _xy.second / _yAxis->getScale();
177  }
178 
179  return Point2DType( x, y );
180 }

184 void IAdaptor::configuring() throw ( ::fwTools::Failed )
185 {
186  SLM_TRACE_FUNC();
187 
188  m_viewInitialSize.first = -1.0f;
189  m_viewInitialSize.second = -1.0f;
190 
191  m_viewportInitialSize.first = -1.0f;
192  m_viewportInitialSize.second = -1.0f;
193 
194  // If the corresponding attributes are present in the config, set the xAxis, yAxis and the adaptor zValue
195  if ( m_configuration->hasAttribute("xAxis") )
196  {
197  m_xAxis = ::scene2D::data::Axis::dynamicCast(
198  this->getScene2DRender()->getRegisteredObject(m_configuration->getAttributeValue("xAxis")));
199  }
200  else
201  {
202  m_xAxis = ::scene2D::data::Axis::New();
203  }
204 
205  if ( m_configuration->hasAttribute("yAxis") )
206  {
207  m_yAxis = ::scene2D::data::Axis::dynamicCast(
208  this->getScene2DRender()->getRegisteredObject(m_configuration->getAttributeValue("yAxis")));
209  }
210  else
211  {
212  m_yAxis = ::scene2D::data::Axis::New();
213  }
214 
215  if ( m_configuration->hasAttribute("zValue") )
216  {
217  m_zValue = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("zValue"));
218  }
219 
220  if ( m_configuration->hasAttribute("opacity") )
221  {
222  m_opacity = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("opacity"));
223  }
224 }
