87 void ImageText::doUpdate() throw(::fwTools::Failed)
88 {
89  std::stringstream ss;
90  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
91 
92  if (::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
93  {
94  ::fwComEd::helper::Image imageHelper(image);
95  unsigned int axialIndex = m_axialIndex->value();
96  unsigned int frontalIndex = m_frontalIndex->value();
97  unsigned int sagittalIndex = m_sagittalIndex->value();
98 
99  int min = this->getLevel() - this->getWindow()/2.0;
100  int max = this->getLevel() + this->getWindow()/2.0;
101 
102  double window = max - min;
103  double level = min + window*0.5;
104 
105  ss << ( ::boost::format("[% 3li,% 3li]") % min % max ) << std::endl;
106  ss << ( ::boost::format("W:% 3lg L:% 3lg") % window % level ) << std::endl;
107  ss << ( ::boost::format("(% 4li,% 4li,% 4li): %s") % sagittalIndex % frontalIndex % axialIndex %
108  imageHelper.getPixelAsString(sagittalIndex, frontalIndex, axialIndex ));
109  }
110 
111  this->setText(ss.str());
112 
113  this->setVtkPipelineModified();
114 }
