00082 void ImageText::doUpdate() throw(::fwTools::Failed)
00083 {
00084     std::stringstream ss;
00085     ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
00086 
00087     if (::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
00088     {
00089         ::fwComEd::helper::Image imageHelper(image);
00090         unsigned int axialIndex    = m_axialIndex->value();
00091         unsigned int frontalIndex  = m_frontalIndex->value();
00092         unsigned int sagittalIndex = m_sagittalIndex->value();
00093 
00094         int min = this->getLevel() - this->getWindow()/2.0;
00095         int max = this->getLevel() + this->getWindow()/2.0;
00096 
00097         double window = max - min;
00098         double level = min + window*0.5;
00099 
00100         ss <<  ( ::boost::format("[% 3li,% 3li]") % min % max ) << std::endl;
00101         ss <<  ( ::boost::format("W:% 3lg L:% 3lg") % window % level ) << std::endl;
00102         ss <<  ( ::boost::format("(% 4li,% 4li,% 4li): %s") % sagittalIndex % frontalIndex % axialIndex %
00103                 imageHelper.getPixelAsString(sagittalIndex, frontalIndex, axialIndex ));
00104     }
00105 
00106     this->setText(ss.str());
00107 
00108     this->setVtkPipelineModified();
00109 }
