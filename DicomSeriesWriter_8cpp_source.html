54 ::boost::filesystem::path longestCommonPrefix( const ::fwMedData::DicomSeries::DicomPathContainerType& paths )
55 {
56  ::boost::filesystem::path longestPrefix;
57  if( !paths.empty() )
58  {
59  longestPrefix = paths.begin()->second;
60  }
61 
62  for(const ::fwMedData::DicomSeries::DicomPathContainerType::value_type& value: paths)
63  {
64  std::pair< ::boost::filesystem::path::const_iterator, ::boost::filesystem::path::const_iterator > p
65  = std::mismatch(longestPrefix.begin(), longestPrefix.end(), value.second.begin());
66 
67  if (p.first != longestPrefix.end())
68  {
69  ::boost::filesystem::path newPrefix;
70  BOOST_FOREACH( const ::boost::filesystem::path &subpath, std::make_pair(longestPrefix.begin(), p.first))
71  {
72  newPrefix /= subpath;
73  }
74  longestPrefix = newPrefix;
75  }
76  }
77 
78  return longestPrefix;
79 }

162 void DicomSeriesWriter::processWrite()
163 {
164  ::fwMedData::DicomSeries::sptr dicomSeries = this->getConcreteObject();
165 
166  FW_RAISE_IF("Dicom series should contain binaries.",
167  dicomSeries->getDicomAvailability() == ::fwMedData::DicomSeries::NONE);
168 
169  // Create folder
170  ::boost::filesystem::path folder = this->getFolder();
171 
172  if(m_writeCount)
173  {
174  std::stringstream ss;
175  ss << "fwGdcmIO" << std::setfill('0') << std::setw(3) << m_writeCount;
176  folder /= ss.str();
177  }
178 
179  if (!::boost::filesystem::exists(folder))
180  {
181  ::boost::filesystem::create_directories(folder);
182  }
183 
184  std::size_t nbInstances = dicomSeries->getNumberOfInstances();
185 
186  m_job->setTotalWorkUnits(nbInstances);
187  unsigned int count = 0;
188 
189  // Write binary files
190  if(dicomSeries->getDicomAvailability() == ::fwMedData::DicomSeries::BINARIES)
191  {
192  BOOST_FOREACH(::fwMedData::DicomSeries::DicomBinaryContainerType::value_type value,
193  dicomSeries->getDicomBinaries())
194  {
195  if(m_job->cancelRequested())
196  {
197  return;
198  }
199 
200  const std::string filename = this->getFilename(value.first);
201 
202  ::fwData::Array::sptr array = value.second;
203  ::fwDataTools::helper::Array arrayHelper(array);
204  char* buffer = static_cast<char*>(arrayHelper.getBuffer());
205  size_t size = array->getSizeInBytes();
206 
207  std::ifstream stream;
208  stream.rdbuf()->pubsetbuf(buffer, size);
209 
210  const ::boost::filesystem::path& dest_dir =
211  m_anonymizer ? folder/m_subPath : folder;
212 
213  if(!::boost::filesystem::exists(dest_dir))
214  {
215  ::boost::filesystem::create_directories(dest_dir);
216  }
217 
218  const ::boost::filesystem::path& dest_file = dest_dir/filename;
219 
220  ::boost::filesystem::ofstream fs(dest_file, std::ios::binary|std::ios::trunc);
221  FW_RAISE_IF("Can't open '" << dest_file.string() << "' for write.", !fs.good());
222 
223  this->processStream(stream, fs);
224  stream.close();
225 
226  m_job->doneWork(++count);
227  }
228  }
229  else
230  {
231  ::boost::filesystem::path longestPrefix = longestCommonPrefix(dicomSeries->getLocalDicomPaths()).parent_path();
232  SLM_TRACE("Longest prefix :" + longestPrefix.string());
233 
234  BOOST_FOREACH(const ::fwMedData::DicomSeries::DicomPathContainerType::value_type &value,
235  dicomSeries->getLocalDicomPaths())
236  {
237  if(m_job->cancelRequested())
238  {
239  break;
240  }
241 
242  const ::boost::filesystem::path& src = value.second;
243  const std::string filename = this->getFilename(src.filename().string());
244 
245  const ::boost::filesystem::path& dest_dir = folder /
246  (m_anonymizer ? m_subPath : removePathPrefix(src.parent_path(),
247  longestPrefix));
248 
249  if(!::boost::filesystem::exists(dest_dir))
250  {
251  ::boost::filesystem::create_directories(dest_dir);
252  }
253 
254  // Read file from filesystem
255  ::boost::filesystem::ifstream stream(src, std::ios::binary);
256 
257  const ::boost::filesystem::path& dest_file = dest_dir / filename;
258 
259  ::boost::filesystem::ofstream fs(dest_file, std::ios::binary|std::ios::trunc);
260  FW_RAISE_IF("Can't open '" << dest_file.string() << "' for write.", !fs.good());
261 
262  this->processStream(stream, fs);
263  stream.close();
264 
265  m_job->doneWork(++count);
266  }
267  }
268 
269  m_job->finish();
270 
271  m_writeCount++;
272 }

311

325  SLM_TRACE("Longest prefix :" + longestPrefix.string());
