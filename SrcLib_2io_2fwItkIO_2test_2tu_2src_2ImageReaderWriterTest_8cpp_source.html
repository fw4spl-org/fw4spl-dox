105 {
106  // inr only support image origin (0,0,0)
107  ::fwData::Image::OriginType origin(3,0);
108  image->setOrigin(origin);
109

112  ::boost::filesystem::create_directories( PATH.parent_path() );
113  ::fwItkIO::ImageWriter::sptr myWriter = ::fwItkIO::ImageWriter::New();
114  myWriter->setObject(image);
115  myWriter->setFile(PATH);
116  myWriter->write();
117 
118  // load Image
119  ::fwData::Image::sptr image2 = ::fwData::Image::New();
120  ::fwItkIO::ImageReader::sptr myReader = ::fwItkIO::ImageReader::New();
121  myReader->setObject(image2);
122  myReader->setFile(PATH);
123  myReader->read();
124 
125  ::boost::filesystem::remove_all( PATH.parent_path().string() );
126 
127  ::fwItkIO::ut::helper::roundSpacing(image2);

130  ::fwTest::helper::ExcludeSetType exclude;
131  exclude.insert("window_center");
132  exclude.insert("window_width");
133  // check Image
