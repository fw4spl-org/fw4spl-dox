86 class PlaneShifterCallback : public MeshVtkCommand
87 {
88 public:
89 
90  static PlaneShifterCallback *New( vtkPlane *src, vtkPlane *dst, double factor = 1.)
91  {
92  return new PlaneShifterCallback( src, dst, factor );
93  }
94 
95  //--------------------------------------------------------------------------
96  PlaneShifterCallback( vtkPlane *src, vtkPlane *dst, double factor )
97  : m_planeSrc(src),
98  m_planeDst(dst),
99  m_factor(factor)
100  {
101  m_planeSrc->Register(this);
102  m_planeDst->Register(this);
103  src->AddObserver(vtkCommand::ModifiedEvent, this);
104  this->Execute( 0, vtkCommand::ModifiedEvent, 0);
105  }
106 
107  void Stop()
108  {
109  m_planeSrc->RemoveObserver(this);
110  m_planeSrc->UnRegister(this);
111  m_planeDst->UnRegister(this);
112  }
113 
114 
115  virtual void Execute( vtkObject *caller, unsigned long eventId, void * )
116  {
117  if (eventId == vtkCommand::ModifiedEvent)
118  {
119  double n[3];
120  double o[3];
121 
122  m_planeSrc->GetNormal(n);
123  m_planeSrc->GetOrigin(o);
124 
125  if (m_factor < 0)
126  {
127  for (int i = 0; i<3; i++)
128  {
129  n[i] = -n[i];
130  }
131  }
132 
133  m_planeDst->SetNormal( n );
134  m_planeDst->SetOrigin( o );
135  m_planeDst->Push( m_factor );
136  m_planeDst->Modified();
137  }
138 
139  }
140 
141 protected:
142  vtkPlane *m_planeSrc;
143  vtkPlane *m_planeDst;
144  double m_factor;
145 };

291  virtual void Execute( vtkObject *caller, unsigned long eventId, void * data)
292  {
293  ::visuVTKAdaptor::Mesh::sptr service;
294 
295  if (m_service.expired())
296  {
297  this->Stop();
298  this->Delete();
299  return;
300  }
301 
302  if (eventId == vtkCommand::ModifiedEvent)
303  {
304  service = m_service.lock();
305 
306  this->Clear();
307 
308  vtkPlane *plane = NULL;
309  for ( m_planeCollectionSrc->InitTraversal();
310  (plane = m_planeCollectionSrc->GetNextItem());
311  )
312  {
313  vtkPlane *newPlane = vtkPlane::New();
314  m_planeCallbacks.push_back(PlaneShifterCallback::New(plane, newPlane, m_factor));
315 
316  vtkPlaneCollection *newCollection = vtkPlaneCollection::New();
317  newCollection->AddItem(newPlane);
318 
319 
320  ::fwRenderVTK::IVtkAdaptorService::sptr meshService =
321  ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService > (
322  service->getObject(),
323  "::visuVTKAdaptor::Mesh" );
324 
325  ::visuVTKAdaptor::Mesh::sptr meshAdaptor = Mesh::dynamicCast(meshService);
326 
327  meshAdaptor->setRenderService ( service->getRenderService() );
328  meshAdaptor->setRenderId ( service->getRenderId() );
329  meshAdaptor->setPickerId ( service->getPickerId() );
330  meshAdaptor->setMaterial ( service->getMaterial() );
331  meshAdaptor->setVtkClippingPlanes( newCollection );
332  meshAdaptor->setAutoRender( service->getAutoRender() );
333 
334  meshAdaptor->start();
335  meshAdaptor->updateVisibility ( service->getVisibility() );
336 
337  newPlane->Delete();
338 
339  m_planeCollections.push_back(newCollection);
340  m_meshServices.push_back(meshAdaptor);
341  }
342 
343  bool hasItems = !m_meshServices.empty();
344  service->setActorPropertyToUnclippedMaterial( hasItems );
345  }
346  else if ( eventId == vtkCommand::UserEvent )
347  {
348  for( ::visuVTKAdaptor::Mesh::wptr adaptor : m_meshServices )
349  {
350  if (!adaptor.expired())
351  {
352  adaptor.lock()->updateVisibility(*(bool*)data);
353  }
354  }
355  }
356  }
357 
358 protected:
359 
360  ::visuVTKAdaptor::Mesh::wptr m_service;
361 
362  vtkPlaneCollection *m_planeCollectionSrc;
363  ::fwData::Mesh::sptr m_mesh;
364 
365  std::vector< ::visuVTKAdaptor::Mesh::wptr > m_meshServices;

571 void Mesh::createTransformService()
572 {
573  ::fwData::Mesh::sptr mesh = this->getObject < ::fwData::Mesh >();
574 
575  if(!this->getTransformId().empty())
576  {
577  vtkTransform *t = m_renderService.lock()->getOrAddVtkTransform(m_transformId);
578  m_transform->Concatenate(t);
579  }
580 
581  ::fwData::TransformationMatrix3D::sptr fieldTransform;
582  fieldTransform = mesh->setDefaultField("TransformMatrix", ::fwData::TransformationMatrix3D::New());
583 
584  vtkTransform *vtkFieldTransform = vtkTransform::New();
585  vtkFieldTransform->Identity();
586  m_transformService = ::visuVTKAdaptor::Transform::dynamicCast(
587  ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService > (
588  fieldTransform,
589  "::visuVTKAdaptor::Transform"
590  )
591  );
592  SLM_ASSERT("Transform service NULL", m_transformService.lock());
593  ::visuVTKAdaptor::Transform::sptr transformService = m_transformService.lock();
594 
595  transformService->setRenderService ( this->getRenderService() );
596  transformService->setRenderId ( this->getRenderId() );
597  transformService->setAutoRender ( this->getAutoRender() );
598 
599  transformService->setTransform(vtkFieldTransform);
600  m_transform->Concatenate(vtkFieldTransform);
601  vtkFieldTransform->Delete();
602 
603  m_actor->SetUserTransform(m_transform);
604 }

813  }

817  texCoord->SetInputData(m_polyData);

819  }

821  {
822  vtkSmartPointer<vtkTextureMapToPlane> texCoord = vtkSmartPointer<vtkTextureMapToPlane>::New();
823  texCoord->SetInputData(m_polyData);
824  m_mapper->SetInputConnection(texCoord->GetOutputPort());
825  }
826  else
827  {
828  m_mapper->SetInputData(m_polyData);
829  }
830 
831  if(!m_textureAdaptorUID.empty())
832  {
833  ::fwData::Material::sptr material = this->getMaterial();
834  SLM_ASSERT("Missing material", material);
835 
836  m_sigTextureApplied->asyncEmit(material);
837  }
838

840  {
841  this->getRenderer()->ResetCamera();
842  }
