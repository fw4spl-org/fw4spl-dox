138 void SQueryEditor::queryPatientName()
139 {
140  try
141  {
142  // Initialize connection
143  m_seriesEnquirer->initialize(
144  m_pacsConfiguration->getLocalApplicationTitle(),
145  m_pacsConfiguration->getPacsHostName(),
146  m_pacsConfiguration->getPacsApplicationPort(),
147  m_pacsConfiguration->getPacsApplicationTitle());
148  m_seriesEnquirer->connect();
149  OFList< QRResponse* > responses;
150 
151  // Find series according to patient's name
152  responses = m_seriesEnquirer->findSeriesByPatientName(m_patientNameLineEdit->text().toStdString());
153 
154  // Convert response to DicomSeries
155  ::fwMedData::SeriesDB::ContainerType series =
156  ::fwPacsIO::helper::Series::toFwMedData(responses);
157  ::fwPacsIO::helper::Series::releaseResponses(responses);
158 
159  // Check whether the instance number start at 1 or 0
160  for(::fwMedData::Series::sptr s: series)
161  {
162  ::fwMedData::DicomSeries::sptr dicomSeries = ::fwMedData::DicomSeries::dynamicCast(s);
163  SLM_ASSERT("The PACS response should contain only DicomSeries", dicomSeries);
164  const std::string instanceUID = m_seriesEnquirer->findSOPInstanceUID(dicomSeries->getInstanceUID(), 0);
165  dicomSeries->setFirstInstanceNumber((instanceUID.empty() ? 1 : 0));
166  }
167 
168  m_seriesEnquirer->disconnect();
169 
170  this->updateSeriesDB(series);
171  }
172  catch (::fwPacsIO::exceptions::Base& exception)
173  {
174  this->displayErrorMessage(exception.what());
175  }
176 
177 }

224 void SQueryEditor::updateSeriesDB(::fwMedData::SeriesDB::ContainerType series)
225 {
226  ::fwMedData::SeriesDB::sptr seriesDB = this->getInOut< ::fwMedData::SeriesDB >("seriesDB");
227  ::fwMedDataTools::helper::SeriesDB seriesDBHelper(seriesDB);
228 
229  // Delete old series from the SeriesDB
230  seriesDBHelper.clear();
231 
232  // Push new series in the SeriesDB
233  for(const ::fwMedData::Series::sptr& s: series)
234  {
235  ::fwMedData::DicomSeries::sptr dicomSeries = ::fwMedData::DicomSeries::dynamicCast(s);
236  seriesDBHelper.add(dicomSeries);
237  }
238 
239  // Notify th SeriesDB
240  seriesDBHelper.notify();
241 
242 }
