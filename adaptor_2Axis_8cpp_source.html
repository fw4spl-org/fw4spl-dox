67 void Axis::configuring() throw( ::fwTools::Failed)
68 {
69  SLM_TRACE_FUNC();
70 
71  SLM_ASSERT("\"config\" tag is missing", m_configuration->getName() == "config");
72 
73  this->IAdaptor::configuring(); // Looks for 'xAxis', 'yAxis' and 'zValue'
74 
75 
76  // 'color'
77  if (!m_configuration->getAttributeValue("color").empty())
78  {
79  ::scene2D::data::InitQtPen::setPenColor(m_color, m_configuration->getAttributeValue("color"));
80  }
81 
82 
83  // 'align' attribute configuration
84  m_align = m_configuration->getAttributeValue("align");
85  SLM_ASSERT(
86  "'align' attribute is missing. Please add an 'align' attribute with value 'left', 'right', 'top' or 'bottom'",
87  !m_align.empty());
88  SLM_ASSERT("Unsupported value for 'align' attribute.",
89  m_align == "left" || m_align == "right" || m_align == "top" || m_align == "bottom");
90 
91 
92  // Axis bounds
93  const std::string min = m_configuration->getAttributeValue("min");
94  const std::string max = m_configuration->getAttributeValue("max");
95 
96  SLM_ASSERT("'min' attribute is missing.", !min.empty());
97  SLM_ASSERT("'max' attribute is missing.", !max.empty());
98 
99  m_min = ::boost::lexical_cast< float >( min );
100  m_max = ::boost::lexical_cast< float >( max );
101 
102 
103  // Ticks size
104  const std::string tickSize = m_configuration->getAttributeValue("tickSize");
105  m_tickSize = ( tickSize.empty() ) ? 1.0 : ::boost::lexical_cast< float >( tickSize );
106 
107 
108  // Viewport
109  SLM_ASSERT("A viewport attribute must be specified with 'viewportUID'.",
110  !m_configuration->getAttributeValue("viewportUID").empty());
111 
112  if( !m_configuration->getAttributeValue("viewportUID").empty() )
113  {
114  m_viewportID = m_configuration->getAttributeValue("viewportUID");
115  }
116 
117 
118  // Step
119  const std::string interval = m_configuration->getAttributeValue("interval");
120  m_interval = ( interval.empty() ) ? 1.0f : ::boost::lexical_cast< float >( interval );
121 }

125 void Axis::buildAxis()
126 {
127  m_color.setCosmetic(true);
128  const int nbValues = (m_max - m_min) / m_interval + 1;
129  m_layer = new QGraphicsItemGroup();
130 
131  for(int i = 0; i < nbValues; ++i)
132  {
133  QGraphicsLineItem* tick = new QGraphicsLineItem(0, 0, 0, 0);
134  tick->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
135  tick->setPen( m_color );
136 
137  m_ticks.push_back( tick );
138  m_layer->addToGroup( tick );
139  }
140 
141  m_line = new QGraphicsLineItem();
142  m_line->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
143  m_line->setPen( m_color );
144 
145  if( m_showLine )
146  {
147  m_layer->addToGroup( m_line );
148  }
149 
150  // Adjust the layer's position and zValue depending on the associated axis
151  m_layer->setPos( m_xAxis->getOrigin(), m_yAxis->getOrigin() );
152  m_layer->setZValue( m_zValue );
153 
154  // Add to the scene the unique item which gather the whole set of rectangle graphic items:
155  this->getScene2DRender()->getScene()->addItem( m_layer );
156 }

174 void Axis::doUpdate() throw( ::fwTools::Failed)
175 {
176  this->initializeViewSize();
177  this->initializeViewportSize();
178 
179  Scene2DRatio ratio = this->getRatio();
180 
181  ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
182  const double viewportHeight = viewport->getHeight();
183  const double viewportWidth = viewport->getWidth();
184 
185  const double viewportSizeRatio = viewportHeight / viewportWidth;
186  const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
187 
188  double viewportWidthRatio = this->getViewportSizeRatio().first;
189 
190  double scaleX = m_tickSize;
191  double scaleY = m_tickSize * viewportSizeRatio;
192  scaleY /= viewportWidthRatio;
193  scaleY *= viewInitialSizeRatio;
194 
195  scaleX = scaleX * ratio.first;
196  scaleY = scaleY * ratio.second;
197 
198  const size_t nbValues = m_ticks.size();
199 
200  const double min = this->getStartVal();
201  const double max = this->getEndVal();
202 
203  float pos;
204  std::pair<double, double> tickSize;
205  std::pair<double, double> tickPos;
206 
207  const std::pair<double, double> viewportSize = this->mapAdaptorToScene(
208  std::pair<double, double>(viewportWidth, viewportHeight), m_xAxis, m_yAxis);
209 
210  if(m_align == "bottom")
211  {
212  tickSize = this->mapAdaptorToScene(std::pair<double, double>(0, m_tickSize), m_xAxis, m_yAxis);
213 
214  const double tickPosY = m_viewport->getY();
215 
216  for(int i = 0; i < nbValues; ++i)
217  {
218  pos = min + i * m_interval;
219  tickPos = this->mapAdaptorToScene(std::pair<double, double>(pos, tickPosY), m_xAxis, m_yAxis);
220  m_ticks.at(i)->setLine(
221  tickPos.first, tickPos.second,
222  tickPos.first, tickPos.second - tickSize.second * scaleY);
223  }
224 
225  m_line->setLine(min, tickPos.second, max, tickPos.second);
226  }
227  else if(m_align == "top")
228  {
229  tickSize = this->mapAdaptorToScene(std::pair<double, double>(0, m_tickSize), m_xAxis, m_yAxis);
230 
231  const double tickPosY = m_viewport->getHeight() * 0.9;
232 
233  for(int i = 0; i < nbValues; ++i)
234  {
235  pos = min + i * m_interval;
236  tickPos = this->mapAdaptorToScene(std::pair<double, double>(pos, tickPosY), m_xAxis, m_yAxis);
237 
238  m_ticks.at(i)->setLine(
239  tickPos.first, tickPos.second,
240  tickPos.first, tickPos.second - tickSize.second * scaleY);
241  }
242 
243  m_line->setLine(min, tickPos.second, max, tickPos.second);
244  }
245  else if(m_align == "left")
246  {
247  tickSize = this->mapAdaptorToScene(std::pair<double, double>(m_tickSize, 0), m_xAxis, m_yAxis);
248 
249  const double tickPosX = m_viewport->getX();
250 
251  for(int i = 0; i < nbValues; ++i)
252  {
253  pos = min + i * m_interval;
254  tickPos = this->mapAdaptorToScene(std::pair<double, double>(tickPosX, pos), m_xAxis, m_yAxis);
255  m_ticks.at(i)->setLine(
256  tickPos.first, tickPos.second,
257  tickPos.first + tickSize.first * scaleX, tickPos.second);
258  }
259 
260  m_line->setLine( tickPos.first, min, tickPos.first, tickPos.second);
261  }
262  else if(m_align == "right")
263  {
264  tickSize = this->mapAdaptorToScene(
265  std::pair<double, double>(m_tickSize, 0), m_xAxis, m_yAxis);
266 
267  const double tickPosX = m_viewport->getX() + m_viewport->getWidth();
268 
269  for(int i = 0; i < nbValues; ++i)
270  {
271  pos = min + i * m_interval;
272 
273  tickPos = this->mapAdaptorToScene(
274  std::pair<double, double>(tickPosX, pos), m_xAxis, m_yAxis);
275 
276  m_ticks.at(i)->setLine(
277  tickPos.first - tickSize.first * scaleX, tickPos.second,
278  tickPos.first, tickPos.second);
279  }
280 
281  m_line->setLine(tickPos.first, min, tickPos.first, tickPos.second);
282  }
283 }
