00060 void Activater::apply()
00061 {
00062     ::boost::shared_ptr< Bundle >  bundle = Runtime::getDefault()->findBundle(m_identifier, m_version);
00063     OSLM_FATAL_IF("Unable to activate Bundle " << m_identifier << "_" << m_version << ". Not found.", bundle==0);
00064 
00065     bundle->setEnable( true );
00066 
00067     // Managment of parameter configuration
00068     for( ParameterContainer::const_iterator i = m_parameters.begin();
00069             i != m_parameters.end();
00070             ++i )
00071     {
00072         bundle->addParameter( i->first, i->second );
00073     }
00074 
00075     // Disable extension point for this bundle
00076     for( DisableExtensionPointContainer::const_iterator id = m_disableExtensionPoints.begin();
00077             id != m_disableExtensionPoints.end();
00078             ++id )
00079     {
00080         if( bundle->hasExtensionPoint(*id) )
00081         {
00082             bundle->setEnableExtensionPoint( *id, false );
00083         }
00084         else
00085         {
00086             OSLM_ERROR("Unable to disable Extension Point " << *id << " defined in the Bundle " << m_identifier << ". Not found.");
00087         }
00088     }
00089 
00090     // Disable extension for this bundle
00091     for( DisableExtensionContainer::const_iterator id = m_disableExtensions.begin();
00092             id != m_disableExtensions.end();
00093             ++id )
00094     {
00095         if( bundle->hasExtension(*id) )
00096         {
00097             bundle->setEnableExtension( *id, false );
00098         }
00099         else
00100         {
00101             OSLM_ERROR("Unable to disable Extension " << *id << " defined in the Bundle " << m_identifier << ". Not found.");
00102         }
00103     }
00104 }
