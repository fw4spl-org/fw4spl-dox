45 void SExtractObj::configuring() throw( ::fwTools::Failed )
46 {
47  typedef ::fwRuntime::ConfigurationElement::sptr ConfigurationType;
48 
49  if(this->isVersion2())
50  {
51  const ConfigurationType inoutCfg = m_configuration->findConfigurationElement("inout");
52  SLM_ASSERT("At one 'inout' tag is required.", inoutCfg);
53 
54  const std::vector< ConfigurationType > extractCfg = inoutCfg->find("extract");
55  SLM_ASSERT("At least one 'extract' tag is required.", !extractCfg.empty());
56 
57  const ConfigurationType outCfg = m_configuration->findConfigurationElement("out");
58 
59  const std::vector< ConfigurationType > outKeyCfg = outCfg->find("key");
60  SLM_ASSERT("You must have as many 'from' tags as 'out' keys.", extractCfg.size() == outKeyCfg.size());
61 
62  for(ConfigurationType cfg : extractCfg)
63  {
64  SLM_ASSERT("Missing attribute 'from'.", cfg->hasAttribute("from"));
65  const std::string from = cfg->getAttributeValue("from");
66 
67  m_sourcePaths.push_back(from);
68  }
69  }
70  else
71  {
72  std::vector< ConfigurationType > extractCfg = m_configuration->find("extract");
73  SLM_ASSERT("At least one 'extract' tag is required.", !extractCfg.empty());
74 
75  for(ConfigurationType cfg : extractCfg)
76  {
77  SLM_ASSERT("Missing attribute 'from'.", cfg->hasAttribute("from"));
78  SLM_ASSERT("Missing attribute 'to'.", cfg->hasAttribute("to"));
79 
80  std::string from = cfg->getAttributeValue("from");
81  std::string to = cfg->getAttributeValue("to");
82 
83  SLM_ASSERT("'from' attribute must begin with '@'", from.substr(0,1) == "@");
84 
85  m_extract[from] = to;
86  }
87 
88  const ConfigurationType modeConfig = m_configuration->findConfigurationElement("mode");
89  if (modeConfig)
90  {
91  auto mode = modeConfig->getValue();
92  if(mode == "extractOnStart")
93  {
94  m_mode = ModeType::START;
95  }
96  else if(mode == "extractOnUpdate")
97  {
98  m_mode = ModeType::UPDATE;
99  }
100  else
101  {
102  SLM_ERROR("Mode " + mode + " unknown. It should be either 'extractOnStart' or 'extractOnUpdate'");
103  }
104  }
105  }
106 }

145 void SExtractObj::extract()
146 {
147  if(this->isVersion2())
148  {
149  auto sourceObject = this->getInOut< ::fwData::Object >("source");
150 
151  size_t index = 0;
152  for(auto path : m_sourcePaths)
153  {
154  const std::string from = path;
155 
156  ::fwData::Object::sptr object;
157  try
158  {
159  object = ::fwDataCamp::getObject( sourceObject, from, true );
160  }
161  catch(::fwDataCamp::exception::NullPointer& np)
162  {
163  this->setOutput("target", nullptr, index);
164  }
165  catch(::fwDataCamp::exception::ObjectNotFound& nf)
166  {
167  SLM_WARN("Object from '"+ from +"' not found");
168  }
169  catch(std::exception& e)
170  {
171  OSLM_FATAL("Unhandled exception: " << e.what());
172  }
173 
174  SLM_WARN_IF("Object from '"+ from +"' not found", !object);
175  if(object)
176  {
177  this->setOutput("target", object, index);
178  }
179  ++index;
180  }
181  }
182  else
183  {
184  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
185 
186  ::fwData::mt::ObjectWriteLock lock(composite);
187  ::fwDataTools::helper::Composite compHelper(composite);
188 
189  for(ExtractMapType::value_type elt : m_extract)
190  {
191  std::string from = elt.first;
192  std::string to = elt.second;
193 
194  try
195  {
196  ::fwData::Object::sptr object = ::fwDataCamp::getObject( composite, from, true );
197 
198  SLM_WARN_IF("Object from '"+ from +"' not found", !object);
199  if(object)
200  {
201  if (composite->find(to) == composite->end())
202  {
203  compHelper.add(to, object);
204  }
205  else
206  {
207  compHelper.swap(to, object);
208  }
209  }
210  }
211  catch(::fwDataCamp::exception::NullPointer& np)
212  {
213  if (composite->find(to) != composite->end())
214  {
215  compHelper.remove(to);
216  }
217 
218  }
219  catch(::fwDataCamp::exception::ObjectNotFound& nf)
220  {
221  SLM_WARN("Object from '"+ from +"' not found");
222  }
223  catch(std::exception& e)
224  {
225  OSLM_FATAL("Unhandled exception: " << e.what());
226  }
227  }
228  compHelper.notify();
229  }
230 }
