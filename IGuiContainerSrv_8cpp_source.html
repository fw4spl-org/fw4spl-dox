00074 void IGuiContainerSrv::create()
00075 {
00076     SLM_ASSERT("ViewRegistrar must be initialized.", m_viewRegistrar);
00077     ::fwGui::container::fwContainer::sptr parent = m_viewRegistrar->getParent();
00078     SLM_ASSERT("Parent container is unknown.", parent);
00079 
00080     ::fwGui::GuiBaseObject::sptr guiObj = ::fwGui::factory::New(::fwGui::builder::IContainerBuilder::REGISTRY_KEY);
00081     m_containerBuilder = ::fwGui::builder::IContainerBuilder::dynamicCast(guiObj);
00082 
00083     OSLM_ASSERT("ClassFactoryRegistry failed for class "<< ::fwGui::builder::IContainerBuilder::REGISTRY_KEY,
00084                 m_containerBuilder);
00085     m_containerBuilder->createContainer(parent);
00086 
00087     ::fwGui::container::fwContainer::sptr container = m_containerBuilder->getContainer();
00088 
00089     if ( m_viewLayoutManagerIsCreated )
00090     {
00091         if (m_hasToolBar)
00092         {
00093             m_toolBarBuilder->createToolBar(parent);
00094             m_viewRegistrar->manageToolBar(m_toolBarBuilder->getToolBar());
00095         }
00096 
00097         m_viewLayoutManager->createLayout(container);
00098         m_viewRegistrar->manage(m_viewLayoutManager->getSubViews());
00099     }
00100 }
