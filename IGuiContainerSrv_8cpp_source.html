51 void IGuiContainerSrv::initialize()
52 {
53  SLM_ASSERT("Service hasn't configuration", m_configuration);
54 
55  // Create view registrar
56  m_viewRegistrar = ::fwGui::registrar::ViewRegistrar::New( this->getID() );
57  // find ViewRegistrar configuration
58  std::vector < ConfigurationType > vectViewMng = m_configuration->find("registry");
59  if ( !vectViewMng.empty() )
60  {
61  m_viewRegistrarConfig = vectViewMng.at(0);
62  m_viewRegistrar->initialize(m_viewRegistrarConfig);
63  }
64 
65  // Create initializeLayoutManager
66  // find gui configuration
67  std::vector < ConfigurationType > vectGui = m_configuration->find("gui");
68  if( !vectGui.empty() )
69  {
70  SLM_ASSERT("No <registry> tag is allowed in the <gui> section", vectGui.at(0)->find("registry").empty());
71 
72  // find view LayoutManager configuration
73  std::vector < ConfigurationType > vectLayoutMng = vectGui.at(0)->find("layout");
74  if( !vectLayoutMng.empty() )
75  {
76  m_viewLayoutConfig = vectLayoutMng.at(0);
77  this->initializeLayoutManager( m_viewLayoutConfig );
78  m_viewLayoutManagerIsCreated = true;
79  }
80 
81  // find toolBarBuilder configuration
82  std::vector < ConfigurationType > vectTBBuilder = vectGui.at(0)->find("toolBar");
83  if(!vectTBBuilder.empty())
84  {
85  m_toolBarConfig = vectTBBuilder.at(0);
86  this->initializeToolBarBuilder(m_toolBarConfig);
87 
88  m_hasToolBar = true;
89  }
90 
91  // find slideView configuration
92  std::vector < ConfigurationType > vectSlideCfg = vectGui.at(0)->find("slideView");
93  for (const auto& slideCfg : vectSlideCfg)
94  {
95  this->initializeSlideViewBuilder(slideCfg);
96  }
97  }
98 }

102 void IGuiContainerSrv::create()
103 {
104  SLM_ASSERT("ViewRegistrar must be initialized.", m_viewRegistrar);
105  ::fwGui::container::fwContainer::sptr parent = m_viewRegistrar->getParent();
106  SLM_ASSERT("Parent container is unknown.", parent);
107 
108  ::fwGui::registry::worker::get()->postTask< void >([this, &parent]
109  {
110  SLM_ASSERT("ViewRegistrar must be initialized.", m_viewRegistrar);
111 
112  ::fwGui::GuiBaseObject::sptr guiObj =
113  ::fwGui::factory::New(::fwGui::builder::IContainerBuilder::REGISTRY_KEY);
114  m_containerBuilder = ::fwGui::builder::IContainerBuilder::dynamicCast(guiObj);
115 
116  SLM_ASSERT("ClassFactoryRegistry failed for class "+ ::fwGui::builder::IContainerBuilder::REGISTRY_KEY,
117  m_containerBuilder);
118  m_containerBuilder->createContainer(parent);
119 
120  ::fwGui::container::fwContainer::sptr container = m_containerBuilder->getContainer();
121 
122  if ( m_viewLayoutManagerIsCreated )
123  {
124  if (m_hasToolBar)
125  {
126  ::fwGui::registry::worker::get()->postTask<void>([&]
127  {
128  m_toolBarBuilder->createToolBar(parent);
129  }).wait();
130 
131  m_viewRegistrar->manageToolBar(m_toolBarBuilder->getToolBar());
132  }
133 
134  ::fwGui::registry::worker::get()->postTask<void>([&]
135  {
136  m_viewLayoutManager->createLayout(container);
137  }).wait();
138 
139  std::vector< ::fwGui::container::fwContainer::sptr > views = m_viewLayoutManager->getSubViews();
140 
141  for (const auto& slideBuilder : m_slideViewBuilders)
142  {
143  SLM_ASSERT("Slide builder is not instantiated", slideBuilder);
144  ::fwGui::registry::worker::get()->postTask<void>([&]
145  {
146  slideBuilder->createContainer(container);
147  }).wait();
148  views.push_back(slideBuilder->getContainer());
149  }
150 
151  m_viewRegistrar->manage(views);
152  }
153  }).wait();
154 }

158 void IGuiContainerSrv::destroy()
159 {
160  SLM_ASSERT("ViewRegistrar must be initialized.", m_viewRegistrar);
161 
162  if ( m_viewLayoutManagerIsCreated )
163  {
164  if (m_hasToolBar)
165  {
166  m_viewRegistrar->unmanageToolBar();
167  SLM_ASSERT("ToolBarBuilder must be initialized.", m_toolBarBuilder);
168 
169  ::fwGui::registry::worker::get()->postTask<void>([&]
170  {
171  m_toolBarBuilder->destroyToolBar();
172  }).wait();
173  }
174 
175  m_viewRegistrar->unmanage();
176  SLM_ASSERT("ViewLayoutManager must be initialized.", m_viewLayoutManager);
177 
178  ::fwGui::registry::worker::get()->postTask<void>([&]
179  {
180  m_viewLayoutManager->destroyLayout();
181  }).wait();
182  }
183 
184  for (const auto& slideBuilder : m_slideViewBuilders)
185  {
186  SLM_ASSERT("Slide builder is not instantiated", slideBuilder);
187  ::fwGui::registry::worker::get()->postTask<void>([&]
188  {
189  slideBuilder->destroyContainer();
190  }).wait();
191  }
192 
193  m_containerBuilder->destroyContainer();
194 }

198 void IGuiContainerSrv::initializeLayoutManager(ConfigurationType layoutConfig)
199 {
200  SLM_ASSERT("Bad configuration name "+layoutConfig->getName()+ ", must be layout",
201  layoutConfig->getName() == "layout");
202  SLM_ASSERT("<layout> tag must have type attribute", layoutConfig->hasAttribute("type"));
203  const std::string layoutManagerClassName = layoutConfig->getAttributeValue("type");
204 
205  ::fwGui::GuiBaseObject::sptr guiObj = ::fwGui::factory::New(layoutManagerClassName);
206  m_viewLayoutManager = ::fwGui::layoutManager::IViewLayoutManager::dynamicCast(guiObj);
207  SLM_ASSERT("ClassFactoryRegistry failed for class "+ layoutManagerClassName, m_viewLayoutManager);
208 
209  m_viewLayoutManager->initialize(layoutConfig);
210 }
