130 void atomTest(const ::boost::filesystem::path & filePath)
131 {
132  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
133  ::fwRuntime::EConfigurationElement::sptr fileCfg = ::fwRuntime::EConfigurationElement::New("file");
134  fileCfg->setValue(filePath.string());
135  srvCfg->addConfigurationElement(fileCfg);
136 
137  ::fwMedData::SeriesDB::sptr seriesDB = ::fwTest::generator::SeriesDB::createSeriesDB(2,2,2);
138  ::fwData::Composite::sptr workspace = ::fwData::Composite::New();
139  workspace->getContainer()["processingDB"] = ::fwData::Composite::New();
140  workspace->getContainer()["planningDB"] = ::fwData::Composite::New();
141 
142  ::boost::filesystem::create_directories( filePath.parent_path() );
143  writeReadFile< ::fwData::Composite>( srvCfg, workspace, "::ioAtoms::SWriter", "::ioAtoms::SReader" );
144  writeReadFile< ::fwMedData::SeriesDB >( srvCfg, seriesDB, "::ioAtoms::SWriter", "::ioAtoms::SReader" );
145 
146  ::fwMedData::SeriesDB::sptr readSeriesDB;
147 
148  // Default policy
149  readSeriesDB = read< ::fwMedData::SeriesDB >(srvCfg, "::ioAtoms::SReader");
150 
151  {
152  ::fwDataCamp::visitor::CompareObjects visitor;
153  visitor.compare(readSeriesDB, seriesDB);
154  compareLog(visitor);
155  CPPUNIT_ASSERT_MESSAGE("Objects not equal", visitor.getDifferences()->empty() );
156  }
157 
158 
159  // 'Change' UUID policy
160  ::fwRuntime::EConfigurationElement::sptr uuidPolicyCfg = ::fwRuntime::EConfigurationElement::New("uuidPolicy");
161  uuidPolicyCfg->setValue("Change");
162 
163  srvCfg->addConfigurationElement(uuidPolicyCfg);
164 
165  readSeriesDB = read< ::fwMedData::SeriesDB >(srvCfg, "::ioAtoms::SReader");
166 
167  {
168  ::fwDataCamp::visitor::CompareObjects visitor;
169  visitor.compare(readSeriesDB, seriesDB);
170  compareLog(visitor);
171  CPPUNIT_ASSERT_MESSAGE("Objects not equal", visitor.getDifferences()->empty() );
172  }
173 
174 
175  // 'Strict' UUID policy
176  uuidPolicyCfg->setValue("Strict");
177  readSeriesDB = read< ::fwMedData::SeriesDB >(srvCfg, "::ioAtoms::SReader");
178 
179  {
180  // DuplicatedDataUUID exception should have been thrown and catch by reader before any data/atom conversion.
181  // We can only check that loaded data is empty.
182  CPPUNIT_ASSERT_MESSAGE("Written data should not be empty", !seriesDB->empty());
183  CPPUNIT_ASSERT_MESSAGE("Loaded data should be empty", readSeriesDB->empty());
184  }
185 
186 
187  // 'Reuse' UUID policy
188  uuidPolicyCfg->setValue("Reuse");
189 
190  ::fwRuntime::EConfigurationElement::sptr injectCfg = ::fwRuntime::EConfigurationElement::New("inject");
191  injectCfg->setValue("seriesDB");
192  srvCfg->addConfigurationElement(injectCfg);
193 
194  ::fwData::Composite::sptr composite = read< ::fwData::Composite >(srvCfg, "::ioAtoms::SReader");
195 
196  {
197  ::fwDataCamp::visitor::CompareObjects visitor;
198  visitor.compare(workspace, composite);
199  compareLog(visitor);
200  CPPUNIT_ASSERT_MESSAGE("Objects should be different", !visitor.getDifferences()->empty() );
201  CPPUNIT_ASSERT(composite->find("seriesDB") != composite->end());
202 
203  {
204  ::fwMedData::SeriesDB::sptr newSeriesDB = ::fwMedData::SeriesDB::dynamicCast( (*composite)["seriesDB"] );
205  CPPUNIT_ASSERT_MESSAGE("Failed to retrieve SeriesDB inside composite with key 'seriesDB'", newSeriesDB);
206 
207  ::fwDataCamp::visitor::CompareObjects visitor;
208  visitor.compare(seriesDB, newSeriesDB);
209  compareLog(visitor);
210  CPPUNIT_ASSERT_MESSAGE("Objects not equal", visitor.getDifferences()->empty() );
211  }
212  }
