169 IService::SharedFutureType IService::start() //throw( ::fwTools::Failed)
170 {
171  if( !m_associatedWorker || ::fwThread::getCurrentThreadId() == m_associatedWorker->getThreadId() )
172  {
173  OSLM_FATAL_IF("Service "<<this->getID()<<" already started", m_globalState != STOPPED);
174 
175  PackagedTaskType task( ::boost::bind(&IService::starting, this) );
176  UniqueFutureType ufuture = task.get_future();
177 
178  m_globalState = STARTING;
179  task();
180  m_globalState = STARTED;
181 
182  if ( ufuture.has_exception() )
183  {
184  ufuture.get();
185  }
186  return ::boost::move(ufuture);
187  }
188  else
189  {
190  return m_slotStart->asyncRun();
191  }
192 }

255 IService::SharedFutureType IService::swap( ::fwData::Object::sptr _obj ) //throw(::fwTools::Failed)
256 {
257  if( !m_associatedWorker || ::fwThread::getCurrentThreadId() == m_associatedWorker->getThreadId() )
258  {
259  OSLM_ASSERT("Swapping on "<< this->getID() << " with same Object " << _obj->getID(),
260  m_associatedObject.lock() != _obj );
261  OSLM_FATAL_IF(
262  "Service "<< this->getID() << " is not STARTED, no swapping with Object " << _obj->getID(),
263  m_globalState != STARTED);
264 
265  PackagedTaskType task( ::boost::bind(&IService::swapping, this) );
266  UniqueFutureType ufuture = task.get_future();
267 
268  m_globalState = SWAPPING;
269  ::fwServices::OSR::swapService( _obj, this->getSptr() );
270  task();
271  m_globalState = STARTED;
272 
273  if ( ufuture.has_exception() )
274  {
275  ufuture.get();
276  }
277  return ::boost::move(ufuture);
278  }
279  else
280  {
281  return m_slotSwap->asyncRun( _obj );
282  }
