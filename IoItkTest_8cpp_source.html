00228 void IoItkTest::SeriesDBInrTest()
00229 {
00230     /*
00231      * - image.inr.gz : CT, type int16, size: 512x512x134, spacing 0.781:0.781:1.6
00232      * - skin.inr.gz : mask skin, type uint8, size: 512x512x134, spacing 0.781:0.781:1.6
00233      */
00234     const ::boost::filesystem::path imageFile = ::fwTest::Data::dir() / "fw4spl/image/inr/image.inr.gz";
00235     const ::boost::filesystem::path skinFile = ::fwTest::Data::dir() / "fw4spl/image/inr/skin.inr.gz";
00236     // Create Config
00237     ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
00238     ::fwRuntime::EConfigurationElement::sptr fileImageCfg = ::fwRuntime::EConfigurationElement::New("file");
00239     fileImageCfg->setValue(imageFile.string());
00240     srvCfg->addConfigurationElement(fileImageCfg);
00241 
00242     ::fwRuntime::EConfigurationElement::sptr fileSkinCfg = ::fwRuntime::EConfigurationElement::New("file");
00243     fileSkinCfg->setValue(skinFile.string());
00244     srvCfg->addConfigurationElement(fileSkinCfg);
00245 
00246     // load SeriesDB
00247     ::fwMedData::SeriesDB::sptr sdb = ::fwMedData::SeriesDB::New();
00248     executeService( sdb, "::io::IReader", "::ioITK::SInrSeriesDBReader", srvCfg );
00249 
00250     ::fwData::Image::SpacingType spacing = list_of(0.781)(0.781)(1.6);
00251     ::fwData::Image::SizeType size = list_of(512)(512)(134);
00252 
00253     CPPUNIT_ASSERT_EQUAL(size_t(2), sdb->getContainer().size());
00254     ::fwMedData::ImageSeries::sptr imgSeries = ::fwMedData::ImageSeries::dynamicCast(sdb->getContainer()[0]);

00256     CPPUNIT_ASSERT_EQUAL(std::string("OT"), imgSeries->getModality());
00257 
00258     ::fwData::Image::sptr image = imgSeries->getImage();
00259     CPPUNIT_ASSERT(image);
00260     CPPUNIT_ASSERT_EQUAL(std::string("int16"), image->getType().string());
00261     CPPUNIT_ASSERT(size == image->getSize());
00262     CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[0], image->getSpacing()[0], EPSILON);
00263     CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[1], image->getSpacing()[1], EPSILON);
00264     CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[2], image->getSpacing()[2], EPSILON);
00265 
00266     imgSeries = ::fwMedData::ImageSeries::dynamicCast(sdb->getContainer()[1]);
00267     CPPUNIT_ASSERT(imgSeries);
00268     CPPUNIT_ASSERT_EQUAL(std::string("OT"), imgSeries->getModality());

00270 
00271     image = imgSeries->getImage();
00272     CPPUNIT_ASSERT(image);
00273     CPPUNIT_ASSERT_EQUAL(std::string("uint8"), image->getType().string());
00274     CPPUNIT_ASSERT(size == image->getSize());
00275     CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[0], image->getSpacing()[0], EPSILON);
00276     CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[1], image->getSpacing()[1], EPSILON);
