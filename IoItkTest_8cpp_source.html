148 void IoItkTest::testSaveLoadInr()
149 {
150  ::fwData::Image::sptr image = ::fwData::Image::New();
151  ::fwTest::generator::Image::generateRandomImage(image, ::fwTools::Type::create("int16"));
152 
153  // inr only support image origin (0,0,0)
154  ::fwData::Image::OriginType origin(3,0);
155  image->setOrigin(origin);
156 
157  // save image in inr
158  const ::boost::filesystem::path PATH = "imageInrTest/image.inr.gz";
159  ::boost::filesystem::create_directories( PATH.parent_path() );
160 
161  // Create Config
162  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
163  ::fwRuntime::EConfigurationElement::sptr fileCfg = ::fwRuntime::EConfigurationElement::New("file");
164  fileCfg->setValue(PATH.string());
165  srvCfg->addConfigurationElement(fileCfg);
166 
167  // Create and execute service
168  executeService( image, "::io::IWriter", "::ioITK::InrImageWriterService", srvCfg );
169 
170  // load Image
171  ::fwData::Image::sptr image2 = ::fwData::Image::New();
172  executeService( image2, "::io::IReader", "::ioITK::InrImageReaderService", srvCfg );
173 
174  ::boost::filesystem::remove_all( PATH.parent_path().string() );
175 
176  ::fwData::Image::SpacingType spacing = image2->getSpacing();
177  std::transform (spacing.begin(), spacing.end(), spacing.begin(), tolerance);
178  image2->setSpacing(spacing);
179 
180  // check Image
181  ::fwTest::helper::ExcludeSetType exclude;
182  exclude.insert("window_center");
183  exclude.insert("window_width");
184 
185  CPPUNIT_ASSERT(::fwTest::helper::compare(image, image2, exclude));
186 }

235 void IoItkTest::SeriesDBInrTest()
236 {
237  /*
238  * - image.inr.gz : CT, type int16, size: 512x512x134, spacing 0.781:0.781:1.6
239  * - skin.inr.gz : mask skin, type uint8, size: 512x512x134, spacing 0.781:0.781:1.6
240  */
241  const ::boost::filesystem::path imageFile = ::fwTest::Data::dir() / "fw4spl/image/inr/image.inr.gz";
242  const ::boost::filesystem::path skinFile = ::fwTest::Data::dir() / "fw4spl/image/inr/skin.inr.gz";
243  // Create Config
244  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
245  ::fwRuntime::EConfigurationElement::sptr fileImageCfg = ::fwRuntime::EConfigurationElement::New("file");
246  fileImageCfg->setValue(imageFile.string());
247  srvCfg->addConfigurationElement(fileImageCfg);
248 
249  ::fwRuntime::EConfigurationElement::sptr fileSkinCfg = ::fwRuntime::EConfigurationElement::New("file");
250  fileSkinCfg->setValue(skinFile.string());
251  srvCfg->addConfigurationElement(fileSkinCfg);
252 
253  // load SeriesDB
254  ::fwMedData::SeriesDB::sptr sdb = ::fwMedData::SeriesDB::New();
255  executeService( sdb, "::io::IReader", "::ioITK::SInrSeriesDBReader", srvCfg );
256 
257  ::fwData::Image::SpacingType spacing = list_of(0.781)(0.781)(1.6);
258  ::fwData::Image::SizeType size = list_of(512)(512)(134);
259 
260  CPPUNIT_ASSERT_EQUAL(size_t(2), sdb->getContainer().size());
261  ::fwMedData::ImageSeries::sptr imgSeries = ::fwMedData::ImageSeries::dynamicCast(sdb->getContainer()[0]);
262  CPPUNIT_ASSERT(imgSeries);
263  CPPUNIT_ASSERT_EQUAL(std::string("OT"), imgSeries->getModality());
264 
265  ::fwData::Image::sptr image = imgSeries->getImage();
266  CPPUNIT_ASSERT(image);
267  CPPUNIT_ASSERT_EQUAL(std::string("int16"), image->getType().string());
268  CPPUNIT_ASSERT(size == image->getSize());
269  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[0], image->getSpacing()[0], EPSILON);
270  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[1], image->getSpacing()[1], EPSILON);
271  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[2], image->getSpacing()[2], EPSILON);
272 
273  imgSeries = ::fwMedData::ImageSeries::dynamicCast(sdb->getContainer()[1]);
274  CPPUNIT_ASSERT(imgSeries);
275  CPPUNIT_ASSERT_EQUAL(std::string("OT"), imgSeries->getModality());
276  CPPUNIT_ASSERT(imgSeries->getImage());
277 
278  image = imgSeries->getImage();
279  CPPUNIT_ASSERT(image);
280  CPPUNIT_ASSERT_EQUAL(std::string("uint8"), image->getType().string());
281  CPPUNIT_ASSERT(size == image->getSize());
282  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[0], image->getSpacing()[0], EPSILON);
283  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[1], image->getSpacing()[1], EPSILON);
284  CPPUNIT_ASSERT_DOUBLES_EQUAL(spacing[2], image->getSpacing()[2], EPSILON);
285 }
