44 void SComputeHistogram::configuring() throw ( ::fwTools::Failed )
45 {
46  if(!this->isVersion2())
47  {
48  std::vector < ::fwRuntime::ConfigurationElement::sptr > cfg = m_configuration->find("histogramId");
49  SLM_ASSERT("Missing tag 'histogramId'", !cfg.empty());

72 void SComputeHistogram::updating() throw ( ::fwTools::Failed )
73 {
74  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
75 
76  ::fwData::mt::ObjectReadLock imgLock(image);
77 
78  if(::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
79  {
80  ::fwData::Histogram::sptr histogram = this->getHistogram();
81 
82  ::fwData::mt::ObjectWriteLock lock(histogram);
83 
84  ComputeHistogramFunctor::Parameter param;
85  param.image = image;
86  param.histogram = histogram;
87  param.binsWidth = m_binsWidth;
88 
89  ::fwTools::DynamicType type = image->getPixelType();
90  ::fwTools::Dispatcher< ::fwTools::IntrinsicTypes, ComputeHistogramFunctor >::invoke( type, param );
91 
92  auto sig = histogram->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
93  {
94  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
95  sig->asyncEmit();
96  }
97  }
98 }

115 ::fwData::Histogram::sptr SComputeHistogram::getHistogram()
116 {
117  ::fwData::Histogram::sptr histogram;
118  if(!this->isVersion2())
119  {
120  SLM_ASSERT("Object " << m_histogramId << " doesn't exist", ::fwTools::fwID::exist(m_histogramId));
121 
122  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_histogramId);
123  histogram = ::fwData::Histogram::dynamicCast(obj);
124  SLM_ASSERT("Object " << m_histogramId << " is not a '::fwData::Histogram'", histogram);
125  }
126  else
127  {
128  histogram = this->getInOut< ::fwData::Histogram>("histogram");
129  }
130 
131  return histogram;
132 }
