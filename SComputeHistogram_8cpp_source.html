68 void SComputeHistogram::updating() throw ( ::fwTools::Failed )
69 {
70  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
71 
72  ::fwData::mt::ObjectReadLock imgLock(image);
73 
74  if(::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
75  {
76  ::fwData::Histogram::sptr histogram = this->getHistogram();
77 
78  ::fwData::mt::ObjectWriteLock lock(histogram);
79 
80  ComputeHistogramFunctor::Parameter param;
81  param.image = image;
82  param.histogram = histogram;
83  param.binsWidth = m_binsWidth;
84 
85  ::fwTools::DynamicType type = image->getPixelType();
86  ::fwTools::Dispatcher< ::fwTools::IntrinsicTypes, ComputeHistogramFunctor >::invoke( type, param );
87 
88  auto sig = histogram->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
89  {
90  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
91  sig->asyncEmit();
92  }
93  }
94 }
