63  virtual void Execute( vtkObject *caller, unsigned long eventId, void *)
64  {
65  assert(m_priority>=0);
66  SLM_ASSERT("m_adaptor not instanced", m_adaptor);
67  SLM_ASSERT("m_picker not instanced", m_picker);
68  if ( m_mouseMoveObserved ||
69  (m_adaptor->getInteractor()->GetControlKey() && m_adaptor->getInteractor()->GetShiftKey()) )
70  {
71  if ( eventId == vtkCommand::MouseMoveEvent )
72  {
73  SLM_ASSERT("m_mouseMoveObserved not instanced", m_mouseMoveObserved);
74  if(pickSomething())
75  {
76  process();
77  }
78  }
79  else if ( eventId == START_INTERACTION_EVENT)
80  {
81  assert(m_mouseMoveObserved == false );
82  if ( pickSomething() )
83  {
84  m_mouseMoveObserved = true;
85  SetAbortFlag(1);
86  m_adaptor->resetPointList();
87  process();
88  m_adaptor->getInteractor()->AddObserver(vtkCommand::MouseMoveEvent, this, m_priority);
89  }
90  }
91  else if ( eventId == STOP_INTERACTION_EVENT && m_mouseMoveObserved)
92  {
93  SLM_ASSERT("m_mouseMoveObserved not instanced", m_mouseMoveObserved);
94  m_adaptor->getInteractor()->RemoveObservers(vtkCommand::MouseMoveEvent, this);
95  m_mouseMoveObserved = false;
96  }
97  }
98  }
99 
100  bool pickSomething()
101  {
102  int x,y;
103  double display[3];
104 
105  m_adaptor->getInteractor()->GetEventPosition(x, y);
106  display[0] = x;
107  display[1] = y;
108  display[2] = 0;
109 
110  return m_picker->Pick( display, m_adaptor->getRenderer() );
111  }

119  m_adaptor->addPoint( world[0], world[1], world[2] );
120  }
