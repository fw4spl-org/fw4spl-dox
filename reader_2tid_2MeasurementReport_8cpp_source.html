37 void MeasurementReport::readSR(SPTR(::fwGdcmIO::container::sr::DicomSRNode)root)
38 {
39  if(root->getCodedAttribute() ==
40  ::fwGdcmIO::container::DicomCodedAttribute("dd1dd1", "DCM", "Imaging Measurement Report"))
41  {
42  for(const SPTR(::fwGdcmIO::container::sr::DicomSRNode)& node: root->getSubNodeContainer())
43  {
44  // Try to identify a fiducial node
45  if(node->getCodedAttribute() == ::fwGdcmIO::container::DicomCodedAttribute("dd1d93", "DCM", "Fiducials"))
46  {
47  for(const SPTR(::fwGdcmIO::container::sr::DicomSRNode)& subNode: node->getSubNodeContainer())
48  {
49  ::fwGdcmIO::reader::tid::Fiducial fiducial(m_dicomSeries, m_reader, m_instance, m_object, m_logger);
50  fiducial.readNode(subNode);
51  }
52  }
53  // Try to identify a measurement node
54  else if(node->getCodedAttribute() ==
55  ::fwGdcmIO::container::DicomCodedAttribute("dd1d91", "DCM", "Imaging Measurements"))
56  {
57  for(const SPTR(::fwGdcmIO::container::sr::DicomSRNode)& subNode: node->getSubNodeContainer())
58  {
59  ::fwGdcmIO::reader::tid::Measurement measurement(m_dicomSeries, m_reader, m_instance, m_object,
60  m_logger);
61  measurement.readNode(subNode);
62  }
63  }
64  }
65  }
66 }
