48 void ToolBarLayoutManager::createLayout( ::fwGui::container::fwToolBar::sptr parent )
49 {
50  SLM_TRACE_FUNC();
51 
52  m_parent = ::fwGuiQt::container::QtToolBarContainer::dynamicCast(parent);
53  SLM_ASSERT("dynamicCast fwToolBar to QtToolBarContainer failed", m_parent);
54 
55  QToolBar* toolBar = m_parent->getQtToolBar();
56 
57  QActionGroup * actionGroup = 0;
58  unsigned int menuItemIndex = 0;
59  for ( ::fwGui::layoutManager::IToolBarLayoutManager::ActionInfo actionInfo : m_actionInfo)
60  {
61  if (actionInfo.m_isSeparator)
62  {
63  if (actionInfo.m_size > 0)
64  {
65  QWidget * widget = new QWidget(toolBar);
66  widget->setMinimumWidth(actionInfo.m_size);
67  toolBar->addWidget(widget);
68  }
69  else
70  {
71  toolBar->addSeparator();
72  }
73  actionGroup = 0;
74  }
75  else if (actionInfo.m_isSpacer)
76  {
77  QWidget* spacer = new QWidget(toolBar);
78  spacer->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Expanding);
79  toolBar->addWidget(spacer);
80  actionGroup = 0;
81  }
82  else if (actionInfo.m_isMenu)
83  {
84  ::fwGuiQt::container::QtMenuContainer::sptr menu = ::fwGuiQt::container::QtMenuContainer::New();
85  QMenu* qtMenu = new QMenu(toolBar);
86  menu->setQtMenu(qtMenu);
87 
88  QToolButton * toolButton = new QToolButton(toolBar);
89  if (toolBar->orientation() == Qt::Horizontal)
90  {
91  toolButton->setSizePolicy(QSizePolicy::Fixed, QSizePolicy::Expanding);
92  }
93  else
94  {
95  toolButton->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
96  }
97  toolButton->setMenu(qtMenu);
98  toolButton->setPopupMode(QToolButton::InstantPopup);
99  toolButton->setText(QString::fromStdString(actionInfo.m_name));
100  if (!actionInfo.m_icon.empty())
101  {
102  QIcon icon(QString::fromStdString(actionInfo.m_icon));
103  toolButton->setIcon(icon);
104  toolButton->setToolTip(QString::fromStdString(actionInfo.m_name));
105  toolButton->setToolButtonStyle(Qt::ToolButtonTextBesideIcon);
106  }
107  toolBar->addWidget(toolButton);
108  m_menus.push_back(menu);
109  }
110  else if (actionInfo.m_isEditor)
111  {
112  ::fwGuiQt::container::QtContainer::sptr container = ::fwGuiQt::container::QtContainer::New();
113  QWidget* widget = new QWidget(toolBar);
114  container->setQtContainer(widget);
115 
116  if (toolBar->orientation() == Qt::Horizontal)
117  {
118  widget->setSizePolicy(QSizePolicy::Fixed, QSizePolicy::Expanding);
119  }
120  else
121  {
122  widget->setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
123  }
124  widget->adjustSize();
125  toolBar->addWidget(widget);
126 
127  m_containers.push_back(container);
128  }
129  else
130  {
131  ::fwGuiQt::container::QtMenuItemContainer::sptr menuItem = ::fwGuiQt::container::QtMenuItemContainer::New();
132  QAction *action;
133  if (!actionInfo.m_icon.empty())
134  {
135  QIcon icon(QString::fromStdString(actionInfo.m_icon));
136  if (!actionInfo.m_icon2.empty())
137  {
138  icon.addFile(QString::fromStdString(actionInfo.m_icon2), QSize(), QIcon::Normal, QIcon::On);
139  icon.addFile(QString::fromStdString(actionInfo.m_icon2), QSize(), QIcon::Active, QIcon::On);
140  }
141  action = toolBar->addAction( icon, QString::fromStdString(actionInfo.m_name) );
142  }
143  else
144  {
145  action = toolBar->addAction( QString::fromStdString(actionInfo.m_name) );
146  }
147 
148  action->setCheckable(actionInfo.m_isCheckable || actionInfo.m_isRadio);
149 
150  if (actionInfo.m_isRadio)
151  {
152  if (!actionGroup)
153  {
154  actionGroup = new QActionGroup(toolBar);
155  }
156  actionGroup->addAction(action);
157  }
158 
159  // create shortcut
160  if( !actionInfo.m_shortcut.empty() )
161  {
162  action->setShortcut(QKeySequence(QString::fromStdString(actionInfo.m_shortcut)));
163  }
164 
165  menuItem->setQtMenuItem(action);
166 
167  m_menuItems.push_back(menuItem);
168  OSLM_ASSERT("No callback found for menu" << actionInfo.m_name, menuItemIndex < m_callbacks.size());
169  ::fwGui::IMenuItemCallback::sptr callback = m_callbacks.at(menuItemIndex);
170 
171  ::fwGuiQt::ActionCallback::sptr qtCallback = ::fwGuiQt::ActionCallback::dynamicCast(callback);
172  SLM_ASSERT("dynamicCast IMenuItemCallback to ActionCallback failed", qtCallback);
173 
174  QObject::connect( action, SIGNAL(triggered(bool)), qtCallback.get(), SLOT(executeQt(bool)));
175  QObject::connect( action, SIGNAL(toggled(bool)), qtCallback.get(), SLOT(checkQt(bool)));
176  menuItemIndex++;
177  }
178  }
