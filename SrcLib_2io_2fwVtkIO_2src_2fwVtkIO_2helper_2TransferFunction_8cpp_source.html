18 void TransferFunction::toVtkLookupTable(
19  ::fwData::TransferFunction::sptr tf,
20  vtkSmartPointer<vtkLookupTable> lt,
21  bool allowTransparency,
22  unsigned int size )
23 {
24  SLM_WARN_IF( "This transfer function conversion is not managed, transfer function is clamped.",
25  tf->getIsClamped() );
26 
27  // Configures basic parameters
28  lt->SetNumberOfTableValues( size );
29  lt->SetScaleToLinear();
30 
31 
32  ::fwData::TransferFunction::TFValuePairType minMax = tf->getMinMaxTFValues();
33 
34  lt->SetTableRange( minMax.first, minMax.second );
35 
36  double delta = ( minMax.second - minMax.first ) / (double) (size - 1);
37  ::fwData::TransferFunction::TFColor interpolatedColor;
38 
39  if ( allowTransparency )
40  {
41  for( unsigned int k = 0; k<size; ++k )
42  {
43  interpolatedColor = tf->getInterpolatedColor( k*delta + minMax.first );
44  lt->SetTableValue(k, interpolatedColor.r, interpolatedColor.g, interpolatedColor.b, interpolatedColor.a);
45  }
46  }
47  else
48  {
49  for( unsigned int k = 0; k<size; ++k )
50  {
51  interpolatedColor = tf->getInterpolatedColor( k*delta + minMax.first );
52  lt->SetTableValue(k, interpolatedColor.r, interpolatedColor.g, interpolatedColor.b, 1.0);
53  }
54  }
55 
56  lt->Modified();
57 }

73  lt->SetRampToLinear();

76  lt->SetHueRange( 0.0, 0.0 );
77  lt->SetSaturationRange( 0.0, 0.0 );
78  lt->SetValueRange( 0.0, 1.0 );
79 
80  lt->Build();
81 
82  lt->Modified();
83 }
