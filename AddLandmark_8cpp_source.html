00084 void AddLandmark::updating() throw(::fwTools::Failed)
00085 {
00086     SLM_TRACE_FUNC();
00087     ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
00088     if (!::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
00089     {
00090         ::fwGui::dialog::MessageDialog::showMessageDialog(
00091                 "Add landmarks",
00092                 "Sorry, it is impossible to add image landmarks. There is not loaded image in the software.",
00093                 ::fwGui::dialog::IMessageDialog::WARNING);
00094         return;
00095     }
00096 
00097     std::string value;
00098     if ( defineLabel(value) )
00099     {
00100         //get landmarks
00101         ::fwComEd::fieldHelper::MedicalImageHelpers::checkLandmarks(  image );
00102         ::fwData::PointList::sptr landmarks =  image->getField< ::fwData::PointList >( ::fwComEd::Dictionary::m_imageLandmarksId);
00103         SLM_ASSERT("landmarks not instanced", landmarks);
00104 
00105         // create a new point
00106         ::fwData::Point::sptr newPoint = ::fwComEd::fieldHelper::MedicalImageHelpers::getImageSliceIndices( image );
00107         // transform slice to mm
00108         std::transform( newPoint->getRefCoord().begin(),newPoint->getRefCoord().end(),
00109                 image->getSpacing().begin(),
00110                 newPoint->getRefCoord().begin(),
00111                 std::multiplies<double>() );
00112         std::transform( newPoint->getRefCoord().begin(),newPoint->getRefCoord().end(),
00113                 image->getOrigin().begin(),
00114                 newPoint->getRefCoord().begin(),
00115                 std::plus<double>() );
00116         // append to landmark
00117         landmarks->getRefPoints().push_back( newPoint );
00118 
00119         // append to point the label
00120         ::fwData::String::sptr label = ::fwData::String::New();
00121         label->value() = value;
00122         newPoint->setField( ::fwComEd::Dictionary::m_labelId , label );
00123 
00124         image->setField("ShowLandmarks", ::fwData::Boolean::New(true));
00125 
00126         // notify
00127         ::fwComEd::ImageMsg::sptr msg = ::fwComEd::ImageMsg::New();
00128         msg->addEvent( ::fwComEd::ImageMsg::LANDMARK );
00129         ::fwServices::IEditionService::notify(this->getSptr(), image, msg);
00130     }
00131 }
