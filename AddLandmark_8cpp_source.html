86 void AddLandmark::updating() throw(::fwTools::Failed)
87 {
88  SLM_TRACE_FUNC();
89  ::fwData::Image::sptr image;
90  if (this->isVersion2())
91  {
92  image = this->getInOut< ::fwData::Image >("image");
93  }
94  else
95  {
96  image = this->getObject< ::fwData::Image >();
97  }
98  if (!::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
99  {
100  ::fwGui::dialog::MessageDialog::showMessageDialog(
101  "Add landmarks",
102  "It is impossible to add image landmarks. There is no loaded image in the software.",
103  ::fwGui::dialog::IMessageDialog::WARNING);
104  return;
105  }
106 
107  std::string value;
108  if ( defineLabel(value) )
109  {
110  //get landmarks
111  ::fwDataTools::fieldHelper::MedicalImageHelpers::checkLandmarks( image );
112  ::fwData::PointList::sptr landmarks = image->getField< ::fwData::PointList >(
113  ::fwDataTools::fieldHelper::Image::m_imageLandmarksId);
114  SLM_ASSERT("landmarks not instanced", landmarks);
115 
116  // create a new point
117  ::fwData::Point::sptr newPoint = ::fwDataTools::fieldHelper::MedicalImageHelpers::getImageSliceIndices( image );
118  // transform slice to mm
119  std::transform( newPoint->getRefCoord().begin(),newPoint->getRefCoord().end(),
120  image->getSpacing().begin(),
121  newPoint->getRefCoord().begin(),
122  std::multiplies<double>() );
123  std::transform( newPoint->getRefCoord().begin(),newPoint->getRefCoord().end(),
124  image->getOrigin().begin(),
125  newPoint->getRefCoord().begin(),
126  std::plus<double>() );
127  // append to landmark
128  landmarks->getRefPoints().push_back( newPoint );
129 
130  // append to point the label
131  ::fwData::String::sptr label = ::fwData::String::New();
132  label->value() = value;
133  newPoint->setField( ::fwDataTools::fieldHelper::Image::m_labelId, label );
134 
135  image->setField("ShowLandmarks", ::fwData::Boolean::New(true));
136 
137  // notify
138  auto sig = image->signal< ::fwData::Image::LandmarkAddedSignalType >(::fwData::Image::s_LANDMARK_ADDED_SIG);
139  sig->asyncEmit(newPoint);
140  }
141 }
