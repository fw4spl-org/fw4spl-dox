82 void PlaneSelector::selectObject( ::fwData::Object::sptr object )
83 {
84  ::fwData::Object::sptr oldObject = m_currentObject.lock();
85 
86  if (oldObject != object)
87  {
88  if (oldObject)
89  {
90  auto sig = oldObject->signal< ::fwData::Plane::SelectedSignalType >(
91  ::fwData::Plane::s_SELECTED_SIG);
92  sig->asyncEmit(false);
93  }
94 
95  m_currentObject.reset();
96 
97  if (object)
98  {
99  auto sig = object->signal< ::fwData::Plane::SelectedSignalType >(
100  ::fwData::Plane::s_SELECTED_SIG);
101  sig->asyncEmit(true);
102  m_currentObject = object;
103  }
104  }
