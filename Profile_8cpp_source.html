115 void Profile::start()
116 {
117  std::for_each( m_activaters.begin(), m_activaters.end(), Apply< ActivaterContainer::value_type >() );
118 
119  // Check validity of extension
120  Runtime * rntm( Runtime::getDefault() );
121  for( Runtime::ExtensionIterator i = rntm->extensionsBegin(); i != rntm->extensionsEnd(); ++i )
122  {
123  SPTR( Extension ) extension( *i );
124  OSLM_FATAL_IF(
125  "Validation not ok for bundle = '" << extension->getBundle()->getIdentifier() << "' (extension id = '" << extension->getIdentifier() << "' )",
126  extension->getBundle()->isEnable() && extension->validate() == Extension::Invalid );
127  }
128 
129  std::for_each( m_starters.begin(), m_starters.end(), Apply< StarterContainer::value_type >() );
130  OSLM_TRACE( "NB INITIALIZERS" << m_initializers.size() );
131 }

199 void Profile::setParams(const Profile::ParamsContainer &params)
200 {
201  m_params = params;
202 
203  if (m_argv)
204  {
205  delete[] m_argv;
206  }
207 
208  m_argc = m_params.size();
209  // allocate memory for an array of character strings
210  m_argv = new char*[m_params.size()];
211 
212  // for each string, allocate memory in the character array and copy
213  for(size_t i = 0; i<m_params.size(); i++)
214  {
215  size_t paramSize = m_params[i].size();
216  m_argv[i] = new char[paramSize+1];
217 #ifndef _WIN32
218  strncpy(m_argv[i], m_params[i].c_str(), paramSize);
219  m_argv[i][paramSize] = '\0';
220 #else
221  strncpy_s(m_argv[i], paramSize+1, m_params[i].c_str(), paramSize);
222 #endif
223  }
224 }
