65  vtkPlaneDeleteCallBack( ::fwRenderVTK::IVtkAdaptorService* service ) :
66  m_service(service),
67  m_picker( vtkCellPicker::New() ),
68  m_propCollection( vtkPropCollection::New() )

88  void fillPickList()
89  {
90  m_picker->InitializePickList();
91  m_propCollection->RemoveAllItems();
92  m_service->getAllSubProps(m_propCollection);
93  m_propCollection->InitTraversal();
94 
95  vtkProp* prop;
96 
97  while ( (prop = m_propCollection->GetNextProp()) )
98  {
99  m_picker->AddPickList(prop);
100  }
101  }
102 
103  virtual void Execute( vtkObject* caller, unsigned long eventId, void*)
104  {
105  int pos[2];
106  m_service->getInteractor()->GetLastEventPosition(pos);
107 
108  if ( eventId == vtkCommand::RightButtonPressEvent )
109  {
110  std::copy(pos, pos+1, m_lastPos);
111  m_display[0] = pos[0];
112  m_display[1] = pos[1];
113 
114  this->fillPickList();
115  if (m_picker->Pick( m_display, m_service->getRenderer() ) )
116  {
117  if(getSelectedPlane())
118  {
119  SetAbortFlag(1);
120  }
121  else
122  {
123  m_pickedPlane.reset();
124  }
125  }
126  }
127  else if ( eventId == vtkCommand::RightButtonReleaseEvent &&
128  std::equal(pos, pos+1, m_lastPos) && !m_pickedPlane.expired() )
129  {
130  // backup of plane
131  ::fwData::Plane::sptr planeBackup(m_pickedPlane);
132 
133  ::fwData::PlaneList::sptr planeList = m_service->getObject< ::fwData::PlaneList >();
134  planeList->getRefPlanes().erase
135  (
136  std::find( planeList->getRefPlanes().begin(), planeList->getRefPlanes().end(), m_pickedPlane.lock())
137  );
138  notifyDeletePlane(planeList, planeBackup);
139  }
140  }
141  bool getSelectedPlane()
142  {
143  bool isFind = false;
144  vtkPropCollection* propc = m_picker->GetActors();
145  vtkProp* prop;
146 
147  propc->InitTraversal();
148  while ( (prop = propc->GetNextProp()) )
149  {
150  m_pickedPlane = ::fwData::Plane::dynamicCast(m_service->getAssociatedObject(prop,1));
151  if( !m_pickedPlane.expired() )
152  {
153  ::fwData::PlaneList::sptr planeList = m_service->getObject< ::fwData::PlaneList >();
154  if(!planeList->getRefPlanes().empty())
155  {
156  ::fwData::PlaneList::PlaneListContainer::iterator itr = std::find(
157  planeList->getRefPlanes().begin(), planeList->getRefPlanes().end(), m_pickedPlane.lock());
158  if(itr != planeList->getRefPlanes().end() )
159  {
160  isFind = true;
161  break;
162  }
163  }
164  }
165  }
166  return isFind;
167  }

217 void PlaneList::doUpdate() throw(fwTools::Failed)
218 {
219  SLM_TRACE_FUNC();
220  ::fwData::PlaneList::sptr planeList = this->getObject< ::fwData::PlaneList >();
221 
222  bool showPlanes;
223  showPlanes = planeList->getField("ShowPlanes", ::fwData::Boolean::New(true))->value();
224  if(showPlanes)
225  {
226  for( ::fwData::Plane::sptr plane : planeList->getPlanes() )
227  {
228  ::fwRenderVTK::IVtkAdaptorService::sptr servicePlane =
229  ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService >
230  ( plane, "::visuVTKAdaptor::Plane" );
231  SLM_ASSERT("servicePlane not instanced", servicePlane);
232 
233  servicePlane->setRenderService(this->getRenderService());
234  servicePlane->setRenderId( this->getRenderId() );
235  servicePlane->setPickerId( this->getPickerId() );
236  servicePlane->setAutoRender( this->getAutoRender() );
237 
238  if (!m_planeCollectionId.empty())
239  {
240  Plane::dynamicCast(servicePlane)->setVtkPlaneCollection( this->getVtkObject(m_planeCollectionId) );
241  }
242  servicePlane->start();
243 
244  m_planeConnections.connect(servicePlane, Plane::s_INTERACTION_STARTED_SIG, this->getSptr(),
245  s_UPDATE_SELECTION_SLOT);
246 
247  this->registerService(servicePlane);
248  }
249  }
