59 void PaintCommand::prePaint( ::fwData::Image::IndexType index )
60 {
61  ::fwData::Image::sptr image = m_image.lock();
62  unsigned int imageTypeSize = image->getPixelType().sizeOf();
63 
64  m_currentPrepaintIndex = index;
65  SLM_ASSERT("currentPrepaintBuff must be empty. Forgot a postPaint call ?", m_currentPrepaintBuff.empty());
66  ::fwData::Image::BufferType* pixels = static_cast< ::fwData::Image::BufferType* >( m_imageHelper->getBuffer() );
67  ::fwData::Image::BufferType* buf = pixels + index*imageTypeSize;
68  std::copy(buf, buf+imageTypeSize,
69  std::back_insert_iterator<std::vector< ::fwData::Image::BufferType > >(m_currentPrepaintBuff));
70 }

74 void PaintCommand::postPaint()
75 {
76  ::fwData::Image::sptr image = m_image.lock();
77  ::fwData::Image::BufferType* pixels = static_cast< ::fwData::Image::BufferType* >( m_imageHelper->getBuffer() );
78  ::fwData::Image::BufferType* buf = pixels + m_currentPrepaintIndex;
79 
80  unsigned int imageTypeSize = image->getPixelType().sizeOf();
81  ::fwData::Image::BufferIndexType bufIndex = m_currentPrepaintIndex * imageTypeSize;
82 
83  for (unsigned int i = 0; i < imageTypeSize; ++i)
84  {
85  this->paint( bufIndex + i, m_currentPrepaintBuff[i], buf[i]);
86  }
87  m_currentPrepaintBuff.clear();
88 }

101 void PaintCommand::apply()
102 {
103  // start image editing
104  ::fwData::Image::BufferType* pixels = static_cast< ::fwData::Image::BufferType* >( m_imageHelper->getBuffer() );
105  SLM_ASSERT("commandIndexContainer and commandColorContainer must have same size",
106  m_commandColorContainer.size() == m_commandIndexContainer.size());
107 
108  // do each voxel modification
109  ColorContainer::const_iterator indexC( m_commandColorContainer.begin() );
110  for(::fwData::Image::BufferIndexType index : m_commandIndexContainer)
111  {
112  pixels[ index ] = indexC->second;
113  ++indexC;
114  }
115  this->notifyImageModification();
116 }
