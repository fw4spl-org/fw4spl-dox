00082 void Color::setRGBA( std::string  hexaColor )
00083 {
00084     OSLM_ASSERT(
00085             "Color string should start with '#' and followed by 6 ou 8 "
00086             "hexadecimal digits. Given color : " << hexaColor ,
00087             hexaColor[0] == '#'
00088                 && ( hexaColor.length() == 7 || hexaColor.length() == 9)
00089             );
00090 
00091     std::string redString = hexaColor.substr(1, 2);
00092     std::string greenString = hexaColor.substr(3, 2);
00093     std::string blueString = hexaColor.substr(5, 2);
00094     int r,g,b, a = 255;
00095 
00096     std::istringstream iss;
00097     iss.str (redString);
00098     iss >> std::hex >> r;
00099     iss.clear();
00100     iss.str (greenString);
00101     iss >> std::hex >> g;
00102     iss.clear();
00103     iss.str (blueString);
00104     iss >> std::hex >> b;
00105 
00106     if (hexaColor.length() == 9)
00107     {
00108         std::string alphaString = hexaColor.substr(7, 2);
00109         iss.clear();
00110         iss.str (alphaString);
00111         iss >> std::hex >> a;
00112     }
00113 
00114     this->setRGBA(r/255.0f, g/255.0f, b/255.0f, a/255.0f);
00115 
00116 }
