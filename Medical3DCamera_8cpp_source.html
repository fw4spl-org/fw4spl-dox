00077     SLM_TRACE_FUNC();

00080     this->setRenderId( m_configuration->getAttributeValue("renderer") );
00081     if(m_configuration->hasAttribute("sliceIndex"))

00083         std::string  orientation = m_configuration->getAttributeValue("sliceIndex");
00084         SLM_ASSERT("Unknown orientation", m_orientationConversion.find(orientation) != m_orientationConversion.end());
00085         m_orientation = m_orientationConversion[orientation];

00087 
00088     if(m_configuration->hasAttribute("resetAtStart"))
00089     {
00090         std::string reset = m_configuration->getAttributeValue("resetAtStart");
00091         SLM_ASSERT("'resetAtStart' value must be 'yes' or 'no'",
00092                    reset == "yes" || reset == "no");
00093         m_resetAtStart = (reset == "yes");
00094     }

00133 void Medical3DCamera::doReceive( ::fwServices::ObjectMsg::csptr msg) throw(fwTools::Failed)
00134 {
00135     ::fwComEd::ImageMsg::csptr imageMsg = ::fwComEd::ImageMsg::dynamicConstCast( msg );
00136 
00137     if ( imageMsg && imageMsg->hasEvent( "CAMERA_ORIENTATION") )
00138     {
00139         ::fwData::Object::csptr dataInfo = imageMsg->getDataInfo("CAMERA_ORIENTATION");

00144                 m_orientationConversion.find(orientation->value()) != m_orientationConversion.end());
