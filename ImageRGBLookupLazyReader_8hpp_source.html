77  template< typename T, typename U >
78  FWDCMTKIO_API static T* createInstanceBuffer(unsigned int rows, unsigned int columns,
79  const ::boost::filesystem::path& instance, const T* redLookup,
80  const T* greenLookup,
81  const T* blueLookup)
82  {
83  DcmFileFormat fileFormat;
84  OFCondition status;
85  DcmDataset* dataset;
86 
87  // Register codecs
88  ::fwDcmtkIO::helper::Codec::registerCodecs();
89 
90  // Create temporary buffer
91  T* tempoBuffer = new T[rows * columns * 3]; // 3 colors (RGB)
92 
93  // Read instance
94  const std::string filename = instance.string();
95  status = fileFormat.loadFile(filename.c_str());
96  FW_RAISE_IF("Unable to read the file: \""+filename+"\"", status.bad());
97  dataset = fileFormat.getDataset();
98 
99  // Decompress data set if compressed
100  dataset->chooseRepresentation(EXS_LittleEndianExplicit, NULL);
101 
102  // Read pixels
103  const U* pixelData;
104  if(sizeof(U) == 1)
105  {
106  const Uint8* pdata;
107  dataset->findAndGetUint8Array(DCM_PixelData, pdata);
108  pixelData = (U*)pdata;
109  }
110  else
111  {
112  const Uint16* pdata;
113  dataset->findAndGetUint16Array(DCM_PixelData, pdata);
114  pixelData = (U*)pdata;
115  }
116 
117  OSLM_WARN_IF("Unable to read pixel data.", !pixelData);
118 
119  for (unsigned int x = 0; x < columns; ++x)
120  {
121  for (unsigned int y = 0; y < rows; ++y)
122  {
123  unsigned int position = y + (x * rows);
124  U value = pixelData[position];
125  tempoBuffer[position*3] = static_cast< T >((redLookup[value]/(double)0xffff)*256);
126  tempoBuffer[position*3+1] = static_cast< T >((greenLookup[value]/(double)0xffff)*256);
127  tempoBuffer[position*3+2] = static_cast< T >((blueLookup[value]/(double)0xffff)*256);
128  }
129  }
130 
131 
132  // Clean up codecs
133  ::fwDcmtkIO::helper::Codec::cleanup();
134 
135  return tempoBuffer;
136  }
