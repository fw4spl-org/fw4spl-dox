349  template < typename IMAGE >
350  void operator()( Param &param )
351  {
352  ::fwComEd::helper::Image imageLock ( param.image );
353  IMAGE * buffer = static_cast < IMAGE* > (imageLock.getBuffer());
354  const ::fwData::Image::SizeType &size = param.image->getSize();
355  ::fwData::Image::SizeType::value_type len = size[0]*size[1]*size[2];
356 
357  T &min = param.min;
358  T &max = param.max;
359 
360  typedef std::numeric_limits<IMAGE> ImgLimits;
361  IMAGE imin = ImgLimits::max();
362  IMAGE imax = (ImgLimits::is_integer || !ImgLimits::is_signed) ? ImgLimits::min() : -ImgLimits::max();
363 
364  IMAGE * bufEnd = buffer + len;
365  IMAGE currentVoxel;
366 
367  for (IMAGE * voxel = buffer; voxel < bufEnd; ++voxel )
368  {
369  currentVoxel = *voxel;
370 
371  if ( currentVoxel < imin )
372  {
373  imin = currentVoxel;
374  }
375  else if (currentVoxel > imax)
376  {
377  imax = currentVoxel;
378  }
379  }
380 
381  typedef std::numeric_limits<T> TLimits;
382  T minT = (TLimits::is_integer || !ImgLimits::is_signed) ? TLimits::min() : -TLimits::max();
383  T maxT = TLimits::max();
384 
385  min = ( imin < minT ) ? minT : static_cast< T > (imin);
386  max = ( imax > maxT ) ? maxT : static_cast< T > (imax);
387 
388  }
