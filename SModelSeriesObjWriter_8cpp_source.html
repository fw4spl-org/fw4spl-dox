57 void SModelSeriesObjWriter::configureWithIHM()
58 {
59  SLM_TRACE_FUNC();
60  static ::boost::filesystem::path _sDefaultPath("");
61 
62  ::fwGui::dialog::LocationDialog dialog;
63  dialog.setTitle("Choose a directory to save meshes");
64  dialog.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
65  dialog.setOption(::fwGui::dialog::ILocationDialog::WRITE);
66  dialog.setType(::fwGui::dialog::ILocationDialog::FOLDER);
67 
68  ::fwData::location::Folder::sptr result;
69 
70  while (result = ::fwData::location::Folder::dynamicCast( dialog.show() ))
71  {
72  if( ::boost::filesystem::is_empty(result->getFolder()) )
73  {
74  break;
75  }
76  // message box
77  ::fwGui::dialog::MessageDialog messageBox;
78  messageBox.setTitle("Overwrite confirmation");
79  messageBox.setMessage("The selected directory is not empty. Write anyway ?");
80  messageBox.setIcon(::fwGui::dialog::IMessageDialog::QUESTION);
81  messageBox.addButton(::fwGui::dialog::IMessageDialog::YES);
82  messageBox.addButton(::fwGui::dialog::IMessageDialog::CANCEL);
83  if( messageBox.show() == ::fwGui::dialog::IMessageDialog::YES)
84  {
85  break;
86  }
87  }
88 
89  if (result)
90  {
91  _sDefaultPath = result->getFolder().parent_path();
92  dialog.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
93  this->setFolder(result->getFolder());
94  }
95  else
96  {
97  this->clearLocations();
98  }
99 }

124 void SModelSeriesObjWriter::updating() throw(::fwTools::Failed)
125 {
126  SLM_TRACE_FUNC();
127 
128  if( this->hasLocationDefined() )
129  {
130  // Retrieve dataStruct associated with this service
131  ::fwMedData::ModelSeries::sptr modelSeries = this->getObject< ::fwMedData::ModelSeries >();
132  SLM_ASSERT("ModelSeries is not instanced", modelSeries);
133 
134  ::fwVtkIO::ModelSeriesObjWriter::sptr writer = ::fwVtkIO::ModelSeriesObjWriter::New();
135  writer->setObject(modelSeries);
136  writer->setFolder(this->getFolder());
137 
138  ::fwGui::Cursor cursor;
139  cursor.setCursor(::fwGui::ICursor::BUSY);
140 
141  try
142  {
143  m_sigJobCreated->emit(writer->getJob());
144  writer->write();
145  }
146  catch (const std::exception & e)
147  {
148  std::stringstream ss;
149  ss << "Warning during saving : " << e.what();
150 
151  ::fwGui::dialog::MessageDialog messageBox;
152  messageBox.setTitle("Warning");
153  messageBox.setMessage( ss.str() );
154  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
155  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
156  messageBox.show();
157  }
158  catch( ... )
159  {
160  std::stringstream ss;
161  ss << "Warning during saving.";
162 
163  ::fwGui::dialog::MessageDialog messageBox;
164  messageBox.setTitle("Warning");
165  messageBox.setMessage( ss.str() );
166  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
167  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
168  messageBox.show();
169  }
170 
171  cursor.setDefaultCursor();
172  }
173 }
