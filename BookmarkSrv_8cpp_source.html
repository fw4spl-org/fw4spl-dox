00039 void BookmarkSrv::configuring()  throw ( ::fwTools::Failed )
00040 {
00041     ::fwRuntime::ConfigurationElement::sptr config = m_configuration->findConfigurationElement("bookmark");
00042     SLM_ASSERT("Problem with configuration for BookmarkSrv type, one element \"bookmark\" must be present", m_configuration->findAllConfigurationElement("bookmark").size() == 1 );
00043 
00044     m_bookmarkName = "";
00045 
00046     if (config->hasAttribute("fromString"))
00047     {
00048         std::string uid = config->getExistingAttributeValue("fromString");
00049         OSLM_ASSERT("Object '" << uid << "' does not exist", ::fwTools::fwID::exist(uid));
00050         ::fwData::String::sptr str = ::fwData::String::dynamicCast(::fwTools::fwID::getObject(uid));
00051         OSLM_ASSERT("Object '" << uid << "' is not a '::fwData::String'", str);
00052         m_bookmarkName = str->value();
00053     }
00054     if (config->hasAttribute("fromString") && config->hasAttribute("name"))
00055     {
00056         m_bookmarkName += "_";
00057     }
00058     if (config->hasAttribute("name"))
00059     {
00060         m_bookmarkName += config->getExistingAttributeValue("name");
00061     }
