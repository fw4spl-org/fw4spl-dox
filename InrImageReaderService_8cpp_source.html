61  ::fwGui::dialog::LocationDialog dialogFile;

91 bool InrImageReaderService::createImage( const ::boost::filesystem::path &inrFileDir,
92  const ::fwData::Image::sptr &_pImg )
93 {
94  SLM_TRACE_FUNC();
95  ::fwItkIO::ImageReader::sptr myLoader = ::fwItkIO::ImageReader::New();
96  bool ok = true;
97 
98  myLoader->setObject(_pImg);
99  myLoader->setFile(inrFileDir);
100 
101  try
102  {
103  ::fwGui::dialog::ProgressDialog progressMeterGUI("Loading Image ");
104  myLoader->addHandler( progressMeterGUI );
105  myLoader->read();
106  }
107  catch (const std::exception & e)
108  {
109  std::stringstream ss;
110  ss << "Warning during loading : " << e.what();
111  ::fwGui::dialog::MessageDialog::showMessageDialog("Warning",
112  ss.str(),
113  ::fwGui::dialog::IMessageDialog::WARNING);
114  ok = false;
115  }
116  catch( ... )
117  {
118  ::fwGui::dialog::MessageDialog::showMessageDialog("Warning",
119  "Warning during loading",
120  ::fwGui::dialog::IMessageDialog::WARNING);
121  ok = false;
122  }
123  return ok;
124 }
