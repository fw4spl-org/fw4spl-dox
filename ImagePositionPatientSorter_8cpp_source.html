59 ImagePositionPatientSorter::DicomSeriesContainerType ImagePositionPatientSorter::apply(
60  const ::fwMedData::DicomSeries::sptr& series, const ::fwLog::Logger::sptr& logger)
61 const throw(::fwDicomIOFilter::exceptions::FilterFailure)
62 {
63  DicomSeriesContainerType result;
64 
65  typedef std::map< double, std::string > SortedFileMapType;
66  SortedFileMapType sortedFiles;
67 
68  DcmFileFormat fileFormat;
69  OFCondition status;
70  DcmDataset* dataset;
71 
72  for(const ::fwMedData::DicomSeries::DicomPathContainerType::value_type& file : series->getLocalDicomPaths())
73  {
74  const std::string& filename = file.second.string();
75  status = fileFormat.loadFile(filename.c_str());
76  FW_RAISE_IF("Unable to read the file: \""+filename+"\"", status.bad());
77  dataset = fileFormat.getDataset();
78 
79  if(!dataset->tagExists(DCM_ImagePositionPatient) || !dataset->tagExists(DCM_ImageOrientationPatient))
80  {
81  const std::string msg =
82  "Unable to split the series using ImagePositionPatient and ImageOrientationPatient. "
83  "Tag(s) missing.";
84  throw ::fwDicomIOFilter::exceptions::FilterFailure(msg);
85  }
86 
87  fwVec3d imagePosition;
88  for(unsigned int i = 0; i < 3; ++i)
89  {
90  dataset->findAndGetFloat64(DCM_ImagePositionPatient, imagePosition[i], i);
91  }
92 
93  fwVec3d imageOrientationU;
94  fwVec3d imageOrientationV;
95  for(unsigned int i = 0; i < 3; ++i)
96  {
97  dataset->findAndGetFloat64(DCM_ImageOrientationPatient, imageOrientationU[i], i);
98  dataset->findAndGetFloat64(DCM_ImageOrientationPatient, imageOrientationV[i], i+3);
99  }
100 
101  //Compute Z direction (cross product)
102  fwVec3d zVector = ::fwMath::cross(imageOrientationU, imageOrientationV);
103 
104  //Compute dot product to get the index
105  double index = ::fwMath::dot(imagePosition, zVector);
106 
107  sortedFiles[index] = filename;
108  }
109 
110  if(sortedFiles.size() != series->getLocalDicomPaths().size())
111  {
112  const std::string msg =
113  "Unable to sort the series using the ImagePositionPatient tag. Some images have the same "
114  "position meaning this series contains multiple volumes. Try to split the volumes using a different "
115  "filter.";
116  throw ::fwDicomIOFilter::exceptions::FilterFailure(msg);
117  }
118 
119  ::fwMedData::DicomSeries::DicomPathContainerType dicomPathContainer;
120  series->setLocalDicomPaths(dicomPathContainer);
121 
122  unsigned int index = 0;
123  for(SortedFileMapType::value_type file : sortedFiles)
124  {
125  series->addDicomPath(index++, file.second);
126  }
127 
128  result.push_back(series);
129 
130  logger->information("The instances have been sorted using the slices positions.");
131 
132  return result;
133 
134 }
