00076 void Grid2D::draw()
00077 {
00078     SLM_TRACE_FUNC();
00079 
00080     SLM_ASSERT("m_xSpacing can not be equal to 0", m_xSpacing != 0);
00081     SLM_ASSERT("m_ySpacing can not be equal to 0", m_ySpacing != 0);
00082 
00083     this->getScene2DRender()->getScene()->removeItem( m_layer );
00084     m_layer = new QGraphicsItemGroup();
00085 
00086     // Calculate the start, end and step on x for the lines
00087     const float xStartVal = getXStartVal();    // Allows to start drawing the grid from 0 with the correct step
00088     const float xEndVal = getXEndVal();      // Allows to start drawing the grid from 0 with the correct step
00089 
00090     // Calculate the start, end and step on y for the lines
00091     const float yStartVal = getYStartVal();    // Allows to start drawing the grid from 0 with the correct step
00092     const float yEndVal = getYEndVal();        // Allows to start drawing the grid from 0 with the correct step
00093 
00094     // Holds the current computed coordinates:
00095     std::pair< double, double > coord1, coord2;
00096 
00097     // Draw the horizontal lines
00098     for ( float yVal = yStartVal ; yVal <= yEndVal ; yVal += m_ySpacing )
00099     {
00100         coord1 = this->mapAdaptorToScene( std::pair< double, double >( xStartVal, yVal) , m_xAxis, m_yAxis );
00101         coord2 = this->mapAdaptorToScene( std::pair< double, double >( xEndVal, yVal) , m_xAxis, m_yAxis );
00102 
00103         QGraphicsLineItem* line = new QGraphicsLineItem(coord1.first, coord1.second, coord2.first, coord2.second);
00104 
00105         // Set the line the pen and push it back in to the lines vector
00106         line->setPen(m_pen);
00107         m_lines.push_back(line);
00108     }
00109 
00110     // Draw the vertical lines
00111     for ( float xVal = xStartVal ; xVal <= xEndVal ; xVal += m_xSpacing )
00112     {
00113         coord1 = this->mapAdaptorToScene(std::pair< double, double >( xVal, yStartVal) , m_xAxis, m_yAxis);
00114         coord2 = this->mapAdaptorToScene(std::pair< double, double >( xVal, yEndVal) , m_xAxis, m_yAxis);
00115 
00116         QGraphicsLineItem* line = new QGraphicsLineItem(coord1.first, coord1.second, coord2.first, coord2.second);
00117 
00118         // Set the line the pen and push it back in to the lines vector
00119         line->setPen(m_pen);
00120         m_lines.push_back(line);
00121     }
00122 
00123     // Add the lines contained in the lines vector to the layer
00124     for ( unsigned int i = 0 ; i < m_lines.size() ; i++)
00125     {
00126         m_layer->addToGroup(m_lines.at(i));
00127     }
00128 
00129     // Set the layer position (according to the related axis) and zValue
00130     m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
00131     m_layer->setZValue(m_zValue);
00132 
00133     // Add the layer containing grid's lines to the scene
00134     this->getScene2DRender()->getScene()->addItem(m_layer);
00135 }
