37 void Grid2D::configuring() throw ( ::fwTools::Failed )
38 {
39  SLM_TRACE_FUNC();
40 
41  SLM_ASSERT("\"config\" tag missing", m_configuration->getName() == "config");
42 
43  this->IAdaptor::configuring();
44 
45  SLM_TRACE("IAdaptor configuring ok");
46 
47 
48  // Set the x/y min/max values
49  m_xMin = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("xMin") );
50  m_xMax = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("xMax") );
51  m_yMin = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("yMin") );
52  m_yMax = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("yMax") );
53 
54 
55  // If the corresponding attributes are present in the config, set the xSpacing, ySpacing between
56  // the lines and color of the lines:
57  if (!m_configuration->getAttributeValue("xSpacing").empty())
58  {
59  m_xSpacing = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("xSpacing") );
60  }
61 
62  if (!m_configuration->getAttributeValue("ySpacing").empty())
63  {
64  m_ySpacing = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("ySpacing") );
65  }
66 
67  if (!m_configuration->getAttributeValue("color").empty())
68  {
69  ::scene2D::data::InitQtPen::setPenColor(m_pen, m_configuration->getAttributeValue("color"), m_opacity);
70  }
71 }

75 void Grid2D::draw()
76 {
77  SLM_TRACE_FUNC();
78 
79  SLM_ASSERT("m_xSpacing can not be equal to 0", m_xSpacing != 0);
80  SLM_ASSERT("m_ySpacing can not be equal to 0", m_ySpacing != 0);
81 
82  this->getScene2DRender()->getScene()->removeItem( m_layer );
83  m_layer = new QGraphicsItemGroup();
84 
85  // Calculate the start, end and step on x for the lines
86  const float xStartVal = getXStartVal(); // Allows to start drawing the grid from 0 with the correct step
87  const float xEndVal = getXEndVal(); // Allows to start drawing the grid from 0 with the correct step
88 
89  // Calculate the start, end and step on y for the lines
90  const float yStartVal = getYStartVal(); // Allows to start drawing the grid from 0 with the correct step
91  const float yEndVal = getYEndVal(); // Allows to start drawing the grid from 0 with the correct step
92 
93  // Holds the current computed coordinates:
94  std::pair< double, double > coord1, coord2;
95 
96  // Draw the horizontal lines
97  for ( float yVal = yStartVal; yVal <= yEndVal; yVal += m_ySpacing )
98  {
99  coord1 = this->mapAdaptorToScene( std::pair< double, double >( xStartVal, yVal), m_xAxis, m_yAxis );
100  coord2 = this->mapAdaptorToScene( std::pair< double, double >( xEndVal, yVal), m_xAxis, m_yAxis );
101 
102  QGraphicsLineItem* line = new QGraphicsLineItem(coord1.first, coord1.second, coord2.first, coord2.second);
103 
104  // Set the line the pen and push it back in to the lines vector
105  line->setPen(m_pen);
106  m_lines.push_back(line);
107  }
108 
109  // Draw the vertical lines
110  for ( float xVal = xStartVal; xVal <= xEndVal; xVal += m_xSpacing )
111  {
112  coord1 = this->mapAdaptorToScene(std::pair< double, double >( xVal, yStartVal), m_xAxis, m_yAxis);
113  coord2 = this->mapAdaptorToScene(std::pair< double, double >( xVal, yEndVal), m_xAxis, m_yAxis);
114 
115  QGraphicsLineItem* line = new QGraphicsLineItem(coord1.first, coord1.second, coord2.first, coord2.second);
116 
117  // Set the line the pen and push it back in to the lines vector
118  line->setPen(m_pen);
119  m_lines.push_back(line);
120  }
121 
122  // Add the lines contained in the lines vector to the layer
123  for ( unsigned int i = 0; i < m_lines.size(); i++)
124  {
125  m_layer->addToGroup(m_lines.at(i));
126  }
127 
128  // Set the layer position (according to the related axis) and zValue
129  m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
130  m_layer->setZValue(m_zValue);
131 
132  // Add the layer containing grid's lines to the scene
133  this->getScene2DRender()->getScene()->addItem(m_layer);
134 }
