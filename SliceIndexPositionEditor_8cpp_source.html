125 void SliceIndexPositionEditor::configuring() throw(fwTools::Failed)
126 {
127  this->initialize();
128 
129  if( this->m_configuration->size() > 0 )
130  {
131  ::fwRuntime::ConfigurationElementContainer::Iterator iter = this->m_configuration->begin();
132  SLM_ASSERT("Only one xml element \"sliceIndex\" is accepted.",
133  this->m_configuration->size() == 1 && (*iter)->getName() == "sliceIndex" );
134  SLM_ASSERT("The xml element \"sliceIndex\" is empty.", !(*iter)->getValue().empty() );
135  std::string orientation = (*iter)->getValue();
136  ::boost::algorithm::trim(orientation);
137  ::boost::algorithm::to_lower(orientation);
138 
139  if(orientation == "axial" )
140  {
141  m_orientation = Z_AXIS;
142  }
143  else if(orientation == "frontal" )
144  {
145  m_orientation = Y_AXIS;
146  }
147  else if(orientation == "sagittal" )
148  {
149  m_orientation = X_AXIS;
150  }
151  else
152  {
153  SLM_FATAL("The value for the xml element \"sliceIndex\" can only be axial, frontal or sagittal.");
154  }
155  }
156 }

215 void SliceIndexPositionEditor::updateSliceIndexFromImg()
216 {
217  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
218 
219  if (::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
220  {
221  // Get Index
222  std::string fieldID = *SLICE_INDEX_FIELDID[m_orientation];
223  OSLM_ASSERT("Field "<<fieldID<<" is missing", image->getField( fieldID ) );
224  unsigned int index = image->getField< ::fwData::Integer >( fieldID )->value();
225 
226  // Update QSlider
227  int max = 0;
228  if(image->getNumberOfDimensions() > m_orientation)
229  {
230  max = static_cast<int>(image->getSize()[m_orientation]-1);
231  }
232  m_sliceSelectorPanel->setSliceRange( 0, max );
233  m_sliceSelectorPanel->setSliceValue( index );
234  }
235 }

250 void SliceIndexPositionEditor::sliceIndexNotification( unsigned int index)
251 {
252  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
253 
254  std::string fieldID = *SLICE_INDEX_FIELDID[m_orientation];
255  OSLM_ASSERT("Field "<<fieldID<<" is missing", image->getField( fieldID ));
256  image->getField< ::fwData::Integer >( fieldID )->value() = index;
257 
258  auto sig = image->signal< ::fwData::Image::SliceIndexModifiedSignalType >(
259  ::fwData::Image::s_SLICE_INDEX_MODIFIED_SIG);
260  ::fwCom::Connection::Blocker block(sig->getConnection(this->slot(s_UPDATE_SLICE_INDEX_SLOT)));
261  sig->asyncEmit(m_axialIndex->value(), m_frontalIndex->value(), m_sagittalIndex->value());
262 }

266 void SliceIndexPositionEditor::sliceTypeNotification( int _type )
267 {
268  Orientation type = static_cast< Orientation >( _type );
269  OSLM_ASSERT("Bad slice type "<<type, type == X_AXIS ||
270  type == Y_AXIS ||
271  type == Z_AXIS );
272 
273  int oldType = static_cast< int > ( m_orientation );
274  // Change slice type
275  m_orientation = type;
276 
277  // Fire the signal
278  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
279  auto sig = image->signal< ::fwData::Image::SliceTypeModifiedSignalType >(
280  ::fwData::Image::s_SLICE_TYPE_MODIFIED_SIG);
281  {
282  ::fwCom::Connection::Blocker block(sig->getConnection(this->slot(s_UPDATE_SLICE_TYPE_SLOT)));
283  sig->asyncEmit(oldType, _type);
284  }
285  this->updateSliceIndexFromImg();
286 }
