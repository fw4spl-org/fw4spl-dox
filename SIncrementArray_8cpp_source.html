62 void SIncrementArray::updating() throw( ::fwTools::Failed )
63 {
64  ::fwData::Array::sptr array = this->getObject< ::fwData::Array >();
65  ::fwData::mt::ObjectWriteLock writeLock(array);
66 
67  SLM_ASSERT("No array.", array);
68  SLM_ASSERT("Array : bad number of dimensions.", array->getNumberOfDimensions() == 1 );
69 
70  const size_t arraySize = array->getSize()[0];
71 
72  ::fwDataTools::helper::Array arrayHelper(array);
73 
74  unsigned int* buffer = static_cast< unsigned int* >( arrayHelper.getBuffer() );
75 
76  // Increment the array values
77  for (size_t i = 0; i < arraySize; i++)
78  {
79  ++buffer[i];
80  }
81 
82  // Notify that the array is modified
83  ::fwData::Object::ModifiedSignalType::sptr sig
84  = array->signal< ::fwData::Object::ModifiedSignalType>( ::fwData::Object::s_MODIFIED_SIG );
85  {
86  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
87  sig->asyncEmit();
88  }
