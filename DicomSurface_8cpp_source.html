120 void DicomSurface::setFromData(fwData::Reconstruction::csptr reconstruction)
121 {
122  // Get mesh
123  ::fwData::Mesh::sptr mesh = reconstruction->getMesh();
124  FW_RAISE_EXCEPTION_IF(::fwGdcmIO::exception::Failed("Can't save this mesh. It must contain only triangles !"),
125  !::fwDataTools::Mesh::hasUniqueCellType(mesh, ::fwData::Mesh::TRIANGLE));
126 
127  ::fwData::Array::sptr pointArray = mesh->getPointsArray();
128  ::fwDataTools::helper::Array pointArrayHelper(pointArray);
129  ::fwData::Mesh::PointValueType* points = pointArrayHelper.begin< ::fwData::Mesh::PointValueType >();
130 
131  ::fwData::Array::sptr cellData = mesh->getCellDataArray();
132  ::fwDataTools::helper::Array cellDataHelper(cellData);
133  ::fwData::Mesh::CellValueType* cells = cellDataHelper.begin< ::fwData::Mesh::CellValueType >();
134 
135  // Initialize members
136  m_pointCoordSize = mesh->getNumberOfPoints();
137  m_pointCoordData = ::boost::shared_ptr< float >(new float[3 * m_pointCoordSize]);
138 
139  m_pointIndexSize = mesh->getNumberOfCells();
140  m_pointIndexList = ::boost::shared_ptr< uint32_t >(new uint32_t[3 * m_pointIndexSize]);
141 
142  m_normalCoordSize = 0;
143 
144  // Traverse Mesh components
145  float* pointCoordData = m_pointCoordData.get();
146  uint32_t* pointIndexList = m_pointIndexList.get();
147 
148  memcpy(pointCoordData, points, pointArray->getSizeInBytes());
149 
150  for (unsigned long id = 0; id < mesh->getCellDataSize(); ++id)
151  {
152  *pointIndexList = static_cast< uint32_t >(*cells);
153  ++cells;
154  ++pointIndexList;
155  }
156 
157  if (mesh->getPointNormalsArray())
158  {
159  ::fwData::Array::sptr normalArray = mesh->getPointNormalsArray();
160  ::fwDataTools::helper::Array normalArrayHelper(normalArray);
161  ::fwData::Mesh::NormalValueType* normals = normalArrayHelper.begin< ::fwData::Mesh::NormalValueType >();
162 
163  m_normalCoordSize = mesh->getNumberOfPoints();
164  m_normalCoordData = ::boost::shared_ptr< float >(new float[3 * m_normalCoordSize]);
165 
166  float* normalCoordData = m_normalCoordData.get();
167 
168  memcpy(normalCoordData, normals, normalArray->getSizeInBytes());
169  }
170 }

192 void DicomSurface::convertToData(fwData::Reconstruction::sptr reconstruction)
193 {
194  // Get mesh
195  ::fwData::Mesh::sptr mesh = ::fwData::Mesh::New();
196 
197  // Initialize members
198  const unsigned long sizePoints = m_pointCoordSize / 3;
199  const unsigned long sizeCells = m_pointIndexSize / 3;
200  mesh->setNumberOfPoints(sizePoints);
201  mesh->setNumberOfCells(sizeCells);
202  mesh->setCellDataSize(sizeCells * 3);
203  mesh->adjustAllocatedMemory();
204 
205  ::fwData::Array::sptr pointArray = mesh->getPointsArray();
206  ::fwData::Array::sptr cellData = mesh->getCellDataArray();
207  ::fwData::Array::sptr cellDataOffsets = mesh->getCellDataOffsetsArray();
208  ::fwData::Array::sptr cellTypes = mesh->getCellTypesArray();
209 
210  ::fwDataTools::helper::Array pointArrayHelper(pointArray);
211  ::fwDataTools::helper::Array cellDataHelper(cellData);
212  ::fwDataTools::helper::Array cellDataOffsetsHelper(cellDataOffsets);
213  ::fwDataTools::helper::Array cellTypesHelper(cellTypes);
214 
215  ::fwData::Mesh::PointValueType* points = pointArrayHelper.begin< ::fwData::Mesh::PointValueType >();
216  ::fwData::Mesh::CellValueType* cells = cellDataHelper.begin< ::fwData::Mesh::CellValueType >();
217 
218  // Traverse DicomSurface components
219  const float* pointCoordData = m_pointCoordData.get();
220  const uint32_t* pointIndexList = m_pointIndexList.get();
221 
222  memcpy(points, pointCoordData, pointArray->getSizeInBytes());
223 
224  for (unsigned long id = 0; id < sizeCells * 3; ++id)
225  {
226  *cells = static_cast< ::fwData::Mesh::CellValueType >(*pointIndexList);
227  ++cells;
228  ++pointIndexList;
229  }
230 
231  std::fill(cellTypesHelper.begin< ::fwData::Mesh::CellTypes >(), cellTypesHelper.end< ::fwData::Mesh::CellTypes >(),
232  static_cast< ::fwData::Mesh::CellTypes >(::fwData::Mesh::TRIANGLE));
233 
234  cell_data_offset_generator cellDataOffsetGenerator;
235 
236  std::generate(cellDataOffsetsHelper.begin< ::fwData::Mesh::CellDataOffsetType >(),
237  cellDataOffsetsHelper.end< ::fwData::Mesh::CellDataOffsetType >(), cellDataOffsetGenerator);
238 
239  if (m_normalCoordSize > 0)
240  {
241  const float* normalCoordData = m_normalCoordData.get();
242  mesh->allocatePointNormals();
243 
244  ::fwData::Array::sptr normalArray = mesh->getPointNormalsArray();
245  ::fwDataTools::helper::Array normalArrayHelper(normalArray);
246  ::fwData::Mesh::NormalValueType* normals = normalArrayHelper.begin< ::fwData::Mesh::NormalValueType >();
247 
248  memcpy(normals, normalCoordData, normalArray->getSizeInBytes());
249  }
250 
251  reconstruction->setMesh(mesh);
252 }
