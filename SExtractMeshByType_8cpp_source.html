47 void SExtractMeshByType::configuring() throw( ::fwTools::Failed )
48 {
49  typedef ::fwRuntime::ConfigurationElement::sptr ConfigType;
50 
51  if(this->isVersion2())
52  {
53  const ConfigType inoutCfg = m_configuration->findConfigurationElement("inout");
54  SLM_ASSERT("At one 'inout' tag is required.", inoutCfg);
55 
56  const std::vector< ConfigType > extractCfg = inoutCfg->find("extract");
57  SLM_ASSERT("At least one 'extract' tag is required.", !extractCfg.empty());
58 
59  bool ok = false;
60 
61  const std::vector< ConfigType > outCfg = m_configuration->find("out");
62  for (const auto& cfg : outCfg)
63  {
64  if(cfg->hasAttribute("group"))
65  {
66  if(cfg->getAttributeValue("group") == "target")
67  {
68  const std::vector< ConfigType > keyCfg = cfg->find("key");
69  OSLM_ASSERT(
70  "You must have as many 'extract' tags as 'out' keys." << extractCfg.size() << " " << keyCfg.size(),
71  extractCfg.size() == keyCfg.size());
72  ok = true;
73  }
74  }
75  }
76  SLM_ASSERT("Missing 'target' output keys", ok);
77 
78  for(ConfigType cfg : extractCfg)
79  {
80  std::string type = cfg->getAttributeValue("type");
81  std::string regex = cfg->getAttributeValue("matching");
82 
83  m_extract.push_back(std::make_pair(type, regex));
84  }
85  }
86  else
87  {
89  std::vector< ConfigType > extractCfg = m_configuration->find("extract");
90  SLM_ASSERT("At least one 'extract' tag is required.", !extractCfg.empty());
91 
92  for(ConfigType cfg : extractCfg)
93  {
94  SLM_ASSERT("Missing attribute 'from'.", cfg->hasAttribute("from"));
95  SLM_ASSERT("Missing attribute 'to'.", cfg->hasAttribute("to"));
96 
97  std::string from = cfg->getAttributeValue("from");
98  std::string to = cfg->getAttributeValue("to");
99  std::string type = cfg->getAttributeValue("type");
100  std::string regex = cfg->getAttributeValue("matching");
101 
102  SLM_ASSERT("'from' attribute must begin with '@'", from.substr(0,1) == "@");
103 
104  m_extractOld[from] = std::make_tuple(to, type, regex);
105  }
106  }
107 }

117 void SExtractMeshByType::updating() throw( ::fwTools::Failed )
118 {
119  if(this->isVersion2())
120  {
121  ::fwMedData::ModelSeries::sptr modelSeries = this->getInOut< ::fwMedData::ModelSeries>("source");
122  OSLM_ASSERT("ModelSeries not found", modelSeries);
123 
124  size_t index = 0;
125  for(const auto& elt : m_extract)
126  {
127  std::string type = elt.first;
128  std::string regex = elt.second;
129 
130  bool found = false;
131  ::fwMedData::ModelSeries::ReconstructionVectorType recs = modelSeries->getReconstructionDB();
132  for(::fwData::Reconstruction::sptr element : recs)
133  {
134  if(element->getCRefStructureType() == type)
135  {
136  ::boost::regex regSurface(regex);
137  ::boost::smatch match;
138  std::string name = element->getOrganName();
139 
140  if(regex.empty() || ::boost::regex_match(name, match, regSurface))
141  {
142  ::fwData::Mesh::sptr obj = element->getMesh();
143 
144  this->setOutput("target", obj, index);
145  found = true;
146  ++index;
147 
148  break;
149  }
150  }
151  }
152  OSLM_ERROR_IF(
153  "Mesh with organ name matching '" << regex << "' and structure type'" << type << "' didn't find",
154  !found);
155  }
156  }
157  else
158  {
159  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
160  ::fwData::mt::ObjectWriteLock lock(composite);
161  ::fwDataTools::helper::Composite compHelper(composite);
162 
163  for(ExtractMapType::value_type elt : m_extractOld)
164  {
165  std::string from = elt.first;
166  std::string to = std::get<0>(elt.second);
167  std::string type = std::get<1>(elt.second);
168  std::string regex = std::get<2>(elt.second);
169 
170  ::fwData::Object::sptr object = ::fwDataCamp::getObject( composite, from );
171  OSLM_ASSERT("Object from '"<< from <<"' not found", object);
172 
173  ::fwMedData::ModelSeries::sptr modelSeries = ::fwMedData::ModelSeries::dynamicCast(object);
174  OSLM_ASSERT("Object from '"<< from <<"' is not a model series", modelSeries);
175 
176  ::fwMedData::ModelSeries::ReconstructionVectorType recs = modelSeries->getReconstructionDB();
177  for(::fwData::Reconstruction::sptr element : recs)
178  {
179  if(element->getCRefStructureType() == type)
180  {
181  ::boost::regex regSurface(regex);
182  ::boost::smatch match;
183  std::string name = element->getOrganName();
184 
185  if(regex.empty() || ::boost::regex_match(name, match, regSurface))
186  {
187  ::fwData::Mesh::sptr obj = element->getMesh();
188  if (composite->find(to) == composite->end())
189  {
190  compHelper.add(to, obj);
191  }
192  else
193  {
194  OSLM_INFO("A key named '"<< to <<
195  "' already exists in the composite. object doesn't added.");
196  }
197 
198  break;
199  }
200  }
201  }
202  OSLM_INFO_IF(
203  "Mesh with organ name matching '" << regex << "' and structure type'" << type << "' didn't find",
204  composite->find(to) == composite->end());
205  }
206  compHelper.notify();
207  }
208 }
