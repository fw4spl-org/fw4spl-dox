79 void DicomInstance::computeSOPClassUID(::fwMedData::Series::sptr series)
80 {
81  // Retrieve series type
82  ::fwMedData::ImageSeries::csptr imageSeries = ::fwMedData::ImageSeries::dynamicConstCast(series);
83  ::fwMedData::ModelSeries::csptr modelSeries = ::fwMedData::ModelSeries::dynamicConstCast(series);
84 
85  // Create result
86  std::string sopClassUID = "";
87 
88  if(imageSeries)
89  {
90  // Retrieve image from series
91  ::fwData::Image::csptr image = imageSeries->getImage();
92 
93  // Compute instance dimension
94  unsigned int dimension =
95  (this->getIsMultiFiles()) ? 2 : static_cast<unsigned int>(image->getNumberOfDimensions());
96 
97  // Define SOP Class UID from the modality
98  ::gdcm::MediaStorage mediaStorage;
99  mediaStorage.GuessFromModality(series->getModality().c_str(), dimension);
100 
101  // Identify the SOPClassUID from a guess
102  if (mediaStorage != ::gdcm::MediaStorage::MS_END && mediaStorage.GetString() != 0)
103  {
104  sopClassUID = mediaStorage.GetString();
105  }
106  // Force SOPClassUID to be CTImageStorage
107  else
108  {
109  sopClassUID = ::gdcm::MediaStorage::GetMSString(::gdcm::MediaStorage::CTImageStorage);
110  }
111  }
112  else if(modelSeries)
113  {
114  sopClassUID = ::gdcm::MediaStorage::GetMSString(::gdcm::MediaStorage::SurfaceSegmentationStorage);
115  }
116 
117  // Update instance information
118  this->setSOPClassUID(sopClassUID);
119 }

123 void DicomInstance::generateSOPInstanceUIDs(::fwMedData::Series::sptr series)
124 {
125  // Retrieve ImageSeries
126  ::fwMedData::ImageSeries::csptr imageSeries = ::fwMedData::ImageSeries::dynamicConstCast(series);
127 
128  // Compute number of instances
129  std::size_t nbInstances = (imageSeries && m_isMultiFiles) ? (imageSeries->getImage()->getSize()[2]) : (1);
130 
131  // Create generator
132  ::gdcm::UIDGenerator uidGenerator;
133 
134  // Generate UIDs
135  for(std::size_t i = 0; i < nbInstances; ++i)
136  {
137  m_SOPInstanceUIDContainer.push_back(uidGenerator.Generate());
138  }
139 }
