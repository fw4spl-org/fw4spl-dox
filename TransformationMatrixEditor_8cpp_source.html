110 void TransformationMatrixEditor::onSliderChange( int angle )
111 {
112  ::fwData::TransformationMatrix3D::sptr tm3D = this->getObject< ::fwData::TransformationMatrix3D >();
113 
114  double angleRad = angle*M_PI/180.0;
115 
116  double cosAngle = cos(angleRad);
117  double sinAngle = sin(angleRad);
118 
119  tm3D->setCoefficient(0,0, cosAngle); tm3D->setCoefficient(0,1, -sinAngle); tm3D->setCoefficient(0,2, 0);
120  tm3D->setCoefficient(0,3, 0);
121  tm3D->setCoefficient(1,0, sinAngle); tm3D->setCoefficient(1,1, cosAngle); tm3D->setCoefficient(1,2, 0);
122  tm3D->setCoefficient(1,2, 0);
123  tm3D->setCoefficient(2,0, 0); tm3D->setCoefficient(2,1, 0); tm3D->setCoefficient(2,2, 1);
124  tm3D->setCoefficient(2,3, 0);
125  tm3D->setCoefficient(3,0, 0); tm3D->setCoefficient(3,1, 0); tm3D->setCoefficient(3,2, 0);
126  tm3D->setCoefficient(3,3, 1);
127 
128  auto sig = tm3D->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
129  {
130  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
131  sig->asyncEmit();
132  }
133 }
