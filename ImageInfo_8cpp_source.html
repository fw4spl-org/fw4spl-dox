110 void ImageInfo::getInteraction(::fwDataTools::PickingInfo info)
111 {
112  SLM_TRACE_FUNC();
113 
114  if (info.m_eventId == ::fwDataTools::PickingInfo::Event::MOUSE_MOVE)
115  {
116  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
117  bool imageIsValid = ::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity( image );
118  m_valueText->setEnabled(imageIsValid);
119  if (imageIsValid)
120  {
121  double* point = info.m_worldPos;
122 
123  ::fwDataTools::helper::Image imageHelper(image);
124 
125  std::string intensity = imageHelper.getPixelAsString(point[0], point[1], point[2] );
126  m_valueText->setText(QString::fromStdString(intensity));
127  }
128  }
129 }
