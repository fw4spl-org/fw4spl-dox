55  this->actionServiceStarting();

86 void SPushObject::configuring() throw( ::fwTools::Failed)
87 {
88  SLM_TRACE_FUNC();
89  this->initialize();
90 
91  std::vector < ConfigurationType > pushConfig = m_configuration->find("push","","",3);
92  std::string src, src_uid, src_key, key;
93  ::boost::regex re("(.*)\\[(.*)\\]");
94  ::boost::smatch match;
95  for( ConfigurationType pushItem : pushConfig)
96  {
97  SLM_ASSERT("<push> tag must have one attribute key.", pushItem->hasAttribute("key"));
98  key = pushItem->getAttributeValue("key");
99  SLM_ASSERT("<push> tag must have one attribute src.", pushItem->hasAttribute("src"));
100  src = pushItem->getAttributeValue("src");
101  if( ::boost::regex_match(src, match, re) )
102  {
103  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
104  src_uid.assign(match[1].first, match[1].second);
105  src_key.assign(match[2].first, match[2].second);
106  m_key2src[key] = std::make_pair(src_uid, src_key);
107  m_srcMap[src_uid].insert(src_key);
108  SLM_TRACE("Add " + src_uid + "[" + src_key + "]");
109  }
110  else
111  {
112  OSLM_FATAL("The regexp \"" << re << "\" does not match \"" << src << "\"");
113  }
114  }
115 }

119 void SPushObject::updating() throw(::fwTools::Failed)
120 {
121  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
122  std::shared_ptr< ::fwComEd::helper::Composite > compositeHelper ( new ::fwComEd::helper::Composite( composite ) );
123 
124  std::string src_uid("");
125  std::string src_key("");
126  std::string key("");
127  for(DestKeyMapType::value_type elt : m_key2src )
128  {
129  key = elt.first;
130  src_uid = elt.second.first;
131  src_key = elt.second.second;
132  SLM_ASSERT( src_uid + " doesn't exist", ::fwTools::fwID::exist(src_uid) );
133  ::fwData::Composite::sptr composite_src = ::fwData::Composite::dynamicCast( ::fwTools::fwID::getObject(
134  src_uid ) );
135  SLM_ASSERT("fwData::Composite dynamicCast failed for "+src_uid, composite_src);
136 
137 
138  ::fwData::Composite::const_iterator iter = composite_src->find(src_key);
139 
140  SLM_WARN_IF("'" + src_key + "' not found in composite '" + src_uid + "'",iter == composite_src->end());
141  if (iter != composite_src->end())
142  {
143  ::fwData::Object::sptr obj = composite_src->getContainer()[src_key];
144 
145  if ( composite->find(key) != composite->end() )
146  {
147  compositeHelper->remove(key);
148  }
149  else
150  {
151  compositeHelper->add(key, obj);
152  }
153  }
154  }
155  // Notification of message
156  compositeHelper->notify();
157 }
