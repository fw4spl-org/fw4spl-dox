97 void DicomSeries::complete(DicomSeriesContainerType& seriesDB, const SPTR(::fwJobs::Observer)& completeSeriesObserver)
98 {
99  ::gdcm::Scanner seriesScanner;
100  seriesScanner.AddTag(s_SeriesInstanceUIDTag);
101  seriesScanner.AddTag(s_ModalityTag);
102  seriesScanner.AddTag(s_SeriesDateTag);
103  seriesScanner.AddTag(s_SeriesTimeTag);
104  seriesScanner.AddTag(s_SeriesDescriptionTag);
105  seriesScanner.AddTag(s_PerformingPhysicianNameTag);
106  seriesScanner.AddTag(s_SOPClassUIDTag);
107 
108  unsigned int progress = 0;
109 
110  for(auto series: seriesDB)
111  {
112  if(series->getLocalDicomPaths().empty())
113  {
114  SLM_ERROR("DicomSeries doesn't not contain any instance.");
115  break;
116  }
117 
118  const std::string filename = series->getLocalDicomPaths().begin()->second.string();
119  FilenameContainerType filenames;
120  filenames.push_back(filename);
121 
122  bool status = seriesScanner.Scan( filenames );
123  FW_RAISE_IF("Unable to read the files.", !status);
124 
125  //Modality
126  std::string modality = getStringValue( seriesScanner, filename, s_ModalityTag );
127  series->setModality(modality);
128 
129  //Date
130  std::string seriesDate = getStringValue( seriesScanner, filename, s_SeriesDateTag );
131  series->setDate(seriesDate);
132 
133  //Time
134  std::string seriesTime = getStringValue( seriesScanner, filename, s_SeriesTimeTag );
135  series->setTime(seriesTime);
136 
137  //Description
138  std::string seriesDescription = getStringValue( seriesScanner, filename, s_SeriesDescriptionTag );
139  series->setDescription(seriesDescription);
140 
141  //Performing Physicians Name
142  std::string performingPhysicianNamesStr = getStringValue( seriesScanner, filename,
143  s_PerformingPhysicianNameTag );
144 
145  if(!performingPhysicianNamesStr.empty())
146  {
147  ::fwMedData::DicomValuesType performingPhysicianNames;
148  ::boost::split( performingPhysicianNames, performingPhysicianNamesStr, ::boost::is_any_of("\\"));
149  series->setPerformingPhysiciansName(performingPhysicianNames);
150  }
151 
152  // Add the SOPClassUID to the series
153  std::string sopClassUID = getStringValue( seriesScanner, filename, s_SOPClassUIDTag );
154  ::fwMedData::DicomSeries::SOPClassUIDContainerType sopClassUIDContainer = series->getSOPClassUIDs();
155  sopClassUIDContainer.insert(sopClassUID);
156  series->setSOPClassUIDs(sopClassUIDContainer);
157  }
158 
159  this->fillSeries(seriesDB, completeSeriesObserver);
160 
161 }

187 DicomSeries::DicomSeriesContainerType DicomSeries::splitFiles(FilenameContainerType& filenames,
188  const ::fwJobs::Observer::sptr& readerObserver)
189 {
190  ::gdcm::Scanner seriesScanner;
191  seriesScanner.AddTag(s_SeriesInstanceUIDTag);
192  seriesScanner.AddTag(s_ModalityTag);
193  seriesScanner.AddTag(s_SeriesDateTag);
194  seriesScanner.AddTag(s_SeriesTimeTag);
195  seriesScanner.AddTag(s_SeriesDescriptionTag);
196  seriesScanner.AddTag(s_PerformingPhysicianNameTag);
197  seriesScanner.AddTag(s_SOPClassUIDTag);
198 
199  readerObserver->setTotalWorkUnits(filenames.size());
200  readerObserver->doneWork(0);
201 
202  bool status = seriesScanner.Scan( filenames );
203  FW_RAISE_IF("Unable to read the files.", !status);
204 
205  ::gdcm::Directory::FilenamesType keys = seriesScanner.GetKeys();
206  ::gdcm::Directory::FilenamesType::const_iterator it;
207 
208  unsigned int progress = 0;
209 
210  DicomSeriesContainerType seriesDB;
211 
212  //Loop through every files available in the scanner
213  for(it = keys.begin(); it != keys.end() && !readerObserver->cancelRequested(); ++it)
214  {
215  const std::string filename = it->c_str();
216  OSLM_ASSERT("The file \"" << filename << "\" is not a key of the gdcm scanner",
217  seriesScanner.IsKey(filename.c_str()));
218 
219  // Create Series
220  this->createSeries(seriesDB, seriesScanner, filename);
221 
222  readerObserver->doneWork(++progress);
223  }
224 
225  return seriesDB;
226 }

230 void DicomSeries::fillSeries(DicomSeriesContainerType& seriesDB,
231  const ::fwJobs::Observer::sptr& completeSeriesObserver)
232 {
233  m_patientMap.clear();
234  m_studyMap.clear();
235  m_equipmentMap.clear();
236 
237  ::gdcm::Scanner attributeScanner;
238  attributeScanner.AddTag(s_PatientIDTag);
239  attributeScanner.AddTag(s_PatientNameTag);
240  attributeScanner.AddTag(s_PatientBirthDateTag);
241  attributeScanner.AddTag(s_PatientSexTag);
242  attributeScanner.AddTag(s_StudyInstanceUIDTag);
243  attributeScanner.AddTag(s_StudyDateTag);
244  attributeScanner.AddTag(s_StudyTimeTag);
245  attributeScanner.AddTag(s_ReferringPhysicianNameTag);
246  attributeScanner.AddTag(s_StudyDescriptionTag);
247  attributeScanner.AddTag(s_PatientAgeTag);
248  attributeScanner.AddTag(s_InstitutionNameTag);
249  attributeScanner.AddTag(s_SeriesInstanceUIDTag);
250 
251  completeSeriesObserver->setTotalWorkUnits(seriesDB.size());
252  std::uint64_t progress = 0;
253 
254  // Fill series
255  for(::fwMedData::DicomSeries::sptr series: seriesDB)
256  {
257  // Compute number of instances
258  auto size = series->getLocalDicomPaths().size();
259  series->setNumberOfInstances(size);
260 
261  if(!size)
262  {
263  SLM_ERROR("The DicomSeries doesn't contain any instance.");
264  break;
265  }
266 
267 
268  // Get first instance filename
269  const std::string filename = series->getLocalDicomPaths().begin()->second.string();
270 
271  // Load first instance
272  std::vector< std::string > firstInstanceContainer;
273  firstInstanceContainer.push_back(filename);
274  bool status = attributeScanner.Scan( firstInstanceContainer );
275  FW_RAISE_IF("Unable to read the files.", !status);
276  OSLM_ASSERT("The file \"" << filename << "\" is not a key of the gdcm scanner",
277  attributeScanner.IsKey(filename.c_str()));
278 
279  // Create data objects from first instance
280  ::fwMedData::Patient::sptr patient = this->createPatient(attributeScanner, filename);
281  ::fwMedData::Study::sptr study = this->createStudy(attributeScanner, filename);
282  ::fwMedData::Equipment::sptr equipment = this->createEquipment(attributeScanner, filename);
283 
284  // Fill series
285  series->setPatient(patient);
286  series->setStudy(study);
287  series->setEquipment(equipment);
288 
289  completeSeriesObserver->doneWork(++progress);
290 
291  if(completeSeriesObserver->cancelRequested())
292  {
293  break;
294  }
295  }
296 }

300 void DicomSeries::createSeries(DicomSeriesContainerType& seriesDB, const ::gdcm::Scanner& scanner,
301  const std::string& filename)
302 {
303  ::fwMedData::DicomSeries::sptr series = ::fwMedData::DicomSeries::sptr();
304 
305  // Get Series Instance UID
306  std::string seriesInstanceUID = getStringValue( scanner, filename, s_SeriesInstanceUIDTag );
307 
308  // Check if the series already exists
309  for(::fwMedData::DicomSeries::sptr dicomSeries: seriesDB)
310  {
311  if(dicomSeries->getInstanceUID() == seriesInstanceUID)
312  {
313  series = dicomSeries;
314  break;
315  }
316  }
317 
318  // If the series doesn't exist we create it
319  if(!series)
320  {
321  series = ::fwMedData::DicomSeries::New();
322  series->setDicomAvailability(::fwMedData::DicomSeries::PATHS);
323 
324  seriesDB.push_back(series);
325 
326  //Instance UID
327  series->setInstanceUID(seriesInstanceUID);
328 
329  //Modality
330  std::string modality = getStringValue( scanner, filename, s_ModalityTag );
331  series->setModality(modality);
332 
333  //Date
334  std::string seriesDate = getStringValue( scanner, filename, s_SeriesDateTag );
335  series->setDate(seriesDate);
336 
337  //Time
338  std::string seriesTime = getStringValue( scanner, filename, s_SeriesTimeTag );
339  series->setTime(seriesTime);
340 
341  //Description
342  std::string seriesDescription = getStringValue( scanner, filename, s_SeriesDescriptionTag );
343  series->setDescription(seriesDescription);
344 
345  //Performing Physicians Name
346  std::string performingPhysicianNamesStr = getStringValue( scanner, filename,
347  s_PerformingPhysicianNameTag );
348 
349  if(!performingPhysicianNamesStr.empty())
350  {
351  ::fwMedData::DicomValuesType performingPhysicianNames;
352  ::boost::split( performingPhysicianNames, performingPhysicianNamesStr, ::boost::is_any_of("\\"));
353  series->setPerformingPhysiciansName(performingPhysicianNames);
354  }
355  }
356 
357  // Add the SOPClassUID to the series
358  std::string sopClassUID = getStringValue( scanner, filename, s_SOPClassUIDTag );
359  ::fwMedData::DicomSeries::SOPClassUIDContainerType sopClassUIDContainer = series->getSOPClassUIDs();
360  sopClassUIDContainer.insert(sopClassUID);
361  series->setSOPClassUIDs(sopClassUIDContainer);
362 
363  // Add the instance to the series
