00094             const std::string& extension = ext->getValue();

00096             SLM_ASSERT("Extension must begin with '.'", extension[0] == '.');
00097 
00098             m_customExts[extension] = backend;
00099             m_allowedExtLabels[extension] = ext->getAttributeValue("label");
00100         }
00101     }
00102 
00103     ConfigurationElementContainer extensionsList = m_configuration->find("extensions");
00104     SLM_ASSERT("The <extensions> element can be set at most once.", extensionsList.size() <= 1);
00105 
00106     if(extensionsList.size() == 1)
00107     {
00108         m_allowedExts.clear();
00109 
00110         ConfigurationElementContainer extensions = extensionsList.at(0)->find("extension");
00111         BOOST_FOREACH(ConfigurationElement extension, extensions)
00112         {
00113             const std::string& ext = extension->getValue();
00114 
00115             // The extension must be found either in custom extensions list or in known extensions
00116             FileExtension2NameType::const_iterator itKnown = SReader::s_EXTENSIONS.find(ext);
00117             FileExtension2NameType::const_iterator itCustom = m_customExts.find(ext);
00118 
00119             const bool extIsKnown = (itKnown != SReader::s_EXTENSIONS.end() || itCustom != m_customExts.end());
00120             OSLM_ASSERT("Extension '" << ext << "' is not allowed in configuration", extIsKnown);
00121 
00122             if(extIsKnown)
00123             {
00124                 m_allowedExts.insert(m_allowedExts.end(), ext);
00125                 m_allowedExtLabels[ext] = extension->getAttributeValue("label");
00126             }
00127         }
00128     }
00129     else
00130     {
00131         m_allowedExts.clear();
00132 
00133         BOOST_FOREACH(FileExtension2NameType::value_type ext, m_customExts)
00134         {
00135             m_allowedExts.insert(m_allowedExts.end(), ext.first);
00136         }
00137 
00138         BOOST_FOREACH(SReader::FileExtension2NameType::value_type ext, SReader::s_EXTENSIONS)
00139         {
00140             m_allowedExts.insert(m_allowedExts.end(), ext.first);
00141             m_allowedExtLabels[ext.first] = ext.second;
00142         }
00143     }

00147     if (patcher.size() == 1)
