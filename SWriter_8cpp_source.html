84 void SWriter::configuring() throw(::fwTools::Failed)
85 {
86  ::io::IWriter::configuring();
87 
88  typedef SPTR (::fwRuntime::ConfigurationElement) ConfigurationElement;
89  typedef std::vector < ConfigurationElement > ConfigurationElementContainer;
90 
91  m_customExts.clear();
92  m_allowedExtLabels.clear();
93 
94  ConfigurationElementContainer customExtsList = m_configuration->find("archive");
95  for(ConfigurationElement archive : customExtsList)
96  {
97  const std::string& backend = archive->getAttributeValue("backend");
98  SLM_ASSERT("No backend attribute given in archive tag", backend != "");
99  SLM_ASSERT("Unsupported backend '" + backend + "'",
100  SReader::s_EXTENSIONS.find("." + backend) != SReader::s_EXTENSIONS.end());
101 
102  ConfigurationElementContainer exts = archive->find("extension");
103  for(ConfigurationElement ext : exts)
104  {
105  const std::string& extension = ext->getValue();
106  SLM_ASSERT("No extension given for backend '" + backend + "'", !extension.empty());
107  SLM_ASSERT("Extension must begin with '.'", extension[0] == '.');
108 
109  m_customExts[extension] = backend;
110  m_allowedExtLabels[extension] = ext->getAttributeValue("label");
111  }
112  }
113 
114  ConfigurationElementContainer extensionsList = m_configuration->find("extensions");
115  SLM_ASSERT("The <extensions> element can be set at most once.", extensionsList.size() <= 1);
116 
117  if(extensionsList.size() == 1)
118  {
119  m_allowedExts.clear();
120 
121  ConfigurationElementContainer extensions = extensionsList.at(0)->find("extension");
122  for(ConfigurationElement extension : extensions)
123  {
124  const std::string& ext = extension->getValue();
125 
126  // The extension must be found either in custom extensions list or in known extensions
127  FileExtension2NameType::const_iterator itKnown = SReader::s_EXTENSIONS.find(ext);
128  FileExtension2NameType::const_iterator itCustom = m_customExts.find(ext);
129 
130  const bool extIsKnown = (itKnown != SReader::s_EXTENSIONS.end() || itCustom != m_customExts.end());
131  OSLM_ASSERT("Extension '" << ext << "' is not allowed in configuration", extIsKnown);
132 
133  if(extIsKnown)
134  {
135  m_allowedExts.insert(m_allowedExts.end(), ext);
136  m_allowedExtLabels[ext] = extension->getAttributeValue("label");
137  }
138  }
139  }
140  else
141  {
142  m_allowedExts.clear();
143 
144  for(FileExtension2NameType::value_type ext : m_customExts)
145  {
146  m_allowedExts.insert(m_allowedExts.end(), ext.first);
147  }
148 
149  for(SReader::FileExtension2NameType::value_type ext : SReader::s_EXTENSIONS)
150  {
151  m_allowedExts.insert(m_allowedExts.end(), ext.first);
152  m_allowedExtLabels[ext.first] = ext.second;
153  }
154  }
155 
156  ConfigurationElementContainer patcher = m_configuration->find("patcher");
157  SLM_ASSERT("The <patcher> element can be set at most once.", patcher.size() <= 1 );
158  if (patcher.size() == 1)
159  {
160  m_context = patcher.at(0)->getExistingAttributeValue("context");
161  m_version = patcher.at(0)->getExistingAttributeValue("version");
162  m_exportedVersion = m_version;
163  m_useAtomsPatcher = true;
164  }
165 }

169 bool SWriter::versionSelection()
170 {
171  using namespace boost::assign;
172 
173  ::fwAtomsPatch::VersionsGraph::sptr vg = ::fwAtomsPatch::VersionsManager::getDefault()->getGraph(m_context);
174 
175  // have some information about this format
176  if ( vg )
177  {
178  std::vector< std::string > versions = vg->getConnectedVersions(m_version);
179  if ( versions.size() == 0 )
180  {
181  m_exportedVersion = m_version;
182  return true;
183  }
184  else
185  {
186  versions.push_back(m_version);
187  ::fwGui::dialog::SelectorDialog dialogVersion;
188 
189  dialogVersion.setTitle("Archive version");
190  dialogVersion.setMessage("Select an archive version");
191 
192  dialogVersion.setSelections(versions);
193  std::string result = dialogVersion.show();
194  if ( !result.empty() )
195  {
196  m_exportedVersion = result;
197  }
198  return !result.empty();
199  }
200  }
201  else
202  {
203  m_exportedVersion = m_version;
204  return true;
205  }

210 void SWriter::updating() throw(::fwTools::Failed)
211 {
212  if(!this->hasLocationDefined())
213  {
214  return;
215  }
216 
217  ::fwData::Object::sptr obj = this->getObject< ::fwData::Object >();
218 
219  ::fwGui::Cursor cursor;
220  cursor.setCursor(::fwGui::ICursor::BUSY);
221 
222  const ::boost::filesystem::path& requestedFilePath = this->getFile();
223  ::boost::filesystem::path filePath = requestedFilePath;
224 
225  if( ::boost::filesystem::exists( requestedFilePath ) )
226  {
227  FW_RAISE_IF( "can't write to : " << requestedFilePath << ", it is a directory.",
228  ::boost::filesystem::is_directory(requestedFilePath)
229  );
230 
231  filePath.replace_extension("%%%%%%" + filePath.extension().string() );
232  filePath = ::boost::filesystem::unique_path(filePath);
233  }
234 
235 
236  const ::boost::filesystem::path folderPath = filePath.parent_path();
237  const ::boost::filesystem::path filename = filePath.filename();
238  std::string extension = ::boost::filesystem::extension(filePath);
239 
240  FW_RAISE_IF("The file extension '" << extension << "' is not managed",
241  m_allowedExts.find(extension) == m_allowedExts.end());
242 
243  if(m_customExts.find(extension) != m_customExts.end())
244  {
245  extension = "." + m_customExts[extension];
246  }
247  FW_RAISE_IF( "Extension is empty", extension.empty() );
248 
249  // Mutex data lock
250  ::fwDataCamp::visitor::RecursiveLock recursiveLock (obj);
251 
252  ::fwAtoms::Object::sptr atom;
253  const unsigned int progressBarOffset = 10;
254 
255  // Convert data to atom : job 1
256  ::fwJobs::Job::sptr convertJob = ::fwJobs::Job::New("Writing " + extension + " file",
257  [ =, &atom](::fwJobs::Job& runningJob)
258  {
259  runningJob.doneWork(progressBarOffset);
260 
261  atom = ::fwAtomConversion::convert(obj);
262  runningJob.done();
263  }, m_associatedWorker );
264 
265  // Path atom : job 2
266  ::fwJobs::Job::sptr patchingJob = ::fwJobs::Job::New("Writing " + extension + " file",
267  [ =, &atom](::fwJobs::Job& runningJob)
268  {
269 
270  atom->setMetaInfo("context", m_context);
271  atom->setMetaInfo("version_name", m_version);
272 
273  if(runningJob.cancelRequested())
274  {
275  return;
276  }
277 
278  runningJob.doneWork(progressBarOffset);
279 
280  if( m_useAtomsPatcher )
281  {
282  ::fwAtomsPatch::PatchingManager globalPatcher( atom );
283  atom = globalPatcher.transformTo( m_exportedVersion );
284  }
285 
286  runningJob.done();
287  },
288  m_associatedWorker
289  );
290 
291  // Writing file : job 3
292  ::fwJobs::Job::sptr writeJob = ::fwJobs::Job::New("Writing " + extension + " file",
293  [ =, &atom](::fwJobs::Job& runningJob)
294  {
295 
296  runningJob.doneWork(progressBarOffset);
297 
298  // Write atom
299  ::fwZip::IWriteArchive::sptr writeArchive;
300  ::fwAtomsBoostIO::FormatType format;
301  ::boost::filesystem::path archiveRootName;
302  if ( extension == ".json" )
303  {
304  writeArchive = ::fwZip::WriteDirArchive::New(folderPath.string());
305  archiveRootName = filename;
306  format = ::fwAtomsBoostIO::JSON;
307  }
308  else if ( extension == ".jsonz" )
309  {
310  if ( ::boost::filesystem::exists( filePath ) )
311  {
312  ::boost::filesystem::remove( filePath );
313  }
314  writeArchive = ::fwZip::WriteZipArchive::New(filePath.string());
315  archiveRootName = "root.json";
316  format = ::fwAtomsBoostIO::JSON;
317  }
318  else if ( extension == ".xml" )
319  {
320  writeArchive = ::fwZip::WriteDirArchive::New(folderPath.string());
321  archiveRootName = filename;
322  format = ::fwAtomsBoostIO::XML;
323  }
324  else if ( extension == ".xmlz" )
325  {
326  if ( ::boost::filesystem::exists( filePath ) )
327  {
328  ::boost::filesystem::remove( filePath );
329  }
330  writeArchive = ::fwZip::WriteZipArchive::New(filePath.string());
331  archiveRootName = "root.xml";
332  format = ::fwAtomsBoostIO::XML;
333  }
334  else
335  {
336  FW_RAISE( "This file extension '" << extension << "' is not managed" );
337  }
338 
339  ::fwAtomsBoostIO::Writer(atom).write( writeArchive, archiveRootName, format );
340  writeArchive.reset();
341 
342  if (filePath != requestedFilePath)
343  {
344  ::boost::filesystem::rename(filePath, requestedFilePath);
345  }
346 
347  runningJob.done();
348  }, m_associatedWorker );
349 
350  ::fwJobs::Aggregator::sptr jobs = ::fwJobs::Aggregator::New(extension + " writer");
351  jobs->add(convertJob);
352  jobs->add(patchingJob);
353  jobs->add(writeJob);
354  jobs->setCancelable(false);
355 
356  m_sigJobCreated->emit(jobs);
357 
358  try
359  {
360  jobs->run().get();
361  }
362  catch( std::exception & e )
363  {
364  OSLM_ERROR( e.what() );
365  ::fwGui::dialog::MessageDialog::showMessageDialog("Medical data writer failed",
366  e.what(),
367  ::fwGui::dialog::IMessageDialog::CRITICAL);
368  }
369  catch( ... )
370  {
371  ::fwGui::dialog::MessageDialog::showMessageDialog("Medical data writer failed",
372  "Writing process aborted",
373  ::fwGui::dialog::IMessageDialog::CRITICAL);
374  }
375 
376  cursor.setDefaultCursor();
377 }

388 void SWriter::configureWithIHM()
389 {
390  static ::boost::filesystem::path _sDefaultPath;
391 
392  if( !m_useAtomsPatcher || versionSelection() )
393  {
394  ::fwGui::dialog::LocationDialog dialogFile;
395  dialogFile.setTitle("Enter file name");
396  dialogFile.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
397  dialogFile.setOption(::fwGui::dialog::ILocationDialog::WRITE);
398  dialogFile.setType(::fwGui::dialog::LocationDialog::SINGLE_FILE);
399 
400  dialogFile.addFilter("Medical data", "*" + ::boost::algorithm::join(m_allowedExts, " *"));
401 
402  for(const std::string& ext : m_allowedExts)
403  {
404  dialogFile.addFilter(m_allowedExtLabels[ext], "*" + ext);
405  }
406 
407  ::fwData::location::SingleFile::sptr result
408  = ::fwData::location::SingleFile::dynamicCast( dialogFile.show() );
409 
410  if (result)
411  {
412  _sDefaultPath = result->getPath();
413  this->setFile( _sDefaultPath );
414  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath.parent_path()) );
415  }
416  else
417  {
418  this->clearLocations();
419  }
420 
421  }
