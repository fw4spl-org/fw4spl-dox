118 ::fwMedData::SeriesDB::sptr SSeriesDBReader::createSeriesDB(const ::boost::filesystem::path& dicomDir)
119 {
120  SLM_TRACE_FUNC();
121  ::vtkGdcmIO::SeriesDBReader::sptr reader = ::vtkGdcmIO::SeriesDBReader::New();
122  ::fwMedData::SeriesDB::sptr dummy = ::fwMedData::SeriesDB::New();
123  reader->setObject(dummy);
124  reader->setFolder(dicomDir);
125 
126  m_sigJobCreated->emit(reader->getJob());
127 
128  try
129  {
130  reader->read();
131  }
132  catch (const std::exception & e)
133  {
134  std::stringstream ss;
135  ss << "Warning during loading : " << e.what();
136  ::fwGui::dialog::MessageDialog::showMessageDialog(
137  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
138  }
139  catch( ... )
140  {
141  ::fwGui::dialog::MessageDialog::showMessageDialog(
142  "Warning", "Warning during loading", ::fwGui::dialog::IMessageDialog::WARNING);
143  }
144 
145  return reader->getConcreteObject();
146 }

150 void SSeriesDBReader::updating() throw(::fwTools::Failed)
151 {
152  SLM_TRACE_FUNC();
153  if( this->hasLocationDefined() )
154  {
155  ::fwMedData::SeriesDB::sptr seriesDB = createSeriesDB( this->getFolder() );
156  if( seriesDB->size() > 0 )
157  {
158  if(this->isStopped()) // FIXME service might be stopped while updating in a worker
159  {
160  return;
161  }
162 
163  // Retrieve dataStruct associated with this service
164  ::fwMedData::SeriesDB::sptr associatedSeriesDB = this->getObject< ::fwMedData::SeriesDB >();
165  SLM_ASSERT("associated SeriesDB not instanced", associatedSeriesDB);
166  associatedSeriesDB->shallowCopy( seriesDB );
167 
168  ::fwGui::Cursor cursor;
169  cursor.setCursor(::fwGui::ICursor::BUSY);
170  this->notificationOfDBUpdate();
171  cursor.setDefaultCursor();
172  }
173  else
174  {
175  ::fwGui::dialog::MessageDialog::showMessageDialog(
176  "Image Reader","This file can not be read. Retry with another file reader.",
177  ::fwGui::dialog::IMessageDialog::WARNING);
178  }
179  }
180 }

184 void SSeriesDBReader::notificationOfDBUpdate()
185 {
186  SLM_TRACE_FUNC();
187  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
188  SLM_ASSERT("Unable to get seriesDB", seriesDB);
189 
190  ::fwMedData::SeriesDB::ContainerType addedSeries;
191  for( ::fwMedData::Series::sptr s : seriesDB->getContainer() )
192  {
193  addedSeries.push_back(s);
194  }
195 
196  auto sig = seriesDB->signal< ::fwMedData::SeriesDB::AddedSeriesSignalType >(
197  ::fwMedData::SeriesDB::s_ADDED_SERIES_SIG);
198  sig->asyncEmit(addedSeries);
199 }
