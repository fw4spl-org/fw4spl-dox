00156     SLM_TRACE_FUNC();
00157     if( this->hasLocationDefined() )
00158     {
00159         ::fwMedData::SeriesDB::sptr seriesDB = createSeriesDB( this->getFolder() );

00161         {
00162             if(this->isStopped()) // FIXME service might be stopped while updating in a worker
00163             {
00164                 return;

00166

00169             SLM_ASSERT("associated SeriesDB not instanced", associatedSeriesDB);
00170             associatedSeriesDB->shallowCopy( seriesDB ) ;
00171 
00172             ::fwGui::Cursor cursor;

00179             ::fwGui::dialog::MessageDialog::showMessageDialog(
