95 inline void TransformationMatrix3D::setTF3DFromMatrix(::fwData::TransformationMatrix3D::sptr& _trf,
96  const ::glm::dmat4x4& _input)
97 {
98  // TransformationMatrix3D is stored row-major
99  // glm matrices are stored column-major
100  auto& coefs = _trf->getRefCoefficients();
101  for (size_t i = 0; i < 4; ++i)
102  {
103  const size_t rowDst = i * 4;
104  const ::glm::length_t rowSrc = static_cast< ::glm::length_t>(i);
105  for (size_t j = 0; j < 4; ++j)
106  {
107  const ::glm::length_t colSrc = static_cast< ::glm::length_t>(j);
108  coefs[rowDst + j] = _input[colSrc][rowSrc];
109  }
110  }
