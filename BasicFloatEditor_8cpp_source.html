127 void BasicFloatEditor::onModifyValue(QString value)
128 {
129  SLM_TRACE_FUNC();
130  ::fwData::Float::sptr floatObj = this->getObject< ::fwData::Float >();
131  ::fwData::Float::sptr oldValue;
132  oldValue = ::fwData::Object::copy(floatObj);
133 
134  std::string strValue = value.toStdString();
135 
136  if (strValue.empty() || strValue == "-")
137  {
138  floatObj->value() = 0;
139  }
140  else if (value.endsWith( QString(",") ))
141  {
142  m_valueCtrl->setText( m_valueCtrl->text().replace(QChar(','), QChar('.')) );
143  }
144  else
145  {
146  int pos = 0;
147  QString str = m_valueCtrl->text();
148  bool isValid = (m_valueCtrl->validator()->validate(str, pos) == QValidator::Acceptable);
149 
150  if (isValid)
151  {
152  floatObj->value() = ::boost::lexical_cast< ::fwData::Float::ValueType >(strValue);
153  }
154  else
155  {
156  this->updating();
157  }
158  }
159 
160  if ( *oldValue != *floatObj )
161  {
162  OSLM_TRACE(floatObj->getID() << " new value : " << *floatObj);
163 
164  auto sig = floatObj->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
165  {
166  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
167  sig->asyncEmit();
168  }
169  }
170 }
