124 void ObjectService::internalRegisterService( ::fwData::Object::sptr object, ::fwServices::IService::sptr service)
125 {
126  OSLM_ASSERT("This service "<< service->getClassname()
127  << " is not valid for the object " << object->getClassname(),
128  ServiceFactory::getDefault()->checkServiceValidity(object->getClassname(), service->getClassname())
129  );
130 
131  OSLM_ASSERT("Service "<< service->getID()<<" already registered",
132  m_container.right.find(service) == m_container.right.end());
133  OSLM_ASSERT("Service "<< service->getID()<<" has already a valid associated object",
134  service->m_associatedObject.expired() || object != service->m_associatedObject.lock());
135 
136  service->m_associatedObject = object;
137  m_container.insert( ServiceContainerType::value_type( object->getOSRKey()->getLogicStamp(), service ) );
138 }

142 void ObjectService::swapService( ::fwData::Object::sptr objDst, ::fwServices::IService::sptr service )
143 {
144  ::fwCore::mt::WriteLock lock(m_containerMutex);
145  OSLM_ASSERT("Object "<< service->getObject()->getID()<<" is not registered in OSR",

148  OSLM_ASSERT("Service "<< service->getID()<<" is not registered in OSR",
149  m_container.right.find(service) != m_container.right.end());
150 
151  m_container.right.erase(service);

179 ObjectService::ObjectVectorType ObjectService::getObjects() const
180 {
181  ObjectVectorType objects;
182  ::fwCore::mt::ReadLock lock(m_containerMutex);
183  const ServiceContainerType::right_map & right = m_container.right;
184  for( const ServiceContainerType::right_map::value_type &elt : right)
185  {
186  if ( std::find(objects.begin(), objects.end(), elt.first->getObject()) == objects.end() )
187  {
188  objects.push_back( elt.first->getObject() );
189  }
190  }
191  SLM_WARN_IF( "No object registered for the requested type of service", objects.empty() );
192  return objects;
193 }

198 {
199  std::stringstream info;
200  ::fwCore::LogicStamp::LogicStampType previousKey = -1;
201  ::fwCore::mt::ReadLock lock(m_containerMutex);
202  for( const ServiceContainerType::left_map::value_type &objSrvMap : m_container.left)
203  {
204  // TODO FIXME getObject() fail if there are expired object in OSR
205  ::fwCore::LogicStamp::LogicStampType key = objSrvMap.first;

207  if ( previousKey != key )
208  {
209  info << "Object ( key = "<<key<<" ) has "
210  << m_container.left.count(key) <<" services." << std::endl;
211  previousKey = key;
212  }
213  ::fwServices::IService::sptr service = objSrvMap.second;
214  info << " srv : uid = "<< service->getID() <<" , classname = "<< service->getClassname()

216  }

260 ObjectService::ServiceVectorType ObjectService::getServices( ::fwData::Object::sptr obj ) const
261 {
262  ServiceVectorType services;
263  ::fwCore::mt::ReadLock lock(m_containerMutex);
264  if(m_container.left.find(obj->getOSRKey()->getLogicStamp()) != m_container.left.end())
265  {
266  ServiceContainerType::left_map::const_iterator iter;
267  ::fwCore::LogicStamp::LogicStampType key = obj->getOSRKey()->getLogicStamp();
268  ServiceContainerType::left_map::const_iterator firstElement = m_container.left.find(key);
269  ServiceContainerType::left_map::const_iterator lastElement = m_container.left.upper_bound(key);
270  for (iter = firstElement; iter != lastElement; ++iter)
271  {
272  services.push_back( iter->second );
273  }
274  }
275  return services;
276 }

281 {
282  bool hasServices = false;
283  ::fwCore::mt::ReadLock lock(m_containerMutex);
284  if( m_container.left.find(obj->getOSRKey()->getLogicStamp()) != m_container.left.end())
285  {
286  ServiceContainerType::left_map::const_iterator iter;
287  ::fwCore::LogicStamp::LogicStampType key = obj->getOSRKey()->getLogicStamp();
288  ServiceContainerType::left_map::const_iterator firstElement = m_container.left.find(key);
289  ServiceContainerType::left_map::const_iterator lastElement = m_container.left.upper_bound(key);
290  for (iter = firstElement; iter != lastElement; ++iter)
291  {
292  if( iter->second->isA(srvType))
293  {
294  hasServices = true;
295  break;
296  }
297  }
298  }
299  return hasServices;
