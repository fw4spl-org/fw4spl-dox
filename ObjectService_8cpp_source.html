167 std::string ObjectService::getRegistryInformation() const
168 {
169  std::stringstream info;
170  ::fwCore::LogicStamp::LogicStampType previousKey = ~0ul;
171  ::fwCore::mt::ReadLock lock(m_containerMutex);
172  for( const ServiceContainerType::left_map::value_type& objSrvMap : m_container.left)
173  {
174  // TODO FIXME getObject() fail if there are expired object in OSR
175  ::fwCore::LogicStamp::LogicStampType key = objSrvMap.first;
176 
177  if ( previousKey != key )
178  {
179  info << "Object ( key = "<<key<<" ) has "
180  << m_container.left.count(key) <<" services." << std::endl;
181  previousKey = key;
182  }
183  ::fwServices::IService::sptr service = objSrvMap.second;
184  info << " srv : uid = "<< service->getID() <<" , classname = "<< service->getClassname()
185  <<" , service is stopped = "<< ( service->isStopped() ? "yes" : "no" ) << std::endl;
186  }
187  return info.str();
188 }

251 void ObjectService::unregisterService(const ::fwServices::IService::KeyType& objKey,
252  ::fwServices::IService::AccessType access,
253  ::fwServices::IService::sptr service )
254 {
255  ::fwCore::mt::WriteLock writeLock(m_containerMutex);
256 
257  ::fwData::Object::cwptr obj;
258 
259  if(access == ::fwServices::IService::AccessType::INPUT)
260  {
261  obj = service->m_inputsMap[objKey];
262  }
263  else if(access == ::fwServices::IService::AccessType::INOUT)
264  {
265  obj = service->m_inOutsMap[objKey];
266  }
267  else
268  {
269  obj = service->m_outputsMap[objKey];
270  }
271 
272  SLM_ASSERT("Object key '" + objKey + "' not found in service '" + service->getID() + "'", !obj.expired());
273 
274  // Remove from the left part
275  {
276  auto range = m_container.left.equal_range(obj.lock()->getOSRKey()->getLogicStamp());
277  for(auto it = range.first; it != range.second; )
278  {
279  if(it->second == service)
280  {
281  it = m_container.left.erase(it);
282  }
283  else
284  {
285  ++it;
286  }
287  }
288  }
289 
290  // Remove from the right part
291  {
292  auto range = m_container.right.equal_range(service);
293  for(auto it = range.first; it != range.second; )
294  {
295  if(it->second == obj.lock()->getOSRKey()->getLogicStamp())
296  {
297  it = m_container.right.erase(it);
298  }
299  else
300  {
301  ++it;
302  }
303  }
304  }
305 
306  if(access == ::fwServices::IService::AccessType::INPUT)
307  {
308  service->m_inputsMap.erase(objKey);
309  }
310  else if(access == ::fwServices::IService::AccessType::INOUT)
311  {
312  service->m_inOutsMap.erase(objKey);
313  }
314  else
315  {
316  service->m_outputsMap.erase(objKey);
317  }
318 }

354 void ObjectService::internalRegisterService(::fwData::Object::sptr object, ::fwServices::IService::sptr service,
355  const ::fwServices::IService::KeyType& objKey,
356  ::fwServices::IService::AccessType access)
357 {
358  SLM_ASSERT("Can't register a null service in OSR.", service);
359  SLM_ASSERT("Can't register a null object in OSR.", object);
360 
361  if(objKey.empty())
362  {
363  OSLM_ASSERT("Service "<< service->getID()<<" has already a valid associated object",
364  service->m_associatedObject.expired() || object != service->m_associatedObject.lock());
365 
366  // Old behavior with 1 object -> N Services
367  OSLM_ASSERT("This service "<< service->getClassname()
368  << " is not valid for the object " << object->getClassname(),
369  ServiceFactory::getDefault()->checkServiceValidity(object->getClassname(), service->getClassname())
370  );
371 
372  OSLM_ASSERT("Service "<< service->getID()<<" already registered",
373  m_container.right.find(service) == m_container.right.end());
374 
375  service->m_associatedObject = object;
376  }
377  else
378  {
379  if(service->m_inputsMap.empty() && service->m_inOutsMap.empty() && service->m_outputsMap.empty())
380  {
381  // If we have an appXml2 but with an old service implementation,
382  // we consider that the primary object is the first one we added
383  service->m_associatedObject = object;
384  }
385  // new behavior with N objects -> N Services
386  if(access == ::fwServices::IService::AccessType::INPUT)
387  {
388  service->m_inputsMap[objKey] = object;
389  }
390  else if(access == ::fwServices::IService::AccessType::INOUT)
391  {
392  service->m_inOutsMap[objKey] = object;
393  }
394  else
395  {
396  service->m_outputsMap[objKey] = object;
397  }
398  }
399  m_container.insert( ServiceContainerType::value_type( object->getOSRKey()->getLogicStamp(), service ) );
400 }

472  if(m_container.left.find(obj->getOSRKey()->getLogicStamp()) != m_container.left.end())
