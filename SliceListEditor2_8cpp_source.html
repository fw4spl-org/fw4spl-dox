00158 void SliceListEditor2::receiving( ::fwServices::ObjectMsg::csptr msg ) throw(::fwTools::Failed)
00159 {
00160     SLM_TRACE_FUNC();
00161     ::fwComEd::ImageMsg::csptr imageMsg = ::fwComEd::ImageMsg::dynamicConstCast( msg );
00162     if( imageMsg && imageMsg->hasEvent( "SCAN_SHOW"))
00163     {
00164         ::fwData::Object::csptr dataInfo = imageMsg->getDataInfo("SCAN_SHOW");
00165         SLM_ASSERT("dataInfo is missing", dataInfo);
00166         SLM_ASSERT("m_relatedServiceId is missing", dataInfo->getField( ::fwComEd::Dictionary::m_relatedServiceId ) );
00167         std::string servId = dataInfo->getField< ::fwData::String >(::fwComEd::Dictionary::m_relatedServiceId)->value();
00168         if( servId ==  m_adaptorUID )
00169         {
00170             ::fwData::Boolean::csptr isShowScan = ::fwData::Boolean::dynamicConstCast(dataInfo);
00171             m_noSliceItem->setChecked(!isShowScan->value());
00172             m_oneSliceItem->setChecked(isShowScan->value() && m_nbSlice == 1);
00173             m_threeSlicesItem->setChecked(isShowScan->value() && m_nbSlice == 3);
00174         }
00175     }
