177 {
178  BufferInfo & info = m_bufferInfos[bufferPtr];
179  SLM_ASSERT("Buffer must be allocated or dumped", (*bufferPtr != NULL) || !info.loaded);
180 
181  m_dumpPolicy->reallocateRequest( info, bufferPtr, newSize );
182 
183  try

353 bool BufferManager::dumpBuffer(BufferInfo & info, BufferManager::BufferPtrType bufferPtr)
354 {
355  if ( !info.loaded || info.lockCount() > 0 || info.size == 0 )
356  {
357  return false;
358  }
359 
360  ::boost::filesystem::path tmp = ::fwTools::System::getTemporaryFolder();
361  ::boost::filesystem::path dumpedFile = ::boost::filesystem::unique_path( tmp/"fwMemory-%%%%-%%%%-%%%%-%%%%.raw" );
362 
363  OSLM_TRACE("dumping " << bufferPtr << " " << dumpedFile);
364  info.lockCounter.reset();
365 
366  if ( this->writeBufferImpl(*bufferPtr, info.size, dumpedFile) )
367  {
368  info.fsFile = ::fwMemory::FileHolder(dumpedFile, true);
369  info.fileFormat = ::fwMemory::RAW;
370  info.istreamFactory = std::make_shared< ::fwMemory::stream::in::Raw >(info.fsFile);
371  info.userStreamFactory = false;
372  info.bufferPolicy->destroy(*bufferPtr);
373  *bufferPtr = NULL;
374  info.loaded = false;
375 
376  m_dumpPolicy->dumpSuccess( info, bufferPtr );
377 
378  m_updatedSig->asyncEmit();
379  }
380 
381  return !info.loaded;
382 }

401 bool BufferManager::restoreBuffer(BufferInfo & info,
402  BufferManager::BufferPtrType bufferPtr, BufferManager::SizeType allocSize)
403 {
404 
405  allocSize = ((allocSize) ? allocSize : info.size);
406  if ( !info.loaded )
407  {
408  OSLM_TRACE("Restoring " << bufferPtr);
409 
410  info.bufferPolicy->allocate(*bufferPtr, allocSize);
411 
412  char * charBuf = static_cast< char * >(*bufferPtr);
413  SizeType size = std::min(allocSize, info.size);
414  bool notFailed = false;
415  {
416  SPTR(std::istream) isptr = (*info.istreamFactory)();
417  std::istream &is = *isptr;
418  SizeType read = is.read(charBuf, size).gcount();
419 
420  FW_RAISE_IF(" Bad file size, expected: " << size << ", was: " << read, size - read != 0);
421  notFailed = !is.fail();
422  }
423 
424  if ( notFailed )
425  {
426  info.loaded = true;
427  info.fsFile.clear();
428  info.lastAccess.modified();
429 
430  m_dumpPolicy->restoreSuccess( info, bufferPtr );
431 
432  info.fileFormat = ::fwMemory::OTHER;
433  info.istreamFactory
434  = std::make_shared< ::fwMemory::stream::in::Buffer >(*bufferPtr,
435  allocSize,
436  std::bind(&getLock, this->getSptr(),
437  bufferPtr));
438  info.userStreamFactory = false;
439  m_updatedSig->asyncEmit();
440  return true;
441  }
442 
443  }
444  return false;
445 }

592 void BufferManager::setIStreamFactoryImpl(BufferPtrType bufferPtr,
593  const SPTR(::fwMemory::stream::in::IFactory) &factory,
594  SizeType size,
595  ::fwMemory::FileHolder fsFile,
596  ::fwMemory::FileFormatType format,
597  const ::fwMemory::BufferAllocationPolicy::sptr &policy
598  )
599 {
600  BufferInfoMapType::iterator iterInfo = m_bufferInfos.find(bufferPtr);
601  FW_RAISE_IF("Buffer is not managed by fwMemory::BufferManager.", iterInfo == m_bufferInfos.end() );
602  BufferInfo & info = iterInfo->second;
603 
604  SLM_ASSERT("Buffer is already set", *bufferPtr == NULL && info.loaded);
605 
606  info.istreamFactory = factory;
607  info.userStreamFactory = true;
608  info.size = size;
609  info.fsFile = fsFile;
610  info.fileFormat = format;
611  info.bufferPolicy = policy;
612  info.loaded = false;
613 
614  switch(m_loadingMode)
615  {
616  case BufferManager::DIRECT:
617  this->restoreBuffer(bufferPtr);
618  break;
619  case BufferManager::LAZY:
620  m_dumpPolicy->dumpSuccess( info, bufferPtr );
621  m_updatedSig->asyncEmit();
622  break;
623  default:
624  SLM_ASSERT("You shall not pass", 0);
625  }
