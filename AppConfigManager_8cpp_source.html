00621 {

00624     typedef std::vector< SlotInfoType > SlotInfoContainerType;
00625 
00626     SignalInfoType signalInfo;

00628 
00629     ::boost::regex re("(.*)/(.*)");

00632

00634     {

00637         {

00639             uid.assign(match[1].first, match[1].second);
00640             key.assign(match[2].first, match[2].second);
00641 
00642             OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName() ,
00643                         !uid.empty() && !key.empty());
00644 
00645             if (elem->getName() == "signal")

00647                 SLM_ASSERT("There must be only one signal by connection",
00648                            signalInfo.first.empty() && signalInfo.second.empty());

00651             else if (elem->getName() == "slot")
00652             {

00654             }
00655         }
00656     }
00657

00659     ::fwCom::HasSignals::sptr hasSignals = ::boost::dynamic_pointer_cast< ::fwCom::HasSignals >(sigSource);
00660 
00661     SLM_ASSERT("Signal source not found" << signalInfo.first , sigSource);
00662     SLM_ASSERT("invalid signal source " << signalInfo.first , hasSignals);
00663 
00664     BOOST_FOREACH(SlotInfoType slotInfo,  slotInfos)
00665     {
