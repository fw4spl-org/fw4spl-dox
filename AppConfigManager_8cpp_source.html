31 ::fwData::Object::sptr AppConfigManager::getNewObject(
32  ConfigAttribute type,
33  ConfigAttribute uid,
34  ConfigAttribute id)
35 {
36  // Building object structure
37  SPTR(::fwRuntime::Extension) ext = ::fwRuntime::findExtension(::boost::get<0>(type));
38  if (ext)
39  {
40  std::string className = ::fwCore::getClassname< ::fwData::Object >();
41  SLM_ASSERT("Extension and classname are different.",
42  ext->getPoint() == className);
43 
44  // Start dll to retrieve proxy and register object
45  ext->getBundle()->start();
46  }
47 
48  ::fwData::Object::sptr obj;
49  obj = ::fwData::factory::New(::boost::get<0>(type));
50  OSLM_ASSERT("Factory failed to build object : " << ::boost::get<0>(type), obj);
51 
52  if (::boost::get<1>(uid))
53  {
54  SLM_ASSERT("Object already has an UID.", !obj->hasID());
55  OSLM_FATAL_IF("UID " << ::boost::get<0>(uid) << " already exists.",
56  ::fwTools::fwID::exist(::boost::get<0>(uid)));
57  obj->setID(::boost::get<0>(uid));
58  }
59 
60  if (::boost::get<1>(id))
61  {
62  obj->setName(::boost::get<0>(id));
63  }
64  return obj;
65 }

196 {
197  std::vector< ::fwServices::IService::SharedFutureType > futures;
198 
199  for(::fwRuntime::ConfigurationElement::csptr elem : m_cfgElem->getElements())
200  {
201  if (elem->getName() == "start")
202  {
203  SLM_ERROR_IF("\"type\" attribute on \"start\" node is deprecated. Use UID instead.",
204  elem->hasAttribute("type"));
205 
206  // Uid
207  SLM_ASSERT("Missing attribute \"uid\".", elem->hasAttribute("uid"));
208  std::string uid = elem->getExistingAttributeValue("uid");
209 
210  OSLM_ASSERT("Service with UID \"" << uid << "\" doesn't exist.", ::fwTools::fwID::exist(uid));
211 
212  ::fwServices::IService::sptr srv = ::fwServices::get(uid);
213 
214  OSLM_ASSERT("No service registered with UID \"" << uid << "\".", srv);
215 
216  futures.push_back(srv->start());
217  m_startedSrv.push_back(srv);
218  }
219  }
220  ::boost::wait_for_all(futures.begin(), futures.end());
221 }

225 void AppConfigManager::processUpdateItems()
226 {
227  std::vector< ::fwServices::IService::SharedFutureType > futures;
228 
229  for(::fwRuntime::ConfigurationElement::csptr elem : m_cfgElem->getElements())
230  {
231  if (elem->getName() == "update")
232  {
233  // Type
234  if (elem->hasAttribute("type"))
235  {
236  std::string type = elem->getExistingAttributeValue("type");
237 
238  auto servicesToUpdate = ::fwServices::OSR::getServices(type);
239 
240  OSLM_ASSERT("No services of type \"" << type << "\" found.", !servicesToUpdate.empty());
241 
242  for(::fwServices::IService::sptr srv : servicesToUpdate)
243  {
244  futures.push_back(srv->update());
245  }
246  }
247  else
248  {
249  SLM_ASSERT("Missing attribute \"uid\" or \"type\".", elem->hasAttribute("uid"));
250  std::string uid = elem->getExistingAttributeValue("uid");
251 
252  OSLM_ASSERT("Service with UID \"" << uid << "\" doesn't exist.", ::fwTools::fwID::exist(uid));
253 
254  futures.push_back(::fwServices::get(uid)->update());
255  }
256  }
257  }
258 
259  ::boost::wait_for_all(futures.begin(), futures.end());
260 }

264 ::fwData::Object::sptr AppConfigManager::createObject()
265 {
266  OSLM_ASSERT("Bad configuration element.", m_cfgElem->getName() == "object");
267 
268  // Get attributes
269 
270  // Uid
271  ConfigAttribute uid("", false);
272  if (m_cfgElem->hasAttribute("uid"))
273  {
274  ::boost::get<0>(uid) = m_cfgElem->getExistingAttributeValue("uid");
275  SLM_ASSERT("\"uid\" attribute is empty.", !::boost::get<0>(uid).empty());
276  ::boost::get<1>(uid) = true;
277  }
278 
279  // Id
280  ConfigAttribute id("", false);
281  if (m_cfgElem->hasAttribute("id"))
282  {
283  ::boost::get<0>(id) = m_cfgElem->getExistingAttributeValue("id");
284  SLM_ASSERT("\"id\" attribute is empty.", !::boost::get<0>(id).empty());
285  ::boost::get<1>(id) = true;
286  }
287 
288  // Type
289  ConfigAttribute type("", false);
290  if (m_cfgElem->hasAttribute("type"))
291  {
292  ::boost::get<0>(type) = m_cfgElem->getExistingAttributeValue("type");
293  SLM_ASSERT("\"type\" attribute is empty.", !::boost::get<0>(type).empty());
294  SLM_ASSERT("\"type\" must be a rooted namespace.",
295  ::boost::get<0>(type).substr(0, 2) == "::");
296  ::boost::get<1>(type) = true;
297  }
298 
299  // Buildmode
300  ConfigAttribute buildMode("", false);
301  if (m_cfgElem->hasAttribute("src"))
302  {
303  ::boost::get<0>(buildMode) = m_cfgElem->getExistingAttributeValue("src");
304  SLM_ASSERT("\"src\" attribute is empty.", !::boost::get<0>(buildMode).empty());
305 
306  SLM_ASSERT("Unhandled build mode (bad \"src\" attribute). "
307  "Must be \"new\" or \"ref\".",
308  ::boost::get<0>(buildMode) == "ref" ||
309  ::boost::get<0>(buildMode) == "src");
310  ::boost::get<1>(buildMode) = true;
311  }
312 
313  // Creation of a new object
314  ::fwData::Object::sptr obj;
315 
316  // Create new or get the referenced object
317  if (::boost::get<1>(buildMode) && ::boost::get<0>(buildMode) == "ref")
318  {
319  SLM_ASSERT("Missing attribute \"uid\".", ::boost::get<1>(uid));
320  obj = getRefObject(type, ::boost::get<0>(uid), id);
321  }
322  else
323  {
324  obj = getNewObject(type, uid, id);
325  }
326 
327  // Get the object parser associated with the object type
328  ::fwServices::registry::ServiceFactory::sptr srvFactory = ::fwServices::registry::ServiceFactory::getDefault();
329 
330  std::string srvImpl = srvFactory->getDefaultImplementationIdFromObjectAndType(
331  obj->getClassname(), "::fwServices::IXMLParser");
332 
333  ::fwServices::IService::sptr srv = srvFactory->create("::fwServices::IXMLParser", srvImpl);
334  m_objectParser = ::fwServices::IXMLParser::dynamicCast(srv);
335  m_objectParser->setObjectConfig(m_cfgElem);
336  m_objectParser->createConfig(obj);
337  return obj;
338 }

376 void AppConfigManager::bindService(::fwRuntime::ConfigurationElement::csptr srvElem)
377 {
378  SLM_ASSERT("Configuration element is not a \"service\" node.", srvElem->getName() == "service");
379 
380  // Get attributes
381 
382  // Uid
383  ConfigAttribute uid("", false);
384  if (srvElem->hasAttribute("uid"))
385  {
386  ::boost::get<0>(uid) = srvElem->getExistingAttributeValue("uid");
387  OSLM_ASSERT("\"uid\" attribute is empty.", !::boost::get<0>(uid).empty());
388  ::boost::get<1>(uid) = true;
389  }
390 
391  // Type
392  std::string type = srvElem->getAttributeValue("type");
393 
394  // Implementation
395  ConfigAttribute implType("", false);
396  if (srvElem->hasAttribute("impl"))
397  {
398  ::boost::get<0>(implType) = srvElem->getExistingAttributeValue("impl");
399  SLM_ASSERT("\"impl\" attribute is empty.", !::boost::get<0>(implType).empty());
400 
401  ::boost::get<1>(implType) = true;
402  }
403 
404  // Type
405  std::string workerKey = srvElem->getAttributeValue("worker");
406 
407  // Config
408  ConfigAttribute config("", false);
409  if (srvElem->hasAttribute("config"))
410  {
411  ::boost::get<0>(config) = srvElem->getExistingAttributeValue("config");
412  SLM_ASSERT("\"config\" attribute is empty.", !::boost::get<0>(config).empty());
413 
414  ::boost::get<1>(config) = true;
415  }
416 
417  // autoConnect
418  const ::fwRuntime::ConfigurationElement::AttributePair attribAutoConnect =
419  srvElem->getSafeAttributeValue("autoConnect");
420  std::string autoConnect = attribAutoConnect.second;
421  SLM_ASSERT("\"autoConnect\" attribute must be either \"yes\" or \"no\".",
422  (!attribAutoConnect.first) || autoConnect == "yes" || autoConnect == "no");
423 
424  // Priority
425  ConfigAttribute priority("", false);
426  if (srvElem->hasAttribute("priority"))
427  {
428  ::boost::get<0>(priority) = srvElem->getExistingAttributeValue("priority");
429  SLM_ASSERT("\"priority\" attribute is empty.", !::boost::get<0>(priority).empty());
430 
431  ::boost::get<1>(priority) = true;
432  }
433 
434  // TODO: have a default implementation of service
435  if (!::boost::get<1>(implType))
436  {
437  OSLM_FATAL("Attribute \"impl\" is required for service " <<
438  (::boost::get<1>(uid) ? ::boost::get<0>(uid) : ".") );
439  }
440 
441  // Create and bind service
442  ::fwServices::IService::sptr srv = this->getNewService(type, uid, implType);
443  ::fwServices::OSR::registerService(m_configuredObject, srv);
444  m_createdSrv.push_back(srv);
445 
446  if (!workerKey.empty())
447  {
448  ::fwServices::registry::ActiveWorkers::sptr activeWorkers = ::fwServices::registry::ActiveWorkers::getDefault();
449  ::fwThread::Worker::sptr worker;
450  worker = activeWorkers->getWorker(workerKey);
451  if (!worker)
452  {
453  worker = ::fwThread::Worker::New();
454  activeWorkers->addWorker(workerKey, worker);
455  }
456  srv->setWorker(worker);
457  }
458 
459  // Get service configuration
460  ::fwRuntime::ConfigurationElement::csptr cfgElem = srvElem;
461 
462  ::fwServices::registry::ServiceConfig::sptr srvCfgFactory = ::fwServices::registry::ServiceConfig::getDefault();
463  if (::boost::get<1>(config))
464  {
465  cfgElem = srvCfgFactory->getServiceConfig(::boost::get<0>(config), ::boost::get<0>(implType));
466  }
467 
468  // Configure
469  SLM_ASSERT("ConstCast failed.", ::fwRuntime::ConfigurationElement::constCast(cfgElem));
470  srv->setConfiguration(::fwRuntime::ConfigurationElement::constCast(cfgElem));
471  srv->configure();
472 
473  // Communication channel
474  if (autoConnect == "yes")
475  {
476  this->autoSigSlotConnection(m_configuredObject, srv, priority);
477  }
478 
479  // Check if user did not bind a service to another service
480  for(::fwRuntime::ConfigurationElement::csptr elem : cfgElem->getElements())
481  {
482  SLM_ASSERT("Cannot bind a service to another service.",
483  elem->getName() != "service" &&
484  elem->getName() != "serviceList");
485  }
486 }

615 void AppConfigManager::createProxy(::fwRuntime::ConfigurationElement::csptr config)
616 {
617  ::boost::regex re("(.*)/(.*)");
618  ::boost::smatch match;
619  std::string src, uid, key;
620 
621  ::fwServices::registry::Proxy::sptr proxy = ::fwServices::registry::Proxy::getDefault();
622 
623  SLM_ASSERT("Missing 'channel' attribute", config->hasAttribute("channel"));
624  std::string channel = config->getAttributeValue("channel");
625  ProxyConnections proxyCnt(channel);
626 
627  for(::fwRuntime::ConfigurationElement::csptr elem : config->getElements())
628  {
629  src = elem->getValue();
630  if( ::boost::regex_match(src, match, re) )
631  {
632  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
633  uid.assign(match[1].first, match[1].second);
634  key.assign(match[2].first, match[2].second);
635 
636  OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName(),
637  !uid.empty() && !key.empty());
638 
639  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(uid);
640 
641  if (elem->getName() == "signal")
642  {
643  ::fwCom::HasSignals::sptr hasSignals = std::dynamic_pointer_cast< ::fwCom::HasSignals >(obj);
644  ::fwCom::SignalBase::sptr sig = hasSignals->signal(key);
645  proxy->connect(channel, sig);
646  proxyCnt.addSignalConnection(uid, key);
647  }
648  else if (elem->getName() == "slot")
649  {
650  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
651  ::fwCom::SlotBase::sptr slot = hasSlots->slot(key);
652  proxy->connect(channel, slot);
653  proxyCnt.addSlotConnection(uid, key);
654  }
655  }
656  }
657  m_vectProxyCtns.push_back(proxyCnt);
658 }

662 void AppConfigManager::destroyProxies()
663 {
664  ::fwServices::registry::Proxy::sptr proxy = ::fwServices::registry::Proxy::getDefault();
665  for(ProxyConnectionsVectType::value_type proxyConnections : m_vectProxyCtns)
666  {
667  for(ProxyConnections::ProxyEltType signalElt : proxyConnections.m_signals)
668  {
669  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(signalElt.first);
670  ::fwCom::HasSignals::sptr hasSignals = std::dynamic_pointer_cast< ::fwCom::HasSignals >(obj);
671  ::fwCom::SignalBase::sptr sig = hasSignals->signal(signalElt.second);
672  proxy->disconnect(proxyConnections.m_channel, sig);
673  }
674  for(ProxyConnections::ProxyEltType slotElt : proxyConnections.m_slots)
675  {
676  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(slotElt.first);
677  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
678  ::fwCom::SlotBase::sptr slot = hasSlots->slot(slotElt.second);
679  proxy->disconnect(proxyConnections.m_channel, slot);
680  }
681  }
682  m_vectProxyCtns.clear();
683 }
