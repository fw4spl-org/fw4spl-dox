85 void SRenderer::stopping() throw(fwTools::Failed)
86 {
87  if( m_render == 0 )
88  {
89  return;
90  }
91 
92  if(m_bPipelineIsInit)
93  {
94  m_negatoSagittal->Delete();
95  m_negatoFrontal->Delete();
96  m_negatoAxial->Delete();
97  m_outline->Delete();
98  }
99 
100  SLM_ASSERT("m_render not instanced", m_render);
101  m_render->Delete();
102  m_render = 0;
103 
104  m_interactorManager->uninstallInteractor();
105  m_interactorManager.reset();
106 
107 // this->getContainer()->clean();
108  this->destroy();
109 }

122 void SRenderer::refresh()
123 {
124  ::fwData::Image::sptr img = this->getObject< ::fwData::Image >();
125  bool imageIsValid = ::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity( img );
126  if(imageIsValid )
127  {
128  if(!m_bPipelineIsInit)
129  {
130  initVTKPipeline();
131  m_bPipelineIsInit = true;
132  }
133  else
134  {
135  updateVTKPipeline();
136  }
137 
138 
139  //
140  int axialIndex = static_cast<int>(img->getSize()[2]/2);
141  int frontalIndex = static_cast<int>(img->getSize()[1]/2);
142  int sagittalIndex = static_cast<int>(img->getSize()[0]/2);
143 
144  m_negatoAxial->SetSliceIndex( axialIndex );
145  m_negatoFrontal->SetSliceIndex( frontalIndex );
146  m_negatoSagittal->SetSliceIndex( sagittalIndex );
147  m_interactorManager->getInteractor()->Render();
148  }
149 }

153 void SRenderer::initVTKPipeline()
154 {
155  vtkSmartPointer< vtkImageData > vtk_img = vtkSmartPointer< vtkImageData >::New();
156  ::fwVtkIO::toVTKImage( this->getObject< ::fwData::Image >(), vtk_img);
157 
158  m_outline = vtkOutlineFilter::New();
159  m_outline->SetInputData(vtk_img);
160 
161  vtkPolyDataMapper* outlineMapper = vtkPolyDataMapper::New();
162  outlineMapper->SetInputConnection(m_outline->GetOutputPort());
163 
164  vtkActor* outlineActor = vtkActor::New();
165  outlineActor->SetMapper( outlineMapper);
166 
167  vtkCellPicker* picker = vtkCellPicker::New();
168  picker->SetTolerance(0.005);
169 
170  //assign default props to the ipw's texture plane actor
171 
172  m_negatoSagittal = vtkImagePlaneWidget::New();
173  m_negatoSagittal->SetInteractor( m_interactorManager->getInteractor() );
174  m_negatoSagittal->SetKeyPressActivationValue('x');
175  m_negatoSagittal->SetPicker(picker);
176  m_negatoSagittal->GetPlaneProperty()->SetColor(1,0,0);
177  m_negatoSagittal->TextureInterpolateOn();
178  m_negatoSagittal->SetInputData(vtk_img);
179  m_negatoSagittal->SetPlaneOrientationToXAxes();
180  m_negatoSagittal->DisplayTextOn();
181  m_negatoSagittal->On();
182  m_negatoSagittal->InteractionOn();
183 
184  m_negatoFrontal = vtkImagePlaneWidget::New();
185  m_negatoFrontal->SetInteractor( m_interactorManager->getInteractor() );
186  m_negatoFrontal->SetKeyPressActivationValue('y');
187  m_negatoFrontal->SetPicker(picker);
188  m_negatoFrontal->GetPlaneProperty()->SetColor(0,1,0);
189  m_negatoFrontal->TextureInterpolateOn();
190  m_negatoFrontal->SetInputData(vtk_img);
191  m_negatoFrontal->SetPlaneOrientationToYAxes();
192  m_negatoFrontal->SetLookupTable( m_negatoSagittal->GetLookupTable());
193  m_negatoFrontal->DisplayTextOn();
194  m_negatoFrontal->UpdatePlacement();
195  m_negatoFrontal->On();
196 
197  m_negatoAxial = vtkImagePlaneWidget::New();
198  m_negatoAxial->SetInteractor( m_interactorManager->getInteractor() );
199  m_negatoAxial->SetKeyPressActivationValue('z');
200  m_negatoAxial->SetPicker(picker);
201  m_negatoAxial->GetPlaneProperty()->SetColor(0,0,1);
202  m_negatoAxial->TextureInterpolateOn();
203  m_negatoAxial->SetInputData(vtk_img);
204  m_negatoAxial->SetPlaneOrientationToZAxes();
205  m_negatoAxial->SetLookupTable( m_negatoSagittal->GetLookupTable());
206  m_negatoAxial->DisplayTextOn();
207  m_negatoAxial->On();
208 
209  // Add the actors
210  m_render->AddActor( outlineActor);
211 
212  // Repaint and resize window
213  m_render->ResetCamera();
214 
215  picker->Delete();
216  outlineActor->Delete();
217  outlineMapper->Delete();
218 }
