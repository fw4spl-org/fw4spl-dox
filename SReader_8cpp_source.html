00115         }
00116     }

00119     SLM_ASSERT("The <extensions> element can be set at most once.", extensionsList.size() <= 1);
00120 
00121     if(extensionsList.size() == 1)
00122     {
00123         m_allowedExts.clear();

00131             FileExtension2NameType::const_iterator itKnown = s_EXTENSIONS.find(ext);
00132             FileExtension2NameType::const_iterator itCustom = m_customExts.find(ext);

00141             }
00142         }
00143     }
00144     else
00145     {
00146         m_allowedExts.clear();
00147 
00148         BOOST_FOREACH(FileExtension2NameType::value_type ext, m_customExts)
00149         {
00150             m_allowedExts.insert(m_allowedExts.end(), ext.first);
00151         }
00152 
00153         BOOST_FOREACH(SReader::FileExtension2NameType::value_type ext, SReader::s_EXTENSIONS)
00154         {
00155             m_allowedExts.insert(m_allowedExts.end(), ext.first);
00156             m_allowedExtLabels[ext.first] = ext.second;
00157         }
00158     }
00159 
00160     ConfigurationElementContainer inject = m_configuration->find("inject");
00161     SLM_ASSERT("The <inject> element can be set at most once.", inject.size() <= 1);

00183

00185                    ("Reuse" == m_uuidPolicy && !m_inject.empty()) || "Reuse" != m_uuidPolicy

00191     if (patcher.size() == 1)
00192     {
00193         m_context = patcher.at(0)->getExistingAttributeValue("context");

00207         ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();

00216                 size_t aliveMemory = stats.totalManaged - stats.totalDumped;
00217                 size_t freeMemory = ::fwMemory::tools::MemoryMonitorTools::estimateFreeMem() / 2;
00218                 size_t barrier = std::max( aliveMemory, std::max( freeMemory, static_cast<size_t>(500L * 1024 * 1024) ) );
00219 
00220                 newDumpPolicy->setBarrier( barrier );
00221                 manager->setDumpPolicy( newDumpPolicy );
00222                 m_oldPolicy = policy;
00223             }
00224         }
00225     }
00226

00228     {
00229         try
00230         {
00231             ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
00232             if( manager && m_oldPolicy )
00233             {
00234                 manager->setDumpPolicy( m_oldPolicy );
00235                 m_oldPolicy.reset();
00236             }
00237         }
00238         catch(...)
00239         {
00240             SLM_ASSERT("Failed to restore old policy", 0);
00241         }
00242     }
