99  typedef SPTR (::fwRuntime::ConfigurationElement) ConfigurationElement;
100  typedef std::vector < ConfigurationElement > ConfigurationElementContainer;
101 
102  m_customExts.clear();
103  m_allowedExtLabels.clear();
104 
105  ConfigurationElementContainer customExtsList = m_configuration->find("archive");
106  for(ConfigurationElement archive : customExtsList)
107  {
108  const std::string& backend = archive->getAttributeValue("backend");
109  SLM_ASSERT("No backend attribute given in archive tag", backend != "");
110  SLM_ASSERT("Unsupported backend '" + backend + "'", s_EXTENSIONS.find("." + backend) != s_EXTENSIONS.end());
111 
112  ConfigurationElementContainer exts = archive->find("extension");
113  for(ConfigurationElement ext : exts)
114  {
115  const std::string& extension = ext->getValue();
116  SLM_ASSERT("No extension given for backend '" + backend + "'", !extension.empty());
117  SLM_ASSERT("Extension must begin with '.'", extension[0] == '.');
118 
119  m_customExts[extension] = backend;
120  m_allowedExtLabels[extension] = ext->getAttributeValue("label");
121  }
122  }
123 
124  ConfigurationElementContainer extensionsList = m_configuration->find("extensions");
125  SLM_ASSERT("The <extensions> element can be set at most once.", extensionsList.size() <= 1);
126 
127  if(extensionsList.size() == 1)
128  {
129  m_allowedExts.clear();
130 
131  ConfigurationElementContainer extensions = extensionsList.at(0)->find("extension");
132  for(ConfigurationElement extension : extensions)
133  {
134  const std::string& ext = extension->getValue();
135 
136  // The extension must be found either in custom extensions list or in known extensions
137  FileExtension2NameType::const_iterator itKnown = s_EXTENSIONS.find(ext);
138  FileExtension2NameType::const_iterator itCustom = m_customExts.find(ext);
139 
140  const bool extIsKnown = (itKnown != SReader::s_EXTENSIONS.end() || itCustom != m_customExts.end());
141  SLM_ASSERT("Extension '" + ext + "' is not allowed in configuration", extIsKnown);
142 
143  if(extIsKnown)
144  {
145  m_allowedExts.insert(m_allowedExts.end(), ext);
146  m_allowedExtLabels[ext] = extension->getAttributeValue("label");
147  }
148  }
149  }
150  else
151  {
152  m_allowedExts.clear();
153 
154  for(FileExtension2NameType::value_type ext : m_customExts)
155  {
156  m_allowedExts.insert(m_allowedExts.end(), ext.first);
157  }
158 
159  for(SReader::FileExtension2NameType::value_type ext : SReader::s_EXTENSIONS)
160  {
161  m_allowedExts.insert(m_allowedExts.end(), ext.first);
162  m_allowedExtLabels[ext.first] = ext.second;
163  }
164  }
165 
166  ConfigurationElementContainer inject = m_configuration->find("inject");
167  SLM_ASSERT("The <inject> element can be set at most once.", inject.size() <= 1);
168  if (inject.size() == 1)
169  {
170  m_inject = inject.at(0)->getValue();
171  }
172 
173  ConfigurationElementContainer filter = m_configuration->find("filter");
174  SLM_ASSERT("The <filter> element can be set at most once.", filter.size() <= 1);
175  if (filter.size() == 1)
176  {
177  m_filter = filter.at(0)->getValue();
178  }
179 
180  ConfigurationElementContainer uuidPolicy = m_configuration->find("uuidPolicy");
181  SLM_ASSERT("The <uuidPolicy> element can be set at most once.", uuidPolicy.size() <= 1);
182  if (uuidPolicy.size() == 1)
183  {
184  m_uuidPolicy = uuidPolicy.at(0)->getValue();
185  SLM_ASSERT("Unknown policy : '"
186  + m_uuidPolicy +
187  "', available policies : 'Strict','Change' or 'Reuse'.",
188  "Strict" == m_uuidPolicy || "Change" == m_uuidPolicy || "Reuse" == m_uuidPolicy );
189 
190  SLM_ASSERT("'Reuse' policy is available only with inject mode",
191  ("Reuse" == m_uuidPolicy && !m_inject.empty()) || "Reuse" != m_uuidPolicy
192  );
193  }
194 
195  ConfigurationElementContainer patcher = m_configuration->find("patcher");
196  SLM_ASSERT("The <patcher> element can be set at most once.", patcher.size() <= 1 );
197  if (patcher.size() == 1)
198  {
199  m_context = patcher.at(0)->getExistingAttributeValue("context");
200  m_version = patcher.at(0)->getExistingAttributeValue("version");
201  m_useAtomsPatcher = true;

209 struct SetDumpPolicy
210 {
211  SetDumpPolicy()
212  {
213  ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
214  m_lock = ::fwCore::mt::WriteLock( manager->getMutex() );
215  if( manager )
216  {
217  ::fwMemory::IPolicy::sptr policy = manager->getDumpPolicy();
218  if( ::fwMemory::policy::NeverDump::dynamicCast(policy) )
219  {
220  ::fwMemory::policy::BarrierDump::sptr newDumpPolicy = ::fwMemory::policy::BarrierDump::New();
221  ::fwMemory::BufferManager::BufferStats stats = manager->getBufferStats().get();
222 
223  size_t aliveMemory = stats.totalManaged - stats.totalDumped;
224  size_t freeMemory = ::fwMemory::tools::MemoryMonitorTools::estimateFreeMem() / 2;
225  size_t barrier =
226  std::max( aliveMemory, std::max( freeMemory, static_cast<size_t>(500L * 1024 * 1024) ) );
227 
228  newDumpPolicy->setBarrier( barrier );
229  manager->setDumpPolicy( newDumpPolicy );
230  m_oldPolicy = policy;
231  }
232  }
233  }
234 
235  ~SetDumpPolicy()
236  {
237  try
238  {
239  ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
240  if( manager && m_oldPolicy )
241  {
242  manager->setDumpPolicy( m_oldPolicy );
243  m_oldPolicy.reset();
244  }
245  }
246  catch(...)
247  {
248  SLM_ASSERT("Failed to restore old policy", 0);
249  }
250  }
251 
252  ::fwMemory::IPolicy::sptr m_oldPolicy;
253  ::fwCore::mt::WriteLock m_lock;
254 };

258 void SReader::updating() throw(::fwTools::Failed)
259 {
260  if(this->hasLocationDefined())
261  {
262  SetDumpPolicy policy;
263 
264  ::fwData::Object::sptr data = this->getObject< ::fwData::Object >();
265 
266  ::fwGui::Cursor cursor;
267  cursor.setCursor(::fwGui::ICursor::BUSY);
268 
269  try
270  {
271  const ::boost::filesystem::path& filePath = this->getFile();
272  const ::boost::filesystem::path folderPath = filePath.parent_path();
273  const ::boost::filesystem::path filename = filePath.filename();
274  std::string extension = ::boost::filesystem::extension(filePath);
275 
276  FW_RAISE_IF( "Unable to guess file format (missing extension)", extension.empty() );
277 
278 
279  if(m_customExts.find(extension) != m_customExts.end())
280  {
281  extension = "." + m_customExts[extension];
282  }
283 
284  ::fwAtoms::Object::sptr atom;
285 
286  const unsigned int progressBarOffset = 10;
287 
288  // Reading file : job 1
289  ::fwJobs::Job::sptr fileReadingJob = ::fwJobs::Job::New("Reading " + extension + " file",
290  [ =, &atom](::fwJobs::Job& runningJob)
291  {
292  runningJob.doneWork(progressBarOffset);
293 
294  // Read atom
295  ::fwZip::IReadArchive::sptr readArchive;
296  ::boost::filesystem::path archiveRootName;
297  ::fwAtomsBoostIO::FormatType format = ::fwAtomsBoostIO::UNSPECIFIED;
298 
299  if ( extension == ".json" )
300  {
301  readArchive = ::fwZip::ReadDirArchive::New(folderPath.string());
302  archiveRootName = filename;
303  format = ::fwAtomsBoostIO::JSON;
304  }
305  else if ( extension == ".jsonz" )
306  {
307  readArchive = ::fwZip::ReadZipArchive::New(filePath.string());
308  archiveRootName = "root.json";
309  format = ::fwAtomsBoostIO::JSON;
310  }
311  else if ( extension == ".xml" )
312  {
313  readArchive = ::fwZip::ReadDirArchive::New(folderPath.string());
314  archiveRootName = filename;
315  format = ::fwAtomsBoostIO::XML;
316  }
317  else if ( extension == ".xmlz" )
318  {
319  readArchive = ::fwZip::ReadZipArchive::New(filePath.string());
320  archiveRootName = "root.xml";
321  format = ::fwAtomsBoostIO::XML;
322  }
323  else
324  {
325  FW_RAISE( "This file extension '" << extension << "' is not managed" );
326  }
327 
328  ::fwAtomsBoostIO::Reader reader;
329  atom = ::fwAtoms::Object::dynamicCast( reader.read( readArchive, archiveRootName, format ) );
330 
331  FW_RAISE_IF( "Invalid atoms file :'" << filePath << "'", !atom );
332 
333  runningJob.doneWork(progressBarOffset);
334 
335  runningJob.done();
336 
337  }, m_associatedWorker);
338 
339 
340  // patching atom : job 2
341  ::fwJobs::Job::sptr patchingJob = ::fwJobs::Job::New("Reading " + extension + " file",
342  [ =, &atom](::fwJobs::Job& runningJob)
343  {
344  if(runningJob.cancelRequested())
345  {
346  return;
347  }
348 
349  runningJob.doneWork(progressBarOffset);
350 
352  if ( m_useAtomsPatcher )
353  {
354  FW_RAISE_IF( "Unable to load data, found '" << atom->getMetaInfo("context")
355  << "' context, but '" << m_context <<
356  "' was excepted.",
357  atom->getMetaInfo("context") != m_context);
358 
359  ::fwAtomsPatch::PatchingManager globalPatcher(atom);
360  atom = globalPatcher.transformTo( m_version );
361  }
362 
363  if(!m_filter.empty())
364  {
365  ::fwAtomsFilter::IFilter::sptr filter = ::fwAtomsFilter::factory::New(m_filter);
366  OSLM_ASSERT("Failed to create IFilter implementation '" << m_filter << "'", filter);
367  filter->apply(atom);
368  }
369  runningJob.done();
370  }, m_associatedWorker);
371 
372  ::fwData::Object::sptr newData;
373 
374  // convert to fwData : job 3
375  ::fwJobs::Job::sptr atomToDataJob = ::fwJobs::Job::New("Reading " + extension + " file",
376  [ =, &newData, &atom](::fwJobs::Job& runningJob)
377  {
378  runningJob.doneWork(progressBarOffset);
379  if(runningJob.cancelRequested())
380  {
381  return;
382  }
383  if("Strict" == m_uuidPolicy)
384  {
385  newData = ::fwAtomConversion::convert(atom, ::fwAtomConversion::AtomVisitor::StrictPolicy());
386  }
387  else if("Reuse" == m_uuidPolicy)
388  {
389  newData = ::fwAtomConversion::convert(atom, ::fwAtomConversion::AtomVisitor::ReusePolicy());
390  }
391  else
392  {
393  newData = ::fwAtomConversion::convert(atom, ::fwAtomConversion::AtomVisitor::ChangePolicy());
394  }
395 
396  runningJob.done();
397  }, m_associatedWorker);
398 
399  ::fwJobs::Aggregator::sptr jobs = ::fwJobs::Aggregator::New(extension + " reader");
400  jobs->add(fileReadingJob);
401  jobs->add(patchingJob);
402  jobs->add(atomToDataJob);
403 
404  m_sigJobCreated->emit(jobs);
405 
406  jobs->run().get();
407 
408  if(jobs->getState() == ::fwJobs::IJob::CANCELED)
409  {
410  return;
411  }
412 
413  FW_RAISE_IF( "Unable to load '" << filePath << "' : invalid data.", !newData );
414 
415  if(m_inject.empty())
416  {
417  FW_RAISE_IF( "Unable to load '" << filePath
418  << "' : trying to load a '" << newData->getClassname()
419  << "' where a '" << data->getClassname() << "' was expected",
420  newData->getClassname() != data->getClassname() );
421 
422  data->shallowCopy(newData);
423  }
424  else
425  {
426  ::fwData::Composite::sptr composite = ::fwData::Composite::dynamicCast(data);
427  SLM_ASSERT("Inject mode works only on a Composite object", composite );
428 
429  ::fwDataTools::helper::Composite helper(composite);
430  helper.add(m_inject, newData);
431  helper.notify();
432  }
433 
434  this->notificationOfUpdate();
435  }
436  catch( std::exception& e )
437  {
438  OSLM_ERROR( e.what() );
439  ::fwGui::dialog::MessageDialog::showMessageDialog("Atoms reader failed", e.what(),
440  ::fwGui::dialog::MessageDialog::CRITICAL);
441  }
442  catch( ... )
443  {
444  ::fwGui::dialog::MessageDialog::showMessageDialog("Atoms reader failed", "Aborting operation.",
445  ::fwGui::dialog::MessageDialog::CRITICAL);
446  }
447 
448  cursor.setDefaultCursor();
449 
450  }
451 }
