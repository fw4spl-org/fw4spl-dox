00082   vtkProp *prop;
00083   vtkCamera *camera;
00084   vtkAbstractMapper3D *mapper = NULL;

00100   double cameraDOP[3];
00101   
00102   //  Initialize picking process
00103   this->Initialize();
00104   this->Renderer = renderer;
00105   this->SelectionPoint[0] = selectionX;
00106   this->SelectionPoint[1] = selectionY;
00107   this->SelectionPoint[2] = selectionZ;
00108 
00109   // Invoke start pick method if defined
00110   this->InvokeEvent(vtkCommand::StartPickEvent,NULL);
00111 
00112   if ( renderer == NULL )
00113     {
00114     vtkErrorMacro(<<"Must specify renderer!");
00115     return 0;
00116     }
00117 
00118   // Get camera focal point and position. Convert to display (screen) 
00119   // coordinates. We need a depth value for z-buffer.
00120   //
00121   camera = renderer->GetActiveCamera();
00122   camera->GetPosition(cameraPos);
00123   cameraPos[3] = 1.0;
00124   camera->GetFocalPoint(cameraFP);
00125   cameraFP[3] = 1.0;
00126 
00127   renderer->SetWorldPoint(cameraFP[0],cameraFP[1],cameraFP[2],cameraFP[3]);
00128   renderer->WorldToDisplay();
00129   displayCoords = renderer->GetDisplayPoint();
00130   selectionZ = displayCoords[2];
00131 
00132   // Convert the selection point into world coordinates.
00133   //
00134   renderer->SetDisplayPoint(selectionX, selectionY, selectionZ);
00135   renderer->DisplayToWorld();
00136   worldCoords = renderer->GetWorldPoint();
00137   if ( worldCoords[3] == 0.0 )
00138     {
00139     vtkErrorMacro(<<"Bad homogeneous coordinates");
00140     return 0;
00141     }
00142   for (i=0; i < 3; i++) 
00143     {
00144     this->PickPosition[i] = worldCoords[i] / worldCoords[3];
00145     }
00146   
00147   //  Compute the ray endpoints.  The ray is along the line running from
00148   //  the camera position to the selection point, starting where this line
00149   //  intersects the front clipping plane, and terminating where this
00150   //  line intersects the back clipping plane.
00151   for (i=0; i<3; i++)
00152     {
00153     ray[i] = this->PickPosition[i] - cameraPos[i];
00154     }
00155   for (i=0; i<3; i++)
00156     {
00157     cameraDOP[i] = cameraFP[i] - cameraPos[i];
00158     }
00159 
00160   vtkMath::Normalize(cameraDOP);
00161 
00162   if (( rayLength = vtkMath::Dot(cameraDOP,ray)) == 0.0 ) 
00163     {
00164     vtkWarningMacro("Cannot process points");
00165     return 0;
00166     } 
00167 
00168   clipRange = camera->GetClippingRange();
00169 
00170   if ( camera->GetParallelProjection() )
00171     {
00172     tF = clipRange[0] - rayLength;
00173     tB = clipRange[1] - rayLength;
00174     for (i=0; i<3; i++) 
00175       {
00176       p1World[i] = this->PickPosition[i] + tF*cameraDOP[i];
00177       p2World[i] = this->PickPosition[i] + tB*cameraDOP[i];
00178       }
00179     }
00180   else
00181     {
00182     tF = clipRange[0] / rayLength;
00183     tB = clipRange[1] / rayLength;
00184     for (i=0; i<3; i++) 
00185       {
00186       p1World[i] = cameraPos[i] + tF*ray[i];
00187       p2World[i] = cameraPos[i] + tB*ray[i];
00188       }
00189     }
00190   p1World[3] = p2World[3] = 1.0;
00191 
00192   // Compute the tolerance in world coordinates.  Do this by
00193   // determining the world coordinates of the diagonal points of the
00194   // window, computing the width of the window in world coordinates, and 
00195   // multiplying by the tolerance.
00196   //
00197   viewport = renderer->GetViewport();
00198   winSize = renderer->GetRenderWindow()->GetSize();
00199   x = winSize[0] * viewport[0];
00200   y = winSize[1] * viewport[1];
00201   renderer->SetDisplayPoint(x, y, selectionZ);
00202   renderer->DisplayToWorld();
00203   renderer->GetWorldPoint(windowLowerLeft);
00204 
00205   x = winSize[0] * viewport[2];
00206   y = winSize[1] * viewport[3];
00207   renderer->SetDisplayPoint(x, y, selectionZ);
00208   renderer->DisplayToWorld();
00209   renderer->GetWorldPoint(windowUpperRight);
00210 
00211   for (tol=0.0,i=0; i<3; i++) 
00212     {
00213     tol += (windowUpperRight[i] - windowLowerLeft[i]) *
00214       (windowUpperRight[i] - windowLowerLeft[i]);
00215     }
00216   
00217   tol = sqrt (tol) * this->Tolerance;
00218 
00219   //  Loop over all props.  Transform ray (defined from position of
00220   //  camera to selection point) into coordinates of mapper (not
00221   //  transformed to actors coordinates!  Reduces overall computation!!!).
00222   //  Note that only vtkProp3D's can be picked by fwVtkPicker.
00223   //
00224   vtkPropCollection *props;
00225   vtkProp *propCandidate;
00226   if ( this->PickFromList ) 
00227     {
00228     props = this->GetPickList();
00229     }
00230   else 
00231     {
00232     props = renderer->GetViewProps();
00233     }
00234 
00235   vtkActor *actor;
00236   vtkLODProp3D *prop3D;
00237   vtkVolume *volume;
00238   vtkImageActor *imageActor = 0;
00239   vtkAssemblyPath *path;
00240   vtkProperty *tempProperty;
00241   this->Transform->PostMultiply();
00242   vtkCollectionSimpleIterator pit;
00243   double scale[3];
00244   for ( props->InitTraversal(pit); (prop=props->GetNextProp(pit)); )
00245     {
00246     for ( prop->InitPathTraversal(); (path=prop->GetNextPath()); )
00247       {
00248       pickable = 0;
00249       actor = NULL;
00250       propCandidate = path->GetLastNode()->GetViewProp();
00251       if ( propCandidate->GetPickable() && propCandidate->GetVisibility() )
00252         {
00253         pickable = 1;
00254         if ( (actor=vtkActor::SafeDownCast(propCandidate)) != NULL )
00255           {
00256           mapper = actor->GetMapper();
00257           if ( actor->GetProperty()->GetOpacity() <= 0.0 )
00258             {
00259             pickable = 0;
00260             }
00261           }
00262         else if ( (prop3D=vtkLODProp3D::SafeDownCast(propCandidate)) != NULL )
00263           {
00264           LODId = prop3D->GetPickLODID();
00265           mapper = prop3D->GetLODMapper(LODId);
00266 
00267           // if the mapper is a vtkMapper (as opposed to a vtkVolumeMapper), 
00268           // then check the transparency to see if the object is pickable
00269           if ( vtkMapper::SafeDownCast(mapper) != NULL)
00270             {
00271             prop3D->GetLODProperty(LODId, &tempProperty);
00272             if ( tempProperty->GetOpacity() <= 0.0 )
00273               {
00274               pickable = 0;
00275               }
00276             }
00277           }
00278         else if ( (volume=vtkVolume::SafeDownCast(propCandidate)) != NULL )
00279           {
00280           mapper = volume->GetMapper();
00281           }
00282         else if ( (imageActor=vtkImageActor::SafeDownCast(propCandidate)) )
00283           {
00284           mapper = 0;
00285           }
00286         else 
00287           {
00288           pickable = 0; //only vtkProp3D's (actors and volumes) can be picked
00289           }
00290         }
00291 
00292       //  If actor can be picked, get its composite matrix, invert it, and
00293       //  use the inverted matrix to transform the ray points into mapper
00294       //  coordinates. 
00295       if ( pickable  &&  mapper != NULL )
00296         {
00297         vtkMatrix4x4 *lastMatrix = path->GetLastNode()->GetMatrix();
00298         if (lastMatrix == NULL)
00299           {
00300           vtkErrorMacro (<< "Pick: Null matrix.");
00301           return 0;
00302           }
00303         this->Transform->SetMatrix(lastMatrix);
00304         this->Transform->Push();
00305         this->Transform->Inverse();
00306         this->Transform->GetScale(scale); //need to scale the tolerance
00307 
00308         this->Transform->TransformPoint(p1World,p1Mapper);
00309         this->Transform->TransformPoint(p2World,p2Mapper);
00310 
00311         for (i=0; i<3; i++) 
00312           {
00313           ray[i] = p2Mapper[i] - p1Mapper[i];
00314           }
00315 
00316         this->Transform->Pop();
00317 
00318         //  Have the ray endpoints in mapper space, now need to compare this
00319         //  with the mapper bounds to see whether intersection is possible.
00320         //
00321         //  Get the bounding box of the modeller.  Note that the tolerance is
00322         //  added to the bounding box to make sure things on the edge of the
00323         //  bounding box are picked correctly.
00324         mapper->GetBounds(bounds);
00325         bounds[0] -= tol; bounds[1] += tol; 
00326         bounds[2] -= tol; bounds[3] += tol; 
00327         bounds[4] -= tol; bounds[5] += tol; 
00328         if ( vtkBox::IntersectBox(bounds, p1Mapper, 
00329                                   ray, hitPosition, t) )
00330           {
00331           t = this->IntersectWithLine(p1Mapper, p2Mapper, 
00332                                       tol*0.333*(scale[0]+scale[1]+scale[2]), path, 
00333                                       static_cast<vtkProp3D *>(propCandidate), mapper);
00334           if ( t < VTK_DOUBLE_MAX )
00335             {
00336             picked = 1;
00337             if ( ! this->Prop3Ds->IsItemPresent(prop) )
00338               {
00339               this->Prop3Ds->AddItem(static_cast<vtkProp3D *>(prop));
00340               }
00341             this->PickedPositions->InsertNextPoint
00342               ((1.0 - t)*p1World[0] + t*p2World[0],
00343                (1.0 - t)*p1World[1] + t*p2World[1],
00344                (1.0 - t)*p1World[2] + t*p2World[2]);
00345 
00346             // backwards compatibility: also add to this->Actors
00347             if (actor)
00348               {
00349               this->Actors->AddItem(actor);
00350               }
00351             }
00352           }
00353         }
00354       else if ( pickable && imageActor )
00355         { // special case for imageActor, which has no mapper
00356         vtkMatrix4x4 *lastMatrix = path->GetLastNode()->GetMatrix();
00357         if (lastMatrix == NULL)
00358           {
00359           vtkErrorMacro (<< "Pick: Null matrix.");
00360           return 0;
00361           }
00362         this->Transform->SetMatrix(lastMatrix);
00363         this->Transform->Push();
00364         this->Transform->Inverse();
00365 
00366         this->Transform->TransformPoint(p1World,p1Mapper);
00367         this->Transform->TransformPoint(p2World,p2Mapper);
00368 
00369         this->Transform->Pop();
00370 
00371         //  Have the ray endpoints in data space, now need to compare this
00372         //  with the displayed image bounds.
00373         imageActor->GetDisplayBounds(bounds);
00374 
00375         t = VTK_DOUBLE_MAX;
00376 
00377         for (i = 0; i < 3; i++)
00378           {
00379           if (bounds[2*i] == bounds[2*i+1] && p2Mapper[i] != p1Mapper[i])
00380             {
00381 //            t = (p2World[i] - bounds[2*i])/(p2Mapper[i] - p1Mapper[i]);
00382             t = (p2Mapper[i] - bounds[2*i])/(p2Mapper[i] - p1Mapper[i]);
00383             break;
00384             }
00385           }
00386 
00387         if (t < VTK_DOUBLE_MAX)
00388           {
00389 //          hitPosition[0] = (1.0 - t)*p1Mapper[0] + t*p2Mapper[0];
00390 //          hitPosition[1] = (1.0 - t)*p1Mapper[1] + t*p2Mapper[1];
00391 //          hitPosition[2] = (1.0 - t)*p1Mapper[2] + t*p2Mapper[2];
00392             hitPosition[0] = t*p1Mapper[0] + (1.0 - t)*p2Mapper[0];
00393             hitPosition[1] = t*p1Mapper[1] + (1.0 - t)*p2Mapper[1];
00394             hitPosition[2] = t*p1Mapper[2] + (1.0 - t)*p2Mapper[2];
00395           if ((bounds[0] == bounds[1] || (hitPosition[0] >= bounds[0]-tol &&
00396                                           hitPosition[0] <= bounds[1]+tol)) &&
00397               (bounds[2] == bounds[3] || (hitPosition[1] >= bounds[2]-tol &&
00398                                           hitPosition[1] <= bounds[3]+tol)) &&
00399               (bounds[4] == bounds[5] || (hitPosition[2] >= bounds[4]-tol &&
00400                                           hitPosition[2] <= bounds[5]+tol)))
00401             {
00402             picked = 1;
00403 
00404             // the following code is handled in MarkPicked for other Prop3Ds
00405             this->Mapper = mapper; // mapper is null
00406             this->DataSet = imageActor->GetInput();
00407             this->MapperPosition[0] = hitPosition[0];
00408             this->MapperPosition[1] = hitPosition[1];
00409             this->MapperPosition[2] = hitPosition[2];
00410             this->Transform->TransformPoint(hitPosition,this->PickPosition);
00411             imageActor->Pick();
00412             this->InvokeEvent(vtkCommand::PickEvent,NULL);
00413 
00414             this->Prop3Ds->AddItem(imageActor);
00415 //            this->PickedPositions->InsertNextPoint
00416 //              ((1.0 - t)*p1World[0] + t*p2World[0],
00417 //               (1.0 - t)*p1World[1] + t*p2World[1],
00418 //               (1.0 - t)*p1World[2] + t*p2World[2]);
00419             this->PickedPositions->InsertNextPoint
00420                 (t*p1World[0] + (1.0 - t)*p2World[0],
00421                  t*p1World[1] + (1.0 - t)*p2World[1],
00422                  t*p1World[2] + (1.0 - t)*p2World[2]);
00423             }
00424           }
00425         }//if visible and pickable not transparent and has mapper
00426       }//for all parts
00427     }//for all actors
00428 
00429   // Invoke end pick method if defined
00430   this->InvokeEvent(vtkCommand::EndPickEvent,NULL);
00431 
00432   return picked;
00433 }
