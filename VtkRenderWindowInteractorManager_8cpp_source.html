00041  bool DropFilter::eventFilter(QObject *obj, QEvent *event)
00042  {
00043      if( event->type() == QEvent::DragEnter)
00044      {
00045          QDragEnterEvent *dragEvent=dynamic_cast< QDragEnterEvent* >(event);
00046          if (dragEvent->mimeData()->hasFormat("text/plain"))
00047          {
00048              QString data = dragEvent->mimeData()->text();
00049              dragEvent->acceptProposedAction();
00050          }
00051      }
00052      else if (event->type() == QEvent::Drop )
00053      {
00054          QDropEvent* dropEvent = dynamic_cast< QDropEvent* >(event);
00055          QString data = dropEvent->mimeData()->text();
00056          ::fwServices::ObjectMsg::sptr message = ::fwServices::ObjectMsg::New();
00057          message->addEvent("DROPPED_UUID", ::fwData::String::New(data.toStdString()));
00058          ::fwServices::IService::sptr service = m_service.lock();
00059          ::fwServices::IEditionService::notify( service, service->getObject< ::fwData::Object >(), message );
00060      } else {
00061          // standard event processing
00062          return QObject::eventFilter(obj, event);
00063      }
00064      return true;
00065  }
