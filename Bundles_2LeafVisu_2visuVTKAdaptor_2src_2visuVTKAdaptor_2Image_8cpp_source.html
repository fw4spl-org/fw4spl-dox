199 void Image::updateImageOpacity()
200 {
201  if (m_imagePortId>= 0)
202  {
203  ::fwData::Image::sptr img = this->getObject< ::fwData::Image >();
204  if(img->getField( "TRANSPARENCY" ) )
205  {
206  ::fwData::Integer::sptr transparency = img->getField< ::fwData::Integer >( "TRANSPARENCY" );
207  m_imageOpacity = (100 - (*transparency) ) / 100.0;
208  }
209  if(img->getField( "VISIBILITY" ) )
210  {
211  ::fwData::Boolean::sptr visible = img->getField< ::fwData::Boolean >( "VISIBILITY" );
212  m_imageOpacity = (*visible) ? m_imageOpacity : 0.0;
213  }
214  vtkImageBlend* imageBlend = vtkImageBlend::SafeDownCast(m_imageRegister);
215  imageBlend->SetOpacity(m_imagePortId, m_imageOpacity);
216  OSLM_TRACE(
217  "vtkImageBlend " << this->m_imageRegisterId << " opacity :" << m_imagePortId << "," << m_imageOpacity );
218  this->setVtkPipelineModified();
219  this->requestRender();
220  }
221 }

225 void Image::buildPipeline( )
226 {
227  SLM_TRACE_FUNC();
228  m_map2colors->SetInputData(m_imageData);
229  m_map2colors->SetLookupTable(m_lut);
230  m_map2colors->SetOutputFormatToRGBA();
231 
232  if (!m_imageRegisterId.empty())
233  {
234  m_imageRegister = this->getVtkObject(m_imageRegisterId);
235  }
236 
237  vtkImageAlgorithm* algorithm = vtkImageAlgorithm::SafeDownCast(m_imageRegister);
238  vtkImageData* imageData = vtkImageData::SafeDownCast(m_imageRegister);
239  vtkImageBlend* imageBlend = vtkImageBlend::SafeDownCast(m_imageRegister);
240 
241  SLM_ASSERT("Invalid vtk image register", algorithm||imageData||imageBlend );
242  if (imageBlend)
243  {
244  SLM_TRACE("Register is a vtkImageBlend");
245  if (m_imagePortId < 0)
246  {
247  m_imagePortId = imageBlend->GetNumberOfInputConnections(0);
248  imageBlend->AddInputConnection(m_map2colors->GetOutputPort());
249  OSLM_TRACE(this->getID() << ": Added image " << m_imagePortId << " on vtkImageBlend");
250  }
251  }
252  else if (algorithm)
253  {
254  SLM_TRACE("Register is a vtkImageAlgorithm");
255  algorithm->SetInputConnection(m_map2colors->GetOutputPort());
256  }
257  else if (imageData)
258  {
259  SLM_TRACE("Register is a vtkImageData");
260  m_map2colors->SetOutput(imageData);
261  m_map2colors->Update();
262  }
263 
264  this->setVtkPipelineModified();
265 }

269 void Image::destroyPipeline( )
270 {
271  vtkImageAlgorithm* algorithm = vtkImageAlgorithm::SafeDownCast(m_imageRegister);
272  vtkImageData* imageData = vtkImageData::SafeDownCast(m_imageRegister);
273  vtkImageBlend* imageBlend = vtkImageBlend::SafeDownCast(m_imageRegister);
274 
275  if (imageBlend)
276  {
277  if (m_imagePortId >= 0)
278  {
279  //Warning : only the removal of the last input connection in the image blend is safe.
280  imageBlend->RemoveInputConnection(0, m_map2colors->GetOutputPort());
281  m_imagePortId = -1;
282  }
283  }
284  else if (algorithm)
285  {
286  algorithm->RemoveInputConnection(0, m_map2colors->GetOutputPort());
287  }
288  else if (imageData)
289  {
290  m_map2colors->SetOutput(0);
291  }
292 
293  this->setVtkPipelineModified();
294 }
