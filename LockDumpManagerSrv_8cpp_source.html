51 void LockDumpManagerSrv::setDumpLockOnImages( LockMapType &lockMap, ::fwData::Composite::ContainerType objects,
52  bool _isLocked )
53 {
54  for( ::fwData::Composite::value_type objectId : objects )
55  {
56  if ( m_managedKeys.empty()
57  || std::find(m_managedKeys.begin(), m_managedKeys.end(), objectId.first) != m_managedKeys.end() )
58  {
59  if (_isLocked)
60  {
61  lockMap.insert( LockMapType::value_type( objectId.first, ::fwData::ObjectLock( objectId.second ) ) );
62  OSLM_TRACE("Locking Key : " << objectId.first);
63  }
64  else
65  {
66  SLM_ASSERT( "ERROR image not locked !!!", lockMap.find(objectId.first) != lockMap.end() );
67  lockMap.erase( objectId.first );
68  }
69  }
70  }
71 }

92  ::fwRuntime::ConfigurationElementContainer updaters = m_configuration->findAllConfigurationElement("manageLockOn");

94  m_managedKeys.clear();
