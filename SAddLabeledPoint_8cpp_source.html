85 void SAddLabeledPoint::updating() throw(::fwTools::Failed)
86 {
87  SLM_TRACE_FUNC();
88  ::fwData::PointList::sptr landmarks = this->getObject< ::fwData::PointList >();
89  SLM_ASSERT("landmarks not instanced", landmarks);
90 
91  std::string value;
92  if ( this->defineLabel(value) )
93  {
94  // create a new point
95  ::fwData::Point::sptr newPoint = ::fwData::Point::New();
96 
97  // append to landmark
98  landmarks->getRefPoints().push_back( newPoint );
99 
100  // append to point the label
101  ::fwData::String::sptr label = ::fwData::String::New();
102  label->value() = value;
103  newPoint->setField( ::fwDataTools::fieldHelper::Image::m_labelId, label );
104 
105  // notify
106  auto sig =
107  landmarks->signal< ::fwData::PointList::PointAddedSignalType >(::fwData::PointList::s_POINT_ADDED_SIG);
108  {
109  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
110  sig->asyncEmit(newPoint);
111  }
112  }
