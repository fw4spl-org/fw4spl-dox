00072 void CrossTypeAction::configuring() throw(fwTools::Failed)
00073 {
00074     SLM_TRACE_FUNC();
00075     ::fwGui::IActionSrv::initialize();
00076 
00077     if( this->m_configuration->size() > 0 )
00078     {
00079         std::vector < ConfigurationType > vectConfig = this->m_configuration->find("crossType");
00080         SLM_ASSERT("Missing <crossType> tag!", !vectConfig.empty());
00081         m_crossType = vectConfig.at(0)->getValue();
00082         ::boost::algorithm::trim(m_crossType);
00083         ::boost::algorithm::to_lower(m_crossType);
00084         SLM_ASSERT("Unknown crossType", m_scaleConversion.find(m_crossType) != m_scaleConversion.end());
00085     }
00086 }
