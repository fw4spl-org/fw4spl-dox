142 {
143  if ( this->getRenderService()->isStarted() && this->getRenderService()->isShownOnScreen()
144  && m_vtkPipelineModified && m_autoRender )

242 ::fwData::Object::sptr IVtkAdaptorService::getAssociatedObject(vtkProp* prop, int depth)
243 {
244  ::fwData::Object::sptr obj;
245 
246  if (prop)
247  {
248  if ( m_propCollection->IsItemPresent(prop) )
249  {
250  obj = this->getObject();
251  }
252  else
253  {
254  ::fwData::Object::sptr res;
255  for( ServiceVector::value_type service : m_subServices)
256  {
257  if(!service.expired())
258  {
259  res = service.lock()->getAssociatedObject(prop, depth - 1 );
260  if (res)
261  {
262  break;
263  }
264  }
265  }
266  obj = ( res && depth == 0 ) ? this->getObject() : res;
267  }
268  }
269  return obj;
270 }

319 void IVtkAdaptorService::getAllSubProps(vtkPropCollection* propc, int depth)
320 {
321  vtkProp* prop;
322 
323  m_propCollection->InitTraversal();
324  while ( (prop = m_propCollection->GetNextProp()) )
325  {
326  getProps(propc, prop);
327  }
328 
329  if(depth != 0)
330  {
331  for( ServiceVector::value_type service : m_subServices)
332  {
333  if(!service.expired())
334  {
335  service.lock()->getAllSubProps( propc, depth - 1 );
336  }
337  }
338  }
339 }
