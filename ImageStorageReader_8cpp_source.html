49 ::fwMedData::Series::sptr ImageStorageReader::read(::fwMedData::DicomSeries::sptr series)
50 {
51  ::fwMedData::DicomSeries::SOPClassUIDContainerType sopClassUIDContainer = series->getSOPClassUIDs();
52  std::string sopClassUID = dcmFindNameOfUID(sopClassUIDContainer.begin()->c_str());
53 
54  ::fwMedData::ImageSeries::sptr imageSeries = ::fwDicomTools::Series::convertToImageSeries(series);
55  DicomPathContainerType instances = series->getLocalDicomPaths();
56 
57  ::fwData::Image::sptr image = ::fwData::Image::New();
58  DcmFileFormat fileFormat;
59  OFCondition status;
60  DcmDataset* dataset;
61 
62  //Get informations from the first instance
63  std::string firstInstance = instances.begin()->second.string();
64  status = fileFormat.loadFile(firstInstance.c_str());
65  DicomImage dicomImage(firstInstance.c_str());
66 
67  FW_RAISE_IF("Unable to read the file: \""+firstInstance+"\"", status.bad() || (
68  dicomImage.getStatus() != EIS_Normal
69  && dicomImage.getStatus() != EIS_MissingAttribute
70  && dicomImage.getStatus() != EIS_NotSupportedValue
71  ));
72  dataset = fileFormat.getDataset();
73 
74  // Decompress data set if compressed
75  dataset->chooseRepresentation(EXS_LittleEndianExplicit, NULL);
76 
77  if(dicomImage.getStatus() != EIS_MissingAttribute)
78  {
79  SLM_WARN("Some informations are missing. The file may have not been read properly.");
80  }
81 
82  //Spacing
83  double spacing[3];
84  dataset->findAndGetFloat64(DCM_PixelSpacing,spacing[0],0);
85  dataset->findAndGetFloat64(DCM_PixelSpacing,spacing[1],1);
86  dataset->findAndGetFloat64(DCM_SliceThickness,spacing[2]);
87 
88  if(series->hasComputedValues("SliceThickness"))
89  {
90  spacing[2] = ::boost::lexical_cast< double >(series->getComputedTagValues().at("SliceThickness"));
91  }
92 
93  if(spacing[0] == 0 || spacing[1] == 0 || spacing[2] == 0)
94  {
95  spacing[0] = spacing[1] = spacing[2] = 1;
96  OSLM_WARN("Invalid value for pixel spacing. Assuming pixel value is 1.");
97  }
98 
99 
100  image->setSpacing(std::vector< double >(spacing, spacing+3));
101 
102  //Origin
103  //TODO: Compute the correct origin
104  double imagePosition[3];
105  dataset->findAndGetFloat64(DCM_ImagePositionPatient, imagePosition[0], 0);
106  dataset->findAndGetFloat64(DCM_ImagePositionPatient, imagePosition[1], 1);
107  dataset->findAndGetFloat64(DCM_ImagePositionPatient, imagePosition[2], 2);
108  image->setOrigin(::boost::assign::list_of(imagePosition[0])(imagePosition[1])(imagePosition[2]));
109 
110  //Size
111  unsigned short rows, columns;
112  dataset->findAndGetUint16(DCM_Rows,rows);
113  dataset->findAndGetUint16(DCM_Columns,columns);
114 
115  uint32_t depth;
116  if(instances.size() == 1)
117  {
118  OFString sframesNumber = "";
119 
120  if(dataset->findAndGetOFString(DCM_NumberOfFrames, sframesNumber).good())
121  {
122  depth = static_cast<uint32_t>(std::stoi(sframesNumber.c_str()));
123  }
124  else
125  {
126  depth = 1;
127  }
128  }
129  else
130  {
131  depth = static_cast<unsigned short>(instances.size());
132  }
133 
134  //FIXME: Remove depth for 2D images ?
135  image->setSize(::boost::assign::list_of(columns)(rows)(depth));
136 
137  //Window Center
138  double windowCenter = 0;
139  dataset->findAndGetFloat64(DCM_WindowCenter,windowCenter);
140  image->setWindowCenter(windowCenter);
141 
142  //Window Width
143  double windowWidth = 0;
144  dataset->findAndGetFloat64(DCM_WindowWidth,windowWidth);
145  image->setWindowWidth(windowWidth);
146 
147  //Number of components
148  OFString data;
149  dataset->findAndGetOFStringArray(DCM_PhotometricInterpretation, data);
150  std::string photometricInterpretation = data.c_str();
151  dataset->findAndGetOFStringArray(DCM_PixelPresentation, data);
152  std::string pixelPresentation = data.c_str();
153 
154  if(photometricInterpretation == "MONOCHROME2")
155  {
156  image->setNumberOfComponents(1);
157  }
158  else if(photometricInterpretation == "RGB" || photometricInterpretation == "YBR")
159  {
160  image->setNumberOfComponents(3);
161  }
162  else if(photometricInterpretation == "ARGB" || photometricInterpretation == "CMYK")
163  {
164  image->setNumberOfComponents(4);
165  }
166  else if(photometricInterpretation == "PALETTE COLOR" || pixelPresentation == "COLOR")
167  {
168  image->setNumberOfComponents(3);
169  }
170  else
171  {
172  FW_RAISE ( "The photometric interpretation \"" << photometricInterpretation << "\" is not supported.");
173  }
174 
175  //Rescale Slope
176  double rescaleSlope;
177  double rescaleIntercept;
178  status = dataset->findAndGetFloat64(DCM_RescaleSlope,rescaleSlope);
179  rescaleSlope = (status.bad()) ? 1 : rescaleSlope;
180  status = dataset->findAndGetFloat64(DCM_RescaleIntercept,rescaleIntercept);
181  rescaleIntercept = (status.bad()) ? 0 : rescaleIntercept;
182 
183  //Type
184  unsigned short samplesPerPixel = 1;
185  unsigned short bitsAllocated = 8;
186  unsigned short bitsStored = 8;
187  unsigned short highBit = 7;
188  unsigned short pixelRepresentation = 0;
189 
190  dataset->findAndGetUint16(DCM_SamplesPerPixel,samplesPerPixel);
191  dataset->findAndGetUint16(DCM_BitsAllocated,bitsAllocated);
192  dataset->findAndGetUint16(DCM_BitsStored,bitsStored);
193  dataset->findAndGetUint16(DCM_HighBit,highBit);
194  dataset->findAndGetUint16(DCM_PixelRepresentation,pixelRepresentation);
195 
196  //Using lookup tables
197  if(photometricInterpretation == "COLOR" || photometricInterpretation == "PALETTE COLOR")
198  {
199  unsigned short colorBitsAllocated = 0;
200  dataset->findAndGetUint16(DCM_RedPaletteColorLookupTableDescriptor, colorBitsAllocated, 2);
201  bitsStored = bitsAllocated = colorBitsAllocated;
202  highBit = static_cast<unsigned short>(colorBitsAllocated-1);
203  }
204 
205  //Find image type
206  ::fwDicomTools::Image imageHelper(
207  samplesPerPixel,bitsAllocated,bitsStored, highBit, pixelRepresentation, rescaleSlope, rescaleIntercept);
208  ::fwTools::Type imageType = imageHelper.findImageTypeFromMinMaxValues();
209 
210  //Set image type
211  image->setType(imageType);
212 
213 
214  //Direct reading mode
215  if(::fwMemory::BufferManager::getDefault()->getLoadingMode() == ::fwMemory::BufferManager::DIRECT)
216  {
217  SLM_INFO("Reading using DIRECT mode.");
218 
219  //Default read
220  if(photometricInterpretation != "PALETTE COLOR" && pixelPresentation != "COLOR")
221  {
222  this->directRead(image, instances, rows, columns, depth, rescaleSlope, rescaleIntercept,
223  pixelRepresentation, imageType);
224  }
225  //RGB lookup read
226  else
227  {
228  this->directRGBLookupRead(image, *dataset, instances, rows, columns, depth, bitsAllocated);
229  }
230 
231  }
232  //Lazy reading mode
233  else
234  {
235  SLM_INFO("Reading using LAZY mode.");
236 
237  //Default read
238  if(photometricInterpretation != "PALETTE COLOR" && pixelPresentation != "COLOR")
239  {
240  this->lazyRead(image, series, rows, columns, depth, rescaleSlope, rescaleIntercept, pixelRepresentation,
241  imageType);
242  }
243  //RGB lookup read
244  else
245  {
246  this->lazyRGBLookupRead(image, series, *dataset, instances, rows, columns, depth, bitsAllocated, imageType);
247  }
248  }
249 
250  // Add the image to the series
251  imageSeries->setImage(image);
252 
253  return imageSeries;
254 }

276 void ImageStorageReader::directRGBLookupRead(::fwData::Image::sptr image, DcmDataset& dataset,
277  DicomPathContainerType instances, unsigned short rows,
278  unsigned short columns, int depth,
279  unsigned short bitsAllocated)
280 {
281  //Allocate image
282  image->allocate();
283  ::fwData::Array::sptr array = image->getDataArray();
284  ::fwDataTools::helper::Array arrayHelper(array);
285 
286  unsigned short pixelValueBitsAllocated = 8;
287  dataset.findAndGetUint16(DCM_BitsAllocated,pixelValueBitsAllocated);
288 
289  // 16 bits allocated
290  if(bitsAllocated == 16)
291  {
292  const Uint16* redLookup;
293  const Uint16* greenLookup;
294  const Uint16* blueLookup;
295  // Those values are freed by the dataset destructor
296  dataset.findAndGetUint16Array(DCM_RedPaletteColorLookupTableData, redLookup);
297  dataset.findAndGetUint16Array(DCM_GreenPaletteColorLookupTableData, greenLookup);
298  dataset.findAndGetUint16Array(DCM_BluePaletteColorLookupTableData, blueLookup);
299 
300  if(pixelValueBitsAllocated == 16)
301  {
302  ::fwDcmtkIO::reader::rgblookup::ImageRGBLookupReader::fillImageBuffer<Uint16, Uint16>(rows,
303  columns, depth,
304  instances,
305  arrayHelper.getBuffer(), redLookup, greenLookup,
306  blueLookup);
307  }
308  else
309  {
310  ::fwDcmtkIO::reader::rgblookup::ImageRGBLookupReader::fillImageBuffer<Uint16, Uint8>(rows,
311  columns, depth,
312  instances,
313  arrayHelper.getBuffer(), redLookup, greenLookup,
314  blueLookup);
315  }
316 
317  }
318  // 8 bits allocated
319  else
320  {
321  const Uint8* redLookup;
322  const Uint8* greenLookup;
323  const Uint8* blueLookup;
324  // Those values are freed by the dataset destructor
325  dataset.findAndGetUint8Array(DCM_RedPaletteColorLookupTableData, redLookup);
326  dataset.findAndGetUint8Array(DCM_GreenPaletteColorLookupTableData, greenLookup);
327  dataset.findAndGetUint8Array(DCM_BluePaletteColorLookupTableData, blueLookup);
328 
329  if(pixelValueBitsAllocated == 16)
330  {
331  ::fwDcmtkIO::reader::rgblookup::ImageRGBLookupReader::fillImageBuffer<Uint8, Uint16>(rows,
332  columns, depth,
333  instances,
334  arrayHelper.getBuffer(), redLookup, greenLookup,
335  blueLookup);
336  }
337  else
338  {
339  ::fwDcmtkIO::reader::rgblookup::ImageRGBLookupReader::fillImageBuffer<Uint8, Uint8>(rows,
340  columns, depth,
341  instances,
342  arrayHelper.getBuffer(), redLookup, greenLookup,
343  blueLookup);
344  }
345  }
