36  ::fwServices::IService::sptr prefSrv;

40 
41  try

44  ::boost::filesystem::path prefFile = ::fwPreferences::getPreferencesFile();
45  prefConfig->setValue(prefFile.string());

47  prefSrv->setConfiguration(prefConfig);
