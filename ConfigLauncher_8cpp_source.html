00038 void ConfigLauncher::parseConfig(const ::fwServices::IService::ConfigType& config)
00039 {
00040     if(config.get_child("service").count("config") > 0)
00041     {
00042         SLM_ASSERT("Sorry you must have one (and only one) <config/> element.",
00043                 config.get_child("service").count("config") == 1 );
00044         const ::fwServices::IService::ConfigType srvconfig = config.get_child("service");
00045         const ::fwServices::IService::ConfigType &config = srvconfig.get_child("config");
00046 
00047         if(config.count("appConfig") == 1 )
00048         {
00049             const ::fwServices::IService::ConfigType &appConfig = config.get_child("appConfig");
00050             m_appConfig = ::fwActivities::registry::ActivityAppConfig(appConfig);
00051         }
00052         OSLM_ASSERT("At most 1 <appConfig> tag is allowed", config.count("appConfig") < 2);
00053     }

00058 void ConfigLauncher::startConfig(::fwServices::IService::sptr srv)
00059 {
00060     ::fwData::Object::sptr currentObj = srv->getObject();
00061     typedef ::fwActivities::registry::ActivityAppConfig AppConfig;
00062     ::fwServices::registry::AppConfig::FieldAdaptorType replaceMap;
00063 
00064     // Generate generic UID
00065     const std::string genericUidAdaptor = ::fwServices::registry::AppConfig::getUniqueIdentifier( srv->getID() );
00066     replaceMap[ConfigLauncher::s_GENERIC_UID_KEY] = genericUidAdaptor;
00067 
00068     BOOST_FOREACH(const AppConfig::ActivityAppConfigParamsType::value_type& param, m_appConfig.parameters)
00069     {
00070         if(!param.isSeshat())
00071         {
00072             std::string by = param.by;
00073             if(by == ConfigLauncher::s_SELF_KEY)
00074             {
00075                 by = currentObj->getID();
00076             }
00077             replaceMap[param.replace] = by;
00078         }
00079         else
00080         {
00081             std::string parameterToReplace = param.by;
00082             if (parameterToReplace.substr(0,1) == "!")
00083             {
00084                 parameterToReplace.replace(0, 1, "@");
00085             }
00086 
00087             ::fwData::Object::sptr obj = ::fwDataCamp::getObject(currentObj, param.by);
00088             OSLM_ASSERT("Invalid seshat path : '"<<param.by<<"'", obj);
00089             ::fwData::String::sptr stringParameter = ::fwData::String::dynamicCast(obj);
00090 
00091             std::string parameterValue = obj->getID();
00092 
00093             if(stringParameter && param.by.substr(0,1) == "!")
00094             {
00095                 parameterValue = stringParameter->getValue();
00096             }
00097             replaceMap[param.replace] = parameterValue;
00098         }
00099     }
00100 
00101     // Init manager
00102     ::fwRuntime::ConfigurationElement::csptr config =
00103             ::fwServices::registry::AppConfig::getDefault()->getAdaptedTemplateConfig( m_appConfig.id, replaceMap );
00104     m_appConfigManager = ::fwServices::AppConfigManager::New();
00105     m_appConfigManager->setConfig( config );
00106 
00107     // Launch config
00108     m_appConfigManager->launch();
00109 
00110     // Add connection
00111     this->connectToConfigRoot(srv);
00112 
00113     m_configIsRunning = true;
00114 }
