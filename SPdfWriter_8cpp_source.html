62 void SPdfWriter::configuring() throw(::fwTools::Failed)
63 {
64  this->::io::IWriter::configuring();
65  if(!this->isVersion2())
66  {
67  ::fwRuntime::ConfigurationElementContainer imagesConfig = m_configuration->findAllConfigurationElement(
68  s_IMAGE_INPUT);
69 
71  for(const auto& imageCfg : imagesConfig.getElements())
72  {
73  const std::string id = imageCfg->getValue();
74  m_imagesUIDs.push_back( id );
75  }
76  ::fwRuntime::ConfigurationElementContainer containersConfig = m_configuration->findAllConfigurationElement(
77  s_CONTAINER_INPUT);
79  for(const auto& containerCfg : containersConfig.getElements())
80  {
81  const std::string id = containerCfg->getValue();
82  m_containersIDs.push_back( id );
83  }
84  }
85  else
86  {
87  typedef ::fwRuntime::ConfigurationElement::sptr ConfigurationType;
88  const ConfigurationType containersConfig = m_configuration->findConfigurationElement("container");
89  if (containersConfig)
90  {
91  const std::vector< ConfigurationType > containersCfg = containersConfig->find(s_CONTAINER_INPUT);
92  for (const auto& cfg : containersCfg)
93  {
94  SLM_ASSERT("Missing attribute 'uid'.", cfg->hasAttribute("uid"));
95  std::string id = cfg->getAttributeValue("uid");
96  m_containersIDs.push_back( id );
97  }
98  }
99  }
100 }

131 void SPdfWriter::updating() throw(::fwTools::Failed)
132 {
133  if( !this->hasLocationDefined() )
134  {
135  configureWithIHM();
136  }
137  if( this->hasLocationDefined() )
138  {
139  QPdfWriter pdfWriter( this->getLocations().front().string().c_str() );
140  QPainter painter( &pdfWriter );
141  pdfWriter.setPageSize( QPagedPaintDevice::A4 );
142 
143  // Scale value to fit the images to a PDF page
144  const int scale = static_cast<const int>(pdfWriter.logicalDpiX() * 8);
145  ::fwThread::Pool& pool = ::fwThread::getDefaultPool();
146 
147  // Adding fwImage from generic scene to the list of images to scale
148  ImagesScaledListType imagesToScale;
149  std::vector< ::std::shared_future< QImage > > futuresQImage;
150  for( const ::fwData::Image::sptr& fwImage : m_imagesToExport )
151  {
152  std::shared_future< QImage > future;
153  future = pool.post(&SPdfWriter::convertFwImageToQImage, fwImage);
154  futuresQImage.push_back( future );
155  }
156  std::for_each(futuresQImage.begin(), futuresQImage.end(), std::mem_fn(&::std::shared_future< QImage >::wait));
157  for (auto& future : futuresQImage)
158  {
159  QImage imageToDraw = future.get();
160  imagesToScale.push_back(imageToDraw);
161  }
162 
163  // Adding QImage from Qt containers to the list of images to scale
164  for( QWidget*& qtContainer : m_containersToExport )
165  {
166  QImage imageToDraw = qtContainer->grab().toImage();
167  imagesToScale.push_back(imageToDraw);
168  }
169 
170  // Scales images to fit the A4 format
171  std::vector< ::std::shared_future< void > > futures;
172  const size_t sizeImagesToScale = imagesToScale.size();
173  for( size_t idx = 0; idx < sizeImagesToScale; ++idx )
174  {
175  std::shared_future<void> future;
176  future = pool.post(&SPdfWriter::scaleQImage, std::ref(imagesToScale[idx]), scale);
177  futures.push_back( future );
178  }
179  std::for_each(futures.begin(), futures.end(), std::mem_fn(&::std::shared_future<void>::wait));
180 
181  // Draws images onto the PDF.
182  for( QImage& qImage : imagesToScale )
183  {
184  if ( pdfWriter.newPage() )
185  {
186  pdfWriter.setPageSize( QPagedPaintDevice::A4 );
187  if ( !qImage.isNull() && qImage.bits() != nullptr )
188  {
189  painter.drawImage( 0, 0, qImage);
190  }
191  }
192  }
193  painter.end();
194  }
195 }

199 void SPdfWriter::starting() throw(::fwTools::Failed)
200 {
201  if (!this->isVersion2())
202  {
203  for(const std::string& id : m_imagesUIDs)
204  {
205  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(id);
206  SLM_ASSERT("Object '" + id + "' is not found", obj);
207  ::fwData::Image::sptr image = ::fwData::Image::dynamicCast(obj);
208  SLM_ASSERT("Object '" + id + "' is not an ::fwData::Image (" + obj->getClassname() + ")",
209  image);
210  m_imagesToExport.push_back(image);
211  }
212  }
213  else
214  {
215  const size_t groupImageSize = this->getKeyGroupSize(s_IMAGE_INPUT);
216  for (size_t idxImage = 0; idxImage < groupImageSize; ++idxImage)
217  {
218  ::fwData::Image::sptr image = this->getInOut< ::fwData::Image >(s_IMAGE_INPUT, idxImage);
219  m_imagesToExport.push_back(image);
220  }
221  }
222  for(const auto& id : m_containersIDs)
223  {
224  ::fwGuiQt::container::QtContainer::sptr containerElt;
225  ::fwGui::container::fwContainer::sptr fwContainerFromConfig;
226  if ( ::fwGui::GuiRegistry::hasSIDContainer( id ) )
227  {
228  fwContainerFromConfig = ::fwGui::GuiRegistry::getSIDContainer( id );
229  }
230  else
231  {
232  fwContainerFromConfig = ::fwGui::GuiRegistry::getWIDContainer( id );
233  }
234  if (fwContainerFromConfig)
235  {
236  containerElt = ::fwGuiQt::container::QtContainer::dynamicCast(fwContainerFromConfig);
237  m_containersToExport.push_back(containerElt->getQtContainer());
238  }
239  }
240 }

271 QImage SPdfWriter::convertFwImageToQImage(::fwData::Image::sptr fwImage)
272 {
273  if (fwImage->getNumberOfComponents() == 3
274  && fwImage->getType().string() == "uint8"
275  && fwImage->getSize()[2] == 1)
276  {
277  // Initialize QImage parameters
278  const ::fwData::Image::SizeType dimension = fwImage->getSize();
279  const int width = static_cast<int>(dimension[0]);
280  const int height = static_cast<int>(dimension[1]);
281 
282  QImage qImage(width, height, QImage::Format_ARGB32);
283  ::boost::uint8_t* qImageBuffer = qImage.bits();
284 
285  ::fwDataTools::helper::Image imgHelper(fwImage);
286  const ::boost::uint8_t* fwImageBuffer = reinterpret_cast< const ::boost::uint8_t*>( imgHelper.getBuffer() );
287 
288  const unsigned int size = static_cast<unsigned int>( width * height) * 4;
289  for(unsigned int idx = 0; idx < size; idx += 4)
290  {
291  qImageBuffer[idx+3] = (255 & 0xFF);
292  qImageBuffer[idx+2] = (*fwImageBuffer++ & 0xFF);
293  qImageBuffer[idx+1] = (*fwImageBuffer++ & 0xFF);
294  qImageBuffer[idx+0] = (*fwImageBuffer++ & 0xFF);
295  }
296  return qImage.mirrored(0, 1);
297  }
298  return QImage();
299 }
