113 void HistogramValue::doUpdate() throw( ::fwTools::Failed)
114 {
115  this->initializeViewSize();
116  this->initializeViewportSize();
117 
118  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
119  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
120  const float histogramMinValue = histogram->getMinValue();
121  const float histogramBinsWidth = histogram->getBinsWidth();
122 
123  // Event coordinates in scene
124  ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( m_coord );
125 
126  int histIndex = (int) sceneCoord.getX();
127  int index = (histIndex - histogramMinValue) / histogramBinsWidth;
128 
129  if(index >= 0 && index < (int)values.size() && m_isInteracting) // avoid std out_of_range on Windows
130  {
131  ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
132  const double viewportHeight = viewport->getHeight();
133  const double viewportWidth = viewport->getWidth();
134 
135  const double viewportSizeRatio = viewportHeight / viewportWidth;
136  const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
137 
138  Scene2DRatio ratio = this->getRatio(); // Total ratio
139  double viewportWidthRatio = this->getViewportSizeRatio().first;
140 
141  double diameterH = m_fontSize;
142  double diameterV = m_fontSize * viewportSizeRatio;
143 
144  diameterV /= viewportWidthRatio;
145  diameterV *= viewInitialSizeRatio;
146 
147  diameterH *= ratio.first;
148  diameterV *= ratio.second;
149 
150  m_text->setText( QString::number( histIndex ) );
151 
152  double scaleX = m_fontSize;
153  double scaleY = m_fontSize * viewportSizeRatio;
154 
155  scaleY /= viewportWidthRatio;
156  scaleY *= viewInitialSizeRatio;
157 
158  scaleX = scaleX * ratio.first;
159  scaleY = scaleY * ratio.second;
160 
161  QTransform transform;
162  transform.scale(scaleX, scaleY);
163 
164  ::fwData::Point::sptr point = ::fwData::Point::dynamicCast( ::fwTools::fwID::getObject( m_histogramPointUID ) );
165 
166  m_text->setTransform( transform );
167  m_text->setPos( point->getRefCoord()[0] + diameterH * 2, point->getRefCoord()[1] - diameterV * 2 );
168  m_text->setVisible( true );
169  }
170  else
171  {
172  m_text->setVisible( false );
173  }
174 }

184 void HistogramValue::processInteraction( ::scene2D::data::Event::sptr _event )
185 {
186  this->initializeViewSize();
187  this->initializeViewportSize();
188 
189  if( _event->getType() == ::scene2D::data::Event::MouseMove )
190  {
191  m_coord = _event->getCoord();
192  }
193  else if( _event->getType() == ::scene2D::data::Event::MouseButtonPress )
194  {
195  m_isInteracting = true;
196  }
197  else if( _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
198  {
199  m_isInteracting = false;
200  }
201 
202  doUpdate();
203 }
