00118 void HistogramValue::doUpdate() throw( ::fwTools::Failed)
00119 {
00120     SLM_TRACE_FUNC();
00121 
00122     ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
00123     ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
00124     const float histogramMinValue = histogram->getMinValue();
00125     const float histogramBinsWidth = histogram->getBinsWidth();
00126 
00127     // Event coordinates in scene
00128     ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( m_coord );
00129 
00130     int histIndex = (int) sceneCoord.getX();
00131     int index = (histIndex - histogramMinValue) / histogramBinsWidth;
00132 
00133     if(index >= 0 && index < (int)values.size() && m_isInteracting) // avoid std out_of_range on Windows
00134     {
00135         ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
00136         const double viewportHeight = viewport->getHeight();
00137         const double viewportWidth = viewport->getWidth();
00138 
00139         const double viewportSizeRatio = viewportHeight / viewportWidth;
00140         const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
00141 
00142         Scene2DRatio ratio = this->getRatio();  // Total ratio
00143         double viewportWidthRatio = this->getViewportSizeRatio().first;
00144 
00145         double diameterH  = m_fontSize;
00146         double diameterV  = m_fontSize * viewportSizeRatio;
00147 
00148         diameterV /= viewportWidthRatio;
00149         diameterV *= viewInitialSizeRatio;
00150 
00151         diameterH *= ratio.first;
00152         diameterV *= ratio.second;
00153 
00154         m_text->setText( QString::number( histIndex ) );
00155 
00156         double scaleX = m_fontSize;
00157         double scaleY = m_fontSize * viewportSizeRatio;
00158 
00159         scaleY /= viewportWidthRatio;
00160         scaleY *= viewInitialSizeRatio;
00161 
00162         scaleX = scaleX * ratio.first;
00163         scaleY = scaleY * ratio.second;
00164 
00165         QTransform transform;
00166         transform.scale(scaleX, scaleY);
00167 
00168         ::fwData::Point::sptr point = ::fwData::Point::dynamicCast( ::fwTools::fwID::getObject( m_histogramPointUID ) );
00169 
00170         m_text->setTransform( transform );
00171         m_text->setPos( point->getRefCoord()[0] + diameterH * 2, point->getRefCoord()[1] - diameterV * 2 );
00172         m_text->setVisible( true );
00173     }
00174     else
00175     {
00176         m_text->setVisible( false );
00177     }
00178 }
