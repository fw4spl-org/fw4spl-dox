73 int Plugin::run() throw()
74 {
75  ::fwData::Array::sptr array = ::fwData::Array::New();
76 
77  ::fwServices::registry::ServiceFactory::sptr serviceFactory = ::fwServices::registry::ServiceFactory::getDefault();
78 
79  // Create the 3 services to read, show and increment the values.
80  ::fwServices::IService::sptr srvRead =
81  serviceFactory->create("::fwServices::IController", "::Tuto15MultithreadCtrl::SReadArray");
82  ::fwServices::IService::sptr srvShow =
83  serviceFactory->create("::fwServices::IController", "::Tuto15MultithreadCtrl::SShowArray");
84  ::fwServices::IService::sptr srvIncrement =
85  serviceFactory->create("::fwServices::IController", "::Tuto15MultithreadCtrl::SIncrementArray");
86 
87  // Register the services
88  ::fwServices::OSR::registerService(array, srvRead);
89  ::fwServices::OSR::registerService(array, srvShow);
90  ::fwServices::OSR::registerService(array, srvIncrement);
91 
92  SLM_ASSERT("Failed to create service", srvRead);
93  SLM_ASSERT("Failed to create service", srvShow);
94  SLM_ASSERT("Failed to create service", srvIncrement);
95 
96  // Create 3 workers for each service
97  ::fwThread::Worker::sptr worker1 = ::fwThread::Worker::New();
98  ::fwThread::Worker::sptr worker2 = ::fwThread::Worker::New();
99  ::fwThread::Worker::sptr worker3 = ::fwThread::Worker::New();
100 
101  // Associate the workers to the services
102  srvRead->setWorker(worker1);
103  srvShow->setWorker(worker2);
104  srvIncrement->setWorker(worker3);
105 
106  // Connect the array "modified" signal to the services show and increment.
107  auto sig = array->signal< ::fwData::Object::ModifiedSignalType>( ::fwData::Object::s_MODIFIED_SIG );
108 
109  ::fwCom::Connection showConnection = sig->connect(srvShow->slot( ::fwServices::IService::s_UPDATE_SLOT) );
110  ::fwCom::Connection incrementConnection = sig->connect(srvIncrement->slot( "startTimer" ) );
111 
112  // Start the services
113  srvRead->start().wait();
114  srvShow->start().wait();
115  srvIncrement->start().wait();
116 
117  // Update the reader
118  srvRead->update().wait();
119 
120  unsigned long long count = 1<<30;
121  double d = 4<<2;
122 
123  OSLM_INFO("Computing " << count << " square roots.");
124 
125  for (unsigned long long i = 0; i < count; ++i)
126  {
127  d = std::sqrt(d);
128  }
129 
130  OSLM_INFO("Done computing " << count << " square roots : " << d);
131 
132 
133  // Disconnect the signals and slots
134  showConnection.disconnect();
135  incrementConnection.disconnect();
136 
137  // Stop the services
138  srvRead->stop().wait();
139  srvShow->stop().wait();
140  srvIncrement->stop().wait();
141 
142  // Unregister the services
143  ::fwServices::OSR::unregisterService(srvRead);
144  ::fwServices::OSR::unregisterService(srvShow);
145  ::fwServices::OSR::unregisterService(srvIncrement);
146 
147  srvRead.reset();
148  srvShow.reset();
149  srvIncrement.reset();
150 
151  // Stop the workers
152  worker1->stop();
153  worker2->stop();
154  worker3->stop();
155 
156  return 0;
157 }
