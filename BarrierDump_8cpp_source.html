00137 size_t BarrierDump::dump(size_t nbOfBytes)
00138 {
00139     size_t dumped = 0;
00140 
00141     ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
00142     if(manager)
00143     {
00144         const ::fwMemory::BufferManager::BufferInfoMapType bufferInfos = manager->getBufferInfos().get();
00145 
00146         typedef std::pair<
00147             ::fwMemory::BufferManager::BufferInfoMapType::key_type,
00148             ::fwMemory::BufferManager::BufferInfoMapType::mapped_type
00149                 > BufferInfosPairType;
00150         typedef std::vector< BufferInfosPairType > BufferVectorType;
00151 
00152         BufferVectorType buffers;
00153 
00154         BOOST_FOREACH(const ::fwMemory::BufferManager::BufferInfoMapType::value_type &elt, bufferInfos)
00155         {
00156             const ::fwMemory::BufferInfo &info = elt.second;
00157             if( ! ( info.size == 0 || info.lockCount() > 0 || !info.loaded )  )
00158             {
00159                 buffers.push_back(elt);
00160             }
00161         }
00162 
00163 
00164         BOOST_FOREACH(const BufferVectorType::value_type &pair, bufferInfos)
00165         {
00166             if(dumped < nbOfBytes)
00167             {
00168                 if( manager->dumpBuffer(pair.first).get() )
00169                 {
00170                     dumped += pair.second.size;
00171                 }
00172             }
00173             else
00174             {
00175                 break;
00176             }
00177         }
00178     }
00179 
00180     return dumped;
00181 }
