83 void ShowHelpContents::configuring() throw(::fwTools::Failed)
84 {
85  /*
86  * .qhp/.qch (source/binary) : Contains a table of contents,
87  * an index of items in the documentation, and a file manifest.
88  * .qhcp/.qhc (source/binary): Contains information that is used to customize
89  * the appearance and available features of Qt Assistant.
90  */
91  this->::fwGui::IActionSrv::initialize();
92  if( m_configuration->findConfigurationElement("filename") )
93  {
94  std::string filename = m_configuration->findConfigurationElement("filename")->getExistingAttributeValue("id");
95  m_fsHelpPath = ::boost::filesystem::path( filename );
96  m_bServiceIsConfigured = ::boost::filesystem::exists(m_fsHelpPath);
97  OSLM_WARN_IF("Help file " <<filename<< " doesn't exist", !m_bServiceIsConfigured);
98  OSLM_TRACE("Filename found " << filename );
99  }
100 }

104 void ShowHelpContents::updating() throw(::fwTools::Failed)
105 {
106  SLM_TRACE_FUNC();
107  SLM_ASSERT("The Help service isn't configured properly.", m_bServiceIsConfigured);
108 
109  QDialog* dialog = new QDialog(qApp->activeWindow());
110  dialog->setWindowTitle(QString("Help"));
111  QHelpEngine* helpEngine = new QHelpEngine(QString::fromStdString(m_fsHelpPath.string()), dialog);
112  if (!helpEngine->setupData())
113  {
114  OSLM_ERROR("HelpEngine error: " << helpEngine->error().toStdString());
115  ::fwGui::dialog::MessageDialog messageBox;
116  messageBox.setTitle("Warning");
117  messageBox.setMessage( "Help file is missing or not correct." );
118  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
119  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
120  messageBox.show();
121  // Setup help engine information failed.
122  // qhc (Qt Help Collection) or qch (Qt Compressed Help) file is not correct.
123  }
124  else
125  {
126  QSplitter *helpPanel = new QSplitter(Qt::Horizontal);
127  HelpBrowser *helpBrowser = new HelpBrowser(helpEngine, dialog);
128  helpPanel->insertWidget(0, helpEngine->contentWidget());
129  helpPanel->insertWidget(1, helpBrowser);
130  helpPanel->setStretchFactor(1, 1);
131 
132  QHBoxLayout *hLayout = new QHBoxLayout();
133  hLayout->addWidget(helpPanel);
134  dialog->setLayout( hLayout );
135  QObject::connect(helpEngine->contentWidget(), SIGNAL(linkActivated(const QUrl &)), helpBrowser,
136  SLOT(setSource(const QUrl &)));
137 
138  dialog->exec();
139  }
140 }
