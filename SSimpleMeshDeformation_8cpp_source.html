72 void SSimpleMeshDeformation::updating() throw(fwTools::Failed)
73 {
74  SLM_TRACE_FUNC();
75  ::fwData::Mesh::sptr mesh = this->getObject< ::fwData::Mesh >();
76 
77  ::fwData::mt::ObjectReadToWriteLock lock(mesh);
78  if ( mesh->getNumberOfPoints() > 0 )
79  {
80  m_hiRestimer.reset();
81  m_hiRestimer.start();
82  this->computeDeformation(m_mesh,m_transformMesh);
83  m_hiRestimer.stop();
84  OSLM_INFO("Deformation time (milli sec) = " << m_hiRestimer.getElapsedTimeInMilliSec());
85 
86  lock.upgrade();
87  m_hiRestimer.reset();
88  m_hiRestimer.start();
89  copyMesh(m_transformMesh,mesh);
90  m_hiRestimer.stop();
91  OSLM_INFO("Copy time (milli sec) = " << m_hiRestimer.getElapsedTimeInMilliSec());
92  lock.downgrade();
93 
94  ::fwData::Mesh::VertexModifiedSignalType::sptr sig;
95  sig = mesh->signal< ::fwData::Mesh::VertexModifiedSignalType >( ::fwData::Mesh::s_VERTEX_MODIFIED_SIG );
96  sig->asyncEmit();
97  }

145 void SSimpleMeshDeformation::computeDeformation (
146  const ::fwData::Mesh::sptr & refMesh,
147  const ::fwData::Mesh::sptr & transformMesh,
148  float deformationPercent )
149 {
150  SLM_ASSERT("Deformation range must be equal to [0,1]", 0 <= deformationPercent && deformationPercent <= 1 );
151 
152  const float maxDeformation = 15/100.0f;
153  const float center = 2/3.0f;
154 
155  ::fwComEd::helper::Mesh meshHelper(refMesh);
156  ::fwComEd::helper::Mesh transformMeshHelper(transformMesh);
157 
158  ::fwData::Mesh::PointsMultiArrayType points = meshHelper.getPoints();
159  ::fwData::Mesh::PointsMultiArrayType pointsTransform = transformMeshHelper.getPoints();
160  ::fwData::Mesh::PointColorsMultiArrayType colorTransform = transformMeshHelper.getPointColors();
161 
162  size_t nbPts = refMesh->getNumberOfPoints();
163 
164  // Compute limits
165  float ymin = points[0][1];
166  float ymax = points[0][1];
167  float val;
168  for(size_t i = 0; i!=nbPts; ++i)
169  {
170  val = points[i][1];
171  if ( val < ymin )
172  {
173  ymin = val;
174  }
175  else if ( val > ymax )
176  {
177  ymax = val;
178  }
179  }
180 
181  // Compute deformation
182  float sizeRef = ymax-ymin;
183  float yref = sizeRef * center + ymin;
184  float strafe = maxDeformation * sizeRef;
185  float currentStrafe = deformationPercent * strafe;
186 
187  for(size_t i = 0; i<nbPts; ++i )
188  {
189  float y = points[i][1];
190  if( y < yref )
191  {
192  float val = ( yref - y ) / ( yref - ymin ) * currentStrafe;
193  pointsTransform[i][1] = y - val;
194  colorTransform[i][0] = 255;
195  colorTransform[i][1] = static_cast<fwData::Mesh::ColorValueType>(255 - 255 * ( val / strafe ));
196  colorTransform[i][2] = static_cast<fwData::Mesh::ColorValueType>(255 - 255 * ( val / strafe ));
197  }
198  else
199  {
200  colorTransform[i][0] = 255;
201  colorTransform[i][1] = 255;
202  colorTransform[i][2] = 255;
203  }
204  }
205 
206  ::fwDataTools::Mesh::generatePointNormals(transformMesh);
207 }

211 void SSimpleMeshDeformation::computeDeformation(
212  const ::fwData::Mesh::sptr & refMesh,
213  const ::fwData::Mesh::sptr & transformMesh )
214 {
215  const int step = 5;
216 
217  m_currentDeformation += m_currentIncrement;
218  if ( m_currentDeformation == 100 )
219  {
220  m_currentIncrement = -step;
221  }
222  else if ( m_currentDeformation == 0 )
223  {
224  m_currentIncrement = step;
225  }
226  this->computeDeformation( refMesh, transformMesh, m_currentDeformation/100.0f );
227 }

231 void SSimpleMeshDeformation::initMeshBackup()
232 {
233  SLM_WARN_IF("Data already init", m_transformMesh || m_mesh);
234  if (!m_mesh && !m_transformMesh)
235  {
236  m_currentIncrement = 0;
237  m_currentDeformation = 0;
238 
239  ::fwData::Mesh::sptr mesh = this->getObject< ::fwData::Mesh >();
240  ::fwData::mt::ObjectReadToWriteLock lock(mesh);
241 
242 
243  lock.upgrade();
244  ::fwDataTools::Mesh::generatePointNormals(mesh);
245  mesh->allocatePointColors( ::fwData::Mesh::RGB );
246  lock.downgrade();
247 
248  m_mesh = ::fwData::Object::copy( mesh );
249 
250  m_transformMesh = ::fwData::Object::copy( mesh );
251  }
