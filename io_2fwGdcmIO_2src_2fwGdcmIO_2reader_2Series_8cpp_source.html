51 ::fwMedData::Series::sptr Series::read(::fwMedData::DicomSeries::sptr dicomSeries)
52 throw(::fwGdcmIO::exception::Failed)
53 {
54  SLM_ASSERT("DicomSeries should not be null.", dicomSeries);
55  SLM_ASSERT("Logger should not be null.", m_logger);
56 
57  // Create instance
58  SPTR(::fwGdcmIO::container::DicomInstance) instance =
59  std::make_shared< ::fwGdcmIO::container::DicomInstance >(dicomSeries);
60 
61  m_cancelled = false;
62 
63  // Create result
64  ::fwMedData::Series::sptr result;
65 
66  if(!dicomSeries->getLocalDicomPaths().empty())
67  {
68  // Get SOPClassUID
69  ::fwMedData::DicomSeries::SOPClassUIDContainerType sopClassUIDContainer = dicomSeries->getSOPClassUIDs();
70  std::string sopClassUID = *sopClassUIDContainer.begin();
71 
72  // If the DicomSeries contains an image (ImageSeries)
73  if (::gdcm::MediaStorage::IsImage(::gdcm::MediaStorage::GetMSType(sopClassUID.c_str())) &&
74  ::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) != ::gdcm::MediaStorage::SpacialFiducialsStorage)
75  {
76  // Read the image
77  ::fwMedData::ImageSeries::sptr imageSeries = ::fwDicomTools::Series::convertToImageSeries(dicomSeries);
78  ::fwData::Image::sptr image = ::fwData::Image::New();
79  imageSeries->setImage(image);
80 
81  // Create IOD Reader
82  ::fwGdcmIO::reader::iod::CTMRImageIOD iod(dicomSeries, instance, m_logger, m_callback, m_cancelled);
83  iod.setBufferRotationEnabled(m_enableBufferRotation);
84 
85  try
86  {
87  iod.read(imageSeries);
88  }
89  catch (const ::fwGdcmIO::exception::Failed& e)
90  {
91  // NOTE : if there is no image, reading is stopped.
92  m_logger->critical(e.what());
93  imageSeries = ::fwMedData::ImageSeries::sptr();
94  }
95 
96  // Set result
97  result = imageSeries;
98  }
99 
100  // Get the RT file names (ModelSeries)
101  else if (::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) ==
102  ::gdcm::MediaStorage::SurfaceSegmentationStorage)
103  {
104  ::fwMedData::ModelSeries::sptr modelSeries = ::fwDicomTools::Series::convertToModelSeries(dicomSeries);
105 
106  // Create IOD Reader
107  ::fwGdcmIO::reader::iod::SurfaceSegmentationIOD iod(dicomSeries, instance, m_logger, m_callback,
108  m_cancelled);
109 
110 
111  try
112  {
113  iod.read(modelSeries);
114  }
115  catch (const ::fwGdcmIO::exception::Failed& e)
116  {
117  // NOTE : if there is no image, reading is stopped.
118  m_logger->critical(e.what());
119  modelSeries = ::fwMedData::ModelSeries::sptr();
120  }
121 
122  // Set result
123  result = modelSeries;
124  }
125  // If the DicomSeries contains a Spatial Fiducials
126  else if (::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) == ::gdcm::MediaStorage::SpacialFiducialsStorage)
127  {
128  // Retrieve referenced image instance
129  SPTR(::fwGdcmIO::container::DicomInstance) imageInstance =
130  this->getSpatialFiducialsReferencedSeriesInstance(dicomSeries);
131 
132  if(imageInstance)
133  {
134  ::fwMedData::ImageSeries::sptr imageSeries =
135  ::fwMedData::ImageSeries::dynamicCast(m_seriesContainerMap[imageInstance]);
136 
137  // Create IOD Reader
138  ::fwGdcmIO::reader::iod::SpatialFiducialsIOD iod(dicomSeries, instance, m_logger, m_callback,
139  m_cancelled);
140 
141  try
142  {
143  iod.read(imageSeries);
144  }
145  catch (const ::fwGdcmIO::exception::Failed& e)
146  {
147  //NOTE: no throw for reading error for SR and RT doc
148  m_logger->critical("Spatial Fiducials reading failed: " + std::string(e.what()));
149  }
150  }
151  else
152  {
153  m_logger->critical(
154  "The spatial fiducials series \""+dicomSeries->getInstanceUID()+
155  "\" could not be read as it refers to an unknown series UID.");
156  }
157  }
158  // If the DicomSeries contains a SR
159  else if (::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) == ::gdcm::MediaStorage::EnhancedSR ||
160  ::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) == ::gdcm::MediaStorage::ComprehensiveSR ||
161  sopClassUID == "1.2.840.10008.5.1.4.1.1.88.34") // FIXME Replace hard coded string by "::gdcm::MediaStorage::GetMSType(sopClassUID.c_str()) == ::gdcm::MediaStorage::Comprehensive3DSR"
162  {
163  // Retrieve referenced image instance
164  SPTR(::fwGdcmIO::container::DicomInstance) imageInstance =
165  this->getStructuredReportReferencedSeriesInstance(dicomSeries);
166 
167  ::fwMedData::ImageSeries::sptr imageSeries =
168  ::fwMedData::ImageSeries::dynamicCast(m_seriesContainerMap[imageInstance]);
169 
170  if(imageInstance && imageSeries)
171  {
172  // Create readers
173  ::fwGdcmIO::reader::iod::ComprehensiveSRIOD iod(dicomSeries, imageInstance, m_logger, m_callback,
174  m_cancelled);
175 
176  try
177  {
178  iod.read(imageSeries);
179  }
180  catch (const ::fwGdcmIO::exception::Failed& e)
181  {
182  //NOTE: no throw for reading error for SR and RT doc
183  m_logger->critical("Structured Report reading failed: " + std::string(e.what()));
184  }
185  }
186  else
187  {
188  m_logger->critical(
189  "The structured report series \""+dicomSeries->getInstanceUID()+
190  "\" could not be read as it refers to an unknown series UID.");
191  }
192 
193  }
194  else
195  {
196  m_logger->critical("DICOM SOP Class UID \"" + sopClassUID +"\" is not supported by the selected reader.");
197  }
198 
199  }
200 
201  // Store series in instance map
202  m_seriesContainerMap[instance] = result;
203 
204  return result;
205 }

209 SPTR(::fwGdcmIO::container::DicomInstance) Series::getSpatialFiducialsReferencedSeriesInstance(
210  ::fwMedData::DicomSeries::sptr dicomSeries)
211 {
212  SPTR(::fwGdcmIO::container::DicomInstance) result;
213 
214  // Path container
215  ::fwMedData::DicomSeries::DicomPathContainerType pathContainer = dicomSeries->getLocalDicomPaths();
216  const std::string filename = pathContainer.begin()->second.string();
217 
218  // Create Reader
219  ::boost::shared_ptr< ::gdcm::Reader > reader = ::boost::shared_ptr< ::gdcm::Reader >( new ::gdcm::Reader );
220  reader->SetFileName( filename.c_str() );
221 
222  // Series Instance UID of the referenced Series
223  std::string seriesInstanceUID = "";
224 
225  if(reader->Read())
226  {
227  // Retrieve dataset
228  const ::gdcm::DataSet& datasetRoot = reader->GetFile().GetDataSet();
229 
230  if(datasetRoot.FindDataElement(::gdcm::Tag(0x0008, 0x1115)))
231  {
232  // Get the content sequence
233  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > sequence =
234  datasetRoot.GetDataElement(::gdcm::Tag(0x0008, 0x1115)).GetValueAsSQ();
235 
236  if(sequence->GetNumberOfItems() > 0)
237  {
238  ::gdcm::Item referencedSeriesItem = sequence->GetItem(1);
239  const ::gdcm::DataSet& referencedSeriesItemDataset = referencedSeriesItem.GetNestedDataSet();
240 
241  // Series Instance UID - Type 1
242  seriesInstanceUID =
243  ::fwGdcmIO::helper::DicomData::getTrimmedTagValue< 0x0020, 0x000E >(referencedSeriesItemDataset);
244  }
245  }
246  }
247 
248  if(!seriesInstanceUID.empty())
249  {
250  for(SeriesContainerMapType::value_type v: m_seriesContainerMap)
251  {
252  if(v.first->getSeriesInstanceUID() == seriesInstanceUID)
253  {
254  result = v.first;
255  break;
256  }
257  }
258  }
259 
260  return result;
261 }

266  ::fwMedData::DicomSeries::sptr dicomSeries)
267 {
268 
269  SPTR(::fwGdcmIO::container::DicomInstance) result;
270 
271  // Path container
272  ::fwMedData::DicomSeries::DicomPathContainerType pathContainer = dicomSeries->getLocalDicomPaths();
273  const std::string filename = pathContainer.begin()->second.string();
274 
275  // Create Reader
276  ::boost::shared_ptr< ::gdcm::Reader > reader = ::boost::shared_ptr< ::gdcm::Reader >( new ::gdcm::Reader );
277  reader->SetFileName( filename.c_str() );
278 
279  // Series Instance UID of the referenced Series
280  std::string seriesInstanceUID = "";
281 
282  if(reader->Read())
283  {
284  // Retrieve dataset
285  const ::gdcm::DataSet& datasetRoot = reader->GetFile().GetDataSet();
286 
287  // Pertinent Other Evidence Sequence - Type 1C
288  if(datasetRoot.FindDataElement(::gdcm::Tag(0x0040, 0xa385)))
289  {
290  // Get the content sequence
291  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > sequence =
292  datasetRoot.GetDataElement(::gdcm::Tag(0x0040, 0xa385)).GetValueAsSQ();
293 
294  if(sequence->GetNumberOfItems() > 0)
295  {
296  ::gdcm::Item studyItem = sequence->GetItem(1);
297  const ::gdcm::DataSet& studyItemDataset = studyItem.GetNestedDataSet();
298 
299  if(studyItemDataset.FindDataElement(::gdcm::Tag(0x0008, 0x1115)))
300  {
301  // Get the series sequence
302  ::gdcm::SmartPointer< ::gdcm::SequenceOfItems > seriesSequence =
303  studyItemDataset.GetDataElement(::gdcm::Tag(0x0008, 0x1115)).GetValueAsSQ();
304 
305  if(seriesSequence->GetNumberOfItems() > 0)
306  {
307  ::gdcm::Item seriesItem = seriesSequence->GetItem(1);
308  const ::gdcm::DataSet& seriesItemDataset = seriesItem.GetNestedDataSet();
309  seriesInstanceUID = ::fwGdcmIO::helper::DicomData::getTrimmedTagValue< 0x0020, 0x000E >(
310  seriesItemDataset);
311  }
312  }
313 
314  }
315  }
316  }
317 
318  if(!seriesInstanceUID.empty())
319  {
320  for(SeriesContainerMapType::value_type v: m_seriesContainerMap)
321  {
322  if(v.first->getSeriesInstanceUID() == seriesInstanceUID)
323  {
324  result = v.first;
325  break;
326  }
327  }
328  }
329 
330  return result;
331 }
