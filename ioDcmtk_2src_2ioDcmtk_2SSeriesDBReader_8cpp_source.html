55 void SSeriesDBReader::configureWithIHM()
56 {
57  static ::boost::filesystem::path _sDefaultPath;
58 
59  ::fwGui::dialog::LocationDialog dialogFile;
60  dialogFile.setTitle(this->getSelectorDialogTitle());
61  dialogFile.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
62  dialogFile.setOption(::fwGui::dialog::ILocationDialog::READ);
63  dialogFile.setType(::fwGui::dialog::LocationDialog::FOLDER);
64 
65  ::fwData::location::Folder::sptr result;
66  result = ::fwData::location::Folder::dynamicCast( dialogFile.show() );
67  if (result)
68  {
69  _sDefaultPath = result->getFolder();
70  this->setFolder( result->getFolder() );
71  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
72  }
73 
74  // Select filter
75  if(!m_filterSelectorSrvConfig.empty())
76  {
77  // Get the config
78  ::fwRuntime::ConfigurationElement::csptr filterSelectorConfig;
79  filterSelectorConfig = ::fwServices::registry::ServiceConfig::getDefault()->getServiceConfig(
80  m_filterSelectorSrvConfig, "::ioDicom::SFilterSelectorDialog");
81 
82  SLM_ASSERT("There is no service configuration "
83  + m_filterSelectorSrvConfig
84  + " for ::ioDicom::SFilterSelectorDialog", filterSelectorConfig);
85 
86  // Init and execute the service
87  ::fwServices::IService::sptr filterSelectorSrv;
88  ::fwData::String::sptr key = ::fwData::String::New();
89  filterSelectorSrv = ::fwServices::add(key,
90  "::gui::editor::IDialogEditor",
91  "::ioDicom::SFilterSelectorDialog");
92  filterSelectorSrv->setConfiguration( ::fwRuntime::ConfigurationElement::constCast(filterSelectorConfig) );
93  filterSelectorSrv->configure();
94  filterSelectorSrv->start();
95  filterSelectorSrv->update();
96  filterSelectorSrv->stop();
97  ::fwServices::OSR::unregisterService( filterSelectorSrv );
98 
99  m_filterType = key->getValue();
100 
101  }
102 }

106 void SSeriesDBReader::configuring() throw (fwTools::Failed)
107 {
108  ::io::IReader::configuring();
109 
110  // Use filter selector
111  ::fwRuntime::ConfigurationElement::sptr selectorConfig =
112  m_configuration->findConfigurationElement("FilterSelectorSrvConfig");
113  if(selectorConfig)
114  {
115  SLM_ASSERT("Missing 'name' attribute", selectorConfig->hasAttribute("name"));
116  m_filterSelectorSrvConfig = selectorConfig->getAttributeValue("name");
117  }
118 
119  // Get SOP Class selection
120  const ::fwRuntime::ConfigurationElement::sptr sopClassSelection =
121  m_configuration->findConfigurationElement("SOPClassSelection");
122  if(sopClassSelection)
123  {
124  ::fwRuntime::ConfigurationElementContainer sopClassElements =
125  sopClassSelection->findAllConfigurationElement("SOPClass");
126 
127  for(::fwRuntime::ConfigurationElementContainer::Iterator it = sopClassElements.begin();
128  it != sopClassElements.end(); ++it)
129  {
130  const ::fwRuntime::ConfigurationElement::AttributePair attributePair = (*it)->getSafeAttributeValue("uid");
131  if(attributePair.first)
132  {
133  SLM_TRACE("New SOP class supported : " + attributePair.second);
134  m_supportedSOPClassSelection.push_back(attributePair.second);
135  }
136  }
137  }
138 
139  // Set filter
140  ::fwRuntime::ConfigurationElement::sptr config = m_configuration->findConfigurationElement("config");
141  if(config)
142  {
143  m_filterType = config->getAttributeValue("filterType");
144  }
145 }

178 ::fwMedData::SeriesDB::sptr SSeriesDBReader::createSeriesDB(const ::boost::filesystem::path& dicomDir)
179 {
180  SLM_TRACE_FUNC();
181  ::fwDcmtkIO::SeriesDBReader::sptr myLoader = ::fwDcmtkIO::SeriesDBReader::New();
182  ::fwMedData::SeriesDB::sptr dummy = ::fwMedData::SeriesDB::New();
183  myLoader->setObject(dummy);
184  myLoader->setFolder(dicomDir);
185  myLoader->setDicomFilterType(m_filterType);
186  myLoader->setSupportedSOPClassContainer(m_supportedSOPClassSelection);
187 
188  if(myLoader->isDicomDirAvailable())
189  {
190  ::fwGui::dialog::MessageDialog messageBox;
191  messageBox.setTitle("Dicomdir file");
192  messageBox.setMessage( "There is a dicomdir file in the root folder. "
193  "Would you like to use it for the reading process ?" );
194  messageBox.setIcon(::fwGui::dialog::IMessageDialog::QUESTION);
195  messageBox.addButton(::fwGui::dialog::IMessageDialog::YES_NO);
196  ::fwGui::dialog::IMessageDialog::Buttons button = messageBox.show();
197 
198  myLoader->setDicomdirActivated(button == ::fwGui::dialog::IMessageDialog::YES);
199  }
200 
201  try
202  {
203  ::fwGui::dialog::ProgressDialog progressMeterGUI("Loading Dicom Image");
204  myLoader->addHandler( progressMeterGUI );
205  myLoader->read();
206  }
207  catch (const std::exception& e)
208  {
209  std::stringstream ss;
210  ss << "Warning during loading : " << e.what();
211  ::fwGui::dialog::MessageDialog::showMessageDialog(
212  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
213  }
214  catch( ... )
215  {
216  ::fwGui::dialog::MessageDialog::showMessageDialog(
217  "Warning", "Warning during loading", ::fwGui::dialog::IMessageDialog::WARNING);
218  }
219 
220  return myLoader->getConcreteObject();
221 }

225 void SSeriesDBReader::updating() throw(::fwTools::Failed)
226 {
227  SLM_TRACE_FUNC();
228  if( this->hasLocationDefined() )
229  {
230  ::fwMedData::SeriesDB::sptr seriesDB = createSeriesDB( this->getFolder() );
231 
232  if( seriesDB->size() > 0 )
233  {
234  // Retrieve dataStruct associated with this service
235  ::fwMedData::SeriesDB::sptr associatedSeriesDB;
236  if (this->isVersion2())
237  {
238  associatedSeriesDB = this->getInOut< ::fwMedData::SeriesDB >(::io::s_DATA_KEY);
239  }
240  else
241  {
242  associatedSeriesDB = this->getObject< ::fwMedData::SeriesDB >();
243  }
244  SLM_ASSERT("associated SeriesDB not instanced", associatedSeriesDB);
245  associatedSeriesDB->shallowCopy( seriesDB );
246 
247  ::fwGui::Cursor cursor;
248  cursor.setCursor(::fwGui::ICursor::BUSY);
249  this->notificationOfDBUpdate();
250  cursor.setDefaultCursor();
251  }
252  else
253  {
254  ::fwGui::dialog::MessageDialog::showMessageDialog(
255  "Image Reader","This file can not be read. Retry with another file reader.",
256  ::fwGui::dialog::IMessageDialog::WARNING);
257  }
258  }

263 void SSeriesDBReader::notificationOfDBUpdate()
264 {
265  SLM_TRACE_FUNC();
266  ::fwMedData::SeriesDB::sptr seriesDB;
267  if (this->isVersion2())
268  {
269  seriesDB = this->getInOut< ::fwMedData::SeriesDB >(::io::s_DATA_KEY);
270  }
271  else
272  {
273  seriesDB = this->getObject< ::fwMedData::SeriesDB >();
274  }
275 
276  SLM_ASSERT("Unable to get seriesDB", seriesDB);
277 
278 
279  ::fwMedData::SeriesDB::ContainerType addedSeries;
280  for(const ::fwMedData::Series::sptr& s : seriesDB->getContainer() )
281  {
282  addedSeries.push_back(s);
283  }
284 
285  auto sig = seriesDB->signal< ::fwMedData::SeriesDB::AddedSeriesSignalType >(
286  ::fwMedData::SeriesDB::s_ADDED_SERIES_SIG);
287  sig->asyncEmit(addedSeries);
288 }
