40 void MemoryInfo::updating( ) throw(::fwTools::Failed)
41 {
42  // Memory information
43  ::boost::uint64_t mo = 1024*1024;
44  ::boost::uint64_t totalSystemMemory = ::fwMemory::tools::MemoryMonitorTools::getTotalSystemMemory();
45  ::boost::uint64_t freeSystemMemory = ::fwMemory::tools::MemoryMonitorTools::getFreeSystemMemory();
46  ::boost::uint64_t usedProcessMemory = ::fwMemory::tools::MemoryMonitorTools::getUsedProcessMemory();
47  ::boost::uint64_t estimateFreeMem = ::fwMemory::tools::MemoryMonitorTools::estimateFreeMem();
48 
49  ::fwMemory::BufferManager::SizeType managedBufferSize = 0;
50  ::fwMemory::BufferManager::SizeType dumpedBufferSize = 0;
51  ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
52  if( manager )
53  {
54  ::fwMemory::BufferManager::BufferStats stats = manager->getBufferStats().get();
55  managedBufferSize = stats.totalManaged;
56  dumpedBufferSize = stats.totalDumped;
57  }
58  std::stringstream stream;
59  stream << "Total system memory = " << totalSystemMemory/mo << " Mo" << std::endl;
60  stream << "Free system memory = " << freeSystemMemory/mo << " Mo" << std::endl;
61  stream << "Used process memory = " << usedProcessMemory/mo << " Mo" << std::endl;
62  stream << "Estimed Free memory = " << estimateFreeMem/mo << " Mo" << std::endl;
63  stream << "ManagedBuffer size = " << managedBufferSize/mo << " Mo" << std::endl;
64  stream << "DumpedBuffer size = " << dumpedBufferSize/mo << " Mo" << std::endl;
65 
66  // Information message box
67  ::fwGui::dialog::MessageDialog::showMessageDialog(
68  "MemoryInfo",
69  stream.str(),
70  ::fwGui::dialog::IMessageDialog::INFO);
71 }
