00039 {
00040     // Memory information

00049     ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
00050     if( manager )
00051     {

00058     stream << "Free system memory  = "   << freeSystemMemory/mo  << " Mo" << std::endl;
00059     stream << "Used process memory = "  << usedProcessMemory/mo << " Mo" << std::endl;
00060     stream << "Estimed Free memory = "  << estimateFreeMem/mo << " Mo" << std::endl;
00061     stream << "ManagedBuffer size  = "  << managedBufferSize/mo << " Mo" << std::endl;
00062     stream << "DumpedBuffer size   = "  << dumpedBufferSize/mo << " Mo" << std::endl;
00063 
00064     // Information message box
00065     ::fwGui::dialog::MessageDialog::showMessageDialog(

00067             stream.str(),
00068             ::fwGui::dialog::IMessageDialog::INFO);
