00056 std::string SelectorDialog::show()
00057 {
00058     QWidget *parent = qApp->activeWindow();

00060     QDialog* dialog = new QDialog(parent);
00061     dialog->setWindowTitle(QString::fromStdString(m_title));

00063     QListWidget *selectionList = new QListWidget(dialog);
00064     BOOST_FOREACH( std::string selection, m_selections)
00065     {
00066         selectionList->addItem(QString::fromStdString ( selection ));
00067     }
00068 
00069     QListWidgetItem* firstItem = selectionList->item(0);
00070     selectionList->setCurrentItem(firstItem);
00071 
00072     QPushButton* okButton = new QPushButton(tr("Ok"));
00073     QPushButton* cancelButton = new QPushButton(tr("Cancel"));
00074 
00075     QHBoxLayout *hLayout = new QHBoxLayout();
00076     hLayout->addWidget(okButton);
00077     hLayout->addWidget(cancelButton);
00078 
00079     QVBoxLayout *vLayout = new QVBoxLayout();
00080     if(!m_message.empty())
00081     {
00082         QLabel* msgText = new QLabel(QString::fromStdString(m_message), dialog);
00083         vLayout->addWidget( msgText);
00084     }
00085     vLayout->addWidget(selectionList);
00086     vLayout->addLayout(hLayout);
00087 
00088     dialog->setLayout(vLayout);
00089     QObject::connect(okButton, SIGNAL(clicked()), dialog, SLOT(accept()));
00090     QObject::connect(cancelButton, SIGNAL(clicked()), dialog, SLOT(reject()));
00091     QObject::connect(selectionList, SIGNAL(itemDoubleClicked( QListWidgetItem * )), dialog, SLOT(accept()));
00092 
00093     std::string selection = "";
00094     if(dialog->exec())
00095     {
00096         selection = selectionList->currentItem()->text().toStdString();
00097     }
00098 
00099     return selection;
