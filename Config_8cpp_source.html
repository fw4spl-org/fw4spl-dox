33 void Config::createConnections(
34  ::fwRuntime::ConfigurationElement::csptr connectionCfg,
35  ::fwServices::helper::SigSlotConnection::sptr connections,
36  const SPTR(::fwTools::Object)& obj)
37 {
38  typedef std::pair< std::string, ::fwCom::Signals::SignalKeyType > SignalInfoType;
39  typedef std::pair< std::string, ::fwCom::Slots::SlotKeyType > SlotInfoType;
40  typedef std::vector< SlotInfoType > SlotInfoContainerType;
41 
42  SignalInfoType signalInfo;
43  SlotInfoContainerType slotInfos;
44 
45  ::boost::regex re("(.*)/(.*)");
46  ::boost::smatch match;
47  std::string src, uid, key;
48 
49  for(::fwRuntime::ConfigurationElement::csptr elem : connectionCfg->getElements())
50  {
51  src = elem->getValue();
52  if( ::boost::regex_match(src, match, re) )
53  {
54  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
55  uid.assign(match[1].first, match[1].second);
56  key.assign(match[2].first, match[2].second);
57 
58  OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName(),
59  !uid.empty() && !key.empty());
60 
61  if (elem->getName() == "signal")
62  {
63  SLM_ASSERT("There must be only one signal by connection",
64  signalInfo.first.empty() && signalInfo.second.empty());
65  signalInfo = std::make_pair(uid, key);
66  }
67  else if (elem->getName() == "slot")
68  {
69  slotInfos.push_back( std::make_pair(uid, key) );
70  }
71  }
72  else
73  {
74  SLM_ASSERT("Object uid is not defined, object used to retrieve signal must be present.", obj);
75  uid = obj->getID();
76  key = src;
77  SLM_ASSERT("Element must be a signal or must be written as <fwID/Key>", elem->getName() == "signal");
78  SLM_ASSERT("There must be only one signal by connection",
79  signalInfo.first.empty() && signalInfo.second.empty());
80  signalInfo = std::make_pair(uid, key);
81  }
82  }
83 
84  ::fwTools::Object::sptr sigSource = ::fwTools::fwID::getObject(signalInfo.first);
85  ::fwCom::HasSignals::sptr hasSignals = std::dynamic_pointer_cast< ::fwCom::HasSignals >(sigSource);
86 
87  SLM_ASSERT("Signal source not found" << signalInfo.first, sigSource);
88  SLM_ASSERT("invalid signal source " << signalInfo.first, hasSignals);
89 
90  for(SlotInfoType slotInfo : slotInfos)
91  {
92  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(slotInfo.first);
93  SLM_ASSERT("Failed to retrieve object '" + slotInfo.first + "'", obj);
94  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
95  SLM_ASSERT("invalid slot owner " << slotInfo.first, hasSlots);
96 
97  connections->connect(hasSignals, signalInfo.second, hasSlots, slotInfo.second);
98  }
99 }

103 void Config::createProxy(
104  const std::string &objectKey,
105  CSPTR(::fwRuntime::ConfigurationElement)cfg,
106  Config::ProxyConnectionsMapType &proxyMap,
107  const SPTR(::fwData::Object)& obj)
108 {
109  ::fwServices::registry::Proxy::sptr proxy = ::fwServices::registry::Proxy::getDefault();
110 
111  SLM_ASSERT("Missing 'channel' attribute", cfg->hasAttribute("channel"));
112  const std::string channel = cfg->getExistingAttributeValue("channel");
113  ProxyConnections proxyCnt(channel);
114 
115  ::boost::regex re("(.*)/(.*)");
116  ::boost::smatch match;
117  std::string src, uid, key;
118  for(::fwRuntime::ConfigurationElement::csptr elem : cfg->getElements())
119  {
120  src = elem->getValue();
121  if( ::boost::regex_match(src, match, re) )
122  {
123  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
124  uid.assign(match[1].first, match[1].second);
125  key.assign(match[2].first, match[2].second);
126 
127  OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName(),
128  !uid.empty() && !key.empty());
129 
130  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(uid);
131 
132  if (elem->getName() == "signal")
133  {
134  ::fwCom::HasSignals::sptr hasSignals = std::dynamic_pointer_cast< ::fwCom::HasSignals >(obj);
135  ::fwCom::SignalBase::sptr sig = hasSignals->signal(key);
136  proxy->connect(channel, sig);
137  proxyCnt.addSignalConnection(uid, key);
138  }
139  else if (elem->getName() == "slot")
140  {
141  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
142  ::fwCom::SlotBase::sptr slot = hasSlots->slot(key);
143  proxy->connect(channel, slot);
144  proxyCnt.addSlotConnection(uid, key);
145  }
146  }
147  else
148  {
149  uid = obj->getID();
150  key = src;
151  SLM_ASSERT("Element must be a signal or must be written as <fwID/Key>", elem->getName() == "signal");
152  ::fwCom::SignalBase::sptr sig = obj->signal(key);
153  proxy->connect(channel, sig);
154  proxyCnt.addSignalConnection(uid, key);
155  }
156  }
157  proxyMap[objectKey].push_back(proxyCnt);
158 }

162 void Config::disconnectProxies(const std::string &objectKey, Config::ProxyConnectionsMapType &proxyMap)
163 {
164  ProxyConnectionsMapType::iterator iter = proxyMap.find(objectKey);
165  if (iter != proxyMap.end())
166  {
167  ::fwServices::registry::Proxy::sptr proxy = ::fwServices::registry::Proxy::getDefault();
168 
169  ProxyConnectionsVectType vectProxyConnections = iter->second;
170 
171  for(ProxyConnectionsVectType::value_type proxyConnections : vectProxyConnections)
172  {
173  for(ProxyConnections::ProxyEltType signalElt : proxyConnections.m_signals)
174  {
175  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(signalElt.first);
176  ::fwCom::HasSignals::sptr hasSignals = std::dynamic_pointer_cast< ::fwCom::HasSignals >(obj);
177  ::fwCom::SignalBase::sptr sig = hasSignals->signal(signalElt.second);
178  proxy->disconnect(proxyConnections.m_channel, sig);
179  }
180  for(ProxyConnections::ProxyEltType slotElt : proxyConnections.m_slots)
181  {
182  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(slotElt.first);
183  ::fwCom::HasSlots::sptr hasSlots = std::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
184  ::fwCom::SlotBase::sptr slot = hasSlots->slot(slotElt.second);
185  proxy->disconnect(proxyConnections.m_channel, slot);
186  }
187  }
188  vectProxyConnections.clear();
189  proxyMap.erase(objectKey);
190  }
191 }
