00031 void Config::createConnections(
00032         ::fwRuntime::ConfigurationElement::csptr connectionCfg,
00033         ::fwServices::helper::SigSlotConnection::sptr connections)
00034 {
00035     typedef std::pair< std::string, ::fwCom::Signals::SignalKeyType > SignalInfoType;
00036     typedef std::pair< std::string, ::fwCom::Slots::SlotKeyType > SlotInfoType;
00037     typedef std::vector< SlotInfoType > SlotInfoContainerType;
00038 
00039     SignalInfoType signalInfo;
00040     SlotInfoContainerType slotInfos;
00041 
00042     ::boost::regex re("(.*)/(.*)");
00043     ::boost::smatch match;
00044     std::string src, uid, key;
00045 
00046     BOOST_FOREACH(::fwRuntime::ConfigurationElement::csptr elem,  connectionCfg->getElements())
00047     {
00048         src = elem->getValue();
00049         if( ::boost::regex_match(src, match, re) )
00050         {
00051             OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
00052             uid.assign(match[1].first, match[1].second);
00053             key.assign(match[2].first, match[2].second);
00054 
00055             OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName() ,
00056                         !uid.empty() && !key.empty());
00057 
00058             if (elem->getName() == "signal")
00059             {
00060                 SLM_ASSERT("There must be only one signal by connection",
00061                            signalInfo.first.empty() && signalInfo.second.empty());
00062                 signalInfo = std::make_pair(uid, key);
00063             }
00064             else if (elem->getName() == "slot")
00065             {
00066                 slotInfos.push_back( std::make_pair(uid, key) );
00067             }
00068         }
00069     }
00070 
00071     ::fwTools::Object::sptr sigSource = ::fwTools::fwID::getObject(signalInfo.first);
00072     ::fwCom::HasSignals::sptr hasSignals = ::boost::dynamic_pointer_cast< ::fwCom::HasSignals >(sigSource);
00073 
00074     SLM_ASSERT("Signal source not found" << signalInfo.first , sigSource);
00075     SLM_ASSERT("invalid signal source " << signalInfo.first , hasSignals);
00076 
00077     BOOST_FOREACH(SlotInfoType slotInfo,  slotInfos)
00078     {
00079         ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(slotInfo.first);
00080         SLM_ASSERT("Failed to retrieve object '" + slotInfo.first + "'", obj);
00081         ::fwCom::HasSlots::sptr hasSlots = ::boost::dynamic_pointer_cast< ::fwCom::HasSlots >(obj);
00082         SLM_ASSERT("invalid slot owner " << slotInfo.first , hasSlots);
00083 
00084         connections->connect(hasSignals, signalInfo.second, hasSlots, slotInfo.second);
00085     }
00086 }
