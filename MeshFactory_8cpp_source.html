00057 void MeshFactory::updateMaterial( ::fwData::Material::sptr material )
00058 {

00061         ::fwData::Color::sptr color = material->ambient();

00063         vtkProperty *property = m_actor->GetProperty();

00066                 color->green(),
00067                 color->blue());
00068 
00069         //3DVSP like rendering
00070         property->SetSpecularColor(1.,1.,1.);

00072         property->SetAmbient(.05);
00073         property->SetDiffuse(1.);
00074         property->SetSpecular(1.);
00075 //        property->SetInterpolationToPhong();
00076 
00077 
00078         property->SetOpacity( color->alpha() );
00079 
00081         if(material->getRepresentationMode() == ::fwData::Material::MODE_SURFACE)
00082         {
00083             m_actor->GetProperty()->SetRepresentationToSurface();
00084             m_actor->GetProperty()->EdgeVisibilityOff();
00085         }
00086         else if(material->getRepresentationMode() == ::fwData::Material::MODE_EDGE)
00087         {
00088             m_actor->GetProperty()->SetRepresentationToSurface();
00089             m_actor->GetProperty()->EdgeVisibilityOn();
00090         }
00091         else if(material->getRepresentationMode() == ::fwData::Material::MODE_WIREFRAME)
00092         {
00093             m_actor->GetProperty()->SetRepresentationToWireframe();
00094         }
00095         else if(material->getRepresentationMode() == ::fwData::Material::MODE_POINT)
00096         {
00097             m_actor->GetProperty()->SetRepresentationToPoints();
00098         }
00099 
00101         if(material->getShadingMode() == ::fwData::Material::MODE_PHONG)
00102         {
00103             m_actor->GetProperty()->SetInterpolationToPhong();
00104         }
00105         else if(material->getShadingMode() == ::fwData::Material::MODE_GOURAUD)
00106         {
00107             m_actor->GetProperty()->SetInterpolationToGouraud();
00108         }
00109         else if(material->getShadingMode() == ::fwData::Material::MODE_FLAT)
00110         {
00111             m_actor->GetProperty()->SetInterpolationToFlat();

00114     }
