45 void ImageSeries::doConfigure() throw(fwTools::Failed)
46 {
47  SLM_TRACE_FUNC();
48 
49  SLM_ASSERT("Configuration must begin with <config>", m_configuration->getName() == "config");
50 
51  if (m_configuration->hasAttribute("mode"))
52  {
53  std::string value(m_configuration->getAttributeValue("mode"));
54  std::transform(value.begin(), value.end(), value.begin(), tolower);
55  OSLM_ASSERT("Bad value "<<value<<" for attribute mode, it should either be '3d' or '2d'.",
56  value == "3d" || value == "2d");
57  this->set3dMode(value == "3d");
58  }
59  if (m_configuration->hasAttribute("slices"))
60  {
61  std::string value(m_configuration->getAttributeValue("slices"));
62  std::transform(value.begin(), value.end(), value.begin(), tolower);
63 
64  if(value == "0")
65  {
66  this->setSliceMode(NegatoMPR::NO_SLICE);
67  }
68  else if(value == "1")
69  {
70  this->setSliceMode(NegatoMPR::ONE_SLICE);
71  }
72  else if(value == "3")
73  {
74  this->setSliceMode(NegatoMPR::THREE_SLICES);
75  }
76  }
77  if(m_configuration->hasAttribute("sliceIndex"))
78  {
79  std::string orientation = m_configuration->getAttributeValue("sliceIndex");
80  if(orientation == "axial" )
81  {
82  m_orientation = Z_AXIS;
83  }
84  else if(orientation == "frontal" )
85  {
86  m_orientation = Y_AXIS;
87  }
88  else if(orientation == "sagittal" )
89  {
90  m_orientation = X_AXIS;
91  }
92  }
93  if(m_configuration->hasAttribute("tfalpha") )
94  {
95  this->setAllowAlphaInTF(m_configuration->getAttributeValue("tfalpha") == "yes");
96  }
97  if (m_configuration->hasAttribute("interpolation"))
98  {
99  this->setInterpolation(!(m_configuration->getAttributeValue("interpolation") == "off"));
100  }
101  if (m_configuration->hasAttribute("vtkimagesource"))
102  {
103  this->setVtkImageSourceId( m_configuration->getAttributeValue("vtkimagesource") );
104  }
105  this->parseTFConfig( m_configuration );
106 }

117 void ImageSeries::doUpdate() throw(fwTools::Failed)
118 {
119  ::fwMedData::ImageSeries::sptr series = this->getObject< ::fwMedData::ImageSeries >();
120 
121  doStop();
122 
123  ::fwData::Image::sptr image = series->getImage();
124 
125  ::fwRenderVTK::IVtkAdaptorService::sptr service =
126  ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService >( image, "::visuVTKAdaptor::NegatoMPR" );
127  SLM_ASSERT("service not instanced", service);
128 
129  service->setTransformId( this->getTransformId() );
130  service->setRenderId( this->getRenderId() );
131  service->setPickerId( this->getPickerId() );
132  service->setRenderService(this->getRenderService());
133  service->setAutoRender( this->getAutoRender() );
134  ::visuVTKAdaptor::NegatoMPR::sptr negato = ::visuVTKAdaptor::NegatoMPR::dynamicCast(service);
135  negato->set3dMode(this->is3dModeEnabled());
136  negato->setSliceMode(this->getSliceMode());
137  negato->setOrientation(this->getOrientation());
138  negato->setAllowAlphaInTF(m_allowAlphaInTF);
139  negato->setInterpolation(m_interpolation);
140  negato->setVtkImageSourceId(m_imageSourceId);
141  negato->parseTFConfig(m_configuration);
142  service->start();
143 
144  this->registerService(service);
145 }
