66 void Manager::queue( ICommand::sptr pCmd, const bool execute )
67 {
68  pCmd->setNotifier( m_serviceNotifier.lock() );
69 
70  // Erase commands after the current one.
71  CmdList::iterator i ( m_listCmd.end()-1 );
72  CmdList::iterator iEnd ( m_lastCmd );
73 
74  while( i != iEnd )
75  {
76  ICommand::sptr pCmd = (*i);
77  m_usedMemory -= pCmd->getSize();
78  --i;
79  m_listCmd.pop_back();
80  }
81 
82  // Execute command, if needed
83  if ( execute )
84  {
85  pCmd->apply();
86  }
87 
88  // Queue command.
89  if ( this->getMaxUndoLevel() > 0 )
90  {
91  // Ensure that the command size will not overflow the used memory counter.
92  const ::boost::uint32_t cmdSize = pCmd->getSize();
93 
94  if( (std::numeric_limits< ::boost::uint32_t >::max() - m_usedMemory) <= cmdSize )
95  {
96  throw std::overflow_error( "Command size will overflow memory usage count." );
97  }
98 
99  // Pushes the command
100  if ( cmdSize <= this->getMaxCommandMemory() )
101  {
102 
103  if ( m_listCmd.size() + 1 > this->getMaxUndoLevel() )
104  {
105  this->removeFirstCmd();
106  }
107 
108  // Purges some command in the history.
109  while (m_usedMemory + cmdSize > getMaxUndoMemory() && !m_listCmd.empty() )
110  {
111  this->removeFirstCmd();
112  }
113 
114  m_listCmd.push_back( pCmd );
115  m_usedMemory += cmdSize;
116 
117  m_lastCmd = m_listCmd.end();
118  --m_lastCmd;
119  }
120  }
121  //else nothing to do
122 }
