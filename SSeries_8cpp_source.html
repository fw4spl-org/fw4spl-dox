131 void SSeries::updating() throw(::fwTools::Failed)
132 {
133  ::fwData::Vector::sptr vector = this->getObject< ::fwData::Vector >();
134  SLM_ASSERT("Failed to retrieve vector", vector);

136  m_btnExport->setVisible(m_sigCanExport->getNumberOfConnections() == 0);

140 
141  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_seriesId);

144  SLM_ASSERT("Failed to retrieve a ::fwMedData::Series from object '" << m_seriesId << "'", seriesSrc);

155  m_btnExport->setEnabled(true);
156  m_sigCanExport->asyncEmit(true);

160  m_btnExport->setEnabled(false);

167 void SSeries::onExportClicked()
168 {
169  SLM_TRACE_FUNC();
170 
171  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_seriesId);
172  SLM_ASSERT("Failed to retrieve object with UID '" + m_seriesId + "'", obj);
173  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(obj);
174  SLM_ASSERT("Failed to retrieve a ::fwMedData::Series from object '" << m_seriesId << "'", series);
175 
176  ::fwTools::Object::sptr objDB = ::fwTools::fwID::getObject(m_seriesDBId);
177  SLM_ASSERT("Failed to retrieve object with UID '" + m_seriesDBId + "'", objDB);
178  ::fwMedData::SeriesDB::sptr seriesDB = ::fwMedData::SeriesDB::dynamicCast(objDB);
179  SLM_ASSERT("Failed to retrieve a ::fwMedData::SeriesDB from object '" << m_seriesDBId << "'", seriesDB);
180 
181  if(m_patientEditor->isValid() && m_studyEditor->isValid()
182  && m_equipmentEditor->isValid() && m_seriesEditor->isValid())
183  {
184 
185  ::fwData::Vector::sptr vector = this->getObject< ::fwData::Vector >();
186  SLM_ASSERT("Failed to retrieve vector", vector);
187  ::fwMedData::Series::sptr seriesVec = ::fwMedData::Series::dynamicCast((*vector)[0]);
188  SLM_ASSERT("Failed to retrieve ::fwMedData::Series from vector", seriesVec);
189 
190  ::fwMedData::Patient::sptr patient = m_patientEditor->getPatient();
191  ::fwMedData::Study::sptr study = m_studyEditor->getStudy();
192 
193  // If a new patient is being created but the study edition didn't change a new study instance UID must be set
194  if(patient->getPatientId() != seriesVec->getPatient()->getPatientId())
195  {
196  study->setInstanceUID(::fwMedDataTools::generateStudyInstanceUid());
197  }
198 
199  series->setPatient(patient);
200  series->setStudy(study);
201  series->setEquipment(m_equipmentEditor->getEquipment());
202 
203  ::fwMedData::Series::sptr seriesInfo = m_seriesEditor->getSeries();
204  series->setModality(seriesInfo->getModality());
205  series->setDate(seriesInfo->getDate());
206  series->setTime(seriesInfo->getTime());
207  series->setDescription(seriesInfo->getDescription());
208  series->setPerformingPhysiciansName(seriesInfo->getPerformingPhysiciansName());
209 
210  ::fwMedDataTools::helper::SeriesDB helper(seriesDB);
211  ::fwMedData::SeriesDB::iterator it = std::find(seriesDB->begin(), seriesDB->end(), series);
212  if(it != seriesDB->end())
213  {
214  ::fwGui::dialog::MessageDialog dlg;
215  dlg.setTitle("The series already exists");
216  dlg.setMessage("This series has already been exported in series selector");
217  dlg.setIcon(::fwGui::dialog::IMessageDialog::INFO);
218 
219  dlg.show();
220  }
221  else
222  {
223  helper.add(series);
224  helper.notify();
225  }
226  m_sigSeriesExported->asyncEmit();
227  }
228  else
229  {
230  ::fwGui::dialog::MessageDialog dlg;
231  dlg.setTitle("Error : information validation failed");
232  dlg.setMessage("Given information are not valid : please fix incorrect values before exporting");
233  dlg.setIcon(::fwGui::dialog::IMessageDialog::CRITICAL);
234 
235  dlg.show();
236  }
