330 const ::fwGdcmIO::container::DicomCodedAttribute* DictionarySegment::guessAnatRegionFromLabel(const std::string& label)
331 {
332  // Delete possible space as last character
333  std::string cleanedLabel = ::boost::algorithm::trim_copy(label);
334 
335  if (!cleanedLabel.empty())
336  {
337  // Search and translate label into anatomic region sequence
338  if (s_ANATOMICAL_REGIONS.left.find(cleanedLabel) != s_ANATOMICAL_REGIONS.left.end())
339  {
340  return s_ANATOMICAL_REGIONS.left.at(cleanedLabel);
341  }
342  }
343  return s_TODEFINE;
344 }
