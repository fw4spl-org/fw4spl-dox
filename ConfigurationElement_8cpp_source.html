142 std::vector < ConfigurationElement::sptr > ConfigurationElement::find(
143  std::string name,
144  std::string attribute,
145  std::string attributeValue,
146  int depth
147  )
148 {
149  typedef std::vector < ConfigurationElement::sptr > ConfVector;
150  ConfVector result;
151 
152  bool nameOk = (name.empty() || this->getName() == name);
153  bool attributeOk = (attribute.empty() || this->hasAttribute(attribute));
154  bool attributeValueOk =
155  (attributeValue.empty() ||
156  (this->hasAttribute(attribute) && this->getAttributeValue(attribute) == attributeValue));
157  if (nameOk && attributeOk && attributeValueOk)
158  {
159  result.push_back(this->shared_from_this());
160  }
161 
162  if (depth != 0)
163  {
164  ConfigurationElement::Iterator iter;
165  for (iter = this->begin(); iter != this->end(); ++iter)
166  {
167  ConfVector deepConf;
168  deepConf = (*iter)->find(name, attribute, attributeValue, depth-1);
169  result.insert(result.end(), deepConf.begin(), deepConf.end());
170  }
171  }
172 
173  return result;
174 }
