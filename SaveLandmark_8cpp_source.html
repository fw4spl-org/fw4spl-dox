62 void SaveLandmark::updating() throw(::fwTools::Failed)
63 {
64  SLM_TRACE_FUNC();
65  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
66  if (!::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
67  {
68  ::fwGui::dialog::MessageDialog::showMessageDialog(
69  "Save landmarks",
70  "It is impossible to save image landmarks. There is no loaded image in the software.",
71  ::fwGui::dialog::IMessageDialog::WARNING);
72  return;
73  }
74  static ::boost::filesystem::path _sDefaultPath("");
75  ::fwGui::dialog::LocationDialog dialogFile;
76  dialogFile.setTitle("Choose a file to save landmarks");
77  dialogFile.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
78  dialogFile.addFilter("Landmark file","*.json");
79  dialogFile.setOption(::fwGui::dialog::ILocationDialog::WRITE);
80 
81  ::fwData::location::SingleFile::sptr result;
82  result = ::fwData::location::SingleFile::dynamicCast( dialogFile.show() );
83 
84  if( result )
85  {
86  ::boost::filesystem::path path = result->getPath();
87  _sDefaultPath = path.parent_path();
88  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
89 
90  this->save(path);
91  }
92 }

117 void SaveLandmark::save(const ::boost::filesystem::path& path)
118 {
119  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
120 
121  //get landmarks
122  ::fwDataTools::fieldHelper::MedicalImageHelpers::checkLandmarks( image );
123  ::fwData::PointList::sptr landmarks = image->getField< ::fwData::PointList >(
124  ::fwDataTools::fieldHelper::Image::m_imageLandmarksId);
125  SLM_ASSERT("landmarks not instanced", landmarks);
126 
127  ::fwData::Composite::sptr replaceMap = ::fwData::Composite::New();
128  (*replaceMap)["GENERIC_UID"] = ::fwData::String::New(
129  ::fwServices::registry::AppConfig::getUniqueIdentifier("SaveLandmarkApp")
130  );
131  (*replaceMap)["landmark"] = ::fwData::String::New(landmarks->getID());
132  (*replaceMap)["file"] = ::fwData::String::New(path.string());
133  ::fwRuntime::ConfigurationElement::csptr config =
134  ::fwServices::registry::AppConfig::getDefault()->getAdaptedTemplateConfig("SaveLandmark", replaceMap);
135 
136  ::fwServices::IAppConfigManager::sptr helper = ::fwServices::IAppConfigManager::New();
137  helper->setConfig( config );
138  helper->launch();
139 
140  helper->stopAndDestroy();
141 }
