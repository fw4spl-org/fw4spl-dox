47 void GzBufferImageReader::read()
48 {
49  assert( ::fwData::location::SingleFile::dynamicCast(m_location) );
50  ::boost::filesystem::path file = ::fwData::location::SingleFile::dynamicCast(m_location)->getPath();
51 
52  assert( file.empty() == false );
53 
54  ::fwData::Image::sptr image = getConcreteObject();
55  size_t imageSizeInBytes = image->getSizeInBytes();
56 
57  image->allocate();
58  ::fwComEd::helper::Image helper(image);
59  char *ptr = static_cast<char*>(helper.getBuffer());
60 
61  gzFile rawFile = gzopen(file.string().c_str(), "rb");
62 
63  SLM_ASSERT("rawFile not instanced", rawFile);
64  if ( rawFile == 0 )
65  {
66  std::string str = "Unable to open ";
67  str += file.string();
68  throw std::ios_base::failure(str);
69  }
70 
71  int uncompressedbytesreaded;
72  size_t readBytes = 0;
73 
74  while ( (uncompressedbytesreaded = gzread(rawFile, ptr + readBytes, imageSizeInBytes - readBytes)) > 0 )
75  {
76  readBytes += uncompressedbytesreaded;
77  }
78 
79  gzclose(rawFile);
80 
81  if ( uncompressedbytesreaded == -1 )
82  {
83  std::string str = "Unable to read ";
84  str += file.string();
85  throw std::ios_base::failure(str);
86  }
87 }
