00081 fwID::IDType fwID::getID( Policy policy) const
00082 {
00083     ::fwCore::mt::ReadToWriteLock lock(m_idMutex);
00084     if ( m_id.empty() ) // no id set
00085     {
00086         if ( policy==GENERATE )
00087         {
00088             IDType newID = generate();
00089             ::fwCore::mt::UpgradeToWriteLock writeLock(lock);
00090             const_cast<fwID *>(this)->addIDInDictionary(newID);
00091         }
00092         else if  ( policy == EMPTY )
00093         { /* nothing to do*/ }
00094         else if ( policy == MUST_EXIST )
00095         {
00096             throw fwTools::Failed( "fwID::getID() no id set" );
00097         }
00098     }
00099     return m_id;
00100 }
