00039 void SDumpAll::updating( ) throw(::fwTools::Failed)
00040 {
00041     size_t nbBuffDumped = 0;
00042     ::fwMemory::BufferManager::BufferInfoMapType buffInfoMap;
00043     ::fwMemory::BufferManager::sptr buffManager = ::fwMemory::BufferManager::getDefault();
00044     if(buffManager)
00045     {
00046         buffInfoMap = buffManager->getBufferInfos().get();
00047     }
00048     BOOST_FOREACH(::fwMemory::BufferManager::BufferInfoMapType::value_type elt, buffInfoMap)
00049     {
00050         ::fwMemory::BufferInfo dumpBuffInfo = elt.second;
00051         bool loaded = dumpBuffInfo.loaded;
00052         bool isLock = dumpBuffInfo.lockCount() > 0;
00053         if(loaded && !isLock)
00054         {
00055             bool dumped = buffManager->dumpBuffer(elt.first).get();
00056             if(dumped)
00057             {
00058                 ++nbBuffDumped;
00059             }
00060         }
00061     }
00062     std::stringstream stream;
00063     stream << nbBuffDumped << " buffer dumped (" << nbBuffDumped <<"/"<<buffInfoMap.size()<<").";
00064     ::fwGui::dialog::MessageDialog::showMessageDialog(
00065                         "Dump all",
00066                         stream.str(),
00067                         ::fwGui::dialog::IMessageDialog::INFO);
