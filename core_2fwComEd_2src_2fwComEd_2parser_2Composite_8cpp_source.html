00052 void Composite::createConfig( ::fwTools::Object::sptr _obj )
00053 {
00054     // Declaration of attributes values
00055     const std::string OBJECT_BUILD_MODE = "src";
00056     const std::string BUILD_OBJECT = "new";
00057     const std::string GET_OBJECT = "ref";
00058 
00059     ::fwData::Composite::sptr dataComposite = ::fwData::Composite::dynamicCast(_obj);
00060     SLM_ASSERT("Sorry, object given in parameter is not a fwData::Composite",dataComposite);
00061 
00062     BOOST_FOREACH( ::fwRuntime::ConfigurationElement::csptr elem, m_cfg->getElements() )
00063     {
00064         if( elem->getName() == "item" )
00065         {
00066 
00067             // Test build mode
00068             std::string buildMode = BUILD_OBJECT;
00069 
00070             if ( elem->hasAttribute( OBJECT_BUILD_MODE ) )
00071             {
00072                 buildMode = elem->getExistingAttributeValue( OBJECT_BUILD_MODE );
00073                 OSLM_ASSERT( "Sorry, buildMode \""<< buildMode <<"\" is not supported by the application.", buildMode == BUILD_OBJECT || buildMode == GET_OBJECT );
00074             }
00075 
00076             SLM_ASSERT( "Sorry, the xml element \"item\" must have an attribute named \"key\" .", elem->hasAttribute("key") );
00077             std::string key = elem->getExistingAttributeValue("key");
00078             SLM_ASSERT( "Sorry, the xml element \"item\" must have an attribute named \"key\" not empty.", ! key.empty() );
00079             SLM_ASSERT( "Sorry, xml element item must have one (and only one) xml sub-element \"object\".", elem->size() == 1 && (*elem->getElements().begin())->getName() == "object" );
00080 
00081             if( buildMode == BUILD_OBJECT )
00082             {
00083                 // Test if key already exist in composite
00084                 OSLM_ASSERT("Sorry the key "<< key <<" already exists in the composite.", dataComposite->find( key ) == dataComposite->end() );
00085 
00086                 // Create and manage object config
00087                 ::fwServices::AppConfigManager::sptr ctm = ::fwServices::AppConfigManager::New();
00088                 ctm->setConfig( * ( elem->getElements().begin() ) );
00089                 m_ctmContainer.push_back( ctm );
00090                 ctm->create();
00091                 ::fwData::Object::sptr localObj = ctm->getConfigRoot< ::fwData::Object >();
00092 
00093                 // Add object
00094                 SLM_ASSERT("Sorry an ::fwData::Composite can contain only ::fwData::Object", localObj );
00095                 (*dataComposite)[ key ] = localObj;
00096 
00097             }
00098             else // if( buildMode == GET_OBJECT )
00099             {
00100                 SLM_FATAL("ACH => Todo");
00101             }
00102         }
00103     }
