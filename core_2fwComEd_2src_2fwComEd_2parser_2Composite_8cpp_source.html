56 void Composite::createConfig( ::fwTools::Object::sptr _obj )
57 {
58  // Declaration of attributes values
59  const std::string OBJECT_BUILD_MODE = "src";
60  const std::string BUILD_OBJECT = "new";
61  const std::string GET_OBJECT = "ref";
62 
63  ::fwData::Composite::sptr dataComposite = ::fwData::Composite::dynamicCast(_obj);
64  SLM_ASSERT("The passed object must be a fwData::Composite",dataComposite);
65 
66  for( ::fwRuntime::ConfigurationElement::csptr elem : m_cfg->getElements() )
67  {
68  if( elem->getName() == "item" )
69  {
70 
71  // Test build mode
72  std::string buildMode = BUILD_OBJECT;
73 
74  if ( elem->hasAttribute( OBJECT_BUILD_MODE ) )
75  {
76  buildMode = elem->getExistingAttributeValue( OBJECT_BUILD_MODE );
77  OSLM_ASSERT( "The buildMode \""<< buildMode <<"\" is not supported, it should be either BUILD_OBJECT"
78  "or GET_OBJECT.",
79  buildMode == BUILD_OBJECT || buildMode == GET_OBJECT );
80  }
81 
82  SLM_ASSERT( "The xml element \"item\" must have an attribute named \"key\" .",
83  elem->hasAttribute("key") );
84  std::string key = elem->getExistingAttributeValue("key");
85  SLM_ASSERT( "The xml element \"item\" must have an attribute named \"key\" which is not empty.",
86  !key.empty() );
87  SLM_ASSERT( "The xml element \"item\" must have one (and only one) xml sub-element \"object\".",
88  elem->size() == 1 && (*elem->getElements().begin())->getName() == "object" );
89 
90  if( buildMode == BUILD_OBJECT )
91  {
92  // Test if key already exist in composite
93  OSLM_ASSERT("The key "<< key <<" already exists in the composite.", dataComposite->find(
94  key ) == dataComposite->end() );
95 
96  // Create and manage object config
97  ::fwServices::AppConfigManager::sptr ctm = ::fwServices::AppConfigManager::New();
98  ctm->setConfig( *( elem->getElements().begin() ) );
99  m_ctmContainer.push_back( ctm );
100  ctm->create();
101  ::fwData::Object::sptr localObj = ctm->getConfigRoot< ::fwData::Object >();
102 
103  // Add object
104  SLM_ASSERT("A ::fwData::Composite can contain only ::fwData::Object", localObj );
105  (*dataComposite)[ key ] = localObj;
106 
107  }
108  else // if( buildMode == GET_OBJECT )
109  {
110  SLM_FATAL("ACH => Todo");
111  }
112  }
113  }
114 }
