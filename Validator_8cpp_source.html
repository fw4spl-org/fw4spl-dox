00084 void Validator::initializeContext()
00085 {
00086     if(m_schemaValidContext)
00087     {
00088         return;
00089     }
00090 
00091     if ( !m_schemaParserContext )
00092     {
00093         if (!(m_schemaParserContext = SchemaParserCtxtSptr (
00094                   xmlSchemaNewParserCtxt(m_xsd_content.c_str()),
00095                   xmlSchemaFreeParserCtxt)
00096               ) )
00097         {
00098             return ;
00099         }
00100         // Set the structured error callback
00101         xmlSchemaSetParserStructuredErrors(m_schemaParserContext.get(), Validator::ErrorHandler, this );
00102     }
00103 
00104     // Load XML schema content
00105     if (!m_schema)
00106     {
00107         m_schema = SchemaSptr ( xmlSchemaParse(m_schemaParserContext.get()), xmlSchemaFree );
00108     }
00109     if (!m_schema)
00110     {
00111         return ;
00112     }
00113 
00114     // Create XML schemas validation context
00115     if ( (m_schemaValidContext = SchemaValidCtxtSptr( xmlSchemaNewValidCtxt(m_schema.get()), xmlSchemaFreeValidCtxt)) )
00116     {
00117         // Set the structured error callback
00118         xmlSchemaSetValidStructuredErrors( m_schemaValidContext.get(), Validator::ErrorHandler, this );
00119     }
00120 }

00130     xmlDocPtr xmlDoc = xmlParseFile ( xmlFile.string().c_str () );

00133         throw std::ios_base::failure("Unable to parse the XML file " + xmlFile.string() );
00134     }
00135     xmlNodePtr xmlRoot = xmlDocGetRootElement (xmlDoc);
00136     if (xmlXIncludeProcessTreeFlags (xmlRoot,XML_PARSE_NOBASEFIX) == -1)
00137     {
00138         xmlFreeDoc(xmlDoc);
00139         throw std::ios_base::failure(std::string ("Unable to manage xinclude !"));
00140     }
00141 
00142     if(!m_schemaValidContext)
00143     {
00144         return false;
00145     }
00146 
00147     result = xmlSchemaValidateDoc(m_schemaValidContext.get(), xmlDoc );
00148 
00149     xmlFreeDoc(xmlDoc);
00150 
00151     if ( result !=0 )

00153         OSLM_WARN("Validator::validation NOK, xml = " << xmlFile.string() ) ;
