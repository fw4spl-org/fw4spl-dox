00067 void LaunchBrowserActionService::updating() throw( ::fwTools::Failed )
00068 {
00069     SLM_TRACE_FUNC();
00070 
00071     SLM_WARN_IF("URL is empty.", m_url.empty());
00072     QUrl url(QString::fromStdString(m_url), QUrl::TolerantMode);
00073 
00074     if(url.isRelative()) // no scheme
00075     {
00076         ::boost::filesystem::path path(QCoreApplication::applicationDirPath().toStdString());
00077         path = path.parent_path(); // install folder path
00078         path /= url.path().toStdString();
00079 
00080         QString urlStr = QString::fromStdString("file:///" + path.string());
00081         url = QUrl(urlStr, QUrl::TolerantMode);
00082     }
00083 
00084     bool isSuccess = QDesktopServices::openUrl(url);
00085     SLM_WARN_IF("Browser wasn't successfully launched.", !isSuccess);
00086 }
