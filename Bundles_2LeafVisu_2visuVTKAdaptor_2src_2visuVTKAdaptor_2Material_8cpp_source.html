123 void Material::updateMaterial( SPTR(::fwData::Material)material )
124 {
125  //3DVSP-like rendering
126  m_property->SetLighting(material->getShadingMode() > 0);
127 
128  ::fwData::Color::sptr diffuse = material->diffuse();
129  m_property->SetDiffuseColor(diffuse->red(), diffuse->green(), diffuse->blue());
130 
131  // Use alpha from the diffuse color
132  m_property->SetOpacity( diffuse->alpha() );
133 
134  ::fwData::Color::sptr ambient = material->ambient();
135  m_property->SetAmbientColor(ambient->red(), ambient->green(), ambient->blue());
136 
137  m_property->SetSpecularColor(1.,1.,1.);
138  m_property->SetSpecularPower(100.); //Shininess
139 
140  // set texture
141  ::fwData::Image::sptr diffTex = material->getDiffuseTexture();
142 
143  if(diffTex != NULL)
144  {
145  ::fwData::mt::ObjectReadLock lock(diffTex);
146 
147  if (diffTex->getSizeInBytes() != 0)
148  {
149  vtkSmartPointer< vtkImageData > vtkImage = vtkSmartPointer< vtkImageData >::New();
150  ::fwVtkIO::toVTKImage( diffTex, vtkImage );
151 
152  vtkSmartPointer<vtkTexture> vtkTex = vtkSmartPointer< vtkTexture >::New();
153  vtkTex->SetInputData(vtkImage);
154 
155  ::fwData::Material::WrappingType wrapping = material->getDiffuseTextureWrapping();
156  vtkTex->SetRepeat( wrapping == ::fwData::Material::REPEAT );
157  vtkTex->SetEdgeClamp( wrapping == ::fwData::Material::CLAMP );
158  m_property->RemoveTexture("diffuse");
159  m_property->SetTexture("diffuse", vtkTex);
160  }
161  }
162 
163  switch(material->getRepresentationMode())
164  {
165  case ::fwData::Material::SURFACE:
166  m_property->SetRepresentationToSurface();
167  m_property->EdgeVisibilityOff();
168  break;
169 
170  case ::fwData::Material::EDGE:
171  m_property->SetRepresentationToSurface();
172  m_property->EdgeVisibilityOn();
173  break;
174 
175  case ::fwData::Material::WIREFRAME:
176  m_property->SetRepresentationToWireframe();
177  break;
178 
179  case ::fwData::Material::POINT:
180  m_property->SetRepresentationToPoints();
181  break;
182 
183  default:
184  OSLM_ASSERT("Unknown material representation mode : " << material->getRepresentationMode(), false );
185  }
186 
187  switch(material->getShadingMode())
188  {
190  case ::fwData::Material::AMBIENT:
191  break;
192  case ::fwData::Material::PHONG:
193  m_property->SetInterpolationToPhong();
194  break;
195 
196  case ::fwData::Material::GOURAUD:
197  m_property->SetInterpolationToGouraud();
198  break;
199 
200  case ::fwData::Material::FLAT:
201  m_property->SetInterpolationToFlat();
202  break;
203 
204  default:
205  OSLM_ASSERT("Unknown shading mode : " << material->getShadingMode(), false );
206  }
207 
208  m_property->Modified();
209  this->setVtkPipelineModified();
210 }
