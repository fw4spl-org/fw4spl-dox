224 ::fwThread::Worker::FutureType WorkerQt::getFuture()
225 {
226  if (!m_future.valid() )
227  {
228  SLM_ASSERT("WorkerQt loop shall be created and ran from main thread ",
229  !m_future.valid() && ::fwThread::getCurrentThreadId() == this->getThreadId() );
230 
231  ::boost::packaged_task< ExitReturnType > task( std::bind(&QApplication::exec) );
232 
233  ::boost::future< ExitReturnType > ufuture = task.get_future();
234 
235  m_future = ::boost::move(ufuture);
236 
237  task();
238  }
239 
240  return m_future;
241 }
