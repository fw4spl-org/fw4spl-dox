00048 ::boost::uint64_t Win32MemoryMonitorTools::estimateFreeMem()
00049 {
00050     ::boost::uint64_t freeMemory = 0;
00051 #ifdef _M_X64
00052     freeMemory = getFreeSystemMemory();
00053 #else
00054     ::boost::uint64_t windowsLimitForAProcess = 1.2 * 1024 * 1024 * 1024; // 1.5 Go
00055     ::boost::uint64_t freeMem = std::min(getFreeSystemMemory(), windowsLimitForAProcess - getUsedProcessMemory());

00063 void Win32MemoryMonitorTools::printProcessMemoryInformation()
00064 {
00065     DWORD processID = GetCurrentProcessId();
00066     HANDLE hProcess;
00067 
00068 
00069     hProcess = OpenProcess( PROCESS_QUERY_INFORMATION |PROCESS_VM_READ,
00070                             FALSE,
00071                             processID );
00072 
00073     if ( NULL != hProcess )
00074     {
00075         // Get the process name.
00076         int nameSize = 100;
00077         char * name = new char[nameSize];
00078 
00079         HMODULE hMod;
00080         DWORD cbNeeded;
00081 
00082         SLM_INFO( "-- Process memory information --" );
00083         if ( EnumProcessModules( hProcess, &hMod, sizeof(hMod), &cbNeeded) )
00084         {
00085             GetModuleBaseNameW( hProcess, hMod, (LPWSTR)name, nameSize );
00086             OSLM_INFO( "    Name : " << name << "  ( PID : " << processID << " )");
00087         }
00088         else
00089         {
00090             OSLM_INFO( "    Name : not found ( PID : " << processID << " )");
00091         }
00092 
00093         PROCESS_MEMORY_COUNTERS_EX pmc;
00094         if ( GetProcessMemoryInfo( hProcess, reinterpret_cast<PROCESS_MEMORY_COUNTERS*>(&pmc), sizeof(pmc)) )
00095         {
00096             OSLM_INFO("    PageFaultCount : "               << (int)pmc.PageFaultCount );
00097             OSLM_INFO("    PeakWorkingSetSize : "           << (int)pmc.PeakWorkingSetSize );
00098             OSLM_INFO("    WorkingSetSize : "               << (int)pmc.WorkingSetSize );
00099             OSLM_INFO("    QuotaPeakPagedPoolUsage : "      << (int)pmc.QuotaPeakPagedPoolUsage );
00100             OSLM_INFO("    QuotaPagedPoolUsage : "          << (int)pmc.QuotaPagedPoolUsage );
00101             OSLM_INFO("    QuotaPeakNonPagedPoolUsage : "   << (int)pmc.QuotaPeakNonPagedPoolUsage );
00102             OSLM_INFO("    QuotaNonPagedPoolUsage : "       << (int)pmc.QuotaNonPagedPoolUsage );
00103             OSLM_INFO("    PagefileUsage : "                << (int)pmc.PagefileUsage );
00104             OSLM_INFO("    PeakPagefileUsage : "            << (int)pmc.PeakPagefileUsage );
00105             OSLM_INFO("    PrivateUsage : "                 << (int)pmc.PrivateUsage );
00106         }
00107         else
00108         {
00109             SLM_WARN("GetProcessMemoryInfo failed !");
00110         }
00111 
00112         delete[] name;
00113         CloseHandle( hProcess );
00114     }
00115 }
