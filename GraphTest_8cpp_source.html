50 
51  ::fwData::Node::sptr n1( ::fwData::Node::New() );
52  ::fwData::Port::sptr p1( ::fwData::Port::New() );
53  p1->setIdentifier("sizex");
54  p1->setType("float");
55  n1->addOutputPort( p1 );
56 
57  ::fwData::Node::sptr n2( ::fwData::Node::New() );
58  ::fwData::Port::sptr p2( ::fwData::Port::New() );
59  p2->setIdentifier("threshold");
60  p2->setType("float");
61  n2->addInputPort( p2 );
62 
63  ::fwData::Edge::sptr e( ::fwData::Edge::New() );
64  e->setIdentifiers("sizex","threshold");
65 
66  g->addNode(n1);
67  g->addNode(n2);
68  g->addEdge(e,n1,n2);
69 
70  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getInputEdges(n1).size() );
71  CPPUNIT_ASSERT_EQUAL( (size_t)1, g->getOutputEdges(n1).size() );
72  CPPUNIT_ASSERT_EQUAL( e, g->getOutputEdges(n1).at(0) );
73  CPPUNIT_ASSERT_EQUAL( (size_t)1, g->getInputEdges(n2).size() );
74  CPPUNIT_ASSERT_EQUAL( e, g->getInputEdges(n2).at(0) );
75  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getOutputEdges(n2).size() );
76 
77  CPPUNIT_ASSERT_EQUAL( n1, g->getSourceNode(e) );
78  CPPUNIT_ASSERT_EQUAL( n2, g->getDestinationNode(e) );
79 
80  // Remove to obtain empty graph
81  CPPUNIT_ASSERT_EQUAL( (size_t)2, g->getNbNodes() );
82  CPPUNIT_ASSERT_EQUAL( (size_t)1, g->getNbEdges() );
83  g->removeEdge(e);
84  CPPUNIT_ASSERT_EQUAL( (size_t)2, g->getNbNodes() );
85  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getNbEdges() );
86  g->removeNode(n1);
87  CPPUNIT_ASSERT_EQUAL( (size_t)1, g->getNbNodes() );
88  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getNbEdges() );
89  g->removeNode(n2);
90  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getNbNodes() );
91  CPPUNIT_ASSERT_EQUAL( (size_t)0, g->getNbEdges() );
