120 QTextDocument* ShowRevInfo::generateReport()
121 {
122  QTextDocument* document = new QTextDocument();
123  QTextCursor cursor(document);
124 
125  QTextCharFormat defaultFormat;
126  defaultFormat.setFontPointSize(12);
127 
128  QTextCharFormat boldFormat = defaultFormat;
129  boldFormat.setFontWeight(QFont::Bold);
130 
131  QTextCharFormat titleFormat = boldFormat;
132 
133  titleFormat.setFontPointSize(20);
134 
135  QTextBlockFormat centerFormat;
136  centerFormat.setAlignment(Qt::AlignHCenter);
137 
138  cursor.setBlockFormat(centerFormat);
139  cursor.insertText(QString("Revision info"), titleFormat);
140  cursor.insertBlock();
141  cursor.insertBlock();
142 
143  QTextBlockFormat leftFormat;
144  leftFormat.setAlignment(Qt::AlignLeft);
145  cursor.setBlockFormat(leftFormat);
146 
147  QTextTableFormat tableFormat;
148  tableFormat.setCellPadding(5);
149  tableFormat.setMargin(0);
150  tableFormat.setBorder(1);
151  tableFormat.setPadding(0);
152  tableFormat.setHeaderRowCount(0);
153  tableFormat.setBorderStyle(QTextFrameFormat::BorderStyle_Solid);
154  tableFormat.setWidth(QTextLength(QTextLength::PercentageLength, 100));
155 
156  ::boost::filesystem::path bundlePath = ::boost::filesystem::current_path() /
157  ::boost::filesystem::path(BUNDLE_PREFIX);
158  ::boost::filesystem::path libPath = ::boost::filesystem::current_path() / "share";
159  const FindMapType& bundleMap = this->findRevInfo(bundlePath);
160  const FindMapType& libMap = this->findRevInfo(libPath);
161 
162  RevMapType revMap;
163  this->getRevMap(bundleMap, revMap, true);
164  this->getRevMap(libMap, revMap, false);
165 
166  cursor.movePosition(QTextCursor::EndOfBlock);
167  cursor.movePosition(QTextCursor::NextBlock);
168 
169  cursor.insertBlock();
170  cursor.insertTable( revMap.size()+1, 3, tableFormat);
171 
172  cursor.setBlockFormat(centerFormat);
173  cursor.insertText(tr("Revision"), boldFormat);
174  cursor.movePosition(QTextCursor::NextCell);
175 
176  cursor.setBlockFormat(centerFormat);
177  cursor.insertText(tr("Bundle"), boldFormat);
178  cursor.movePosition(QTextCursor::NextCell);
179 
180  cursor.setBlockFormat(centerFormat);
181  cursor.insertText(tr("Lib"), boldFormat);
182  cursor.movePosition(QTextCursor::NextCell);
183 
184  for(RevMapType::value_type val : revMap)
185  {
186  cursor.insertHtml(QString::fromStdString(val.first).replace("\n","<br/>"));
187  cursor.movePosition(QTextCursor::NextCell);
188  cursor.setBlockFormat(centerFormat);
189  for(std::string name : val.second.first)
190  {
191  cursor.insertHtml(QString::fromStdString("<br/>" + name));
192  }
193 
194  cursor.movePosition(QTextCursor::NextCell);
195  cursor.setBlockFormat(centerFormat);
196  for(std::string name : val.second.second)
197  {
198  cursor.insertHtml(QString::fromStdString("<br/>" + name));
199  }
200  cursor.movePosition(QTextCursor::NextCell);
201  }
202  return document;
203 }

208 {

229 void ShowRevInfo::getRevMap(const ShowRevInfo::FindMapType& findMap, ShowRevInfo::RevMapType& revMap, bool isBundle)
230 {
231  for(FindMapType::value_type val : findMap)
232  {
233  size_t length;
234  std::string buf;
235  std::ifstream file;
236  file.open(val.second.string().c_str(), std::ios::binary );
237 
238  if (!file.is_open())
239  {
240  OSLM_ERROR( "RevInfo file loading error for " << val.second.string());
241  throw std::ios_base::failure("Unable to open " + val.second.string());
242  }
243 
244  file.seekg (0, std::ios::end);
245  length = file.tellg();
246  file.seekg (0, std::ios::beg);
247  buf.resize(length);
248  char* buffer = &buf[0];
249 
250  file.read (buffer, length);
251  file.close();
252 
253  if(isBundle)
254  {
255  revMap[buf].first.insert(val.first);
256  }
257  else
258  {
259  revMap[buf].second.insert(val.first);
260  }
261  }
262 }

266 void ShowRevInfo::saveReport()
267 {
268  static ::boost::filesystem::path _sDefaultPath("");
269  ::fwGui::dialog::LocationDialog dialogFile;
270  dialogFile.setTitle("Choose a location to save revision info report");
271  dialogFile.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
272  dialogFile.addFilter("ODT","*.odt");
273  dialogFile.addFilter("HTML","*.html");
274  dialogFile.setOption(::fwGui::dialog::ILocationDialog::WRITE);
275 
276  ::fwData::location::SingleFile::sptr result = ::fwData::location::SingleFile::dynamicCast( dialogFile.show() );
277 
278  if( result )
279  {
280  ::boost::filesystem::path filename = result->getPath();
281  _sDefaultPath = filename.parent_path();
282  dialogFile.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
283 
284  QTextDocument* document = this->generateReport();
285 
286  QTextDocumentWriter writer(QString::fromStdString(filename.string()));
287  if( ::boost::filesystem::extension(filename) == ".odt" )
288  {
289  writer.setFormat("ODF");
290  }
291  else
292  {
293  writer.setFormat("HTML");
294  }
295  writer.write(document);
296  delete document;
297  }
