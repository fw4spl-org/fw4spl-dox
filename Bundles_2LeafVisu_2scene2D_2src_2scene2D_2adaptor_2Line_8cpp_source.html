41 void Line::configuring() throw ( ::fwTools::Failed )
42 {
43  SLM_TRACE_FUNC();
44 
45  SLM_ASSERT("\"config\" tag missing", m_configuration->getName() == "config");
46 
47  this->IAdaptor::configuring();
48 
49  SLM_TRACE("IAdaptor configuring ok");
50 
51  // Set the beginning and ending coordinates values
52  m_x1 = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("x1") );
53  m_x2 = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("x2") );
54  m_y1 = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("y1") );
55  m_y2 = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("y2") );
56 
57  // If the corresponding attributes are present in the config, set the color of the line
58  if (!m_configuration->getAttributeValue("color").empty())
59  {
60  ::scene2D::data::InitQtPen::setPenColor(m_pen, m_configuration->getAttributeValue("color"));
61  }
62 }

66 void Line::draw()
67 {
68  SLM_TRACE_FUNC();
69 
70  // Draw the line
71  QGraphicsLineItem* line = new QGraphicsLineItem(
72  this->mapAdaptorToScene(std::pair< double, double >( m_x1, m_y1), m_xAxis, m_yAxis).first,
73  this->mapAdaptorToScene(std::pair< double, double >( m_x1, m_y1), m_xAxis, m_yAxis).second,
74  this->mapAdaptorToScene(std::pair< double, double >( m_x2, m_y2), m_xAxis, m_yAxis).first,
75  this->mapAdaptorToScene(std::pair< double, double >( m_x2, m_y2), m_xAxis, m_yAxis).second
76  );
77  // Set the line the pen
78  line->setPen(m_pen);
79 
80  // Add the line to the layer
81  m_layer->addToGroup(line);
82 
83  // Set the layer position (according to the related axis) and zValue
84  m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
85  m_layer->setZValue(m_zValue);
86  // Add the layer to the scene
87  this->getScene2DRender()->getScene()->addItem(m_layer);
88 }
