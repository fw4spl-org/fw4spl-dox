00059 void Line::draw()
00060 {
00061     SLM_TRACE_FUNC();
00062 
00063     // Draw the line
00064     QGraphicsLineItem* line = new QGraphicsLineItem(
00065         this->mapAdaptorToScene(std::pair< double , double >( m_x1, m_y1) , m_xAxis, m_yAxis).first,
00066         this->mapAdaptorToScene(std::pair< double , double >( m_x1, m_y1) , m_xAxis, m_yAxis).second,
00067         this->mapAdaptorToScene(std::pair< double , double >( m_x2, m_y2) , m_xAxis, m_yAxis).first,
00068         this->mapAdaptorToScene(std::pair< double , double >( m_x2, m_y2) , m_xAxis, m_yAxis).second
00069     );
00070     // Set the line the pen
00071     line->setPen(m_pen);
00072 
00073     // Add the line to the layer
00074     m_layer->addToGroup(line);
00075 
00076     // Set the layer position (according to the related axis) and zValue
00077     m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
00078     m_layer->setZValue(m_zValue);
00079     // Add the layer to the scene
00080     this->getScene2DRender()->getScene()->addItem(m_layer);
