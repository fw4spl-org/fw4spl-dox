42 void Plugin::initialize() throw( ::fwRuntime::RuntimeException )
43 {
44  m_image = ::fwData::Image::New();
45 
46  // Reader service
47  m_readerSrv = ::fwServices::add(m_image, "::io::IReader", "::ioVTK::SImageReader");
48  ::fwServices::IService::ConfigType readerCfg;
49  readerCfg.put("service.file", "./TutoData/patient1.vtk");
50  m_readerSrv->setConfiguration( readerCfg );
51  m_readerSrv->configure();
52 
53  // Render service
54  m_renderSrv = ::fwServices::add(m_image, "::fwRender::IRender", "::vtkSimpleNegato::SRenderer",
55  "myRenderingTuto");
56  m_renderSrv->configure();
57 
58  // Frame service
59  m_frameSrv = ::fwServices::add(m_image, "::fwGui::IFrameSrv", "::gui::frame::SDefaultFrame");
60 
61  ::fwServices::IService::ConfigType frameConfig;
62 
63  frameConfig.put("service.gui.frame.name", "tutoDataServiceBasic");
64  frameConfig.put("service.gui.frame.icon", "Bundles/Tuto02DataServiceBasicCtrl_0-1/tuto.ico");
65  frameConfig.put("service.gui.frame.minSize.<xmlattr>.width", "800");
66  frameConfig.put("service.gui.frame.minSize.<xmlattr>.height", "600");
67 
68  frameConfig.put("service.registry.view.<xmlattr>.sid", "myRenderingTuto");
69 
70  m_frameSrv->setConfiguration( frameConfig );
71  m_frameSrv->configure();
72 
73  // Start app
74  m_frameSrv->start();
75  m_readerSrv->start();
76  m_renderSrv->start();
77 
78  // Update
79  m_readerSrv->update();
80  m_renderSrv->update();
81 }
