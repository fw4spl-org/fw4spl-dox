45 ::fwData::Composite::sptr vectorToComposite(::fwData::Vector::sptr vector,
46  const ::fwActivities::registry::ActivityRequirement &req)
47 {
48  namespace ActReg = ::fwActivities::registry;
49  ::fwData::Composite::sptr composite = ::fwData::Composite::New();
50 
51  OSLM_ASSERT("Each possible items in requirement need to have a matching key", req.keys.size() >= req.maxOccurs );
52 
53  ActReg::ActivityRequirement::KeyType::const_iterator iter = req.keys.begin();
54 
55  for(const ::fwData::Object::sptr &obj : *vector)
56  {
57  const ActReg::ActivityRequirementKey &keyTag = (*iter++);
58  if(keyTag.path.empty())
59  {
60  (*composite)[keyTag.key] = obj;
61  }
62  else
63  {
64  (*composite)[keyTag.key] = ::fwDataCamp::getObject( obj, keyTag.path );
65  }
66  }
67 
68  return composite;
69 }

73 ::fwMedData::ActivitySeries::sptr ActivitySeries::buildData(
74  const ::fwActivities::registry::ActivityInfo& activityInfo,
75  ::fwData::Vector::sptr currentSelection ) const
76 {
77  ::fwMedData::ActivitySeries::sptr actSeries = ::fwMedData::ActivitySeries::New();
78 
79  ::fwMedData::Series::sptr series;
80  for(const ::fwData::Object::sptr& obj : *currentSelection)
81  {
82  series = ::fwMedData::Series::dynamicCast(obj);
83  if(series)
84  {
85  break;
86  }
87  }
88 
89  if(series)
90  {
91  actSeries->setPatient( ::fwData::Object::copy(series->getPatient()) );
92  actSeries->setStudy( ::fwData::Object::copy(series->getStudy()) );
93  actSeries->setEquipment( ::fwData::Object::copy(series->getEquipment()) );
94  }
95 
96  actSeries->setModality("OT");
97  actSeries->setInstanceUID("fwActivities." + ::fwTools::UUID::generateUUID() );
98 
99  ::boost::posix_time::ptime now = ::boost::posix_time::second_clock::local_time();
100  actSeries->setDate(::fwTools::getDate(now));
101  actSeries->setTime(::fwTools::getTime(now));
102 
103  actSeries->setActivityConfigId(activityInfo.id);
104  ::fwData::Composite::sptr data = actSeries->getData();
105 
106  namespace ActReg = ::fwActivities::registry;
107 
108  ActReg::ActivityInfo::RequirementsType reqVect = activityInfo.requirements;
109  for(const ActReg::ActivityRequirement& req : reqVect)
110  {
111  ::fwData::Vector::sptr vectorType = this->getType(currentSelection, req.type);
112  // param is optional (minOccurs==0) or required (minOccurs==1), but is single (maxOccurs == 1)
113  if(req.maxOccurs == 1 && req.minOccurs == 1)
114  {
115  OSLM_ASSERT("No param name "<<req.name<<" with type "<<req.type, !vectorType->empty());
116  (*data)[req.name] = (*vectorType)[0];
117  }
118  else
119  {
120  SLM_ASSERT("Unknown specified container: '"+req.container+"'.",
121  req.container.empty() ||
122  req.container == "vector" ||
123  req.container == "composite");
124  if(req.container == "vector")
125  {
126  (*data)[req.name] = vectorType;
127  }
128  else if(req.container == "composite" || req.container.empty())
129  {
130  (*data)[req.name] = vectorToComposite(vectorType, req);
131  }
132  }
133  }
134 
135  return actSeries;
136 }
