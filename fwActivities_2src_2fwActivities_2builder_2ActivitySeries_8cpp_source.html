00045 ::fwData::Composite::sptr vectorToComposite(::fwData::Vector::sptr vector,
00046                                             const ::fwActivities::registry::ActivityRequirement &req)
00047 {
00048     namespace ActReg = ::fwActivities::registry;
00049     ::fwData::Composite::sptr composite = ::fwData::Composite::New();
00050 
00051     OSLM_ASSERT("Each possible items in requirement need to have a matching key", req.keys.size() >= req.maxOccurs );
00052 
00053     ActReg::ActivityRequirement::KeyType::const_iterator iter = req.keys.begin();
00054 
00055     BOOST_FOREACH(const ::fwData::Object::sptr &obj, *vector)
00056     {
00057         const ActReg::ActivityRequirementKey &keyTag = (*iter++);
00058         if(keyTag.path.empty())
00059         {
00060             (*composite)[keyTag.key] = obj;
00061         }
00062         else
00063         {
00064             (*composite)[keyTag.key] = ::fwDataCamp::getObject( obj, keyTag.path );
00065         }
00066     }
00067 
00068     return composite;
00069 }

00085             break;

00098 
00099     ::boost::posix_time::ptime now = ::boost::posix_time::second_clock::local_time();
00100     actSeries->setDate(::fwTools::getDate(now));
00101     actSeries->setTime(::fwTools::getTime(now));
00102 
00103     actSeries->setActivityConfigId(activityInfo.id);
00104     ::fwData::Composite::sptr data = actSeries->getData();
00105 
00106     namespace ActReg = ::fwActivities::registry;
00107 
00108     ActReg::ActivityInfo::RequirementsType reqVect = activityInfo.requirements;
00109     BOOST_FOREACH(const ActReg::ActivityRequirement& req, reqVect)
00110     {
00111         ::fwData::Vector::sptr vectorType = this->getType(currentSelection, req.type);
00112         // param is optional (minOccurs==0) or required (minOccurs==1), but is single (maxOccurs == 1)
00113         if(req.maxOccurs == 1 && req.minOccurs == 1)
00114         {
00115             OSLM_ASSERT("No param name "<<req.name<<" with type "<<req.type, !vectorType->empty());
00116             (*data)[req.name] = (*vectorType)[0];
00117         }
00118         else
00119         {
00120             SLM_ASSERT("Unknown specified container: '"+req.container+"'.",
00121                        req.container.empty() ||
00122                        req.container == "vector" ||
00123                        req.container == "composite");
00124             if(req.container == "vector")
00125             {
00126                 (*data)[req.name] = vectorType;
00127             }
00128             else if(req.container == "composite" || req.container.empty())
00129             {
00130                 (*data)[req.name] = vectorToComposite(vectorType, req);
00131             }
00132         }
00133     }
00134 
00135     return actSeries;
00136 }
