00061 Numeric::ValueType Numeric::valueFromString(const std::string &s, Numeric::NumericType type)
00062 {
00063     Numeric::ValueType res;
00064     SLM_ASSERT("Invalid variant type requested", EMPTY <= type && type <= DOUBLE);
00065 
00066     if ( type == EMPTY)
00067     {
00068         FW_RAISE_EXCEPTION_IF(
00069             ::fwAtoms::Exception( std::string("Unable to get numeric from '") + s + "'"),
00070             !(lexicalCast< ::boost::int64_t >(res, s)
00071               || lexicalCast< ::boost::uint64_t >(res, s)
00072               || lexicalCast< double >(res, s))
00073             );
00074     }
00075     else if ( type == INT)
00076     {
00077         FW_RAISE_EXCEPTION_IF(
00078             ::fwAtoms::Exception( std::string("Unable to get int64 numeric from '") + s + "'"),
00079             !lexicalCast< ::boost::int64_t >(res, s)
00080             );
00081     }
00082     else if ( type == UINT)
00083     {
00084         FW_RAISE_EXCEPTION_IF(
00085             ::fwAtoms::Exception( std::string("Unable to get uint64 numeric from '") + s + "'"),
00086             !lexicalCast< ::boost::uint64_t >(res, s)
00087             );
00088     }
00089     else if ( type == DOUBLE)
00090     {
00091         FW_RAISE_EXCEPTION_IF(
00092             ::fwAtoms::Exception( std::string("Unable to get double numeric from '") + s + "'"),
00093             !lexicalCast< double >(res, s)
00094             );
00095     }
00096 
00097     return res;
00098 }
