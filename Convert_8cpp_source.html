46 
47  std::string nodeValue = _cfgElement->getValue();

49  {
50  xmlNodeSetContent (pNode, reinterpret_cast<const xmlChar *>(nodeValue.c_str()));
51  }
52 
53  std::map<std::string, std::string> attr_cfe = _cfgElement->getAttributes();
54 
55 
56  for ( std::map<std::string, std::string>::iterator iter_attr_cfe = attr_cfe.begin();
57  iter_attr_cfe!= attr_cfe.end();
58  ++iter_attr_cfe)
59  {
60  //ATTRIBUTES + VALUES
61  xmlSetProp(pNode, xmlCharStrdup((iter_attr_cfe->first).c_str()),
62  xmlCharStrdup((iter_attr_cfe->second).c_str()) );
63  if ((iter_attr_cfe->first) == std::string("class"))
64  {
65  xmlSetProp(pNode, xmlCharStrdup((iter_attr_cfe->first).c_str()),
66  xmlCharStrdup((iter_attr_cfe->second).c_str()) );
67  }
68  }
69 
70  SLM_ASSERT( "ConfigurationElement should not have children("
71  << _cfgElement->size() <<") and a value ("
72  << nodeValue <<") at the same time.",
73  !(!nodeValue.empty() && _cfgElement->size()) );
74 
75  for ( std::vector< ::fwRuntime::ConfigurationElement::sptr >::iterator iter_cfeC = _cfgElement->begin();
76  iter_cfeC != _cfgElement->end();
77  ++iter_cfeC )
78  {
79  fromConfigurationElementToXml( (*iter_cfeC), pNode);
80  }

86 xmlNodePtr Convert::runningBundlesToXml( )
87 {
88  xmlNodePtr node_root = xmlNewNode( NULL, xmlCharStrdup( "Bundles" ) );
89  std::set< std::shared_ptr< ::fwRuntime::Bundle > > ::iterator iter_bundles;
90  ::fwRuntime::Runtime * tmp_runtime = ::fwRuntime::Runtime::getDefault();
91 
92  xmlNodePtr activated_Node = xmlNewNode( NULL, xmlCharStrdup( "Activated" ) );
93  xmlAddChild(node_root, activated_Node );
94 
95  xmlNodePtr inactivated_Node = xmlNewNode( NULL, xmlCharStrdup( "Inactivated" ) );
96  xmlAddChild(node_root, inactivated_Node );
97 
98  bool enable_Value = false; // the 'do while' loop stop if enable_Value==false.
99  do
100  {
101  enable_Value = !enable_Value;
102  for (iter_bundles = tmp_runtime->bundlesBegin();
103  iter_bundles != tmp_runtime->bundlesEnd();
104  ++iter_bundles)
105  {
106  //BUNDLE
107 
108  xmlNodePtr bundleNode = xmlNewNode( NULL, xmlCharStrdup( (*iter_bundles)->getIdentifier().c_str() ) );
109  if (enable_Value)
110  {
111  xmlAddChild(activated_Node, bundleNode );
112  }
113  else
114  {
115  xmlAddChild(inactivated_Node, bundleNode );
116  }
117 
118  //EXTENSIONS POINTS
119  xmlNodePtr extensionPoint_activated_list_Node = xmlNewNode( NULL, xmlCharStrdup( "Extensions_Points" ) );
120  xmlAddChild(bundleNode, extensionPoint_activated_list_Node );
121 
122  for ( std::set< std::shared_ptr< ::fwRuntime::ExtensionPoint > >::const_iterator iter_extensionPoints =
123  (*iter_bundles)->extensionPointsBegin();
124  iter_extensionPoints != (*iter_bundles)->extensionPointsEnd();
125  ++iter_extensionPoints)
126  {
127  //EXTENSIONS POINTS
128  std::string str = "Identifier : "+(*iter_extensionPoints)->getIdentifier();
129  //-----DEBUG------
130  xmlNodePtr extensionPointsNode = xmlNewNode( NULL, xmlCharStrdup( str.c_str() ) );
131  //xmlNodePtr extensionPointsNode = xmlNewNode( NULL, xmlCharStrdup( (str.substr(str.find_last_of("::")+1)).c_str() ) ) ;
132  //-----DEBUG------
133  if (((*iter_extensionPoints)->isEnable())==enable_Value)
134  {
135  xmlAddChild(extensionPoint_activated_list_Node, extensionPointsNode );
136  }
137 
138  std::vector< ::fwRuntime::ConfigurationElement::sptr > AllConfigElement =
139  ((*iter_extensionPoints)->getAllConfigurationElements());
140  for (std::vector< ::fwRuntime::ConfigurationElement::sptr >::const_iterator iter_config_elem =
141  AllConfigElement.begin();
142  iter_config_elem != AllConfigElement.end();
143  ++iter_config_elem)
144  {
145  //CONFIGURATIONS ELEMENTS
146  Convert::fromConfigurationElementToXml( (*iter_config_elem), extensionPointsNode);
147  }
148  }//end extensionsPoints parsing
149 
150  //Extensions
151  xmlNodePtr extension_activated_list_Node = xmlNewNode( NULL, xmlCharStrdup( "Extensions" ) );
152  xmlAddChild(bundleNode, extension_activated_list_Node );
153 
154  for ( std::set< std::shared_ptr< ::fwRuntime::Extension > >::const_iterator iter_extension =
155  (*iter_bundles)->extensionsBegin();
156  iter_extension != (*iter_bundles)->extensionsEnd();
157  ++iter_extension)
158  {
159  if (((*iter_extension)->isEnable())==enable_Value)
160  {
161  std::string str = (*iter_extension)->getPoint();
162  xmlNodePtr extensionNode = xmlNewNode( NULL, xmlCharStrdup( (str.c_str() ) ) );
163 
164  // Adds node if not exist
165  xmlNodePtr node;
166  bool found_node = false;
167  for ( node = extension_activated_list_Node->children;
168  node;
169  node = node->next)
170  {
171  if ( (node->type == XML_ELEMENT_NODE) &&
172  !(xmlStrcmp(node->name, xmlCharStrdup( (str.c_str() ) ) ) ) )
173  {
174  extensionNode = node;
175  found_node = true;
176  break;
177  }
178  }
179 
180  if (!found_node)
181  {
182  xmlAddChild(extension_activated_list_Node, extensionNode );
183  }
184  //end adds node
185 
186  for (std::vector< ::fwRuntime::ConfigurationElement::sptr >::iterator iter_cfe_extension =
187  (*iter_extension)->begin();
188  iter_cfe_extension != (*iter_extension)->end();
189  ++iter_cfe_extension )
190  {
191  Convert::fromConfigurationElementToXml( (*iter_cfe_extension), extensionNode);
192  }
193  }
194  }//end Extensions parsing
195 
196  //cleaning : delete empty node (or if contain empty node)
197  // xmlKeepBlanksDefault(0) don't work; only for TEXT NODES
198  // xmlIsBlankNode too, work only for Text-Nodes
199  if (!(extensionPoint_activated_list_Node->children))
200  {
201  xmlUnlinkNode(extensionPoint_activated_list_Node);
202  xmlFreeNode(extensionPoint_activated_list_Node);
203  }
204 
205  if (!(extension_activated_list_Node->children))
206  {
207  xmlUnlinkNode(extension_activated_list_Node);
208  xmlFreeNode(extension_activated_list_Node);
209  }
210 
211  if (!(bundleNode->children))
212  {
213  xmlUnlinkNode(bundleNode);
214  xmlFreeNode(bundleNode);
215  }
216  //end cleaning
217  }//end bundles iterator
218  }
219  while ( enable_Value );
220 
221  return node_root;
222 }//runningBundlesToXml

248 ::boost::property_tree::ptree Convert::toPropertyTree( ::fwRuntime::ConfigurationElement::sptr _cfgElement)
249 {
250  ::boost::property_tree::ptree pt;
251  ::boost::property_tree::ptree ptAttr;
252 
253  std::string propertyName = _cfgElement->getName();
254  std::string propertyValue = _cfgElement->getValue();
255 
256  typedef std::map<std::string, std::string> AttributeMatType;
257 
258  if(!propertyValue.empty())
259  {
260  pt.put(propertyName, propertyValue);
261  }
262 
263  AttributeMatType attr = _cfgElement->getAttributes();
264 
265  for ( AttributeMatType::iterator iter = attr.begin(); iter!= attr.end(); ++iter)
266  {
267  ptAttr.put(iter->first,iter->second);
268  }
269 
270  if(!ptAttr.empty())
271  {
272  pt.put_child(propertyName + ".<xmlattr>", ptAttr);
273  }
274 
275  for ( ::fwRuntime::ConfigurationElementContainer::Iterator iterElement = _cfgElement->begin();
276  iterElement != _cfgElement->end();
277  ++iterElement )
278  {
279  std::string childName = (*iterElement)->getName();
280  ::boost::property_tree::ptree ptChild;
281 
282  ptChild = toPropertyTree (*iterElement);
283 
284  boost::optional< ::boost::property_tree::ptree & > child = ptChild.get_child_optional(childName);
285 
286  if(child)
287  {
288  ptChild = *child;
289  }
290 
291  pt.add_child( propertyName + "." + childName, ptChild );
292  }
293  return pt;
294 }
