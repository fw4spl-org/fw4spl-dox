62 struct SetFromConfig

65  {
66  if(config)

69  if(it != config->end())
70  {
71  ::fwData::String::sptr obj = ::fwData::String::dynamicCast(it->second);
72  attribute = T(::boost::lexical_cast<CAST_T>(obj->value()));
73  }
74  }
75  }

149  SLM_ASSERT("It should be a \"retrieveMethod\" tag in the "

161  // Set information from xml and update PacsConfiguration

194 void SPacsConfigurationInitializer::updating() throw(::fwTools::Failed)
195 {
196  SLM_TRACE_FUNC();
197  ::fwPacsIO::data::PacsConfiguration::sptr pacsConfiguration =
198  this->getObject< ::fwPacsIO::data::PacsConfiguration >();
199 
200  // Check if the user has changed the Pacs configuration and update the local var
201  if(pacsConfiguration->getLocalApplicationTitle () != m_localApplicationTitle
202  || pacsConfiguration->getPacsHostName () != m_pacsHostName
203  || pacsConfiguration->getPacsApplicationTitle () != m_pacsApplicationTitle
204  || pacsConfiguration->getPacsApplicationPort () != m_pacsApplicationPort
205  || pacsConfiguration->getMoveApplicationTitle () != m_moveApplicationTitle
206  || pacsConfiguration->getMoveApplicationPort () != m_moveApplicationPort
207  || pacsConfiguration->getRetrieveMethod () != m_retrieveMethod
208  )
209  {
210  m_localApplicationTitle = pacsConfiguration->getLocalApplicationTitle ();
211  m_pacsHostName = pacsConfiguration->getPacsHostName ();
212  m_pacsApplicationTitle = pacsConfiguration->getPacsApplicationTitle ();
213  m_pacsApplicationPort = pacsConfiguration->getPacsApplicationPort ();
214  m_moveApplicationTitle = pacsConfiguration->getMoveApplicationTitle ();
215  m_moveApplicationPort = pacsConfiguration->getMoveApplicationPort ();
216  m_retrieveMethod = pacsConfiguration->getRetrieveMethod ();
217  }
218 
219  // If a preference key is set, save the local var to the preferences
220  if(!m_preferenceKey.empty())
221  {
222  std::pair< ::fwServices::IService::sptr, ::fwData::Composite::sptr> prefs_pair = getPreferences();
223  ::fwServices::IService::sptr prefService = prefs_pair.first;
224  ::fwData::Composite::sptr prefs = prefs_pair.second;
225  if(prefs && (prefs->find(m_preferenceKey) == prefs->end() || !(*prefs)[m_preferenceKey]))
226  {
227  (*prefs)[m_preferenceKey] = ::fwData::Composite::New();
228  }
229  if(prefs)
230  {
231  ::fwData::Composite::sptr config = ::fwData::Composite::dynamicCast((*prefs)[m_preferenceKey]);
232  (*config)["LocalApplicationTitle"] = ::fwData::String::New(m_localApplicationTitle);
233  (*config)["PacsHostName" ] = ::fwData::String::New(m_pacsHostName);
234  (*config)["PacsApplicationTitle" ] = ::fwData::String::New(m_pacsApplicationTitle);
235  (*config)["PacsApplicationPort" ] =
236  ::fwData::String::New(::boost::lexical_cast<std::string>(m_pacsApplicationPort));
237  (*config)["MoveApplicationTitle" ] = ::fwData::String::New(m_moveApplicationTitle);
238  (*config)["MoveApplicationPort" ] =
239  ::fwData::String::New(::boost::lexical_cast<std::string>(m_moveApplicationPort));
240  (*config)["RetrieveMethod" ] =
241  ::fwData::String::New(::boost::lexical_cast<std::string>(m_retrieveMethod));
242  }
243  if(prefService)
244  {
245  prefService->update();
246  }
247  }
248 }
