52 void MenuBarRegistrar::initialize( ::fwRuntime::ConfigurationElement::sptr configuration)
53 {
54  OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be registry",
55  configuration->getName() == "registry");
56 
57  // index represents associated menu with position in menus vector
58  unsigned int index = 0;
59  // initialize m_menuSids map with configuration
60  std::vector < ConfigurationType > vectMenus = configuration->find("menu");
61  for( ConfigurationType menu : vectMenus)
62  {
63  SLM_ASSERT("<menu> tag must have sid attribute", menu->hasAttribute("sid"));
64 
65  if(menu->hasAttribute("sid"))
66  {
67  bool start = false;
68  if(menu->hasAttribute("start"))
69  {
70  std::string startValue = menu->getAttributeValue("start");
71  SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
72  startValue == "yes" || startValue == "no");
73  start = (startValue=="yes");
74  }
75  std::string sid = menu->getAttributeValue("sid");
76  m_menuSids[sid] = SIDMenuMapType::mapped_type(index, start);
77  }
78  index++;
79  }
80 }

84 void MenuBarRegistrar::manage(std::vector< ::fwGui::container::fwMenu::sptr > menus )
85 {
86  ::fwGui::container::fwMenu::sptr menu;
87  for( SIDMenuMapType::value_type sid : m_menuSids)
88  {
89  OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!",
90  sid.second.first < menus.size());
91  menu = menus.at( sid.second.first );
92  ::fwGui::GuiRegistry::registerSIDMenu(sid.first, menu);
93  if(sid.second.second) //service is auto started?
94  {
95  OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
96  ::fwServices::IService::sptr service = ::fwServices::get( sid.first );
97  service->start();
98  }
99  else
100  {
101  bool service_exists = ::fwTools::fwID::exist(sid.first );
102  if (!service_exists || ::fwServices::get( sid.first )->isStopped())
103  {
104  ::fwGui::GuiRegistry::actionServiceStopping(sid.first);
105  }
106  }
107  }
108 }
