00050 void MenuBarRegistrar::initialize( ::fwRuntime::ConfigurationElement::sptr configuration)
00051 {
00052     OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be registry",
00053             configuration->getName() == "registry");
00054 
00055     // index represents associated menu with position in menus vector
00056     unsigned int index = 0;
00057     // initialize m_menuSids map with configuration
00058     std::vector < ConfigurationType > vectMenus = configuration->find("menu");
00059     BOOST_FOREACH( ConfigurationType menu, vectMenus)
00060     {
00061         SLM_ASSERT("<menu> tag must have sid attribute", menu->hasAttribute("sid"));
00062 
00063         if(menu->hasAttribute("sid"))
00064         {
00065             bool start = false;
00066             if(menu->hasAttribute("start"))
00067             {
00068                 std::string startValue = menu->getAttributeValue("start");
00069                 SLM_ASSERT("Wrong value '"<< startValue <<"' for 'start' attribute (require yes or no)",
00070                         startValue == "yes" || startValue == "no");
00071                 start = (startValue=="yes");
00072             }
00073             std::string sid = menu->getAttributeValue("sid");
00074             m_menuSids[sid] = SIDMenuMapType::mapped_type(index, start);
00075         }
00076         index++;
00077     }
00078 }

00082 void MenuBarRegistrar::manage(std::vector< ::fwGui::container::fwMenu::sptr > menus )
00083 {
00084     ::fwGui::container::fwMenu::sptr menu;
00085     BOOST_FOREACH( SIDMenuMapType::value_type sid, m_menuSids)
00086     {
00087         OSLM_ASSERT("Container index "<< sid.second.first <<" is bigger than subViews size!", sid.second.first < menus.size());
00088         menu = menus.at( sid.second.first );
00089         ::fwGui::GuiRegistry::registerSIDMenu(sid.first, menu);
00090         if(sid.second.second) //service is auto started?
00091         {
00092             OSLM_ASSERT("Service "<<sid.first <<" not exists.", ::fwTools::fwID::exist(sid.first ) );
00093             ::fwServices::IService::sptr service = ::fwServices::get( sid.first ) ;
00094             service->start();
00095         }
00096         else
00097         {
00098             bool service_exists = ::fwTools::fwID::exist(sid.first );
00099             if (!service_exists || ::fwServices::get( sid.first )->isStopped())
00100             {
00101                 ::fwGui::GuiRegistry::actionServiceStopping(sid.first);
00102             }
00103         }
00104     }
