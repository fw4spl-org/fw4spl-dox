328 void moveDirectory(::boost::filesystem::path input, ::boost::filesystem::path output)
329 {
330  ::boost::system::error_code ec;
331  ::boost::filesystem::copy_directory(input, output, ec);
332  FW_RAISE_IF("copy_directory " << input.string() << " " << output.string()
333  << " error : " << ec.message(), ec.value());
334 
335  ::boost::filesystem::directory_iterator it(input);
336  ::boost::filesystem::directory_iterator end;
337  ::boost::filesystem::permissions(output, ::boost::filesystem::owner_all, ec);
338  OSLM_ERROR_IF("set " << output.string() << " permission error : " << ec.message(), ec.value());
339 
340  for(; it != end; ++it)
341  {
342  ::boost::filesystem::path dest = output / it->path().filename();
343  if(::boost::filesystem::is_directory(*it))
344  {
345  moveDirectory(*it, dest);
346  }
347  else
348  {
349  ::boost::filesystem::rename(*it, dest, ec);
350  FW_RAISE_IF(
351  "rename " << it->path().string() << " " << dest.string() << " error : " << ec.message(),
352  ec.value());
353  }
354 
355  ::boost::filesystem::permissions(dest, ::boost::filesystem::owner_all, ec);
356  OSLM_ERROR_IF("set " << dest.string() << " permission error : " << ec.message(), ec.value());
357  }
358 }

374 void DicomAnonymizer::anonymizationProcess(const ::boost::filesystem::path& dirPath)
375 {
376  // Create temporary directory
377  ::boost::filesystem::path tmpPath = ::fwTools::System::getTemporaryFolder("DicomAnonymizer");
378  tmpPath /= "tmp";
379 
380  // Doesn't use ::boost::filesystem::rename because of potential issues when moving folders across volumes
381  moveDirectory(dirPath, tmpPath);
382 
383  ::boost::system::error_code ec;
384 
385  ::boost::filesystem::directory_iterator it(dirPath);
386  ::boost::filesystem::directory_iterator end;
387

440 void DicomAnonymizer::anonymize(std::istream& inputStream, std::ostream& outputStream)
441 {
442  // File Reader
443  ::gdcm::Reader reader;
444  reader.SetStream(inputStream);
445  FW_RAISE_IF("Unable to anonymize (file read failed)", !reader.Read());
446 
447  // String filter
448  m_stringFilter.SetFile(reader.GetFile());
449 
450  // Objects used to scan groups of elements
451  ::gdcm::Tag tag;
452  ::gdcm::DataElement dataElement;
453  const ::gdcm::File& datasetFile = reader.GetFile();
454  ::gdcm::DataSet dataset = datasetFile.GetDataSet();
455 
456  m_anonymizer.SetFile( reader.GetFile() );
457 
458  m_anonymizer.RemoveGroupLength();
459  m_anonymizer.RemoveRetired();
460 
461  for(ExceptionTagMapType::value_type exception: m_exceptionTagMap)
462  {
463  m_anonymizer.Replace(exception.first, exception.second.c_str());
464  }
465 
466  ::boost::range::for_each(m_actionCodeDTags, [this](const ::gdcm::Tag& tag)
467  {
468  this->applyActionCodeD(tag);
469  });
470  ::boost::range::for_each(m_actionCodeZTags, [this](const ::gdcm::Tag& tag)
471  {
472  this->applyActionCodeZ(tag);
473  });
474  ::boost::range::for_each(m_actionCodeXTags, [this](const ::gdcm::Tag& tag)
475  {
476  this->applyActionCodeX(tag);
477  });
478  ::boost::range::for_each(m_actionCodeKTags, [this](const ::gdcm::Tag& tag)
479  {
480  this->applyActionCodeK(tag);
481  });
482  ::boost::range::for_each(m_actionCodeCTags, [this](const ::gdcm::Tag& tag)
483  {
484  this->applyActionCodeC(tag);
485  });
486  ::boost::range::for_each(m_actionCodeUTags, [this](const ::gdcm::Tag& tag)
487  {
488  this->applyActionCodeU(tag);
489  });
490 
491  auto applyActionCodeXWithException = [this](const ::gdcm::Tag& tag)
492  {
493  if(m_exceptionTagMap.find(tag) == m_exceptionTagMap.end())
494  {
495  this->applyActionCodeX(tag);
496  }
497  };
498 
499  // Curve Data (0x50xx,0xxxxx)
500  dataElement = dataset.FindNextDataElement(::gdcm::Tag(0x5000, 0x0));
501  tag = dataElement.GetTag();
502  while (((tag.GetGroup() >> 8) & 0xff) == 0x50 )
503  {
504  applyActionCodeXWithException(tag); //Curve Data
505  dataElement = dataset.FindNextDataElement(
506  ::gdcm::Tag(tag.GetGroup(), static_cast< uint16_t >(tag.GetElement() + 1)));
507  tag = dataElement.GetTag();
508  }
509 
510  // Overlay Comments (0x60xx,0x4000)
511  dataElement = dataset.FindNextDataElement(::gdcm::Tag(0x6000, 0x4000));
512  tag = dataElement.GetTag();
513  while (((tag.GetGroup() >> 8) & 0xff) == 0x60 )
514  {
515  if(tag.GetElement() == 0x4000)
516  {
517  applyActionCodeXWithException(tag);
518  }
519  dataElement = dataset.FindNextDataElement(
520  ::gdcm::Tag(static_cast< uint16_t >(tag.GetGroup() + 1), 0x4000));
521  tag = dataElement.GetTag();
522  }
523 
524  // Overlay Data (0x60xx,0x3000)
525  dataElement = dataset.FindNextDataElement(::gdcm::Tag(0x6000, 0x3000));
526  tag = dataElement.GetTag();
527  while (((tag.GetGroup() >> 8) & 0xff) == 0x60 )
528  {
529  if(tag.GetElement() == 0x3000)
530  {
531  applyActionCodeXWithException(tag);
532  }
533  dataElement = dataset.FindNextDataElement(::gdcm::Tag(static_cast< uint16_t >(tag.GetGroup() + 1), 0x3000));
534  tag = dataElement.GetTag();
535  }
536 
537  m_anonymizer.RemovePrivateTags(); //Private attributes (X)
538 
539  // Write file
540  ::gdcm::Writer writer;
541  writer.SetStream(outputStream);
542  writer.SetFile(reader.GetFile());
543 
544  FW_RAISE_IF("Unable to anonymize (file write failed)", !writer.Write());
545 }
