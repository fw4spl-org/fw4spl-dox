00064         // VAG attention : ImageFileReader ne notifie AUCUNE progressEvent mais son ImageIO oui!!!! mais ImageFileReader ne permet pas de l'atteindre

00068         // Reader IO (*1*)
00069         typename itk::ImageIOBase::Pointer imageIORead = itk::ImageIOFactory::CreateImageIO( param.m_filename.c_str(), itk::ImageIOFactory::ReadMode);
00070 
00071         // set observation (*2*)
00072         Progressor progress(imageIORead, param.m_fwReader, param.m_filename);
00073 
00074         // the reader
00075         typedef itk::Image< PIXELTYPE, 3 >              ImageType;
00076         typedef itk::ImageFileReader< ImageType >       ReaderType;
00077         typename ReaderType::Pointer reader = ReaderType::New();
00078         reader->SetFileName( param.m_filename.c_str() );
00079         // attach its IO (*3*)
00080         reader->SetImageIO( imageIORead );
00081 
00082         reader->Update();
00083         typename ImageType::Pointer itkimage = reader->GetOutput();

00085     }
