00123 void Square::processInteraction( ::scene2D::data::Event::sptr _event )
00124 {
00125     SLM_TRACE_FUNC();
00126     if ( _event->getType() == ::scene2D::data::Event::MouseButtonPress && _event->getButton() == ::scene2D::data::Event::LeftButton )
00127     {
00128         if ( this->coordViewIsInItem( _event->getCoord(), m_rec ) )
00129         {
00130             SLM_TRACE("Point is captured");
00131             m_pointIsCaptured = true;
00132             m_oldCoord = this->coordViewToCoordItem( _event->getCoord(), m_rec );
00133             m_rec->setBrush( Qt::yellow );
00134             _event->setAccepted(true);
00135         }
00136     }
00137     else if ( m_pointIsCaptured && _event->getType() == ::scene2D::data::Event::MouseMove )
00138     {
00139         ::scene2D::data::Coord newCoord = this->coordViewToCoordItem( _event->getCoord(), m_rec );
00140         m_rec->moveBy( newCoord.getX() - m_oldCoord.getX(), newCoord.getY() - m_oldCoord.getY() );
00141         m_oldCoord = newCoord;
00142         _event->setAccepted(true);
00143     }
00144     else if ( m_pointIsCaptured && _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
00145     {
00146         m_rec->setBrush( m_color );
00147         m_pointIsCaptured = false;
00148         _event->setAccepted(true);
00149     }
00150 
00151 }
