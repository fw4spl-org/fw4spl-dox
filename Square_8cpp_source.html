123 void Square::processInteraction( ::scene2D::data::Event::sptr _event )
124 {
125  SLM_TRACE_FUNC();
126  if ( _event->getType() == ::scene2D::data::Event::MouseButtonPress &&
127  _event->getButton() == ::scene2D::data::Event::LeftButton )
128  {
129  if ( this->coordViewIsInItem( _event->getCoord(), m_rec ) )
130  {
131  SLM_TRACE("Point is captured");
132  m_pointIsCaptured = true;
133  m_oldCoord = this->coordViewToCoordItem( _event->getCoord(), m_rec );
134  m_rec->setBrush( Qt::yellow );
135  _event->setAccepted(true);
136  }
137  }
138  else if ( m_pointIsCaptured && _event->getType() == ::scene2D::data::Event::MouseMove )
139  {
140  ::scene2D::data::Coord newCoord = this->coordViewToCoordItem( _event->getCoord(), m_rec );
141  m_rec->moveBy( newCoord.getX() - m_oldCoord.getX(), newCoord.getY() - m_oldCoord.getY() );
142  m_oldCoord = newCoord;
143  _event->setAccepted(true);
144  }
145  else if ( m_pointIsCaptured && _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
146  {
147  m_rec->setBrush( m_color );
148  m_pointIsCaptured = false;
149  _event->setAccepted(true);
150  }
151 
152 }
