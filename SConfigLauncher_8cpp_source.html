81 void SConfigLauncher::configuring() throw(fwTools::Failed)
82 {
83  this->initialize();
84 
85  if(this->isVersion2())
86  {
87  typedef ::fwRuntime::ConfigurationElement::sptr ConfigType;
88  typedef ::fwRuntime::EConfigurationElement::sptr EditableConfigType;
89 
90  ConfigType cfg = this->getConfiguration();
91 
92  ConfigType appCfg = cfg->findConfigurationElement("appConfig");
93  SLM_ASSERT("Missing 'appConfig' tag.", appCfg);
94  std::string appCfgId = appCfg->getAttributeValue("id");
95 
96 
97  EditableConfigType srvCfg = ::fwRuntime::EConfigurationElement::New( "service" );
98  EditableConfigType newCfg = srvCfg->addConfigurationElement("config");
99  EditableConfigType newAppCfg = newCfg->addConfigurationElement("appConfig");
100  newAppCfg->setAttributeValue("id", appCfgId);
101 
102  EditableConfigType newParamsCfg = newAppCfg->addConfigurationElement("parameters");
103 
104  const std::vector< ConfigType > inoutsCfg = cfg->find("inout");
105  for (const auto& inoutCfg : inoutsCfg)
106  {
107  std::string key = inoutCfg->getAttributeValue("key");
108  SLM_ASSERT("[" + appCfgId + "] Missing 'key' tag.", !key.empty());
109 
110  std::string uid = inoutCfg->getAttributeValue("uid");
111  SLM_ASSERT("[" + appCfgId + "] Missing 'uid' tag.", !uid.empty());
112 
113  EditableConfigType newParamCfg = newParamsCfg->addConfigurationElement("parameter");
114  newParamCfg->setAttributeValue("replace", key);
115 
116  auto obj = this->getInOut< ::fwData::Object>(key);
117  newParamCfg->setAttributeValue("uid", obj->getID());
118  }
119 
120  const std::vector< ConfigType > paramsCfg = cfg->find("parameter");
121  for (const auto& paramCfg : paramsCfg)
122  {
123  std::string replace = paramCfg->getAttributeValue("replace");
124  SLM_ASSERT("[" + appCfgId + "] Missing 'replace' tag.", !replace.empty());
125 
126  EditableConfigType newParamCfg = newParamsCfg->addConfigurationElement("parameter");
127  newParamCfg->setAttributeValue("replace", replace);
128 
129  std::string uid = paramCfg->getAttributeValue("uid");
130  if(!uid.empty())
131  {
132  newParamCfg->setAttributeValue("uid", uid);
133  }
134  else
135  {
136  std::string by = paramCfg->getAttributeValue("by");
137  SLM_ASSERT("[" + appCfgId + "] Missing 'uid' or 'by' tag.", !by.empty());
138 
139  newParamCfg->setAttributeValue("by", by);
140  }
141  }
142 
143  m_configLauncher->parseConfig(::fwRuntime::Convert::toPropertyTree(srvCfg));
144  }
145  else
146  {
147  m_configLauncher->parseConfig(this->getConfigTree());
148  }
149 }

153 void SConfigLauncher::setIsActive(bool isActive)
154 {
155  this->::fwGui::IActionSrv::setIsActive(isActive);
156  if ( isActive )
157  {
158  ::fwServices::registry::Proxy::sptr proxies = ::fwServices::registry::Proxy::getDefault();
159  proxies->connect(m_proxychannel, this->slot(s_STOP_CONFIG_SLOT));
160  ::fwServices::helper::ConfigLauncher::FieldAdaptorType replaceMap;
161  replaceMap[s_CLOSE_CONFIG_CHANNEL_ID] = m_proxychannel;
162  m_configLauncher->startConfig(this->getSptr(), replaceMap);
163  m_sigLaunched->asyncEmit();
164  }
165  else
166  {
167  this->stopConfig();
168  }
169 }
