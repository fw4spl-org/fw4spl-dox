60 void Measurement::createNodes(SPTR(::fwGdcmIO::container::sr::DicomSRNode)parent, bool useSCoord3D)
61 {
62  ::fwData::Vector::sptr distanceVector =
63  m_object->getField< ::fwData::Vector >(::fwDataTools::fieldHelper::Image::m_imageDistancesId);
64  if (distanceVector)
65  {
66  unsigned int id = 1;
67  for(::fwData::Object::sptr object: distanceVector->getContainer())
68  {
69  ::fwData::PointList::sptr pointList = ::fwData::PointList::dynamicCast(object);
70  if(pointList)
71  {
72  this->createMeasurement(parent, pointList, id++, useSCoord3D);
73  }
74  }
75  }

80 void Measurement::createMeasurement(SPTR(::fwGdcmIO::container::sr::DicomSRNode)parent,
81  const ::fwData::PointList::sptr& pointList, unsigned int id, bool useSCoord3D)
82 {
83  const ::fwData::Point::sptr point1 = pointList->getPoints()[0];
84  const ::fwData::Point::sptr point2 = pointList->getPoints()[1];
85 
86  double coordinates[6];
87  coordinates[0] = point1->getCoord()[0];
88  coordinates[1] = point1->getCoord()[1];
89  coordinates[2] = point1->getCoord()[2];
90  coordinates[3] = point2->getCoord()[0];
91  coordinates[4] = point2->getCoord()[1];
92  coordinates[5] = point2->getCoord()[2];
93 
94  double distance = sqrt( (coordinates[0] - coordinates[3]) * (coordinates[0] - coordinates[3]) +
95  (coordinates[1] - coordinates[4]) * (coordinates[1] - coordinates[4]) +
96  (coordinates[2] - coordinates[5]) * (coordinates[2] - coordinates[5]) );
97 
98  // Retrieve Frame Numbers
99  int frameNumber1 = ::fwGdcmIO::helper::DicomData::convertPointToFrameNumber(m_object, point1);
100 
101  // Create Measurement Node
102  SPTR(::fwGdcmIO::container::sr::DicomSRNumNode) numNode =
103  std::make_shared< ::fwGdcmIO::container::sr::DicomSRNumNode >(
104  ::fwGdcmIO::container::DicomCodedAttribute("121206", "DCM", "Distance"), "CONTAINS", distance,
105  ::fwGdcmIO::container::DicomCodedAttribute("mm", "UCUM", "millimeter", "1.4"));
106  parent->addSubNode(numNode);
107 
108  if(useSCoord3D)
109  {
110  // Create SCoord Node
111  const float scoord[] = {
112  static_cast<float>(point1->getCoord()[0]),
113  static_cast<float>(point1->getCoord()[1]),
114  static_cast<float>(point1->getCoord()[2]),
115  static_cast<float>(point2->getCoord()[0]),
116  static_cast<float>(point2->getCoord()[1]),
117  static_cast<float>(point2->getCoord()[2])
118  };
119  std::vector<float> scoordVector (scoord, scoord + 6);
120  SPTR(::fwGdcmIO::container::sr::DicomSRSCoord3DNode) scoordNode =
121  std::make_shared< ::fwGdcmIO::container::sr::DicomSRSCoord3DNode >(
122  ::fwGdcmIO::container::DicomCodedAttribute("121230", "DCM", "Path"),
123  "INFERRED FROM", "POLYLINE", scoordVector, m_instance->getSOPInstanceUIDContainer()[0]);
124  numNode->addSubNode(scoordNode);
125  }
126  else
127  {
128  SLM_ASSERT("Unable to save a 3D distance using a SCOORD object.",
129  frameNumber1 == ::fwGdcmIO::helper::DicomData::convertPointToFrameNumber(m_object, point2));
130 
131  // Create SCoord Node
132  const float scoord[] = {
133  static_cast<float>(point1->getCoord()[0]),
134  static_cast<float>(point1->getCoord()[1]),
135  static_cast<float>(point2->getCoord()[0]),
136  static_cast<float>(point2->getCoord()[1])
137  };
138  std::vector<float> scoordVector (scoord, scoord + 4);
139  SPTR(::fwGdcmIO::container::sr::DicomSRSCoordNode) scoordNode =
140  std::make_shared< ::fwGdcmIO::container::sr::DicomSRSCoordNode >(
141  ::fwGdcmIO::container::DicomCodedAttribute("121230", "DCM", "Path"),
142  "INFERRED FROM", "POLYLINE", scoordVector);
143  numNode->addSubNode(scoordNode);
144 
145  // Create Image Node
146  SPTR(::fwGdcmIO::container::sr::DicomSRImageNode) imageNode =
147  std::make_shared< ::fwGdcmIO::container::sr::DicomSRImageNode >(
148  ::fwGdcmIO::container::DicomCodedAttribute(), "SELECTED FROM", m_instance->getSOPClassUID(),
149  m_instance->getSOPInstanceUIDContainer()[frameNumber1-1], frameNumber1);
150  scoordNode->addSubNode(imageNode);
151  }
