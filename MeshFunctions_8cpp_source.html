00022 bool intersect_triangle(fwVec3d _orig, fwVec3d _dir, fwVec3d _vert0, fwVec3d _vert1, fwVec3d _vert2, double &_t, double &_u, double &_v)
00023 {
00024     const double Epsilon = 0.000001;
00025 
00026     fwVec3d edge1, edge2, tvec, pvec, qvec;
00027 
00028     /* find vectors for two edges sharing vert0 */
00029     edge1 = _vert1 - _vert0;
00030     edge2 = _vert2 - _vert0;
00031 
00032     /* begin calculating determinant - also used to calculate U parameter */
00033     pvec = ::fwMath::cross(_dir, edge2);
00034 
00035     /* if determinant is near zero, ray lies in plane of triangle */
00036     const double Det = ::fwMath::dot(edge1, pvec);
00037 
00038     if (Det > -Epsilon && Det < Epsilon) return false;
00039     const double Inv_det = 1.0 / Det;
00040 
00041     /* calculate distance from vert0 to ray origin */
00042     tvec = _orig - _vert0;
00043 
00044     /* calculate U parameter and test bounds */
00045     _u = Inv_det * ::fwMath::dot(tvec, pvec);
00046     if (_u < 0.0 || _u > 1.0) return false;
00047 
00048     /* prepare to test V parameter */
00049     qvec = ::fwMath::cross(tvec, edge1);
00050 
00051     /* calculate V parameter and test bounds */
00052     _v =  Inv_det * ::fwMath::dot(_dir, qvec);
00053     if (_v < 0.0 || _u + _v > 1.0) return false;
00054 
00055     /* calculate t, ray intersects triangle */
00056     _t = Inv_det * ::fwMath::dot(edge2, qvec);
00057     return true;
00058 }
