00068 void ViewportInteractor::processInteraction( ::scene2D::data::Event::sptr _event )
00069 {
00070     if ( _event->getType() == ::scene2D::data::Event::MouseWheelUp
00071             && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
00072     {
00073         this->zoom(true);
00074     }
00075     else if ( _event->getType() == ::scene2D::data::Event::MouseWheelDown
00076             && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
00077     {
00078         this->zoom(false);
00079     }
00080     else if ( _event->getType() == ::scene2D::data::Event::MouseButtonPress
00081             && _event->getButton() == ::scene2D::data::Event::LeftButton
00082             && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
00083     {
00084         m_viewportIsTranslated = true;
00085         m_lastCoordEvent = _event->getCoord();
00086     }
00087     else if ( m_viewportIsTranslated )
00088     {
00089         if ( _event->getType() == ::scene2D::data::Event::MouseMove )
00090         {
00091             ::scene2D::data::Coord coord = _event->getCoord();
00092             ::scene2D::data::Viewport::sptr sceneViewport = this->getScene2DRender()->getViewport();
00093 
00094             float dx = coord.getX() - m_lastCoordEvent.getX();
00095             float xTrans = dx * sceneViewport->getWidth() / (float) this->getScene2DRender()->getView()->width();
00096 
00097             float dy = coord.getY() - m_lastCoordEvent.getY();
00098             float yTrans = dy * sceneViewport->getHeight() / (float) this->getScene2DRender()->getView()->height();
00099 
00100             sceneViewport->setX( sceneViewport->getX() - xTrans );
00101             sceneViewport->setY( sceneViewport->getY() - yTrans );
00102             this->getScene2DRender()->getView()->updateFromViewport();
00103 
00104             m_lastCoordEvent = coord;
00105         }
00106         else if ( _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
00107         {
00108             m_viewportIsTranslated = false;
00109         }
00110     }
00111 }

00115 void ViewportInteractor::zoom( bool zoomIn )
00116 {
00117     ::scene2D::data::Viewport::sptr sceneViewport = this->getScene2DRender()->getViewport();
00118 
00119     float zoomPercent = 10.f / 100.0f;
00120     float y = sceneViewport->getY();
00121     float x = sceneViewport->getX();
00122     float width = sceneViewport->getWidth();
00123     float height = sceneViewport->getHeight();
00124     float centerX = x + width/2.0f;
00125     float centerY = y + height/2.0f;
00126 
00127     float newWidth;
00128     float newHeight;
00129     if ( zoomIn )
00130     {
00131         newWidth = width * zoomPercent;
00132         newHeight = height * zoomPercent;
00133     }
00134     else
00135     {
00136         newWidth = - width * zoomPercent;
00137         newHeight = - height * zoomPercent;
00138     }
00139 
00140     newWidth += width;
00141     newHeight += height;
00142 
00143     x = centerX - newWidth / 2;
00144     y = centerY - newHeight / 2;
00145 
00146     width = newWidth;
00147     height = newHeight;
00148 
00149     // Set viewport
00150     sceneViewport->setX( x );
00151     sceneViewport->setY( y );
00152     sceneViewport->setWidth( width );
00153     sceneViewport->setHeight( height );
00154     this->getScene2DRender()->getView()->updateFromViewport();
00155 }
