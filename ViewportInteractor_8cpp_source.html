68 void ViewportInteractor::processInteraction( ::scene2D::data::Event::sptr _event )
69 {
70  if ( _event->getType() == ::scene2D::data::Event::MouseWheelUp
71  && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
72  {
73  this->zoom(true);
74  }
75  else if ( _event->getType() == ::scene2D::data::Event::MouseWheelDown
76  && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
77  {
78  this->zoom(false);
79  }
80  else if ( _event->getType() == ::scene2D::data::Event::MouseButtonPress
81  && _event->getButton() == ::scene2D::data::Event::LeftButton
82  && _event->getModifier() == ::scene2D::data::Event::ShiftModifier )
83  {
84  m_viewportIsTranslated = true;
85  m_lastCoordEvent = _event->getCoord();
86  }
87  else if ( m_viewportIsTranslated )
88  {
89  if ( _event->getType() == ::scene2D::data::Event::MouseMove )
90  {
91  ::scene2D::data::Coord coord = _event->getCoord();
92  ::scene2D::data::Viewport::sptr sceneViewport = this->getScene2DRender()->getViewport();
93 
94  float dx = coord.getX() - m_lastCoordEvent.getX();
95  float xTrans = dx * sceneViewport->getWidth() / (float) this->getScene2DRender()->getView()->width();
96 
97  float dy = coord.getY() - m_lastCoordEvent.getY();
98  float yTrans = dy * sceneViewport->getHeight() / (float) this->getScene2DRender()->getView()->height();
99 
100  sceneViewport->setX( sceneViewport->getX() - xTrans );
101  sceneViewport->setY( sceneViewport->getY() - yTrans );
102  this->getScene2DRender()->getView()->updateFromViewport();
103 
104  m_lastCoordEvent = coord;
105  }
106  else if ( _event->getType() == ::scene2D::data::Event::MouseButtonRelease )
107  {
108  m_viewportIsTranslated = false;
109  }
110  }
111 }

115 void ViewportInteractor::zoom( bool zoomIn )
116 {
117  ::scene2D::data::Viewport::sptr sceneViewport = this->getScene2DRender()->getViewport();
118 
119  float zoomPercent = 10.f / 100.0f;
120  float y = sceneViewport->getY();
121  float x = sceneViewport->getX();
122  float width = sceneViewport->getWidth();
123  float height = sceneViewport->getHeight();
124  float centerX = x + width/2.0f;
125  float centerY = y + height/2.0f;
126 
127  float newWidth;
128  float newHeight;
129  if ( zoomIn )
130  {
131  newWidth = width * zoomPercent;
132  newHeight = height * zoomPercent;
133  }
134  else
135  {
136  newWidth = -width * zoomPercent;
137  newHeight = -height * zoomPercent;
138  }
139 
140  newWidth += width;
141  newHeight += height;
142 
143  x = centerX - newWidth / 2;
144  y = centerY - newHeight / 2;
145 
146  width = newWidth;
147  height = newHeight;
148 
149  // Set viewport
150  sceneViewport->setX( x );
151  sceneViewport->setY( y );
152  sceneViewport->setWidth( width );
153  sceneViewport->setHeight( height );
154  this->getScene2DRender()->getView()->updateFromViewport();
155 }
