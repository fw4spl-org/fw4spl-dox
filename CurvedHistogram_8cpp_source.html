135 CurvedHistogram::Points CurvedHistogram::getBSplinePoints( const Points & _points ) const
136 {
137  Points bSplinePoints;
138  point_list list; // see bspline.h
139 
140  // Add again the first point with a higher value in order to prevent B-Spline algorithm from removing
141  // the first value.
142  list.add_point(
143  new point(
144  (float) _points[0].first,
145  (float) _points[0].second * 2) );
146 
147  // Add all the points
148  for(const auto& pt : _points )
149  {
150  list.add_point(
151  new point(
152  (float)(pt.first),
153  (float)(pt.second)) );
154  }
155 
156  // Add again the last point
157  list.add_point(
158  new point(
159  (float) _points.back().first,
160  (float) _points.back().second / 2 ) );
161 
162 
163  // Commpute the points of the B-Spline with external code from AHO (to be integrated here later).
164  cat_curve curve( list );
165  curve.m_precision = static_cast<int>(_points.size() * 5);
166  curve.compute();
167 
168  for(int i = 0; i < curve.m_precision; ++i)
169  {
170  bSplinePoints.push_back( Point( curve.m_curve_point[i].x, curve.m_curve_point[i].y ) );
171  }
172 
173  return bSplinePoints;
174 }

178 CurvedHistogram::Points CurvedHistogram::getResampledBSplinePoints(const Points & _bSplinePoints ) const
179 {
180  Points points;
181  Point point = _bSplinePoints.front();
182 
183  double dx, dy;
184  const double maxLength = 2000;
185  double segmentLength = 0;
186 
187  points.push_back( point );
188 
189  for(Points::const_iterator it = _bSplinePoints.begin() + 1; it != _bSplinePoints.end(); ++it)
190  {
191  dx = abs((*it).first - point.first); // theoretically positive
192  dy = abs((*it).second - point.second);
193 
194  segmentLength += sqrt( dx * dx + dy * dy );
195 
196  if(segmentLength > maxLength)
197  {
198  points.push_back( *it );
199  segmentLength = 0;
200  }
201 
202  point = *it;
203  }
204 
205  points.push_back( _bSplinePoints.back() );
206 
207  OSLM_TRACE("B-Spline points size moved from " << _bSplinePoints.size() << " to " << points.size());
208 
209  return points;
210 }

214 void CurvedHistogram::computePointToPathLengthMapFromBSplinePoints( Points & _bSplinePoints )
215 {
216  Points::iterator it = _bSplinePoints.begin();
217

219  {

223  QPointF prevPt = QPointF(p.first, p.second);
224  m_painterPath->lineTo( p.first, p.second );
225  qreal len = m_painterPath->length();
226  ++it;
227 
228  for(; it != _bSplinePoints.end(); ++it)
229  {

232  m_painterPath->lineTo( p.first, p.second );

242  }

249 {
250  SLM_TRACE_FUNC();
251 
252  this->doStop();
253 
254  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
255 
256  ::fwData::mt::ObjectReadLock lock(histogram);
257 
258  if (!histogram->getValues().empty())
259  {
260  m_layer = new QGraphicsItemGroup();
261 
262  m_painterPath = new QPainterPath();
263 
264  Points controlPoints = this->getControlPoints( histogram );
265  Points bSplinePoints = this->getBSplinePoints( controlPoints );
266 
267  this->computePointToPathLengthMapFromBSplinePoints( bSplinePoints );
268 
269  // Try to remove unnecessary points of the B-Spline points
270  Points resampledBSplinePoints = this->getResampledBSplinePoints( bSplinePoints );
271  bSplinePoints = resampledBSplinePoints;
272 
273  this->buildBSplineFromPoints( bSplinePoints );
274 
275  // Adjust the layer's position and zValue depending on the associated axis
276  m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
277  m_layer->setZValue(m_zValue);
278 
279  // Add to the scene the unique item which gather the whole set of rectangle graphic items:
280  this->getScene2DRender()->getScene()->addItem( m_layer );
281  }

286 void CurvedHistogram::buildBSplineFromPoints(Points & _bSplinePoints )
287 {
288  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
289 
290  const bool useBorderColor = (m_borderColor.color() != Qt::transparent);
291  const bool useInnerColor = (m_innerColor.color() != Qt::transparent);
292 
293  Point currentPoint = this->mapAdaptorToScene( Point(
294  histogram->getMinValue(), _bSplinePoints[0].second), m_xAxis,
295  m_yAxis );
296  Point previousPoint = currentPoint;
297  Points::iterator it;
298 
299  const QPointF startPoint( currentPoint.first, currentPoint.second / 10 ); // divide by 10 to cut meaningless values
300  QPainterPath path( QPointF(startPoint.x(), 0.0) );
301  path.lineTo( startPoint );
302 
303  const int maxLinesPerPath = 10;
304  int lineCount = 0;
305 
306  previousPoint.first = startPoint.x();
307  previousPoint.second = startPoint.y();
308 
309  // Build the path with the B-Spline points
310  for(it = _bSplinePoints.begin() + 1; it != _bSplinePoints.end(); ++it)
311  {
312  currentPoint = this->mapAdaptorToScene(*it, m_xAxis, m_yAxis );
313 
314  path.lineTo( currentPoint.first, currentPoint.second );
315  }
316 
317  // Close the path:
318  m_painterPath->lineTo( histogram->getMaxValue(), _bSplinePoints.back().second);
319 
320  if( useBorderColor )
321  {
322  path.lineTo( currentPoint.first, 0.0 );
323  this->addBorderItem( path );
324  }
325 
326  if( useInnerColor )
327  {
328  path.lineTo( previousPoint.first, 0.0 );
329  this->addInnerItem( path );
330  }
331 }

363 CurvedHistogram::Points CurvedHistogram::quadraticInterpolation(
364  const Point _p0, const Point _p1, const Point _p2 )
365 {
366  Points points;
367  Point p;
368 
369  const double d2 =
370  2 * ((_p2.second - _p1.second) / (_p2.first - _p1.first) - (_p1.second - _p0.second)
371  / (_p1.first - _p0.first))
372  / (_p2.first - _p0.first);
373 
374  const double d1 = (_p2.second - _p1.second) / (_p2.first - _p1.first) - 0.5 * d2 * (_p2.first - _p1.first);
375  const double d0 = _p1.second;
376 
377  points.push_back( _p0 );
378 
379  for(double x = _p0.first; x < _p2.first; x += 0.5)
380  {
381  p.first = x;
382  p.second = 0.5 * d2 * pow(x - _p1.first, 2) + d1 * (x - _p1.first) + d0;
383 
384  points.push_back( p );
385  }
386 
387  return points;
388 }

392 CurvedHistogram::Points CurvedHistogram::cosinusInterpolation(const Point _p0, const Point _p1)
393 {
394  Points points;
395  Point p;
396  double mu2;
397  double deltaX = _p1.first - _p0.first;
398 
399  for(double mu = 0.0; mu < 1.0; mu += 0.05)
400  {
401  mu2 = (1 - std::cos(mu * PI)) / 2;
402 
403  p.first = _p0.first + mu * deltaX;
404  p.second = _p0.second * (1 - mu2) + _p1.second * mu2;
405 
406  points.push_back( p );
407  }
408 
409  return points;
410 }

414 CurvedHistogram::Points CurvedHistogram::cubicInterpolation(
415  const Point _p0, const Point _p1, const Point _p2, const Point _p3 )
416 {
417  Points points;
418  Point p;
419  double a0, a1, a2, a3, mu2;
420  const double deltaX = _p2.first - _p1.first;
421  for(double mu = 0.0; mu < 1.0; mu += 0.01)
422  {
423  mu2 = mu * mu;
424 
425  /*
426  a0 = _p3.second - _p2.second - _p0.second + _p1.second;
427  a1 = _p0.second - _p1.second - a0;
428  a2 = _p2.second - _p0.second;
429  a3 = _p1.second;
430  */
431 
432  // Smoother curves (Catmull-Rom s_plines)
433  a0 = -0.5*_p0.second + 1.5*_p1.second - 1.5*_p2.second + 0.5*_p3.second;
434  a1 = _p0.second - 2.5*_p1.second + 2*_p2.second - 0.5*_p3.second;
435  a2 = -0.5*_p0.second + 0.5*_p2.second;
436  a3 = _p1.second;
437 
438  p.first = _p1.first + mu * deltaX;
439  p.second = a0 * mu * mu2 +a1 * mu2 + a2 * mu + a3;
440 
441  points.push_back( p );
442  }
443 
444  return points;
445 }

449 void CurvedHistogram::updateCurrentPoint(const ::scene2D::data::Event::sptr& _event )
450 {
451  SLM_ASSERT("m_histogramPointUID must be defined in order to update the related ::fwData::Point data.",
452  !m_histogramPointUID.empty());
453 
454  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
455  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
456  const float histogramMinValue = histogram->getMinValue();
457  const float histogramBinsWidth = histogram->getBinsWidth();
458 
459  // Event coordinates in scene
460  ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( _event->getCoord() );
461 
462  const int histIndex = (int) sceneCoord.getX();
463  const int index = histIndex - histogramMinValue;
464  const int nbValues = (int)values.size() * histogramBinsWidth;
465 
466  if(index >= 0 && index < nbValues && m_positionsToPathLength.find( histIndex ) != m_positionsToPathLength.end())
467  {
468  double key = m_positionsToPathLength[ histIndex ];
469  const double percent = m_painterPath->percentAtLength( key );
470  QPointF qPoint = m_painterPath->pointAtPercent( percent );
471 
472  ::fwData::Point::sptr point = this->getSafeInOut< ::fwData::Point>( m_histogramPointUID );
473  SLM_ASSERT("m_histogramPointUID can't be null here.", point);
474 
475  point->getRefCoord()[0] = sceneCoord.getX();
476  point->getRefCoord()[1] = qPoint.y() * m_scale;
477  }
478 }

528 void CurvedHistogram::processInteraction( ::scene2D::data::Event::sptr _event)
529 {
530  SLM_TRACE_FUNC();
531 
532  bool updatePointedPos = false;
533 
534  // Vertical scaling
535  if( _event->getType() == ::scene2D::data::Event::MouseWheelUp )
536  {
537  m_scale *= SCALE;
538  m_layer->setTransform(QTransform::fromScale(1, SCALE), true);
539 
540  updatePointedPos = true;
541  }
542  else if( _event->getType() == ::scene2D::data::Event::MouseWheelDown )
543  {
544  m_scale /= SCALE;
545  m_layer->setTransform(QTransform::fromScale(1, 1 / SCALE), true);
546 
547  updatePointedPos = true;
548  }
549  else if( _event->getType() == ::scene2D::data::Event::MouseMove )
550  {
551  updatePointedPos = true;
552  }
553 
554  if( !m_histogramPointUID.empty() )
555  {
556  updateCurrentPoint( _event );
557  }
558 }
