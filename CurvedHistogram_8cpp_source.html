00349 CurvedHistogram::Points CurvedHistogram::quadraticInterpolation(
00350         const Point _p0, const Point _p1, const Point _p2 )
00351 {
00352     Points points;
00353     Point p;
00354 
00355     const double d2 =
00356         2 * ((_p2.second - _p1.second) / (_p2.first - _p1.first) - (_p1.second - _p0.second)
00357                 / (_p1.first - _p0.first))
00358             / (_p2.first - _p0.first);
00359 
00360     const double d1 = (_p2.second - _p1.second) / (_p2.first - _p1.first) - 0.5 * d2 * (_p2.first - _p1.first);
00361     const double d0 = _p1.second;
00362 
00363     points.push_back( _p0 );
00364 
00365     for(double x = _p0.first; x < _p2.first; x += 0.5)
00366     {
00367         p.first = x;
00368         p.second = 0.5 * d2 * pow(x - _p1.first, 2) + d1 * (x - _p1.first) + d0;
00369 
00370         points.push_back( p );
00371     }
00372 
00373     return points;
00374 }

00378 CurvedHistogram::Points CurvedHistogram::cosinusInterpolation(const Point _p0, const Point _p1)
00379 {
00380     Points points;
00381     Point p;
00382     double mu2;
00383     double deltaX = _p1.first - _p0.first;
00384 
00385     for(double mu = 0.0; mu < 1.0; mu += 0.05)
00386     {
00387         mu2 = (1 - std::cos(mu * PI)) / 2;
00388 
00389         p.first = _p0.first + mu * deltaX;
00390         p.second = _p0.second * (1 - mu2) + _p1.second * mu2;
00391 
00392         points.push_back( p );
00393     }
00394 
00395     return points;

00400 CurvedHistogram::Points CurvedHistogram::cubicInterpolation(
00401         const Point _p0, const Point _p1, const Point _p2, const Point _p3 )
00402 {
00403     Points points;
00404     Point p;
00405     double a0, a1, a2, a3, mu2;
00406     const double deltaX = _p2.first - _p1.first;
00407     for(double mu = 0.0; mu < 1.0; mu += 0.01)
00408     {
00409         mu2 = mu * mu;
00410 
00411         /*
00412         a0 = _p3.second - _p2.second - _p0.second + _p1.second;
00413         a1 = _p0.second - _p1.second - a0;
00414         a2 = _p2.second - _p0.second;
00415         a3 = _p1.second;
00416         */
00417 
00418         // Smoother curves (Catmull-Rom s_plines)
00419         a0 = -0.5*_p0.second + 1.5*_p1.second - 1.5*_p2.second + 0.5*_p3.second;
00420         a1 = _p0.second - 2.5*_p1.second + 2*_p2.second - 0.5*_p3.second;
00421         a2 = -0.5*_p0.second + 0.5*_p2.second;
00422         a3 = _p1.second;
00423 
00424         p.first = _p1.first + mu * deltaX;
00425         p.second = a0 * mu * mu2 +a1 * mu2 + a2 * mu + a3;
00426 
00427         points.push_back( p );
00428     }
00429 
00430     return points;
00431 }
