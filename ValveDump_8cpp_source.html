00124 size_t ValveDump::dump(size_t nbOfBytes)
00125 {
00126     size_t dumped = 0;
00127 
00128     ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
00129     if(manager)
00130     {
00131         const ::fwMemory::BufferManager::BufferInfoMapType bufferInfos = manager->getBufferInfos().get();
00132 
00133         typedef std::pair<
00134             fwMemory::BufferManager::BufferInfoMapType::key_type,
00135             fwMemory::BufferManager::BufferInfoMapType::mapped_type
00136                 > BufferInfosPairType;
00137         typedef std::vector< BufferInfosPairType > BufferVectorType;
00138 
00139         BufferVectorType buffers;
00140 
00141         BOOST_FOREACH(const ::fwMemory::BufferManager::BufferInfoMapType::value_type &elt, bufferInfos)
00142         {
00143             const ::fwMemory::BufferInfo &info = elt.second;
00144             if( ! ( info.size == 0 || info.lockCount() > 0 || !info.loaded )  )
00145             {
00146                 buffers.push_back(elt);
00147             }
00148         }
00149 
00150 
00151         BOOST_FOREACH(const BufferVectorType::value_type &pair, bufferInfos)
00152         {
00153             if(dumped < nbOfBytes)
00154             {
00155                 if( manager->dumpBuffer(pair.first).get() )
00156                 {
00157                     dumped += pair.second.size;
00158                 }
00159             }
00160             else
00161             {
00162                 break;
00163             }
00164         }
00165     }
00166 
00167     return dumped;
00168 }
