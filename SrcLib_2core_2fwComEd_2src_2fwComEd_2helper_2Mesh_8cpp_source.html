00170 {
00171     SLM_ASSERT("Bad number of points ("<< nb << ") for cell type: 'NO_CELL'",

00178             type != ::fwData::Mesh::TRIANGLE || nb == 3);
00179     SLM_ASSERT("Bad number of points ("<< nb << ") for cell type: 'QUAD'",
00180             type != ::fwData::Mesh::QUAD || nb == 4);
00181     SLM_ASSERT("Bad number of points ("<< nb << ") for cell type: 'POLY'",
00182             type != ::fwData::Mesh::POLY || nb > 4);
00183 
00184     ::fwData::Mesh::Id cellsDataSize = m_mesh->getCellDataSize();
00185     ::fwData::Mesh::Id nbCells = m_mesh->getNumberOfCells();
00186     ::fwData::Array::sptr cellTypes = m_mesh->getCellTypesArray();
00187     ::fwData::Array::sptr cellDataOffsets = m_mesh->getCellDataOffsetsArray();
00188     ::fwData::Array::sptr cellData = m_mesh->getCellDataArray();
00189 
00190     size_t allocatedCellTypes       = cellTypes->empty() ? 0 : cellTypes->getSize().at(0);
00191     size_t allocatedCellDataOffsets = cellDataOffsets->empty() ? 0 : cellDataOffsets->getSize().at(0);
00192 
00193     if( allocatedCellTypes <= nbCells )
00194     {
00195         cellTypes->resize(list_of(allocatedCellTypes + CELL_REALLOC_STEP), true);
00196     }

00203     size_t allocatedCellData = cellData->empty() ? 0 : cellData->getSize().at(0);
00204 
00205     if( allocatedCellData <= cellsDataSize + nb )
00206     {
00207         cellData->resize(list_of(allocatedCellData + CELLDATA_REALLOC_STEP), true);
00208     }
00209 
00210

00212     m_helperCellTypes->setItem(list_of(nbCells), t);

00216             );
00217     std::copy(cell, cell+nb, buf);
00218 
00219     const ::fwData::Mesh::CellDataOffsetType id[1] = {cellsDataSize};
00220     m_helperCellDataOffsets->setItem(list_of(nbCells), id);

00222 
00223     cellsDataSize += nb;

00225     m_mesh->setNumberOfCells(nbCells + 1);
