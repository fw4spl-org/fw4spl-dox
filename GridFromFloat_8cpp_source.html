00066 void GridFromFloat::draw()
00067 {
00068     SLM_TRACE_FUNC();
00069 
00070     SLM_ASSERT("m_xSpacing can not equal 0", m_xSpacing != 0);
00071     SLM_ASSERT("m_ySpacing can not equal 0", m_ySpacing != 0);
00072 
00073     // Remove all lines from the scene
00074     for (std::vector<QGraphicsItem*>::iterator it = m_lines.begin(); it != m_lines.end(); ++it)
00075     {
00076         this->getScene2DRender()->getScene()->removeItem(*it);
00077     }
00078     // Clear the lines vector
00079     m_lines.clear();
00080 
00081     // Calculate the start, end and step on x for the lines
00082     float xStartVal = (int)( m_xMin / m_xSpacing ) * m_xSpacing;
00083     float xEndVal = (int)( m_xMax / m_xSpacing ) * m_xSpacing;
00084     float xStep = m_xSpacing;
00085 
00086     // Calculate the start, end and step on y for the lines
00087     float yStartVal = (int)( m_yMin / m_ySpacing ) * m_ySpacing;
00088     float yEndVal = (int)( m_yMax / m_ySpacing ) * m_ySpacing;
00089     float yStep = m_ySpacing;
00090 
00091     // Draw the horizontal lines
00092     for ( float yVal = yStartVal ; yVal <= yEndVal ; yVal += yStep )
00093     {
00094         QGraphicsLineItem* line = new QGraphicsLineItem(
00095             this->mapAdaptorToScene(std::pair< double , double >( xStartVal, yVal) , m_xAxis, m_yAxis).first,
00096             this->mapAdaptorToScene(std::pair< double , double >( xStartVal, yVal) , m_xAxis, m_yAxis).second,
00097             this->mapAdaptorToScene(std::pair< double , double >( xEndVal, yVal) , m_xAxis, m_yAxis).first,
00098             this->mapAdaptorToScene(std::pair< double , double >( xEndVal, yVal) , m_xAxis, m_yAxis).second
00099         );
00100         // Set the line the pen and push it back in to the lines vector
00101         line->setPen(m_pen);
00102         m_lines.push_back(line);
00103     }
00104 
00105     // Draw the vertical lines
00106     for ( float xVal = xStartVal ; xVal <= xEndVal ; xVal += xStep )
00107     {
00108         QGraphicsLineItem* line = new QGraphicsLineItem(
00109             this->mapAdaptorToScene(std::pair< double , double >( xVal, yStartVal) , m_xAxis, m_yAxis).first,
00110             this->mapAdaptorToScene(std::pair< double , double >( xVal, yStartVal) , m_xAxis, m_yAxis).second,
00111             this->mapAdaptorToScene(std::pair< double , double >( xVal, yEndVal) , m_xAxis, m_yAxis).first,
00112             this->mapAdaptorToScene(std::pair< double , double >( xVal, yEndVal) , m_xAxis, m_yAxis).second
00113         );
00114         // Set the line the pen and push it back in to the lines vector
00115         line->setPen(m_pen);
00116         m_lines.push_back(line);
00117     }
00118 
00119     // Add the lines contained in the lines vector to the layer
00120     for ( unsigned int i = 0 ; i < m_lines.size() ; i++)
00121     {
00122         m_layer->addToGroup(m_lines.at(i));
00123     }
00124 
00125     // Set the layer position (according to the related axis) and zValue
00126     m_layer->setPos(m_xAxis->getOrigin(), m_yAxis->getOrigin());
00127     m_layer->setZValue(m_zValue);
00128     // Add the layer to the scene
00129     this->getScene2DRender()->getScene()->addItem(m_layer);
