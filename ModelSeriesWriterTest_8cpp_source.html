00150 
00151     const fs::path dir = ::fwTools::System::getTemporaryFolder() / "modelSeries";
00152

00154     {
00155         CPPUNIT_ASSERT_MESSAGE(std::string("Directory ") + dir.string() + " must be empty",
00156                 fs::is_empty(dir));
00157     }
00158     else
00159     {
00160         fs::create_directories(dir);
00161     }
00162 
00163     runModelSeriesSrv(
00164             "::ioVTK::SModelSeriesWriter",
00165             getIOCfgFromFolder(dir),
00166             modelSeries);
00167 
00168     ::fwMedData::SeriesDB::sptr seriesDB = ::fwMedData::SeriesDB::New();
00169 
00170     FileContainerType files;
00171     for(fs::directory_iterator it(dir); it != fs::directory_iterator(); ++it)
00172     {
00173         files.push_back(it->path().string());
00174     }
00175 
00176     CPPUNIT_ASSERT_EQUAL(modelSeries->getReconstructionDB().size(), files.size());
00177 
00178     runModelSeriesSrv(
00179             "::ioVTK::SSeriesDBReader",
00180             getIOCfgFromFiles(files),
00181             seriesDB);
00182 
00183     const ::fwMedData::SeriesDB::ContainerType& series = seriesDB->getContainer();
00184     CPPUNIT_ASSERT_EQUAL((size_t)1, series.size());
00185 
00186     ::fwMedData::ModelSeries::sptr readSeries = ::fwMedData::ModelSeries::dynamicCast(series[0]);
00187     CPPUNIT_ASSERT_MESSAGE("A ModelSeries was expected", readSeries);
00188 
00189     typedef ::fwMedData::ModelSeries::ReconstructionVectorType RecVecType;
00190     const RecVecType& readRecs = readSeries->getReconstructionDB();
00191     CPPUNIT_ASSERT_EQUAL(files.size(), readRecs.size());
00192 
00193     std::set< size_t > refHashes;
00194     std::set< size_t > readHashes;
00195 
00196     const RecVecType& refRecs = modelSeries->getReconstructionDB();
00197     RecVecType::const_iterator itRef = refRecs.begin();
00198     RecVecType::const_iterator itRead = readRecs.begin();
00199 
00200     for(; itRef != refRecs.end(); ++itRef, ++itRead)
00201     {
00202         refHashes.insert(recHash(*itRef));
00203         CPPUNIT_ASSERT_MESSAGE("No valid mesh found in reconstruction", (*itRef)->getMesh());
00204         readHashes.insert(recHash(*itRead));
00205         CPPUNIT_ASSERT_MESSAGE("No valid mesh found in read reconstruction", (*itRead)->getMesh());
00206     }
00207 
00208     CPPUNIT_ASSERT(refHashes == readHashes);
