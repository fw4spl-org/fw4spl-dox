95 void Field::notify(fwServices::IService::sptr _serviceSource)
96 {
97  SLM_ASSERT("Field helper need a non-null object pointer", !m_object.expired());
98 
99  if ( !m_addedFields.empty() )
100  {
101  auto sig = m_object.lock()->signal< ::fwData::Object::AddedFieldsSignalType >(
102  ::fwData::Object::s_ADDED_FIELDS_SIG);
103 
104  sig->asyncEmit(m_addedFields);
105  }
106  if ( !m_newChangedFields.empty() && !m_oldChangedFields.empty() )
107  {
108  auto sig = m_object.lock()->signal< ::fwData::Object::ChangedFieldsSignalType >(
109  ::fwData::Object::s_CHANGED_FIELDS_SIG);
110 
111  sig->asyncEmit(m_newChangedFields, m_oldChangedFields);
112  }
113  if ( !m_removedFields.empty() )
114  {
115  auto sig = m_object.lock()->signal< ::fwData::Object::RemovedFieldsSignalType >(
116  ::fwData::Object::s_REMOVED_FIELDS_SIG);
117 
118  sig->asyncEmit(m_removedFields);
119  }
120  OSLM_INFO_IF("No changes were found on the fields of the object '" + m_object.lock()->getID()
121  + "', nothing to notify.",
122  m_addedFields.empty() && m_newChangedFields.empty() && m_removedFields.empty());
123 }

127 void Field::buildMessage(
128  const ::fwData::Object::FieldMapType &oldFields,
129  const ::fwData::Object::FieldMapType &newFields
130  )
131 {
132  ::fwData::Object::FieldNameVectorType oldFieldNames;
133  ::fwData::Object::FieldNameVectorType newFieldNames;
134 
135  std::transform(
136  oldFields.begin(), oldFields.end(),
137  std::back_inserter(oldFieldNames),
138  std::bind(&::fwData::Object::FieldMapType::value_type::first, std::placeholders::_1)
139  );
140  std::transform(
141  newFields.begin(), newFields.end(),
142  std::back_inserter(newFieldNames),
143  std::bind(&::fwData::Object::FieldMapType::value_type::first, std::placeholders::_1)
144  );
145 
146  std::sort(oldFieldNames.begin(), oldFieldNames.end());
147  std::sort(newFieldNames.begin(), newFieldNames.end());
148 
149  ::fwData::Object::FieldNameVectorType added; // new - old
150  ::fwData::Object::FieldNameVectorType changed; // old & new
151  ::fwData::Object::FieldNameVectorType removed; // old - new
152 
153  std::set_difference(
154  newFieldNames.begin(), newFieldNames.end(),
155  oldFieldNames.begin(), oldFieldNames.end(),
156  std::back_inserter(added)
157  );
158 
159  std::set_intersection(
160  newFieldNames.begin(), newFieldNames.end(),
161  oldFieldNames.begin(), oldFieldNames.end(),
162  std::back_inserter(changed)
163  );
164 
165  std::set_difference(
166  oldFieldNames.begin(), oldFieldNames.end(),
167  newFieldNames.begin(), newFieldNames.end(),
168  std::back_inserter(removed)
169  );
170 
171  for(const ::fwData::Object::FieldNameVectorType::value_type &fieldName : added)
172  {
173  m_addedFields[fieldName] = newFields.find(fieldName)->second;
174  }
175 
176  for(const ::fwData::Object::FieldNameVectorType::value_type &fieldName : changed)
177  {
178  m_newChangedFields[fieldName] = newFields.find(fieldName)->second;
179  m_oldChangedFields[fieldName] = oldFields.find(fieldName)->second;
180  }
181 
182  for(const ::fwData::Object::FieldNameVectorType::value_type &fieldName : changed)
183  {
184  m_removedFields[fieldName] = oldFields.find(fieldName)->second;
185  }
186 }
