95 void Field::notify()
96 {
97  SLM_ASSERT("Field helper need a non-null object pointer", !m_object.expired());
98 
99  if ( !m_addedFields.empty() )
100  {
101  auto sig = m_object.lock()->signal< ::fwData::Object::AddedFieldsSignalType >(
102  ::fwData::Object::s_ADDED_FIELDS_SIG);
103 
104  sig->asyncEmit(m_addedFields);
105  }
106  if ( !m_newChangedFields.empty() && !m_oldChangedFields.empty() )
107  {
108  auto sig = m_object.lock()->signal< ::fwData::Object::ChangedFieldsSignalType >(
109  ::fwData::Object::s_CHANGED_FIELDS_SIG);
110 
111  sig->asyncEmit(m_newChangedFields, m_oldChangedFields);
112  }
113  if ( !m_removedFields.empty() )
114  {
115  auto sig = m_object.lock()->signal< ::fwData::Object::RemovedFieldsSignalType >(
116  ::fwData::Object::s_REMOVED_FIELDS_SIG);
117 
118  sig->asyncEmit(m_removedFields);
119  }
120  OSLM_INFO_IF("No changes were found on the fields of the object '" + m_object.lock()->getID()
121  + "', nothing to notify.",
122  m_addedFields.empty() && m_newChangedFields.empty() && m_removedFields.empty());
123 }

127 void Field::buildMessage(const ::fwData::Object::FieldMapType& oldFields,
128  const ::fwData::Object::FieldMapType& newFields)
129 {
130  ::fwData::Object::FieldNameVectorType oldFieldNames;
131  ::fwData::Object::FieldNameVectorType newFieldNames;
132 
133  std::transform(
134  oldFields.begin(), oldFields.end(),
135  std::back_inserter(oldFieldNames),
136  std::bind(&::fwData::Object::FieldMapType::value_type::first, std::placeholders::_1)
137  );
138  std::transform(
139  newFields.begin(), newFields.end(),
140  std::back_inserter(newFieldNames),
141  std::bind(&::fwData::Object::FieldMapType::value_type::first, std::placeholders::_1)
142  );
143 
144  std::sort(oldFieldNames.begin(), oldFieldNames.end());
145  std::sort(newFieldNames.begin(), newFieldNames.end());
146 
147  ::fwData::Object::FieldNameVectorType added; // new - old
148  ::fwData::Object::FieldNameVectorType changed; // old & new
149  ::fwData::Object::FieldNameVectorType removed; // old - new
150 
151  std::set_difference(
152  newFieldNames.begin(), newFieldNames.end(),
153  oldFieldNames.begin(), oldFieldNames.end(),
154  std::back_inserter(added)
155  );
156 
157  std::set_intersection(
158  newFieldNames.begin(), newFieldNames.end(),
159  oldFieldNames.begin(), oldFieldNames.end(),
160  std::back_inserter(changed)
161  );
162 
163  std::set_difference(
164  oldFieldNames.begin(), oldFieldNames.end(),
165  newFieldNames.begin(), newFieldNames.end(),
166  std::back_inserter(removed)
167  );
168 
169  for(const ::fwData::Object::FieldNameVectorType::value_type& fieldName : added)
170  {
171  m_addedFields[fieldName] = newFields.find(fieldName)->second;
172  }
173 
174  for(const ::fwData::Object::FieldNameVectorType::value_type& fieldName : changed)
175  {
176  m_newChangedFields[fieldName] = newFields.find(fieldName)->second;
177  m_oldChangedFields[fieldName] = oldFields.find(fieldName)->second;
178  }
179 
180  for(const ::fwData::Object::FieldNameVectorType::value_type& fieldName : changed)
181  {
182  m_removedFields[fieldName] = oldFields.find(fieldName)->second;
183  }
184 }
