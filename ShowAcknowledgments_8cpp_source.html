71 //------------------------------------------------------------------------------
72 void ShowAcknowledgments::updating( ) throw(::fwTools::Failed)
73 {
74  SLM_TRACE_FUNC();
75  SLM_ASSERT("The Acknowledgments service isn't configured properly.", m_bServiceIsConfigured);
76 
77  QDialog* dialog = new QDialog(qApp->activeWindow());
78  dialog->setWindowTitle(QString("Acknowledgments"));
79  QUrl url(QString::fromStdString(m_fsAckPath.string()));
80 #if defined(QT_WEBKIT)
81  QWebView* htmlView = new QWebView(dialog);
82  htmlView->page()->setLinkDelegationPolicy(QWebPage::DelegateAllLinks);
83  htmlView->load( url );
84  QObject::connect( htmlView, SIGNAL(linkClicked(const QUrl &)),this, SLOT(onUrlClicked(const QUrl &)));
85 #else
86  QTextBrowser * htmlView = new QTextBrowser (dialog);
87  htmlView->setSource(url);
88  htmlView->setOpenExternalLinks(true);
89  htmlView->setMinimumSize(500, 400);
90 #endif
91  QPushButton* okButton = new QPushButton(QObject::tr("Ok"));
92  QHBoxLayout *hLayout = new QHBoxLayout();
93  hLayout->addStretch();
94  hLayout->addWidget(okButton);
95  hLayout->setContentsMargins(5,5,5,5);
96 
97  QFrame* line = new QFrame(dialog);
98  line->setFrameShape(QFrame::HLine);
99  line->setFrameShadow(QFrame::Sunken);
100 
101  QVBoxLayout* layout = new QVBoxLayout();
102  layout->addWidget(htmlView, 0);
103  layout->addWidget(line, 0);
104  layout->addLayout(hLayout, 0);
105  layout->setContentsMargins(0,0,0,0);
106  layout->setSpacing(0);
107  dialog->setLayout( layout );
108 
109  QObject::connect(okButton, SIGNAL(clicked()), dialog, SLOT(accept()));
110  dialog->exec();
111 }
