00050 void ConfigurationElement2XML(::fwRuntime::ConfigurationElement::sptr _cfgElement, xmlNodePtr pNode)
00051 {
00052     //ATTRIBUTES + VALUES
00053     std::map<std::string, std::string> attr = _cfgElement->getAttributes() ;
00054     for ( std::map<std::string, std::string>::iterator iter_attr_cfe = attr.begin() ; iter_attr_cfe!= attr.end(); ++iter_attr_cfe)
00055     {
00056         xmlSetProp(pNode , xmlCharStrdup((iter_attr_cfe->first).c_str()), xmlCharStrdup((iter_attr_cfe->second).c_str()) ) ;
00057     }
00058     //ELEMENTS
00059     BOOST_FOREACH(::fwRuntime::ConfigurationElement::sptr elt, _cfgElement->getElements())
00060     {
00061         xmlNodePtr child = xmlNewNode( NULL,  xmlCharStrdup( elt->getName().c_str() ) );
00062 
00063         xmlAddChild(pNode, child) ;
00064         // If configuration element is a XML_TEXT_NODE : WARNING : even whitespace (non XML_TEXT_NODE) have been considered as valid XML_TEXT_NODE by BundleDescriptorReader!!!!
00065         if( !elt->getValue().empty() )
00066         {
00067             xmlNodeSetContent(child, xmlCharStrdup( elt->getValue().c_str() ));
00068         }
00069         // If configuration element is a XML_ELEMENT_NODE
00070         else
00071         {
00072             ConfigurationElement2XML(elt, child);
00073         }
00074     }

00079 ConfigurationElement::sptr getCfgAsAnExtension( ConfigurationElement::sptr config , std::string extension_pt )
00080 {
00081     ConfigurationElement::sptr resultConfig;
00082     if( config->hasAttribute("config") )
00083     {
00084         std::string cfgContribution = config->getExistingAttributeValue("config") ;
00085         std::vector< ConfigurationElement::sptr > cfgs = ::fwRuntime::getAllCfgForPoint( extension_pt  ) ;
00086         OSLM_FATAL_IF( "No configuration contribution found for extension point " <<  extension_pt, cfgs.empty());
00087 
00088         // Search for all matching contributions
00089         std::vector< ConfigurationElement::sptr > matchingCfg ;
00090         BOOST_FOREACH(ConfigurationElement::sptr elt, cfgs)
00091         {
00092             if( cfgContribution == elt->getExistingAttributeValue("id") )
00093             {
00094                 matchingCfg.push_back( elt ) ;
00095             }
00096         }
00097 
00098         // If no contribution found
00099         OSLM_FATAL_IF( "No contribution matching the requested requirement (" << cfgContribution
00100                 << " for extension point " << extension_pt << " )", matchingCfg.empty());
00101 
00102         // Normal case : only one matching contribution has been found: matchingCfg.size == 1
00103         resultConfig =  *matchingCfg.begin();
00104 
00105         // If several matching contributions
00106         OSLM_WARN_IF("Several contribution identified by " << cfgContribution << "( for cfg extension point " << extension_pt << " )"
00107                 << " has been found : the first one is returned", (matchingCfg.size() > 1));
00108     }
00109     SLM_WARN_IF("Configuration has no config attribute",  !config->hasAttribute("config"));
00110     return resultConfig ;
00111 }
