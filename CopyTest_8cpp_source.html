121 void CopyTest::severalReferencesCopyTest()
122 {
123  ::fwData::Integer::sptr integer = ::fwData::Integer::New(42);
124  ::fwData::Composite::sptr composite = ::fwData::Composite::New();
125 
126  (*composite)["A"] = integer;
127  (*composite)["B"] = integer;
128  composite->setField("F1", integer);
129  composite->setField("F2", integer);
130 
131  ::fwData::Composite::sptr compositeCopy = ::fwData::Object::copy(composite);
132 
133  CPPUNIT_ASSERT( integer != ::fwData::Integer::dynamicCast((*compositeCopy)["A"]) );
134  CPPUNIT_ASSERT_EQUAL(42, ::fwData::Integer::dynamicCast((*compositeCopy)["A"])->getValue());
135  CPPUNIT_ASSERT_EQUAL((*compositeCopy)["A"], compositeCopy->getField("F1"));
136  CPPUNIT_ASSERT_EQUAL((*compositeCopy)["A"], compositeCopy->getField("F2"));
137  CPPUNIT_ASSERT_EQUAL((*compositeCopy)["A"], (*compositeCopy)["B"]);
138 
139 
140  ::fwData::Vector::sptr vector = ::fwData::Vector::New();
141 
142  vector->getContainer().push_back(composite);
143  vector->getContainer().push_back(composite);
144  vector->setField("F1", composite);
145  vector->setField("F2", composite);
146 
147  ::fwData::Vector::sptr vectorCopy = ::fwData::Object::copy(vector);
148  CPPUNIT_ASSERT( composite != (*vectorCopy)[0] );
149  CPPUNIT_ASSERT_EQUAL((*vectorCopy)[0], vectorCopy->getField("F1"));
150  CPPUNIT_ASSERT_EQUAL((*vectorCopy)[0], vectorCopy->getField("F2"));
151  CPPUNIT_ASSERT_EQUAL((*vectorCopy)[0], (*vectorCopy)[1]);
152 
153 
154 
155  ::fwData::List::sptr list = ::fwData::List::New();
156 
157  list->getContainer().push_back(vector);
158  list->getContainer().push_back(vector);
159  list->setField("F1", vector);
160  list->setField("F2", vector);
161 
162  ::fwData::List::sptr listCopy = ::fwData::Object::copy(list);
163  CPPUNIT_ASSERT( vector != (*listCopy).front() );
164  CPPUNIT_ASSERT_EQUAL((*listCopy).front(), listCopy->getField("F1"));
165  CPPUNIT_ASSERT_EQUAL((*listCopy).front(), listCopy->getField("F2"));
166  CPPUNIT_ASSERT_EQUAL((*listCopy).front(), (*listCopy).back());
167
