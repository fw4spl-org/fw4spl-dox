00166 void visit(const ::fwAtoms::Base::sptr &base, const std::string &path)

00177         if (base)

00191             else if(base->isString())

00193                 type.write( H5std_string("string"), str_type );
00194                 H5::DataSet val  = group.createDataSet( "value", str_type, att_space );

00197             else if(base->isBoolean())

00199                 type.write( H5std_string("boolean"), str_type );
