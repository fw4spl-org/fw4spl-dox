35 void Mesh::fromVTKMesh( vtkSmartPointer<vtkPolyData> polyData, ::fwData::Mesh::sptr mesh )
36 {
37  vtkPoints* points = polyData->GetPoints();
38  if (points)
39  {
40  mesh->clear();
41  vtkIdType numberOfPoints = points->GetNumberOfPoints();
42  vtkIdType numberOfCells = polyData->GetNumberOfCells();
43 
44  mesh->allocate(numberOfPoints, numberOfCells, numberOfCells*3);
45  ::fwDataTools::helper::Mesh meshHelper(mesh);
46 
47  double* point;
48  ::fwData::Mesh::Id idx;
49  for (vtkIdType i = 0; i < numberOfPoints; ++i)
50  {
51  point = points->GetPoint(i);
52  idx = meshHelper.insertNextPoint(point[0], point[1], point[2]);
53  SLM_ASSERT("Mesh index not correspond to VTK index point", idx == i);
54  }
55 
56  vtkCell* cell;
57  vtkIdList* idList;
58  int cellType;
59  for (vtkIdType i = 0; i < numberOfCells; ++i)
60  {
61  cell = polyData->GetCell(i);
62  idList = cell->GetPointIds();
63  cellType = cell->GetCellType();
64 
65  switch (cellType)
66  {
67  case VTK_VERTEX:
68  SLM_ASSERT("Wrong number of ids: "<<idList->GetNumberOfIds(), idList->GetNumberOfIds()==1);
69  meshHelper.insertNextCell( idList->GetId(0));
70  break;
71  case VTK_LINE:
72  SLM_ASSERT("Wrong number of ids: "<<idList->GetNumberOfIds(), idList->GetNumberOfIds()==2);
73  meshHelper.insertNextCell( idList->GetId(0), idList->GetId(1));
74  break;
75  case VTK_TRIANGLE:
76  SLM_ASSERT("Wrong number of ids: "<<idList->GetNumberOfIds(), idList->GetNumberOfIds()==3);
77  meshHelper.insertNextCell( idList->GetId(0), idList->GetId(1), idList->GetId(2));
78  break;
79  case VTK_QUAD:
80  SLM_ASSERT("Wrong number of ids: "<<idList->GetNumberOfIds(), idList->GetNumberOfIds()==4);
81  meshHelper.insertNextCell( idList->GetId(0), idList->GetId(1), idList->GetId(2), idList->GetId(3));
82  break;
83  default:
84  FW_RAISE("VTK Mesh type "<<cellType<< " not supported.");
85  }
86  }
87 
88 
89  if(polyData->GetPointData()->HasArray("Colors"))
90  {
91  vtkSmartPointer<vtkUnsignedCharArray> colors;
92  colors = vtkUnsignedCharArray::SafeDownCast(polyData->GetPointData()->GetArray("Colors"));

94

99  meshHelper.updateLock();
100 
101  ::fwData::Mesh::Id nbPoints = mesh->getNumberOfPoints();
102  for (size_t i = 0; i != nbPoints; ++i)
103  {
104  meshHelper.setPointColor(i, colors->GetPointer(i*nbComponents));
105  }
106  }

108  if(polyData->GetCellData()->HasArray("Colors"))
109  {

387 void Mesh::toVTKMesh( const ::fwData::Mesh::csptr& mesh, vtkSmartPointer<vtkPolyData> polyData)
388 {
389  vtkSmartPointer< vtkPoints > pts = vtkSmartPointer< vtkPoints >::New();
390  polyData->SetPoints(pts);
391  Mesh::updatePolyDataPoints(polyData, mesh);
392 
393  ::fwDataTools::helper::MeshGetter meshHelper(mesh);
394  unsigned int nbCells = mesh->getNumberOfCells();
395 
396  ::fwData::Mesh::ConstCellTypesMultiArrayType cellTypes = meshHelper.getCellTypes();
397  ::fwData::Mesh::ConstCellDataMultiArrayType cellData = meshHelper.getCellData();
398  ::fwData::Mesh::ConstCellDataOffsetsMultiArrayType cellDataOffsets = meshHelper.getCellDataOffsets();
399 
400  polyData->Allocate(4, nbCells);
401 
402  vtkIdType typeVtkCell;
403  vtkIdType cell[4];
404  for(unsigned int i = 0; i<nbCells; ++i )
405  {
406  ::fwData::Mesh::CellTypes cellType = cellTypes[i];
407  ::fwData::Mesh::Id offset = cellDataOffsets[i];
408  switch( cellType )
409  {
410  case ::fwData::Mesh::POINT:
411  typeVtkCell = VTK_VERTEX;
412  cell[0] = cellData[offset];
413  polyData->InsertNextCell( typeVtkCell, 1, cell );
414  break;
415  case ::fwData::Mesh::EDGE:
416  typeVtkCell = VTK_LINE;
417  cell[0] = cellData[offset];
418  cell[1] = cellData[offset+1];
419  polyData->InsertNextCell( typeVtkCell, 2, cell );
420  break;
421  case ::fwData::Mesh::TRIANGLE:
422  typeVtkCell = VTK_TRIANGLE;
423  cell[0] = cellData[offset];
424  cell[1] = cellData[offset+1];
425  cell[2] = cellData[offset+2];
426  polyData->InsertNextCell( typeVtkCell, 3, cell );
427  break;
428  case ::fwData::Mesh::QUAD:
429  typeVtkCell = VTK_QUAD;
430  cell[0] = cellData[offset];
431  cell[1] = cellData[offset+1];
432  cell[2] = cellData[offset+2];
433  cell[3] = cellData[offset+3];
434  polyData->InsertNextCell( typeVtkCell, 4, cell );
435  break;
436  case ::fwData::Mesh::TETRA:
437  typeVtkCell = VTK_LINE;
438  cell[0] = cellData[offset];
439  cell[1] = cellData[offset+1];
440  polyData->InsertNextCell( typeVtkCell, 2, cell );
441 
442  cell[0] = cellData[offset+1];
443  cell[1] = cellData[offset+2];
444  polyData->InsertNextCell( typeVtkCell, 2, cell );
445 
446  cell[0] = cellData[offset+2];
447  cell[1] = cellData[offset+3];
448  polyData->InsertNextCell( typeVtkCell, 2, cell );
449 
450  cell[0] = cellData[offset+3];
451  cell[1] = cellData[offset];
452  polyData->InsertNextCell( typeVtkCell, 2, cell );
453 
454  cell[0] = cellData[offset+2];
455  cell[1] = cellData[offset];
456  polyData->InsertNextCell( typeVtkCell, 2, cell );
457 
458  cell[0] = cellData[offset+1];
459  cell[1] = cellData[offset+3];
460  polyData->InsertNextCell( typeVtkCell, 2, cell );
461 
462  }
463  }
464 
465  Mesh::updatePolyDataPointNormals(polyData, mesh);
466  Mesh::updatePolyDataCellNormals(polyData, mesh);
467 
468  Mesh::updatePolyDataPointColor(polyData, mesh);
469  Mesh::updatePolyDataCellColor(polyData, mesh);
470 
471  Mesh::updatePolyDataPointTexCoords(polyData, mesh);
472  Mesh::updatePolyDataCellTexCoords(polyData, mesh);
473 }

477 vtkSmartPointer<vtkPolyData> Mesh::updatePolyDataPoints(vtkSmartPointer<vtkPolyData> polyDataDst,
478  const ::fwData::Mesh::csptr& meshSrc )
479 {
480  SLM_ASSERT( "vtkPolyData should not be NULL", polyDataDst);
481  ::fwDataTools::helper::MeshGetter meshHelper(meshSrc);
482 
483  vtkPoints* polyDataPoints = polyDataDst->GetPoints();
484  ::fwData::Mesh::Id nbPoints = meshSrc->getNumberOfPoints();
485  ::fwData::Mesh::ConstPointsMultiArrayType points = meshHelper.getPoints();
486 
487  if (nbPoints != polyDataPoints->GetNumberOfPoints())
488  {
489  polyDataPoints->SetNumberOfPoints(nbPoints);
490  }
491 
492  vtkIdType id = 0;
493  typedef ::fwData::Mesh::PointsMultiArrayType::index PointTypesIndex;
494  for (PointTypesIndex i = 0; i != nbPoints; ++i)
495  {
496  polyDataPoints->SetPoint(id++, points[i][0], points[i][1], points[i][2]);
497  }
498 
499  polyDataPoints->Modified();
500  return polyDataDst;
501 }

507 {
508  SLM_ASSERT( "vtkPolyData should not be NULL", polyDataDst);
509 
510  ::fwData::Array::sptr pointColorArray = meshSrc->getPointColorsArray();
511  if(pointColorArray)
512  {
513  ::fwDataTools::helper::Array arrayHelper(pointColorArray);
514 
515  vtkSmartPointer<vtkUnsignedCharArray> colors = vtkSmartPointer<vtkUnsignedCharArray>::New();
516  size_t nbComponents = pointColorArray->getNumberOfComponents();
517  colors->SetNumberOfComponents(nbComponents);
518  colors->SetName("Colors");
519 
520  unsigned char* pointColor = arrayHelper.begin< unsigned char >();
521  unsigned char* pointColorEnd = arrayHelper.end< unsigned char >();
522 
523  for (; pointColor != pointColorEnd; pointColor += nbComponents)
524  {
525  colors->InsertNextTupleValue(pointColor);
526  }
527  polyDataDst->GetPointData()->SetScalars(colors);
528  polyDataDst->Modified();
529  }
530  else
531  {
532  if(polyDataDst->GetPointData()->HasArray("Colors"))
533  {
534  polyDataDst->GetPointData()->RemoveArray("Colors");
535  }
536  polyDataDst->Modified();
537  }
538 
539  return polyDataDst;
540 }

745 double Mesh::computeVolume( const ::fwData::Mesh::csptr& mesh )
746 {
747  vtkSmartPointer< vtkPolyData > vtkMeshRaw = vtkSmartPointer< vtkPolyData >::New();
748  Mesh::toVTKMesh( mesh, vtkMeshRaw );
749 
750  // identify and fill holes in meshes
751  vtkSmartPointer< vtkFillHolesFilter > holesFilter = vtkSmartPointer< vtkFillHolesFilter >::New();
752  holesFilter->SetHoleSize(2000);
753  holesFilter->SetInputData(vtkMeshRaw);
754  holesFilter->Update();
755  if (holesFilter->GetOutput()->GetNumberOfCells() > 0) // Filter return empty mesh when no topological holes are present
756  {
757  vtkMeshRaw = holesFilter->GetOutput();
758  }
759 
760  // compute normals for polygonal mesh
761  vtkSmartPointer< vtkPolyDataNormals > filter = vtkSmartPointer< vtkPolyDataNormals >::New();
762  filter->SetInputData(vtkMeshRaw);
763  filter->AutoOrientNormalsOn();
764  filter->FlipNormalsOff();
765 
766  // estimate volume, area, shape index of triangle mesh
767  vtkSmartPointer< vtkMassProperties > calculator = vtkSmartPointer< vtkMassProperties >::New();
768  calculator->SetInputConnection( filter->GetOutputPort() );
769  calculator->Update();
770 
771  double volume = calculator->GetVolume();
772  OSLM_DEBUG(
773  "GetVolume : " << volume << " vtkMassProperties::GetVolumeProjected = " <<
774  calculator->GetVolumeProjected() );
775  OSLM_DEBUG("Error : " << (calculator->GetVolume()- fabs(calculator->GetVolumeProjected()))*10000);
776  if ( (calculator->GetVolume()- fabs(calculator->GetVolumeProjected()))*10000 > calculator->GetVolume() )
777  {
778  std::stringstream ss;
779  ss << "volume - |volume projected| > volume/10000.0" << std::endl;
780  ss << "[volume = " << volume << "]" << std::endl;
781  ss << "[volume projected = " << calculator->GetVolumeProjected()<<"]";
782  throw (std::out_of_range( ss.str() ));
783  }
784 
785  return volume;
786 }
