00239             typeVtkCell = VTK_QUAD;

00241             cell[1] = cellData[offset+1];

00247             typeVtkCell = VTK_LINE;

00258             polyData->InsertNextCell( typeVtkCell, 2, cell );
00259 
00260             cell[0] = cellData[offset+3];
00261             cell[1] = cellData[offset];
00262             polyData->InsertNextCell( typeVtkCell, 2, cell );
00263 
00264             cell[0] = cellData[offset+2];
00265             cell[1] = cellData[offset];
00266             polyData->InsertNextCell( typeVtkCell, 2, cell );
00267 
00268             cell[0] = cellData[offset+1];

00551 {
00552     vtkSmartPointer< vtkPolyData > vtkMeshRaw = vtkSmartPointer< vtkPolyData >::New();
00553     Mesh::toVTKMesh( mesh, vtkMeshRaw );
00554 
00555     // identify and fill holes in meshes
00556     vtkSmartPointer< vtkFillHolesFilter > holesFilter = vtkSmartPointer< vtkFillHolesFilter >::New();
00557     holesFilter->SetHoleSize(2000);
00558     holesFilter->SetInputData(vtkMeshRaw);
00559     holesFilter->Update();
00560     if (holesFilter->GetOutput()->GetNumberOfCells() > 0) // Filter return empty mesh when no topological holes are present
00561     {
00562         vtkMeshRaw = holesFilter->GetOutput();
00563     }
00564 
00565     // compute normals for polygonal mesh
00566     vtkSmartPointer< vtkPolyDataNormals > filter = vtkSmartPointer< vtkPolyDataNormals >::New();
00567     filter->SetInputData(vtkMeshRaw);
00568     filter->AutoOrientNormalsOn();
00569     filter->FlipNormalsOff();
00570 
00571     // estimate volume, area, shape index of triangle mesh
00572     vtkSmartPointer< vtkMassProperties > calculator = vtkSmartPointer< vtkMassProperties >::New();
00573     calculator->SetInputConnection( filter->GetOutputPort() );
00574     calculator->Update();
00575 
00576     double volume =  calculator->GetVolume();
00577     OSLM_DEBUG("GetVolume : " << volume << " vtkMassProperties::GetVolumeProjected = " << calculator->GetVolumeProjected() );
00578     OSLM_DEBUG("Error : " << (calculator->GetVolume()- fabs(calculator->GetVolumeProjected()))*10000);
00579     if ( (calculator->GetVolume()- fabs(calculator->GetVolumeProjected()))*10000 > calculator->GetVolume() )
00580     {
00581         std::stringstream ss;
