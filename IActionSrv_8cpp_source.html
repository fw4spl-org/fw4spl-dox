61 void IActionSrv::initialize()
62 {
63  m_registrar = ::fwGui::registrar::ActionRegistrar::New(this->getID());
64 
65  OSLM_ASSERT("Depreciated tag <name> in "<< this->getID() << " configuration.", !m_configuration->hasAttribute(
66  "name"));
67  OSLM_ASSERT("Depreciated tag <shortcut> in "<< this->getID() << " configuration.",
68  !m_configuration->hasAttribute("shortcut"));
69  OSLM_ASSERT("Depreciated tag <enable> in "<< this->getID() << " configuration.",
70  !m_configuration->hasAttribute("enable"));
71  OSLM_ASSERT(
72  "Depreciated tag <specialAction> in "<< this->getID() << " configuration.",
73  !m_configuration->hasAttribute("specialAction"));
74  OSLM_ASSERT("Depreciated tag <style> in "<< this->getID() << " configuration.",
75  !m_configuration->hasAttribute("style"));
76  OSLM_ASSERT("Depreciated tag <state> in "<< this->getID() << " configuration.",
77  !m_configuration->hasAttribute("state"));
78 
79  ::fwRuntime::ConfigurationElementContainer::Iterator iter;
80  for( iter = m_configuration->begin(); iter != m_configuration->end(); ++iter )
81  {
82  if( (*iter)->getName() == "state" )
83  {
84  ConfigurationType stateCfg = *iter;
85 
86  if( stateCfg->hasAttribute("inverse") )
87  {
88  std::string invertState = stateCfg->getExistingAttributeValue("inverse");
89  SLM_ASSERT("Wrong attribute value : must be 'true' or 'false'",
90  (invertState == "true") || (invertState == "false"));
91  m_activeStateValue = !(invertState == "true");
92  }
93 
94  if( stateCfg->hasAttribute("active") )
95  {
96  std::string isActive = stateCfg->getExistingAttributeValue("active");
97  SLM_ASSERT("Wrong attribute value : must be 'true' or 'false'",
98  (isActive == "true") || (isActive == "false"));
99  m_isActive = (isActive == "true");
100  }
101 
102  if( stateCfg->hasAttribute("executable") )
103  {
104  std::string isExecutable = stateCfg->getExistingAttributeValue("executable");
105  SLM_ASSERT("Wrong attribute value : must be 'true' or 'false'",
106  (isExecutable == "true") || (isExecutable == "false"));
107  m_isExecutable = (isExecutable == "true");
108  }
109 
110  if( stateCfg->hasAttribute("visible") )
111  {
112  std::string isVisible = stateCfg->getExistingAttributeValue("visible");
113  SLM_ASSERT("Wrong attribute value : must be 'true' or 'false'",
114  (isVisible == "true") || (isVisible == "false"));
115  m_isVisible = (isVisible == "true");
116  }
117  }
118 
119  if( (*iter)->getName() == "confirmation" )
120  {
121  ConfigurationType cfg = *iter;
122 
123  SLM_ASSERT("Missing attribute 'value'", cfg->hasAttribute("value"));
124  std::string confirm = cfg->getExistingAttributeValue("value");
125  SLM_ASSERT("Wrong attribute value : must be 'true' or 'false'",
126  (confirm == "true") || (confirm == "false"));
127  m_confirmAction = (confirm == "true");
128 
129  if( cfg->hasAttribute("message") )
130  {
131  m_confirmMessage = cfg->getExistingAttributeValue("message");
132  }
133 
134  if( cfg->hasAttribute("defaultbutton") )
135  {
136  m_defaultButton = cfg->getExistingAttributeValue("defaultbutton");
137  }
138 
139  }
140  }

247 bool IActionSrv::confirmAction()
248 {
249  bool actionIsConfirmed = true;
250 
251  if (m_confirmAction)
252  {
253  ::fwGui::dialog::MessageDialog dialog;
254  dialog.setTitle("Confirmation");
255  std::stringstream ss;
256  ss << "Do you really want to execute this action ? ";
257  if (!m_confirmMessage.empty())
258  {
259  ss << std::endl << m_confirmMessage;
260  }
261  dialog.setMessage( ss.str() );
262 
263  if(m_defaultButton == "yes")
264  {
265  dialog.setDefaultButton( ::fwGui::dialog::IMessageDialog::YES );
266  }
267  else if(m_defaultButton == "no")
268  {
269  dialog.setDefaultButton( ::fwGui::dialog::IMessageDialog::NO );
270  }
271  else if(!m_defaultButton.empty())
272  {
273  SLM_WARN("unknown button: " + m_defaultButton);
274  }
275 
276  dialog.setIcon( ::fwGui::dialog::IMessageDialog::QUESTION );
277  dialog.addButton( ::fwGui::dialog::IMessageDialog::YES_NO );
278  ::fwGui::dialog::IMessageDialog::Buttons button = dialog.show();
279 
280  actionIsConfirmed = (button == ::fwGui::dialog::IMessageDialog::YES);
281  }
282 
283  return actionIsConfirmed;
284 }
