00049 ::boost::python::object getImageBuffer (::fwData::Image::sptr image)
00050 {
00051     using namespace ::boost::python;
00052 
00053     ::fwComEd::helper::Image imageHelper(image);
00054     if ( imageHelper.getBuffer() )
00055     {
00056         Py_buffer *pybuf = new Py_buffer;
00057         pybuf->obj = NULL;
00058         pybuf->buf = imageHelper.getBuffer();
00059         pybuf->readonly= 0;
00060         pybuf->len = image->getSizeInBytes();
00061         pybuf->format =  (char *)typeCPP2Python[ image->getType() ];
00062         pybuf->itemsize =  image->getType().sizeOf() ;
00063         pybuf->ndim = image->getNumberOfDimensions();
00064         pybuf->shape =  new Py_ssize_t[image->getNumberOfDimensions()];
00065         pybuf->strides =  new Py_ssize_t[image->getNumberOfDimensions()];
00066         pybuf->suboffsets =  new Py_ssize_t[image->getNumberOfDimensions()];
00067         pybuf->internal = NULL;
00068 
00069         std::copy(  image->getSize().begin(), image->getSize().end(), pybuf->shape);
00070         std::fill(pybuf->suboffsets, pybuf->suboffsets+3, -1);
00071 
00072         PyBuffer_FillContiguousStrides(   pybuf->ndim , pybuf->shape, pybuf->strides, pybuf->itemsize, 'F');
00073 
00074         handle<> bufHandle( PyMemoryView_FromBuffer( pybuf ) );
00075         delete pybuf;
00076 
00077         return object( bufHandle );
00078     }
00079 
00080     return object();
00081 }
