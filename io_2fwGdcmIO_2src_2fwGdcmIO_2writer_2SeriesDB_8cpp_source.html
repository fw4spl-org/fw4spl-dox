44 void SeriesDB::write()
45 {
46 // // Disable GDCM Warnings
47 // ::gdcm::Trace::SetWarning(false);
48 
49  ::fwMedData::SeriesDB::sptr seriesDB = this->getConcreteObject();
50  SLM_ASSERT("SeriesDB not instanced", seriesDB);
51 
52  ::fwGdcmIO::writer::Series::sptr writer = ::fwGdcmIO::writer::Series::New();
53  writer->setFiducialsExportMode(m_fiducialsExportMode);
54 
55  // Copy and sort container in order to write ImageSeries before ModelSeries
56  ::fwMedData::SeriesDB::ContainerType seriesContainer = seriesDB->getContainer();
57  std::sort (seriesContainer.begin(), seriesContainer.end(), SeriesDB::seriesComparator);
58 
59  // Write all patients
60  for( ::fwMedData::Series::sptr series: seriesContainer)
61  {
62  // Create a new directory
63  const ::boost::filesystem::path& seriesPath = this->getFolder() / series->getInstanceUID();
64  ::boost::filesystem::create_directory(seriesPath);
65  writer->setObject(series);
66  writer->setFolder(seriesPath);
67 
68  // Forward event progress to its parents
69  ::fwTools::ProgressAdviser::ProgessHandler handler =
70  ::boost::bind( &Series::notifyProgress, this, ::boost::lambda::_1, ::boost::lambda::_2);
71  writer->addHandler(handler);
72 
73  // Write a series
74  writer->write();
75  }
76 }
