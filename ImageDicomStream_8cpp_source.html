00052     vtkSmartPointer< vtkStringArray > fileArray = vtkSmartPointer< vtkStringArray >::New();
00053     fileArray->Initialize();
00054     BOOST_FOREACH( std::string file, m_dcmInfo->m_seriesFiles )
00055     {
00056         OSLM_TRACE("Add " << file << " to vtkGdcmReader");
00057         fileArray->InsertNextValue( file.c_str() );
00058     }
00059 
00060     m_reader = vtkSmartPointer< vtkGDCMImageReader >::New();
00061     m_reader->FileLowerLeftOn();

00063     bool res = false;
00064     if ( filesStillExist( m_dcmInfo->m_seriesFiles ) )
00065     {
00066         m_reader->SetFileNames( fileArray );
00067         try
00068         {
00069             m_reader->Update();
00070             m_inputReader = static_cast<char*>( m_reader->GetOutput()->GetScalarPointer() );
00071             res = true;
00072         }
00073         catch (std::exception &e)
00074         {
00075             m_reader = 0;
00076             m_inputReader = 0;
00077             OSLM_ERROR ( "Error during conversion : " << e.what() );
00078         }
00079         catch (...)
00080         {
00081             m_reader = 0;
00082             m_inputReader = 0;
00083             OSLM_ERROR ( "Unexpected error during conversion" );

00102     streamsize amt = static_cast<streamsize>( m_dcmInfo->m_buffSizeInBytes - m_pos );
00103     streamsize result = std::min(n, amt);
00104 
00105     if (result != 0) {
00106         if( m_success )
00107         {
00108             std::copy( m_inputReader + m_pos, m_inputReader + m_pos + result, s );

00114     {
