110 struct RegexLogCompare
111 {
112  bool operator() (std::string a, std::string b)
113  {
114  boost::regex re(".*(msg n [[:digit:]]+)$");
115  boost::smatch matchA;
116  boost::smatch matchB;
117  bool doMatchA = boost::regex_match(a, matchA, re);
118  bool doMatchB = boost::regex_match(b, matchB, re);
119  CPPUNIT_ASSERT_MESSAGE( std::string("Regex do not match ") + a, doMatchA);
120  CPPUNIT_ASSERT_MESSAGE( std::string("Regex do not match ") + b, doMatchB);
121 
122  std::string strA(matchA[1].first, matchA[1].second);
123  std::string strB(matchB[1].first, matchB[1].second);
124 
125  return strA < strB;
126  }
127 } regex_compare;

170 void SpyLogTest::checkLog(const std::vector<std::string> &logMessagesRef, const std::vector<std::string> &logMessages)
171 {
172  CPPUNIT_ASSERT_EQUAL(logMessagesRef.size(), logMessages.size());
173 
174  const std::string linePattern("(\\[[0-9]+\\])");
175  const std::string timePattern("(\\[.+\\])");
176  const std::string levelPattern("( *\\[[a-z]+\\])");
177  const std::string filePattern("( .*:)");
178  const std::string fileLinePattern("([0-9]+: )");
179  const std::string messagePattern("(.*)$");
180 
181  boost::regex re(
182  linePattern
183  + timePattern
184  + levelPattern
185  + filePattern
186  + fileLinePattern
187  + messagePattern );
188 
189  boost::smatch match;
190  std::string regexMessage;
191  size_t i = 0;
192 
193  for(const std::string &log : logMessages)
194  {
195  bool doMatch = boost::regex_match(log, match, re);
196 
197  CPPUNIT_ASSERT_MESSAGE(log + " don't match regex.", doMatch);
198 
199  regexMessage.assign(match[6].first, match[6].second);
200 
201  CPPUNIT_ASSERT_EQUAL(logMessagesRef[i], regexMessage);
202  ++i;
203  }
204 }
