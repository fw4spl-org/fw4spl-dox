48 void GetObjectTest::getTest()
49 {
50  // Visit 1
51  ::fwData::Image::sptr img1 = ::fwData::Image::New();
52  ::fwTest::generator::Image::generateRandomImage(img1, ::fwTools::Type::create("int16"));
53  ::fwData::Image::sptr img2 = ::fwData::Image::New();
54  ::fwTest::generator::Image::generateRandomImage(img2, ::fwTools::Type::create("uint8"));
55  ::fwData::Composite::sptr composite = ::fwData::Composite::New();
56  composite->getContainer()["img1"] = img1;
57  composite->getContainer()["img2"] = img2;
58  ::fwData::Object::sptr subObj1 = ::fwDataCamp::getObject( composite, "@values.img2" );
59  CPPUNIT_ASSERT_MESSAGE("Image must be equal", subObj1 == img2);
60 
61  // Visit 2
62  ::fwData::Float::sptr zspacing = ::fwDataCamp::getObject< ::fwData::Float >( composite, "@values.img2.spacing.2" );
63  CPPUNIT_ASSERT_MESSAGE("spacing must be equal",
64  img2->getSpacing()[2] - 0.001 < zspacing->value() &&
65  zspacing->value() < img2->getSpacing()[2] + 0.001 );
66 
67  // Visit 3
68  ::fwMedData::Patient::sptr patient1 = ::fwMedData::Patient::New();
69  patient1->setName( "toto" );
70  ::fwData::String::sptr str = ::fwDataCamp::getObject< ::fwData::String >( patient1, "@name" );
71  CPPUNIT_ASSERT_MESSAGE("Name must be equal", patient1->getName() == str->value() );
72 
73  // Visit 4
74  composite->setField("toto", img1);
75  img1->setField("titi", img2);
76  ::fwData::Object::sptr subObj2 = ::fwDataCamp::getObject( composite, "@fields.toto.fields.titi" );
77  CPPUNIT_ASSERT_MESSAGE("Image must be equal", subObj2 == img2 );
78 }
