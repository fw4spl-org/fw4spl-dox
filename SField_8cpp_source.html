76 void SField::stopping() throw ( ::fwTools::Failed )
77 {
78  SLM_TRACE_FUNC();
79 
80  for(SubServicesMapType::value_type elt : m_fieldsSubServices)
81  {
82  SubServicesVecType subServices = elt.second;
83  BOOST_REVERSE_FOREACH( SPTR(SubService) subSrv, subServices )
84  {
85  OSLM_ASSERT("SubService on "<< elt.first <<" expired !", subSrv->getService() );
86 
87  if( subSrv->m_hasAutoConnection )
88  {
89  subSrv->m_connections.disconnect();
90  }
91  subSrv->getService()->stop().wait();
92  ::fwServices::OSR::unregisterService(subSrv->getService());
93  subSrv->m_service.reset();
94  }
95  }
96  m_fieldsSubServices.clear();
97 
98  while( !m_objectConnections.empty())
99  {
100  this->removeConnections(m_objectConnections.begin()->first);
101  }
102  SLM_ASSERT("Connections must be empty", m_objectConnections.empty());
103 
104  while( !m_proxyCtns.empty())
105  {
106  this->disconnectProxies(m_proxyCtns.begin()->first);
107  }
108  SLM_ASSERT("Proxy connections must be empty", m_proxyCtns.empty());
109 }

135 {

137 
138  ::fwData::Object::sptr object = this->getObject();
139  const ::fwData::Object::FieldMapType& fieldsMap = object->getFields();
140  ::fwRuntime::ConfigurationElementContainer::Iterator iter;
141  for (iter = m_managerConfiguration->begin(); iter != m_managerConfiguration->end(); ++iter)
142  {
143  if ((*iter)->getName() == "field")
144  {
145  const std::string fieldName = (*iter)->getAttributeValue("id");
146 
147  if (fieldsMap.find(fieldName) != fieldsMap.end())

151  else if (m_dummyStopMode)

180 ::fwServices::IService::sptr SField::add( ::fwData::Object::sptr obj, ::fwRuntime::ConfigurationElement::sptr _elt )
181 {
182  OSLM_ASSERT("ConfigurationElement node name must be \"service\" not "<<_elt->getName(),
183  _elt->getName() == "service" );
184  SLM_ASSERT("Attribute \"type\" is missing", _elt->hasAttribute("type") );
185  SLM_ASSERT("Attribute \"impl\" is missing", _elt->hasAttribute("impl") );
186 
187  ::fwServices::IService::sptr service;
188 
189  std::string serviceType = _elt->getExistingAttributeValue("type");
190  std::string implementationType = _elt->getExistingAttributeValue("impl");
191 
192  // Add service with possible id
193  if( _elt->hasAttribute("uid") )
194  {
195  service = ::fwServices::add( obj, serviceType, implementationType, _elt->getExistingAttributeValue("uid") );
196  }
197  else
198  {
199  service = ::fwServices::add( obj, serviceType, implementationType );
200  }
201 
202  // Search for configuration : inline or offline
203  ::fwRuntime::ConfigurationElement::sptr cfg = _elt;
204  if( _elt->hasAttribute("config"))
205  {
206  cfg = ::fwRuntime::ConfigurationElement::constCast(
207  ::fwServices::registry::ServiceConfig::getDefault()->getServiceConfig( _elt->
208  getExistingAttributeValue("config"),
209  implementationType ) );
210  }
211 
212  // Set configuration
213  service->setConfiguration( cfg );
214 
215  // Configure
216  service->configure();
217 
218  // Return
219  return service;
220 }

224 void SField::addField( const FieldNameType& fieldName, ::fwData::Object::sptr field )
225 {
226  if(!m_managerConfiguration->find("field", "id", fieldName).empty())
227  {
228  ConfigurationType conf = m_managerConfiguration->find("field", "id", fieldName).at(0);
229  const std::string fieldType = conf->getAttributeValue("type");
230 
231  OSLM_ASSERT("FieldType "<<fieldType<<" does not match ObjectType in Object "<<field->getClassname(),
232  fieldType == field->getClassname());
233  SubServicesVecType subVecSrv;
234  std::vector< ConfigurationType > services = conf->find("service");
235  for( ConfigurationType cfg : services)
236  {
237  ::fwServices::IService::sptr srv = this->add( field, cfg );
238  OSLM_ASSERT("Instantiation Service failed on field "<<fieldName, srv);
239  srv->configure();
240  SPTR(SubService) subSrv = SPTR(SubService)( new SubService());
241  subSrv->m_config = cfg;
242  subSrv->m_service = srv;
243 
244  // Standard communication management
245  if ( cfg->getAttributeValue("autoConnect") == "yes" )
246  {
247  subSrv->m_hasAutoConnection = true;
248  subSrv->m_connections.connect( field, srv, srv->getObjSrvConnections() );
249  }
250 
251 
252  std::string workerKey = cfg->getAttributeValue("worker");
253 
254  if (!workerKey.empty())
255  {
256  ::fwServices::registry::ActiveWorkers::sptr activeWorkers =
257  ::fwServices::registry::ActiveWorkers::getDefault();
258  ::fwThread::Worker::sptr worker;
259  worker = activeWorkers->getWorker(workerKey);
260  if (!worker)
261  {
262  worker = ::fwThread::Worker::New();
263  activeWorkers->addWorker(workerKey, worker);
264  }
265  srv->setWorker(worker);
266  }
267 
268  subVecSrv.push_back(subSrv);
269  subSrv->getService()->start();
270  if (m_mode =="startAndUpdate")
271  {
272  subSrv->getService()->update();
273  }
274  }
275  m_fieldsSubServices[fieldName] = subVecSrv;
276 
277  this->manageConnections(fieldName, field, conf);
278  this->manageProxies(fieldName, field, conf);
279  }
280  else
281  {
282  OSLM_DEBUG("Field "<<fieldName<<" not managed.");
283  }
284 }

298 void SField::swapField(const FieldNameType& fieldName, ::fwData::Object::sptr field)
299 {
300  std::vector< ConfigurationType > fields = m_managerConfiguration->find("field", "id", fieldName);
301  for( ConfigurationType cfg : fields)
302  {
303  SubServicesVecType subServices = m_fieldsSubServices[fieldName];
304  for( SPTR(SubService) subSrv : subServices )
305  {
306  OSLM_ASSERT("SubService on " << fieldName <<" expired !", subSrv->getService() );
307  OSLM_ASSERT( subSrv->getService()->getID() << " is not started ", subSrv->getService()->isStarted());
308 
309  OSLM_TRACE("Swapping subService " << subSrv->getService()->getID() << " on "<< fieldName );
310  if(subSrv->getService()->getObject() != field)
311  {
312  subSrv->getService()->swap(field);
313  subSrv->m_dummy.reset();
314 
315  if (subSrv->m_hasAutoConnection)
316  {
317  subSrv->m_connections.disconnect();
318  subSrv->m_connections.connect( field, subSrv->getService(),
319  subSrv->getService()->getObjSrvConnections() );
320  }
321  }
322  else
323  {
324  OSLM_WARN( subSrv->getService()->getID()
325  << "'s field already is '"
326  << subSrv->getService()->getObject()->getID()
327  << "', no need to swap");
328  }
329  }
330  this->manageConnections(fieldName, field, cfg);
331  this->manageProxies(fieldName, field, cfg);
332  }
333 }

347 void SField::removeField( const FieldNameType& fieldName )
348 {
349  if(!m_managerConfiguration->find("field", "id", fieldName).empty())
350  {
351  ConfigurationType conf = m_managerConfiguration->find("field", "id", fieldName).at(0);
352  const std::string fieldType = conf->getAttributeValue("type");
353 
354  this->removeConnections(fieldName);
355  this->disconnectProxies(fieldName);
356 
357  SubServicesVecType subServices = m_fieldsSubServices[fieldName];
358  ::fwData::Object::sptr dummyObj;
359  dummyObj = ::fwData::factory::New(fieldType);
360  for( SPTR(SubService) subSrv : subServices )
361  {
362  OSLM_ASSERT("SubService on " << fieldName <<" expired !", subSrv->getService() );
363  OSLM_ASSERT( subSrv->getService()->getID() << " is not started ", subSrv->getService()->isStarted());
364  if(m_dummyStopMode)
365  {
366  subSrv->getService()->swap(dummyObj);
367  subSrv->m_dummy = dummyObj;
368  }
369  else
370  {
371  if( subSrv->m_hasAutoConnection )
372  {
373  subSrv->m_connections.disconnect();
374  }
375 
376  subSrv->getService()->stop();
377  ::fwServices::OSR::unregisterService(subSrv->getService());
378  subSrv->m_service.reset();
379  }
380  }
381  if(!m_dummyStopMode)
382  {
383  m_fieldsSubServices.erase(fieldName);
384  }
385  else
386  {
387  this->manageConnections(fieldName, dummyObj, conf);
388  this->manageProxies(fieldName, dummyObj, conf);
389  }
390  }
391  else
392  {
393  OSLM_DEBUG("Object "<<fieldName<<" not managed.");
394  }
