119 void SImageReader::updating() throw ( ::fwTools::Failed )
120 {
121  if( this->hasLocationDefined() )
122  {
123  // Retrieve dataStruct associated with this service
124  ::fwData::Image::sptr pImage = this->getObject< ::fwData::Image >();
125  SLM_ASSERT("pImage not instanced", pImage);
126 
127  // Read new image path and update image. If the reading process is a success, we notify all listeners that image has been modified.
128 
129  ::fwGui::Cursor cursor;
130  cursor.setCursor(::fwGui::ICursor::BUSY);
131  try
132  {
133  if ( this->loadImage( this->getFile(), pImage ) )
134  {
135  this->notificationOfDBUpdate();
136  }
137  }
138  catch(::fwTools::Failed& e)
139  {
140  OSLM_TRACE("Error : " << e.what());
141  FW_RAISE_EXCEPTION(e);
142  }
143 
144  cursor.setDefaultCursor();
145  }
146 }

159 bool SImageReader::loadImage( const ::boost::filesystem::path imgFile, ::fwData::Image::sptr _pImg )
160 {
161  bool ok = true;
162 
163  std::string ext = ::boost::filesystem::extension(imgFile);
164  ::boost::algorithm::to_lower(ext);
165 
166  ::fwDataIO::reader::IObjectReader::sptr imageReader;
167  if(ext == ".vtk")
168  {
169  imageReader = configureReader< ::fwVtkIO::ImageReader >( imgFile );
170  }
171  else if(ext == ".vti")
172  {
173  imageReader = configureReader< ::fwVtkIO::VtiImageReader >( imgFile );
174  }
175  else if(ext == ".mhd")
176  {
177  imageReader = configureReader< ::fwVtkIO::MetaImageReader >( imgFile );
178  }
179  else
180  {
181  FW_RAISE_EXCEPTION(::fwTools::Failed("Only .vtk, .vti and .mhd are supported."));
182  }
183 
184  // Set the image (already created, but empty) that will be modified
185  ::fwData::mt::ObjectWriteLock lock(_pImg);
186  imageReader->setObject(_pImg);
187 
188  m_sigJobCreated->emit(imageReader->getJob());
189 
190  try
191  {
192  imageReader->read();
193  }
194  catch(::fwTools::Failed& e)
195  {
196  std::stringstream ss;
197  ss << "Warning during loading : " << e.what();
198 
199  ::fwGui::dialog::MessageDialog::showMessageDialog(
200  "Warning",
201  ss.str(),
202  ::fwGui::dialog::IMessageDialog::WARNING);
203  ok = false;
204  // Raise exception for superior level
205  FW_RAISE_EXCEPTION(e);
206  }
207  catch (const std::exception & e)
208  {
209  std::stringstream ss;
210  ss << "Warning during loading : " << e.what();
211 
212  ::fwGui::dialog::MessageDialog::showMessageDialog(
213  "Warning",
214  ss.str(),
215  ::fwGui::dialog::IMessageDialog::WARNING);
216  ok = false;
217  }
218  catch( ... )
219  {
220  ::fwGui::dialog::MessageDialog::showMessageDialog(
221  "Warning",
222  "Warning during loading.",
223  ::fwGui::dialog::IMessageDialog::WARNING);
224  ok = false;
225  }
226 
227  return ok;
228 }
