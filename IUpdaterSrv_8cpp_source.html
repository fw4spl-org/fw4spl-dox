30 void IUpdaterSrv::configureManagedEvents(::fwRuntime::ConfigurationElement::sptr configuration)
31 {
32  SLM_TRACE_FUNC();
33 
34  ::fwRuntime::ConfigurationElementContainer handleEvents = configuration->findAllConfigurationElement("update");
35 
36  SLM_ASSERT("Problem with configuration for ObjFromMsgUpdaterSrv type, missing element \"update\"",
37  handleEvents.size() != 0 );
38  OSLM_DEBUG( "handleEvents.size() = " << handleEvents.size() );
39  m_managedEvents.clear();
40  for( ::fwRuntime::ConfigurationElementContainer::Iterator item = handleEvents.begin();
41  item != handleEvents.end();
42  ++item )
43  {
44  SLM_FATAL_IF( "The attribute \"compositeKey\" is missing", !(*item)->hasAttribute("compositeKey") );
45  std::string compositeKey = (*item)->getExistingAttributeValue("compositeKey");
46 
47  SLM_FATAL_IF( "The attribute \"onEvent\" is missing", !(*item)->hasAttribute("onEvent") );
48  std::string onEvent = (*item)->getExistingAttributeValue("onEvent");
49 
50  SLM_FATAL_IF( "The attribute \"fromUID\" is missing", !(*item)->hasAttribute("fromUID") );
51  std::string fromUID = (*item)->getExistingAttributeValue("fromUID");
52 
53  SLM_FATAL_IF( "The attribute \"actionType\" is missing, the actionType must be ADD, ADD_OR_SWAP, SWAP, REMOVE,"
54  "REMOVE_IF_PRESENT or DO_NOTHING", !(*item)->hasAttribute("actionType") );
55  std::string actionType = (*item)->getExistingAttributeValue("actionType");
56 
57  ActionType action;
58  if ( actionType == "ADD" )
59  {
60  action = ADD;
61  }
62  else if ( actionType == "SWAP" )
63  {
64  action = SWAP;
65  }
66  else if ( actionType == "REMOVE" )
67  {
68  action = REMOVE;
69  }
70  else if ( actionType == "ADD_OR_SWAP" )
71  {
72  action = ADD_OR_SWAP;
73  }
74  else if ( actionType == "REMOVE_IF_PRESENT" )
75  {
76  action = REMOVE_IF_PRESENT;
77  }
78  else if ( actionType == "DO_NOTHING" )
79  {
80  action = DO_NOTHING;
81  }
82  else
83  {
84  SLM_FATAL("This type of \"actionType\" is not managed by ObjFromMsgUpdaterSrv type, The actionType"
85  "should be ADD, ADD_OR_SWAP, SWAP, REMOVE, REMOVE_IF_PRESENT and DO_NOTHING");
86  }
87 
88  OSLM_INFO(
89  "Manage event "<< onEvent <<" from this object "<< fromUID <<" and "<< actionType << " "<< compositeKey <<
90  " in my composite.");
91  ::boost::tuple< std::string, std::string, std::string, ActionType > managedEvent (onEvent, fromUID,
92  compositeKey, action);
93  m_managedEvents.push_back( managedEvent );
94  //addNewHandledEvent( onEvent );
95  }

100 void IUpdaterSrv::updateComposite(::fwData::Composite::sptr pComposite, ::fwData::Object::sptr _obj,
101  std::string _compositeKey, ActionType _action )
102 {
103 
104  // Manage special action
105  if ( _action == ADD_OR_SWAP )
106  {
107  if ( pComposite->find(_compositeKey) != pComposite->end() )
108  {
109  _action = SWAP;
110  }
111  else
112  {
113  _action = ADD;
114  }
115  }
116  else if(_action == REMOVE_IF_PRESENT )
117  {
118  if ( pComposite->find(_compositeKey) != pComposite->end() )
119  {
120  _action = REMOVE;
121  }
122  else
123  {
124  _action = DO_NOTHING;
125  }
126  }
127 
128  // Use helper on composite
129  std::shared_ptr< ::fwComEd::helper::Composite > pCompositeHelper ( new ::fwComEd::helper::Composite( pComposite ) );
130 
131  if(_action != DO_NOTHING)
132  {
133  switch ( _action )
134  {
135  case REMOVE:
136  {
137  pCompositeHelper->remove(_compositeKey);
138  break;
139  }
140  case SWAP:
141  {
142  pCompositeHelper->swap(_compositeKey,_obj);
143  break;
144  }
145  case ADD:
146  {
147  pCompositeHelper->add(_compositeKey,_obj);
148  break;
149  }
150  default:
151  {
152  SLM_FATAL("This action type is not managed");
153  break;
154  }
155  }
156 
157  // Notification of message
158  pCompositeHelper->notify();
159  }
160  else
161  {
162  OSLM_INFO("Do nothing for objectID " << _compositeKey);
163  }
164 }
