00049 /*
00050  * @brief  Open a file in the zip archive for writing
00051  * @note Z_BEST_SPEED compression level for '.raw' files,
00052  *       Z_NO_COMPRESSION for 'raw.gz', Z_DEFAULT_COMPRESSION otherwise.
00053  */
00054 std::streamsize openFile(zipFile zipDescriptor, const ::boost::filesystem::path &path)
00055 {
00056     const std::string extension = path.extension().string();
00057     int compressLevel = Z_DEFAULT_COMPRESSION;
00058     if(extension == ".raw")
00059     {
00060         compressLevel = Z_BEST_SPEED;
00061     }
00062     else if(extension == ".raw.gz")
00063     {
00064         compressLevel = Z_NO_COMPRESSION;
00065     }
00066     zip_fileinfo zfi;
00067     zfi.internal_fa = 0;
00068     zfi.external_fa = 0;// @todo FIXME
00069 
00070     ::boost::posix_time::ptime now = boost::posix_time::second_clock::local_time();
00071     std::tm ptm = ::boost::posix_time::to_tm(now);
00072     zfi.dosDate = 0;
00073     zfi.tmz_date.tm_year = ptm.tm_year;
00074     zfi.tmz_date.tm_mon  = ptm.tm_mon;
00075     zfi.tmz_date.tm_mday = ptm.tm_mday;
00076     zfi.tmz_date.tm_hour = ptm.tm_hour;
00077     zfi.tmz_date.tm_min  = ptm.tm_min;
00078     zfi.tmz_date.tm_sec  = ptm.tm_sec;
00079 
00080     std::streamsize nRet = zipOpenNewFileInZip(zipDescriptor,
00081             path.generic_string().c_str(),
00082             &zfi,
00083             NULL,
00084             0,
00085             NULL,
00086             0,
00087             NULL,
00088             Z_DEFLATED,
00089             compressLevel);

00091     return nRet;
