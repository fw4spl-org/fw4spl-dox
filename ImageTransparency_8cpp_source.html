50 void ImageTransparency::starting() throw(::fwTools::Failed)
51 {
52  SLM_TRACE_FUNC();
53  this->::fwGui::IGuiContainerSrv::create();
54  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
55 
56  ::fwGuiQt::container::QtContainer::sptr qtContainer = ::fwGuiQt::container::QtContainer::dynamicCast(
57  this->getContainer() );
58  QWidget* const container = qtContainer->getQtContainer();
59  SLM_ASSERT("container not instanced", container);
60 
61  QHBoxLayout* hLayout = new QHBoxLayout();
62 
63  QLabel* staticText = new QLabel( QObject::tr("Transparency: "), container);
64  hLayout->addWidget( staticText, 0, Qt::AlignVCenter );
65 
66  m_valueSlider = new QSlider( Qt::Horizontal, container );
67  hLayout->addWidget( m_valueSlider, 1, Qt::AlignVCenter );
68  m_valueSlider->setRange(0, 100);
69  m_valueSlider->setMinimumWidth(100);
70 
71  m_valueCheckBox = new QCheckBox( QObject::tr("visible"), container);
72  m_action = new QAction(container);
73  m_action->setCheckable(true);
74  if (!m_shortcut.empty())
75  {
76  m_action->setShortcut(QKeySequence(QString::fromStdString(m_shortcut)));
77  }
78  m_valueCheckBox->addAction(m_action);
79  hLayout->addWidget( m_valueCheckBox, 0, Qt::AlignVCenter );
80 
81  container->setLayout( hLayout );
82 
83  QObject::connect(m_valueSlider, SIGNAL(valueChanged(int)), this, SLOT(onModifyTransparency(int)));
84  QObject::connect(m_valueCheckBox, SIGNAL(stateChanged(int)), this, SLOT(onModifyVisibility(int)));
85  QObject::connect(m_action, SIGNAL(triggered(bool)), this, SLOT(onModifyVisibility(bool)));
86 
87  this->updating();
88 }

122 void ImageTransparency::updating() throw(::fwTools::Failed)
123 {
124  ::fwData::Image::sptr img = this->getObject< ::fwData::Image >();
125 
126  bool imageIsValid = ::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity( img );
127  m_valueSlider->setEnabled(imageIsValid);
128  m_valueCheckBox->setEnabled(imageIsValid);
129  if (imageIsValid)
130  {
131  QObject::disconnect(m_valueSlider, SIGNAL(valueChanged(int)), this, SLOT(onModifyTransparency(int)));
132  QObject::disconnect(m_valueCheckBox, SIGNAL(stateChanged(int)), this, SLOT(onModifyVisibility(int)));
133  QObject::disconnect(m_action, SIGNAL(triggered(bool)), this, SLOT(onModifyVisibility(bool)));
134 
135  if(img->getField( "TRANSPARENCY" ) )
136  {
137  ::fwData::Integer::sptr transparency = img->getField< ::fwData::Integer >( "TRANSPARENCY" );
138  m_valueSlider->setValue( *transparency );
139  }
140  else
141  {
142  img->setField( "TRANSPARENCY", ::fwData::Integer::New(0) );
143  m_valueSlider->setValue( 0 );
144  }
145  if(img->getField( "VISIBILITY" ) )
146  {
147  ::fwData::Boolean::sptr visible = img->getField< ::fwData::Boolean >( "VISIBILITY" );
148  m_valueCheckBox->setChecked( *visible );
149  m_action->setChecked(*visible);
150  }
151  else
152  {
153  img->setField( "VISIBILITY", ::fwData::Boolean::New(true) );
154  m_valueCheckBox->setChecked( true );
155  m_action->setChecked(true);
156  }
157  QObject::connect(m_valueSlider, SIGNAL(valueChanged(int)), this, SLOT(onModifyTransparency(int)));
158  QObject::connect(m_valueCheckBox, SIGNAL(stateChanged(int)), this, SLOT(onModifyVisibility(int)));
159  QObject::connect(m_action, SIGNAL(triggered(bool)), this, SLOT(onModifyVisibility(bool)));
160  }
161 }

179 void ImageTransparency::onModifyTransparency(int value)
180 {
181  SLM_TRACE_FUNC();
182  ::fwData::Image::sptr img = this->getObject< ::fwData::Image >();
183  img->setField( "TRANSPARENCY", ::fwData::Integer::New(value) );
184 
185  auto sig = img->signal< ::fwData::Image::TransparencyModifiedSignalType >(
186  ::fwData::Image::s_TRANSPARENCY_MODIFIED_SIG);
187  {
188  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
189  sig->asyncEmit();
190  }
