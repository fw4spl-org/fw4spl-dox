22 Pool::Pool(size_t _threads) : m_stop(false)
23 {
24  const auto avalaibleCores = std::thread::hardware_concurrency();
25  OSLM_WARN_IF( _threads << " threads were allocated in this thread pool, but you only have " <<
26  avalaibleCores << " physical cores on this CPU",
27  _threads > std::thread::hardware_concurrency());
28 
29  for(size_t i = 0; i<_threads; ++i)
30  {
31  m_workers.emplace_back(
32  [this]
33  {
34  for(;; )
35  {
36  std::function<void()> task;
37 
38  {
39  std::unique_lock<std::mutex> lock(m_queueMutex);
40  m_condition.wait(lock, [this] { return m_stop || !m_tasks.empty(); });
41  if(m_stop && m_tasks.empty())
42  {
43  return;
44  }
45  task = std::move(m_tasks.front());
46  m_tasks.pop();
47  }
48 
49  task();
50  }
51  }
52  );
53  }
54 }
