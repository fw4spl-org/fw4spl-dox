00066 void CameraOrientationAction::configuring() throw(fwTools::Failed)
00067 {
00068     SLM_TRACE_FUNC();
00069     ::fwGui::IActionSrv::initialize();
00070 
00071     if( this->m_configuration->size() > 0 )
00072     {
00073         ::fwRuntime::ConfigurationElementContainer::Iterator iter = this->m_configuration->begin() ;
00074         SLM_ASSERT("Sorry, only one xml element \"orientation\" is accepted.", this->m_configuration->size() == 1 && (*iter)->getName() == "orientation" );
00075         SLM_ASSERT("Sorry, xml element \"orientation\" is empty.", ! (*iter)->getValue().empty() );
00076         m_orientation = (*iter)->getValue();
00077         ::boost::algorithm::trim(m_orientation);
00078         ::boost::algorithm::to_lower(m_orientation);
00079 
00080         if(m_orientation != "axial" && m_orientation != "frontal" && m_orientation != "sagittal")
00081         {
00082             SLM_FATAL("The value for the xml element \"orientation\" can only be axial, frontal or sagittal.");
00083         }
00084     }
00085 
00086 }
