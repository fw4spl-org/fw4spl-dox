260 IExecutable * Runtime::createExecutableInstance( const std::string & type ) throw( RuntimeException )
261 {
262  std::shared_ptr< ExecutableFactory > factory;
263 
264  // Retrieves the executable factory.
265  factory = this->findExecutableFactory( type );
266  if( factory == 0 )
267  {
268  throw RuntimeException( type + ": no executable factory found for that type." );
269  }
270 
271  // Creates the executable instance

274 
275  // Job's done.

280 
281 IExecutable * Runtime::createExecutableInstance( const std::string & type,
282  ConfigurationElement::sptr configurationElement ) throw(
283  RuntimeException )
284 {
285  std::shared_ptr< ExecutableFactory > factory;
286 
287  // Retrieves the executable factory.
288  factory = this->findExecutableFactory( type );
289 
290  // If there is no factory has been found, it is possible that
291  // it has not been registered since the bundle of the given configuration element
292  // is not started.
293  // So we start that bundle and look for the executable factory one more type.
294  if( factory == 0)
295  {
296  configurationElement->getBundle()->start();
297  factory = this->findExecutableFactory( type );
298  }
299 
300  // If we still have not found any executable factory, then notify the problem.
301  if( factory == 0 )
302  {
303  throw RuntimeException( type + ": no executable factory found for that type." );
304  }
305 
306  // Creates the executable instance
307  IExecutable* result( 0 );
308  try
309  {
310  factory->getBundle()->start();
311  result = factory->createExecutable();
312 
313  result->setBundle( factory->getBundle() );
314  result->setInitializationData( configurationElement );
315  }
316  catch( const std::exception & e )
317  {
318  std::string message( "Unable to create an executable instance. " );
319  throw RuntimeException( message + e.what() );
320  }
321  // Job's done.
322  return result;
323 }
