101 ::fwGui::dialog::IMessageDialog::Buttons MessageDialog::show()
102 {
103  MessageDialogQtIconsType::const_iterator iterIcon = messageDialogQtIcons.find(m_icon);
104  SLM_ASSERT("Unknown Icon", iterIcon != messageDialogQtIcons.end());
105 
106  QMessageBox::Icon icon = iterIcon->second;
107  QString title = QString::fromStdString(m_title);
108  QString text = QString::fromStdString(m_message);
109  QMessageBox::StandardButtons buttons = QMessageBox::NoButton;
110 
111  for(MessageDialogQtButtonType::value_type button : messageDialogQtButton)
112  {
113  if ( m_buttons & button.first)
114  {
115  buttons |= button.second;
116  }
117  }
118 
119  QMessageBox box(icon, title, text, buttons, qApp->activeWindow());
120 
121  for(int i = 0; i < m_customButtons.size(); ++i)
122  {
123  box.addButton(m_customButtons[i], QMessageBox::ActionRole);
124  }
125 
126  for(int i = 0; i < m_customButtons.size(); ++i)
127  {
128  box.addButton(m_customButtons[i], QMessageBox::ActionRole);
129  }
130 
131  MessageDialogQtButtonType::const_iterator iter = messageDialogQtButton.find(m_defaultButton);
132  if(iter != messageDialogQtButton.end())
133  {
134  box.setDefaultButton(QMessageBox::StandardButton(static_cast<int>(iter->second)));
135  }
136 
137  ::fwGui::dialog::IMessageDialog::Buttons result = ::fwGui::dialog::IMessageDialog::NOBUTTON;
138 
139  box.exec();
140 
141  for(MessageDialogQtButtonType::value_type button : messageDialogQtButton)
142  {
143  if ( box.standardButton( box.clickedButton() ) == button.second)
144  {
145  result = button.first;
146  break;
147  }
148  }
149  return result;
150 }
