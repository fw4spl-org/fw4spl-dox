43 void AppConfig::parseBundleInformation()
44 {
45  auto extensions = ::fwRuntime::getAllExtensionsForPoint("::fwServices::registry::AppConfig");
46  for( const auto& ext : extensions )
47  {
48  // Get id
49  std::string configId = ext->findConfigurationElement("id")->getValue();
50 
51  // Get group
52  std::string group = "";
53  if ( ext->hasConfigurationElement("group") )
54  {
55  group = ext->findConfigurationElement("group")->getValue();
56  }
57  // Get desc
58  std::string desc = "No description available";
59  if ( ext->hasConfigurationElement("desc") )
60  {
61  desc = ext->findConfigurationElement("desc")->getValue();
62  }
63 
64  AppInfo::ParametersType parameters;
65  if ( ext->hasConfigurationElement("parameters") )
66  {
67  ::fwRuntime::ConfigurationElement::csptr parametersConfig = ext->findConfigurationElement("parameters");
68  ::fwRuntime::ConfigurationElement::Container elements = parametersConfig->getElements();
69  for( ::fwRuntime::ConfigurationElement::sptr paramConfig : elements )
70  {
71  std::string name = paramConfig->getExistingAttributeValue("name");
72 
73  if(paramConfig->hasAttribute("default"))
74  {
75  parameters[name] = paramConfig->getAttributeValue("default");
76  }
77  else
78  {
79  parameters[name] = s_mandatoryParameterIdentifier;
80  }
81 
82  }
83  }
84 
85  // Get config
86  ::fwRuntime::ConfigurationElement::csptr config = *(ext->findConfigurationElement("config")->begin());
87 
88  // Get bundle
89  std::shared_ptr< ::fwRuntime::Bundle> bundle = ext->getBundle();
90  std::string bundleId = bundle->getIdentifier();
91  std::string bundleVersion = bundle->getVersion().string();
92 
93  // Add app info
94  this->addAppInfo( configId, group, desc, parameters, config, bundleId, bundleVersion );
95  }
96 }

143 ::fwRuntime::ConfigurationElement::csptr AppConfig::getAdaptedTemplateConfig(
144  const std::string & configId,
145  const FieldAdaptorType fieldAdaptors ) const
146 {
147  ::fwCore::mt::ReadLock lock(m_registryMutex);
148  // Get config template
149  Registry::const_iterator iter = m_reg.find( configId );
150  SLM_ASSERT("The id " << configId << " is not found in the application configuration registry",
151  iter != m_reg.end());
152 
153  // Adapt config
154  ::fwRuntime::ConfigurationElement::sptr newConfig;
155 
156  FieldAdaptorType fields;
157  AppInfo::ParametersType parameters = iter->second->parameters;
158  for( AppInfo::ParametersType::value_type param : parameters )
159  {
160  FieldAdaptorType::const_iterator iter = fieldAdaptors.find( param.first );
161  std::stringstream key;
162  key << "\\$\\{" << param.first << "\\}";
163  if ( iter != fieldAdaptors.end() )
164  {
165  fields[key.str()] = iter->second;
166  }
167  else if ( param.second != s_mandatoryParameterIdentifier)
168  {
169  fields[key.str()] = param.second;
170  }
171  else
172  {
173  FW_RAISE("Parameter : '" << param.first << "' is needed by the app configuration id='"<< configId <<"'.");
174  }
175  }
176  newConfig = this->adaptConfig( iter->second->config, fields );
177 
178  return newConfig;
179 }

256 ::fwRuntime::EConfigurationElement::sptr AppConfig::adaptConfig( ::fwRuntime::ConfigurationElement::csptr _cfgElem,
257  const FieldAdaptorType & fieldAdaptors ) const
258 {
259  SLM_TRACE_FUNC();
260 
261  ::fwRuntime::EConfigurationElement::sptr result = ::fwRuntime::EConfigurationElement::New( _cfgElem->getName() );
262  result->setValue( this->adaptField( _cfgElem->getValue(), fieldAdaptors ) );
263 
264  typedef std::map<std::string, std::string> MapAttributesType;
265  for( MapAttributesType::value_type attribute : _cfgElem->getAttributes() )
266  {
267  result->setAttributeValue( attribute.first, this->adaptField( attribute.second, fieldAdaptors ) );
268  }
269 
270  for ( ::fwRuntime::ConfigurationElement::csptr subElem : _cfgElem->getElements())
271  {
272  result->addConfigurationElement( this->adaptConfig( subElem, fieldAdaptors ) );
273  }
274 
275  return result;
276 }

280 std::string AppConfig::adaptField( const std::string & _str, const FieldAdaptorType & fieldAdaptors ) const
281 {
282  std::string newStr = _str;
283  if(!_str.empty())
284  {
285  for(FieldAdaptorType::value_type fieldAdaptor : fieldAdaptors)
286  {
287  std::stringstream sstr;
288  sstr << "(.*)" << fieldAdaptor.first << "(.*)";
289  ::boost::regex machine_regex ( sstr.str() );
290  if ( ::boost::regex_match( _str, machine_regex ) )
291  {
292  std::stringstream machine_format;
293  machine_format << "\\1" << fieldAdaptor.second << "\\2";
294  newStr = ::boost::regex_replace( newStr,
295  machine_regex, machine_format.str(),
296  ::boost::match_default | ::boost::format_sed );
297  }
298  }
299  }
300  return newStr;
301 }
