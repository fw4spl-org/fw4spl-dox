54 ConfigurationElement::sptr findConfigurationElement( const std::string & identifier,
55  const std::string & pointIdentifier )
56 {
57  typedef std::vector< ConfigurationElement::sptr > ElementContainer;
58  ConfigurationElement::sptr resultConfig;
59  ElementContainer elements =
60  getAllConfigurationElementsForPoint< ElementContainer >( pointIdentifier );
61  ElementContainer::iterator foundElement = ::std::find_if( elements.begin(),
62  elements.end(),
63  ConfigurationElementIdentifierPredicate(identifier) );
