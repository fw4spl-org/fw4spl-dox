00124 void HistogramCursor::doUpdate() throw( ::fwTools::Failed)
00125 {
00126     SLM_TRACE_FUNC();
00127 
00128     ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
00129     ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
00130     const float histogramMinValue = histogram->getMinValue();
00131     const float histogramBinsWidth = histogram->getBinsWidth();
00132 
00133     // Event coordinates in scene
00134     ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( m_coord );
00135 
00136     int histIndex = (int) sceneCoord.getX();
00137     int index = (histIndex - histogramMinValue) / histogramBinsWidth;
00138 
00139     if(index >= 0 && index < (int)values.size()) // avoid std out_of_range on MS Windows
00140     {
00141         ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
00142         const double viewportHeight = viewport->getHeight();
00143         const double viewportWidth = viewport->getWidth();
00144 
00145         const double viewportSizeRatio = viewportHeight / viewportWidth;
00146         const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
00147 
00148         const Scene2DRatio ratio = this->getRatio();  // Total ratio
00149         const double viewportRatio = this->getViewportSizeRatio().first;
00150 
00151         double diameterH  = m_pointSize;
00152         double diameterV  = m_pointSize * viewportSizeRatio;
00153 
00154         diameterV /= viewportRatio;
00155         diameterV *= viewInitialSizeRatio;
00156 
00157         // Apply the ratio of the scene 2D in order to keep the same size for the circles if viewport's size or
00158         // view's size change:
00159         diameterH *= ratio.first;
00160         diameterV *= ratio.second;
00161 
00162         ::fwData::Point::sptr point = ::fwData::Point::dynamicCast( ::fwTools::fwID::getObject( m_histogramPointUID ) );
00163 
00164         double x = point->getRefCoord()[0] - diameterH / 2;
00165         double y = point->getRefCoord()[1] - diameterV / 2;
00166 
00167         m_index->setRect( x, y, diameterH, diameterV );
00168     }
00169 }

00174 {
00175     ::fwComEd::HistogramMsg::csptr histoMsg = ::fwComEd::HistogramMsg::dynamicConstCast(_msg);
