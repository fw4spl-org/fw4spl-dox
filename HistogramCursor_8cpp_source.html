117 void HistogramCursor::doUpdate() throw( ::fwTools::Failed)
118 {
119  this->initializeViewSize();
120  this->initializeViewportSize();
121 
122  ::fwData::Histogram::sptr histogram = this->getObject< ::fwData::Histogram>();
123  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();
124  const float histogramMinValue = histogram->getMinValue();
125  const float histogramBinsWidth = histogram->getBinsWidth();
126 
127  // Event coordinates in scene
128  ::scene2D::data::Coord sceneCoord = this->getScene2DRender()->mapToScene( m_coord );
129 
130  int histIndex = (int) sceneCoord.getX();
131  int index = (histIndex - histogramMinValue) / histogramBinsWidth;
132 
133  if(index >= 0 && index < (int)values.size()) // avoid std out_of_range on MS Windows
134  {
135  ::scene2D::data::Viewport::sptr viewport = this->getScene2DRender()->getViewport();
136  const double viewportHeight = viewport->getHeight();
137  const double viewportWidth = viewport->getWidth();
138 
139  const double viewportSizeRatio = viewportHeight / viewportWidth;
140  const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
141 
142  const Scene2DRatio ratio = this->getRatio(); // Total ratio
143  const double viewportRatio = this->getViewportSizeRatio().first;
144 
145  double diameterH = m_pointSize;
146  double diameterV = m_pointSize * viewportSizeRatio;
147 
148  diameterV /= viewportRatio;
149  diameterV *= viewInitialSizeRatio;
150 
151  // Apply the ratio of the scene 2D in order to keep the same size for the circles if viewport's size or
152  // view's size change:
153  diameterH *= ratio.first;
154  diameterV *= ratio.second;
155 
156  ::fwData::Point::sptr point = ::fwData::Point::dynamicCast( ::fwTools::fwID::getObject( m_histogramPointUID ) );
157 
158  double x = point->getRefCoord()[0] - diameterH / 2;
159  double y = point->getRefCoord()[1] - diameterV / 2;
160 
161  m_index->setRect( x, y, diameterH, diameterV );
162  }
163 }
