73  template<class PIXELTYPE>
74  void operator()( const Parameter& param )
75  {
76  OSLM_DEBUG( "itk::ImageSeriesWriter with PIXELTYPE "<< fwTools::DynamicType::string<PIXELTYPE>() );
77 
78  ::fwData::Image::sptr image = param.m_dataImage;
79 
80  // VAG attention : ImageFileReader ne notifie AUCUNE progressEvent mais son ImageIO oui!!!! mais ImageFileReader ne permet pas de l'atteindre
81  // car soit mis a la mano ou alors construit lors de l'Update donc trop tard
82  // Il faut dont creer une ImageIO a la mano (*1*): affecter l'observation sur IO (*2*) et mettre le IO dans le reader (voir *3*)
83 
84  // Reader IO (*1*)
85  typename itk::ImageIOBase::Pointer imageIOWrite = itk::ImageIOFactory::CreateImageIO( "image.jpg",
86  itk::ImageIOFactory::WriteMode);
87  assert( imageIOWrite.IsNotNull() );
88 
89  // create writer
90  typedef itk::Image< PIXELTYPE, 3> itkImageType;
91  typedef itk::Image< unsigned char, 2 > Image2DType;
92  typedef typename itk::ImageSeriesWriter< itkImageType, Image2DType > WriterType;
93  typename WriterType::Pointer writer = WriterType::New();
94 
95  // set observation (*2*)
96  itk::LightProcessObject::Pointer castHelper = (itk::LightProcessObject*)(imageIOWrite.GetPointer());
97  assert( castHelper.IsNotNull() );
98  Progressor progress(castHelper, param.m_fwWriter, param.m_filename);
99 
100  // create itk Image
101  typename itkImageType::Pointer itkImage = ::fwItkIO::itkImageFactory<itkImageType>( image );
102 
103  typedef ::itk::IntensityWindowingImageFilter< itkImageType, itkImageType > RescaleFilterType;
104  typename RescaleFilterType::Pointer rescaleFilter = RescaleFilterType::New();
105 
106  double min, max;
107  ::fwData::Composite::sptr poolTF;
108  poolTF =
109  image->getField< ::fwData::Composite>( ::fwDataTools::fieldHelper::Image::m_transferFunctionCompositeId );
110  if(poolTF)
111  {
112  ::fwData::Composite::iterator iter = poolTF->find(::fwData::TransferFunction::s_DEFAULT_TF_NAME);
113  if(iter != poolTF->end())
114  {
115  ::fwData::TransferFunction::sptr tf;
116  tf = ::fwData::TransferFunction::dynamicCast(iter->second);
117  min = tf->getWLMinMax().first;
118  max = tf->getWLMinMax().second;
119  }
120  }
121  else
122  {
123  ::fwDataTools::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
124  }
125 
126  rescaleFilter->SetWindowMinimum( min );
127  rescaleFilter->SetWindowMaximum( max );
128  rescaleFilter->SetOutputMinimum( 0 );
129  rescaleFilter->SetOutputMaximum( 255 );
130  rescaleFilter->InPlaceOff();
131  rescaleFilter->SetInput( itkImage );
132  rescaleFilter->Update();
133 
134  writer->SetInput( rescaleFilter->GetOutput() );
135 
136  typedef itk::NumericSeriesFileNames NameGeneratorType;
137 
138  NameGeneratorType::Pointer nameGenerator = NameGeneratorType::New();
139 
140  std::string format = param.m_filename;
141  format += "/%04d.jpg";
142  nameGenerator->SetSeriesFormat( format.c_str() );
143  nameGenerator->SetStartIndex( 1 );
144  nameGenerator->SetEndIndex( image->getSize()[2] );
145  nameGenerator->SetIncrementIndex( 1 );
146 
147  writer->SetFileNames( nameGenerator->GetFileNames() );
148 
149  writer->SetImageIO( imageIOWrite );
150 
151  // save image;
152  writer->Update();
153  }
