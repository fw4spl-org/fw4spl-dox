00108         poolTF = image->getField< ::fwData::Composite>( ::fwComEd::Dictionary::m_transferFunctionCompositeId );
00109         if(poolTF)
00110         {
00111             ::fwData::Composite::iterator iter = poolTF->find(::fwData::TransferFunction::s_DEFAULT_TF_NAME);
00112             if(iter != poolTF->end())
00113             {
00114                 ::fwData::TransferFunction::sptr tf;
00115                 tf = ::fwData::TransferFunction::dynamicCast(iter->second);
00116                 min = tf->getWLMinMax().first;
00117                 max = tf->getWLMinMax().second;
00118             }
00119         }
00120         else
00121         {
00122             ::fwComEd::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
00123         }
00124

00126         rescaleFilter->SetWindowMaximum( max );
00127         rescaleFilter->SetOutputMinimum( 0 );
00128         rescaleFilter->SetOutputMaximum( 255 );
