96 void MedicalImageAdaptor::getCurrentSliceCenter(double center[3])
97 {
98  ::fwData::Image::sptr image = this->getImage();
99  double imageSize[3];
100  this->getImageSize(imageSize);
101  double origin[3];
102  this->getImageOrigin(origin);
103 
104  ::fwData::Integer::sptr sliceIndex[3];
105  this->getSliceIndex(sliceIndex);
106  double index[3] = {
107  static_cast<double>(sliceIndex[0]->value()),
108  static_cast<double>(sliceIndex[1]->value()),
109  static_cast<double>(sliceIndex[2]->value())
110  };
111 
112  center[0] = origin[0] + (imageSize[0]-1.)/ 2.;
113  center[1] = origin[1] + (imageSize[1]-1.)/ 2.;
114  center[2] = origin[2] + (imageSize[2]-1.)/ 2.;
115 
116  double spacing[3];
117  this->getImageSpacing(spacing);
118  center[m_orientation] = origin[m_orientation] + index[m_orientation]*spacing[m_orientation];
119 }

143 void MedicalImageAdaptor::getPlane( double points[4][3], int sliceNumber)
144 {
145  ::fwData::Image::sptr image = this->getImage();
146  double extent[6];
147  for (char i = 0; i<3; ++i )
148  {
149  extent[2*i] = 0;
150  extent[2*i+1] = image->getSize()[i]*image->getSpacing()[i];
151  }
152  extent[2*m_orientation] = sliceNumber*image->getSpacing()[m_orientation];
153  extent[2*m_orientation+1] = sliceNumber*image->getSpacing()[m_orientation];
154 
155  const int* extentIndex = indexSet[ m_orientation ];
156  for (int p = 0; p<4; ++p)
157  {
158  for (int i = 0; i<3; ++i)
159  {
160  points[p][i] = extent[ *(extentIndex++) ];
161  }
162  }
163 }

181 void MedicalImageAdaptor::worldToSliceIndex(const double world[3], int index[3] )
182 {
183  double spacing[3];
184  this->getImageSpacing(spacing);
185  double origin[3];
186  this->getImageOrigin(origin);
187  for ( int i = 0; i<3; ++i )
188  {
189  // nearest integer
190  index[i] =
191  static_cast<int>( ( (world[i]-origin[i])/spacing[i] ) +
192  ( ( (world[i]-origin[i])/spacing[i] ) >= 0 ? 0.5 : -0.5 ) );
193  }

267 void MedicalImageAdaptor::updateTransferFunction( ::fwData::Image::sptr image )
268 {
269  if ( !m_tfSelectionFwID.empty() )
270  {
271  ::fwData::Composite::sptr tfSelection = m_tfSelection.lock();
272 
273  OSLM_ASSERT( "The object with the fwID '" + m_tfSelectionFwID + "' doesn't exist.", tfSelection );
274  OSLM_ASSERT( "The selectedTFKey must be defined, check your configuration.", !m_selectedTFKey.empty() );
275  if ( tfSelection->find( m_selectedTFKey ) == tfSelection->end() )
276  {
277  ::fwData::TransferFunction::sptr tfGreyLevel = ::fwData::TransferFunction::createDefaultTF();
278  if (image->getWindowWidth() != 0 )
279  {
280  tfGreyLevel->setWindow( image->getWindowWidth() );
281  tfGreyLevel->setLevel( image->getWindowCenter() );
282  }
283  else if(::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
284  {
285  double min, max;
286  ::fwDataTools::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
287  ::fwData::TransferFunction::TFValuePairType wlMinMax(min, max);
288  tfGreyLevel->setWLMinMax(wlMinMax);
289  }
290 
291  ::fwDataTools::helper::Composite compositeHelper(tfSelection);
292  compositeHelper.add(m_selectedTFKey, tfGreyLevel);
293  compositeHelper.notify();
294  }
295  m_tfSelection = tfSelection;
296  }
297  else
298  {
299  const std::string poolFieldName = ::fwDataTools::fieldHelper::Image::m_transferFunctionCompositeId;
300  ::fwData::Composite::sptr tfSelection = image->getField< ::fwData::Composite >(poolFieldName);
301 
302  if ( m_tfSelection.expired() || m_tfSelection.lock() != tfSelection )
303  {
304  const std::string defaultTFName = ::fwData::TransferFunction::s_DEFAULT_TF_NAME;
305 
306  ::fwDataTools::helper::Image helper(image);
307  helper.createTransferFunctionPool(); // do nothing if image tf pool already exist
308 
309  tfSelection = image->getField< ::fwData::Composite >(poolFieldName);
310 
311  m_selectedTFKey = defaultTFName;
312  m_tfSelection = tfSelection;
313  }
314  }
315 }

429 void MedicalImageAdaptor::installTFConnections()
430 {
431  ::fwData::Composite::sptr tfComposite = this->getTransferFunctionSelection();
432  SLM_ASSERT( "Missing transfer function selection composite",tfComposite);
433 
434  ::fwCom::Connection connection;
435  connection = tfComposite->signal(::fwData::Composite::s_ADDED_OBJECTS_SIG)->connect(m_slotAddedObjects);
436  m_tfSelectionConnections.addConnection(connection);
437  connection = tfComposite->signal(::fwData::Composite::s_CHANGED_OBJECTS_SIG)->connect(m_slotChangedObjects);
438  m_tfSelectionConnections.addConnection(connection);
439  connection = tfComposite->signal(::fwData::Composite::s_REMOVED_OBJECTS_SIG)->connect(m_slotRemovedObjects);
440  m_tfSelectionConnections.addConnection(connection);
441 
442  ::fwData::TransferFunction::sptr tf = this->getTransferFunction();
443 
444  connection = tf->signal(::fwData::TransferFunction::s_POINTS_MODIFIED_SIG)->connect(m_slotUpdateTFPoints);
445  m_tfConnections.addConnection(connection);
446  connection = tf->signal(::fwData::TransferFunction::s_WINDOWING_MODIFIED_SIG)->connect(m_slotUpdateTFWindowing);
447  m_tfConnections.addConnection(connection);
