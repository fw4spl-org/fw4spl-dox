99 void MedicalImageAdaptor::getCurrentSliceCenter(double center[3])
100 {
101  ::fwData::Image::sptr image = this->getImage();
102  double imageSize[3];
103  this->getImageSize(imageSize);
104  double origin[3];
105  this->getImageOrigin(origin);
106 
107  ::fwData::Integer::sptr sliceIndex[3];
108  this->getSliceIndex(sliceIndex);
109  double index[3] = {
110  static_cast<double>(sliceIndex[0]->value()),
111  static_cast<double>(sliceIndex[1]->value()),
112  static_cast<double>(sliceIndex[2]->value())
113  };
114 
115  center[0] = origin[0] + (imageSize[0]-1.)/ 2.;
116  center[1] = origin[1] + (imageSize[1]-1.)/ 2.;
117  center[2] = origin[2] + (imageSize[2]-1.)/ 2.;
118 
119  double spacing[3];
120  this->getImageSpacing(spacing);
121  center[m_orientation] = origin[m_orientation] + index[m_orientation]*spacing[m_orientation];
122 }

146 void MedicalImageAdaptor::getPlane( double points[4][3], int sliceNumber)
147 {
148  ::fwData::Image::sptr image = this->getImage();
149  double extent[6];
150  for (char i = 0; i<3; ++i )
151  {
152  extent[2*i] = 0;
153  extent[2*i+1] = image->getSize()[i]*image->getSpacing()[i];
154  }
155  extent[2*m_orientation] = sliceNumber*image->getSpacing()[m_orientation];
156  extent[2*m_orientation+1] = sliceNumber*image->getSpacing()[m_orientation];
157 
158  const int *extentIndex = indexSet[ m_orientation ];
159  for (int p = 0; p<4; ++p)
160  {
161  for (int i = 0; i<3; ++i)
162  {
163  points[p][i] = extent[ *(extentIndex++) ];
164  }
165  }

184 void MedicalImageAdaptor::worldToSliceIndex(const double world[3], int index[3] )
185 {
186  double spacing[3];
187  this->getImageSpacing(spacing);
188  double origin[3];
189  this->getImageOrigin(origin);
190  for ( int i = 0; i<3; ++i )
191  {
192  // nearest integer
193  index[i] =
194  static_cast<int>( ( (world[i]-origin[i])/spacing[i] ) +
195  ( ( (world[i]-origin[i])/spacing[i] ) >= 0 ? 0.5 : -0.5 ) );
196  }

267 void MedicalImageAdaptor::updateTransferFunction( ::fwData::Image::sptr image )
268 {
269  if ( !m_tfSelectionFwID.empty() )
270  {
271  if ( m_tfSelection.expired() )
272  {
273  ::fwData::Composite::sptr tfSelection =
274  ::fwData::Composite::dynamicCast( ::fwTools::fwID::getObject( m_tfSelectionFwID ) );
275  OSLM_ASSERT( "The object with the fwID '" + m_tfSelectionFwID + "' doesn't exist.", tfSelection );
276  OSLM_ASSERT( "The selectedTFKey must be defined, check your configuration.", !m_selectedTFKey.empty() );
277  if ( tfSelection->find( m_selectedTFKey ) == tfSelection->end() )
278  {
279  ::fwData::TransferFunction::sptr tfGreyLevel = ::fwData::TransferFunction::createDefaultTF();
280  if (image->getWindowWidth() != 0 )
281  {
282  tfGreyLevel->setWindow( image->getWindowWidth() );
283  tfGreyLevel->setLevel( image->getWindowCenter() );
284  }
285  else if(::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
286  {
287  double min, max;
288  ::fwComEd::fieldHelper::MedicalImageHelpers::getMinMax(image, min, max);
289  ::fwData::TransferFunction::TFValuePairType wlMinMax(min, max);
290  tfGreyLevel->setWLMinMax(wlMinMax);
291  }
292 
293  ::fwComEd::helper::Composite compositeHelper(tfSelection);
294  compositeHelper.add(m_selectedTFKey, tfGreyLevel);
295  compositeHelper.notify();
296  }
297  m_tfSelection = tfSelection;
298  }
299  }
300  else
301  {
302  const std::string poolFieldName = ::fwComEd::Dictionary::m_transferFunctionCompositeId;
303  ::fwData::Composite::sptr tfSelection = image->getField< ::fwData::Composite >(poolFieldName);
304 
305  if ( m_tfSelection.expired() || m_tfSelection.lock() != tfSelection )
306  {
307  const std::string defaultTFName = ::fwData::TransferFunction::s_DEFAULT_TF_NAME;
308 
309  ::fwComEd::helper::Image helper(image);
310  helper.createTransferFunctionPool(); // do nothing if image tf pool already exist
311 
312  tfSelection = image->getField< ::fwData::Composite >(poolFieldName);
313 
314  m_selectedTFKey = defaultTFName;
315  m_tfSelection = tfSelection;
316  }
317  }
318 }

436 void MedicalImageAdaptor::installTFConnections()
437 {
438  SLM_ASSERT( "TF connections already exist",!m_tfSelectionConnections && !m_tfConnections);
439 
440  ::fwData::Composite::sptr composite = this->getTransferFunctionSelection();
441 
442  m_tfSelectionConnections = ::fwServices::helper::SigSlotConnection::New();
443  m_tfConnections = ::fwServices::helper::SigSlotConnection::New();
444 
445  ::fwCom::Connection connection;
446  connection = composite->signal(::fwData::Composite::s_ADDED_OBJECTS_SIG)->connect(m_slotAddedObjects);
447  m_tfSelectionConnections->addConnection(connection);
448  connection = composite->signal(::fwData::Composite::s_CHANGED_OBJECTS_SIG)->connect(m_slotChangedObjects);
449  m_tfSelectionConnections->addConnection(connection);
450  connection = composite->signal(::fwData::Composite::s_REMOVED_OBJECTS_SIG)->connect(m_slotRemovedObjects);
451  m_tfSelectionConnections->addConnection(connection);
452 
453  ::fwData::TransferFunction::sptr tf = this->getTransferFunction();
454  connection = tf->signal(::fwData::TransferFunction::s_POINTS_MODIFIED_SIG)->connect(
455  m_slotUpdateTFPoints);
456  m_tfConnections->addConnection(connection);
457  connection = tf->signal(::fwData::TransferFunction::s_WINDOWING_MODIFIED_SIG)->connect(
458  m_slotUpdateTFWindowing);
459  m_tfConnections->addConnection(connection);
460 }
