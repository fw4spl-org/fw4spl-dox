102 void ServiceTest::testServiceCreationWithUUID()
103 {
104  const std::string myUUID = "myUUID";
105  const std::string myUUID2 = "myUUID2";
106  const std::string myUUID3 = "myUUID3";
107  size_t nbServices = 0;
108 
109  ::fwData::Integer::sptr obj = ::fwData::Integer::New();
110  ::fwServices::IService::sptr service;
111  ::fwServices::IService::sptr service2;
112 
113  // Test if the object support the service
114  CPPUNIT_ASSERT( ::fwServices::registry::ServiceFactory::getDefault()->support(obj->getClassname(),
115  "::fwServices::ut::TestService") );
116 
117  // Test adding service
118  service = ::fwServices::add(obj, "::fwServices::ut::TestService", "::fwServices::ut::TestServiceImplementation",
119  myUUID);
120  ::fwServices::add(obj, "::fwServices::ut::TestService", "::fwServices::ut::TestServiceImplementation", myUUID2);
121  nbServices = 2;
122  CPPUNIT_ASSERT(::fwServices::OSR::has(obj, "::fwServices::ut::TestService") );
123  CPPUNIT_ASSERT(::fwTools::fwID::exist(myUUID) );
124  CPPUNIT_ASSERT(::fwTools::fwID::exist(myUUID2));
125 
126  // Test getting the service its object
127  service2 = ::fwServices::get(myUUID2);
128  CPPUNIT_ASSERT(service);
129  CPPUNIT_ASSERT(service2);
130  CPPUNIT_ASSERT_EQUAL(obj, service2->getObject< ::fwData::Integer >());
131  CPPUNIT_ASSERT_EQUAL(myUUID2, service2->getID());
132  CPPUNIT_ASSERT( !::fwTools::fwID::exist(myUUID3) );
133  CPPUNIT_ASSERT_EQUAL( nbServices, ::fwServices::OSR::getServices(obj, "::fwServices::ut::TestService").size() );
134 
135  // Test erasing service
136  ::fwServices::OSR::unregisterService(service);
137  nbServices--;
138  CPPUNIT_ASSERT_EQUAL( nbServices, ::fwServices::OSR::getServices(obj, "::fwServices::ut::TestService").size() );
139  ::fwServices::OSR::unregisterService(service2);
140 }

183 void ServiceTest::testCommunication()
184 {
185  registry::ActiveWorkers::sptr activeWorkers = registry::ActiveWorkers::getDefault();
186  activeWorkers->initRegistry();
187 
188  const std::string EVENT = "EVENT";
189  const std::string service1UUID = "service1UUID";
190  const std::string service2UUID = "service2UUID";
191 
192  ::fwData::Composite::sptr obj = ::fwData::Composite::New();
193  ::fwServices::ut::TestService::sptr service1;
194  ::fwServices::ut::TestService::sptr service2;
195 
196  // Add services
197  CPPUNIT_ASSERT( ::fwServices::registry::ServiceFactory::getDefault()->support(obj->getClassname(),
198  "::fwServices::ut::TestService") );
199  ::fwServices::add(obj, "::fwServices::ut::TestService", "::fwServices::ut::TestServiceImplementation",
200  service1UUID);
201  CPPUNIT_ASSERT(::fwServices::OSR::has(obj, "::fwServices::ut::TestService") );
202  service1 = ::fwServices::ut::TestService::dynamicCast( ::fwServices::get(service1UUID) );
203  CPPUNIT_ASSERT(service1);
204 
205  ::fwServices::add(obj, "::fwServices::ut::TestService", "::fwServices::ut::TestServiceImplementation",
206  service2UUID);
207  service2 = ::fwServices::ut::TestService::dynamicCast( ::fwServices::get(service2UUID) );
208  CPPUNIT_ASSERT(service2);
209 
210  // Start services
211  service1->start().wait();
212  service2->start().wait();
213  CPPUNIT_ASSERT(service1->isStarted());
214  CPPUNIT_ASSERT(service2->isStarted());
215 
216  // Register communication channel
217  ::fwServices::helper::SigSlotConnection::sptr comHelper = ::fwServices::helper::SigSlotConnection::New();
218  comHelper->connect( obj, service1, service1->getObjSrvConnections() );
219  comHelper->connect( obj, service2, service2->getObjSrvConnections() );
220  comHelper->connect( service1, ::fwServices::ut::TestServiceImplementation::s_MSG_SENT_SIG,
221  service2, ::fwServices::ut::TestServiceImplementation::s_RECEIVE_MSG_SLOT );
222 
223  // Service1 send notification
224  ::fwServices::ut::TestServiceImplementation::MsgSentSignalType::sptr sig;
225  sig = service1->signal< ::fwServices::ut::TestServiceImplementation::MsgSentSignalType >(
226  ::fwServices::ut::TestServiceImplementation::s_MSG_SENT_SIG);
227  {
228  ::fwCom::SlotBase::sptr slot;
229  slot = service1->slot( ::fwServices::IService::s_UPDATE_SLOT );
230  ::fwCom::Connection::Blocker block(sig->getConnection(slot));
231  sig->asyncEmit(EVENT);
232  }
233 
234  // Test if service2 has received the message
235  service1->stop().wait();
236  service2->stop().wait();
237  CPPUNIT_ASSERT(service2->getIsUpdatedMessage());
238 
239  comHelper->disconnect();
240  comHelper.reset();
241 
242  ::fwServices::OSR::unregisterService(service1);
243  ::fwServices::OSR::unregisterService(service2);
244 
245  activeWorkers->clearRegistry();
246 }
