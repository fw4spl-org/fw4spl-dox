00094 void SMeshModifier::updating() throw( ::fwTools::Failed )
00095 {
00096     SLM_TRACE_FUNC();
00097     ::fwData::Mesh::sptr mesh = this->getObject< ::fwData::Mesh >();
00098     SLM_ASSERT("Mesh dynamicCast failed", mesh);
00099     ::fwComEd::MeshMsg::sptr msg = ::fwComEd::MeshMsg::New();
00100     try
00101     {
00102         if(m_functor == "ShakeMeshPoint")
00103         {
00104             ::fwDataTools::Mesh::shakePoint(mesh);
00105             msg->addEvent( ::fwComEd::MeshMsg::VERTEX_MODIFIED );
00106         }
00107         else if(m_functor == "ColorizeMeshCells")
00108         {
00109             ::fwDataTools::Mesh::colorizeMeshCells(mesh);
00110             msg->addEvent( ::fwComEd::MeshMsg::CELL_COLORS_MODIFIED );
00111         }
00112         else if(m_functor == "ColorizeMeshPoints")
00113         {
00114             ::fwDataTools::Mesh::colorizeMeshPoints(mesh);
00115             msg->addEvent( ::fwComEd::MeshMsg::POINT_COLORS_MODIFIED );
00116         }
00117         else if(m_functor == "ComputeCellNormals")
00118         {
00119             ::fwDataTools::Mesh::generateCellNormals(mesh);
00120             msg->addEvent( ::fwComEd::MeshMsg::CELL_NORMALS_MODIFIED );
00121         }
00122         else if(m_functor == "ComputePointNormals")
00123         {
00124             ::fwDataTools::Mesh::generatePointNormals(mesh);
00125             msg->addEvent( ::fwComEd::MeshMsg::POINT_NORMALS_MODIFIED );
00126         }
00127         else if(m_functor == "ShakeCellNormals")
00128         {
00129             ::fwDataTools::Mesh::shakeCellNormals(mesh);
00130             msg->addEvent( ::fwComEd::MeshMsg::CELL_NORMALS_MODIFIED );
00131         }
00132         else if(m_functor == "ShakePointNormals")
00133         {
00134             ::fwDataTools::Mesh::shakePointNormals(mesh);
00135             msg->addEvent( ::fwComEd::MeshMsg::POINT_NORMALS_MODIFIED );
00136         }
00137         else if(m_functor == "MeshDeformation")
00138         {
00139             m_animator.computeDeformation( mesh, 100, 50 );
00140             msg->addEvent( ::fwComEd::MeshMsg::VERTEX_MODIFIED );
00141             msg->addEvent( ::fwComEd::MeshMsg::POINT_COLORS_MODIFIED );
00142         }
00143     }
00144     catch (const std::exception & e)
00145     {
00146         std::stringstream ss;
00147         ss << "Warning during generating : " << e.what();
00148 
00149         ::fwGui::dialog::MessageDialog::showMessageDialog(
00150                 "Warning",
00151                 ss.str(),
00152                 ::fwGui::dialog::IMessageDialog::WARNING);
00153     }
00154 
00155     ::fwServices::IEditionService::notify(this->getSptr(), mesh, msg);
00156 }
