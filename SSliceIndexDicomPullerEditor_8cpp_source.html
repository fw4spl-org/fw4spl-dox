282 void SSliceIndexDicomPullerEditor::readImage(std::size_t selectedSliceIndex)
283 {
284  // DicomSeries
285  ::fwMedData::DicomSeries::csptr dicomSeries = this->getInOut< ::fwMedData::DicomSeries >("series");
286  SLM_ASSERT("DicomSeries should not be null !", dicomSeries);
287  if( dicomSeries->getModality() != "CT" && dicomSeries->getModality() != "MR" && dicomSeries->getModality() != "XA")
288  {
289  return;
290  }
291 
292  // Clear temporary seriesDB
293  ::fwMedDataTools::helper::SeriesDB sDBTempohelper(m_tempSeriesDB);
294  sDBTempohelper.clear();
295 
296  // Creates unique temporary folder, no need to check if exists before (see ::fwTools::System::getTemporaryFolder)
297  ::boost::filesystem::path path = ::fwTools::System::getTemporaryFolder("dicom");
298  ::boost::filesystem::path tmpPath = path / "tmp";
299 
300  SLM_INFO("Create " + tmpPath.string());
301  ::boost::filesystem::create_directories(tmpPath);
302 
303 
304  SLM_ASSERT("Dicom data shall be available before reading them.",
305  dicomSeries->getDicomAvailability() != ::fwMedData::DicomSeries::NONE
306  || dicomSeries->isInstanceAvailable(selectedSliceIndex));
307 
308  if(dicomSeries->getDicomAvailability() != ::fwMedData::DicomSeries::BINARIES )
309  {
310  ::fwMedData::DicomSeries::DicomPathContainerType paths = dicomSeries->getLocalDicomPaths();
311  ::boost::filesystem::path src = paths[selectedSliceIndex];
312  ::boost::filesystem::path dest = tmpPath / src.filename();
313 
314  ::boost::system::error_code err;
315  ::boost::filesystem::create_hard_link( src, dest, err );
316  if (err.value() != 0)
317  {
318  SLM_INFO("Copying " + src.string() + " to " + dest.string());
319  ::boost::filesystem::copy( src, dest );
320 
321  ::boost::system::error_code errPerm;
322  ::boost::filesystem::permissions(dest, ::boost::filesystem::owner_all, errPerm);
323  SLM_ERROR_IF("set permission error : " + errPerm.message(), errPerm.value());
324  }
325  }
326  else if(dicomSeries->getDicomAvailability() == ::fwMedData::DicomSeries::BINARIES)
327  {
328 
329  const ::fwMedData::DicomSeries::DicomBinaryContainerType& binaries = dicomSeries->getDicomBinaries();
330  ::fwMedData::DicomSeries::DicomBinaryContainerType::const_iterator binary = binaries.begin();
331  std::advance(binary, selectedSliceIndex);
332 
333  ::fwData::Array::sptr array = binary->second;
334  ::fwDataTools::helper::Array arrayHelper(array);
335  char* buffer = static_cast<char*>(arrayHelper.getBuffer());
336  size_t size = array->getSizeInBytes();
337 
338  ::boost::filesystem::path dest = tmpPath / binary->first;
339  ::boost::filesystem::ofstream fs(dest, std::ios::binary|std::ios::trunc);
340  FW_RAISE_IF("Can't open '" << tmpPath << "' for write.", !fs.good());
341 
342  fs.write(buffer, size);
343  fs.close();
344  }
345 
346  // Read image
347 
348  m_dicomReader.lock()->setFolder(tmpPath);
349  if(!m_dicomReader.expired())
350  {
351  m_dicomReader.lock()->update();
352 
353  if(m_dicomReader.expired() || m_dicomReader.lock()->isStopped())
354  {
355  return;
356  }
357  }
358  else
359  {
360  return;
361  }
362 
363  //Copy image
364  ::fwMedData::ImageSeries::sptr imageSeries;
365 
366  if(m_tempSeriesDB->getContainer().size() > 0)
367  {
368  imageSeries = ::fwMedData::ImageSeries::dynamicCast(*(m_tempSeriesDB->getContainer().begin()));
369  }
370 
371  if(imageSeries)
372  {
373  ::fwData::Image::sptr newImage = imageSeries->getImage();
374  ::fwData::Image::SizeType newSize = newImage->getSize();
375 
376  newImage->setField(::fwDataTools::fieldHelper::Image::m_axialSliceIndexId, m_axialIndex);
377  m_frontalIndex->setValue(static_cast<int>(newSize[0]/2));
378  newImage->setField(::fwDataTools::fieldHelper::Image::m_frontalSliceIndexId, m_frontalIndex);
379  m_sagittalIndex->setValue(static_cast<int>(newSize[1]/2));
380  newImage->setField(::fwDataTools::fieldHelper::Image::m_sagittalSliceIndexId, m_sagittalIndex);
381 
382  this->setOutput("image", newImage);
383  }
384 
385  ::boost::system::error_code ec;
386  ::boost::filesystem::remove_all(path, ec);
387  SLM_ERROR_IF("remove_all error for path " + path.string() + ": " + ec.message(), ec.value());
388 }

392 void SSliceIndexDicomPullerEditor::pullInstance()
393 {
394  SLM_ASSERT("Pacs not configured.", m_pacsConfiguration);
395 
396  if( m_pacsConfiguration )
397  {
398  // Catch any errors
399  try
400  {
401  // DicomSeries
402  ::fwMedData::DicomSeries::sptr dicomSeries = this->getInOut< ::fwMedData::DicomSeries >("series");
403  SLM_ASSERT("DicomSeries should not be null !", dicomSeries);
404 
405  // Get selected slice
406  std::size_t selectedSliceIndex = m_sliceIndexSlider->value() + dicomSeries->getFirstInstanceNumber();
407 
408  m_seriesEnquirer->initialize(m_pacsConfiguration->getLocalApplicationTitle(),
409  m_pacsConfiguration->getPacsHostName(),
410  m_pacsConfiguration->getPacsApplicationPort(),
411  m_pacsConfiguration->getPacsApplicationTitle(),
412  m_pacsConfiguration->getMoveApplicationTitle());
413 
414  m_seriesEnquirer->connect();
415  std::string seriesInstanceUID = dicomSeries->getInstanceUID();
416  std::string sopInstanceUID =
417  m_seriesEnquirer->findSOPInstanceUID(seriesInstanceUID, static_cast<unsigned int>(selectedSliceIndex));
418 
419  // Check if an instance with the selected Instance Number is found on the PACS
420  if(!sopInstanceUID.empty())
421  {
422  // Pull Selected Series using C-GET Requests
423  m_seriesEnquirer->pullInstanceUsingGetRetrieveMethod(seriesInstanceUID, sopInstanceUID);
424 
425  // Add path and trigger reading
426  ::boost::filesystem::path path = ::fwTools::System::getTemporaryFolder() / "dicom/";
427  ::boost::filesystem::path filePath = path.string() + seriesInstanceUID + "/" + sopInstanceUID;
428  dicomSeries->addDicomPath(selectedSliceIndex, filePath);
429  m_slotReadImage->asyncRun(selectedSliceIndex);
430  }
431  else
432  {
433  std::stringstream ss;
434  ss << "The selected series does not have an instance matching the selected instance number (" <<
435  selectedSliceIndex << ").";
436  m_slotDisplayMessage->asyncRun(ss.str());
437  }
438 
439  // Close connection
440  m_seriesEnquirer->disconnect();
441 
442  }
443  catch (::fwPacsIO::exceptions::Base& exception)
444  {
445  std::stringstream ss;
446  ss << "Unable to connect to the pacs. Please check your configuration: \n"
447  << "Pacs host name: " << m_pacsConfiguration->getPacsHostName() << "\n"
448  << "Pacs application title: " << m_pacsConfiguration->getPacsApplicationTitle() << "\n"
449  << "Pacs port: " << m_pacsConfiguration->getPacsApplicationPort() << "\n";
450  m_slotDisplayMessage->asyncRun(ss.str());
451  SLM_WARN(exception.what());
452  }
453 
454  }
455  else
456  {
457  SLM_ERROR("Pacs pull aborted : no pacs configuration found.");
458  }
459 }
