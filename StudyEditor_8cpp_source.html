79 SPTR(::fwMedData::Study) StudyEditor::getStudy()
80 {
81  ::fwMedData::Study::sptr srcStudy = m_series->getStudy();
82  ::fwMedData::Study::sptr study = ::fwMedData::Study::New();
83  study->setDescription(m_description->text().trimmed().toStdString());
84  study->setDate(m_date->text().trimmed().toStdString());
85  study->setTime(m_time->text().trimmed().toStdString());
86  study->setReferringPhysicianName(m_referringPhysicianName->text().trimmed().toStdString());
87  study->setInstanceUID(srcStudy->getInstanceUID());
88  study->setPatientAge(srcStudy->getPatientAge());
89 
90  std::string date = srcStudy->getDate();
91  ::boost::algorithm::trim(date);
92  std::string time = srcStudy->getTime();
93  ::boost::algorithm::trim(time);
94  std::string refP = srcStudy->getReferringPhysicianName();
95  ::boost::algorithm::trim(refP);
96  std::string age = srcStudy->getPatientAge();
97  ::boost::algorithm::trim(age);
98 
99  // Study description comparison ommitted here
100  const bool same = study->getDate() == date
101  && study->getTime() == time
102  && study->getReferringPhysicianName() == refP
103  && study->getInstanceUID() == srcStudy->getInstanceUID()
104  && study->getPatientAge() == age;
105 
106  ::fwData::String::sptr fieldNewStudy = srcStudy->getField< ::fwData::String>(s_NEW_STUDY_FIELD_NAME);
107  if(fieldNewStudy || !same)
108  {
109  study->setInstanceUID(::fwMedDataTools::generateStudyInstanceUid());
110  }
111 
112  return study;
113 }
