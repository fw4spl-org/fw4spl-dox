00087 
00088     /* std::map are sorted regarding to key values. Keys of m_zValue2AdaptorID are z-values (float).
00089      *
00090      * The dispatching is performed to the highest z-valued adaptors first, so we start iterating
00091      * from the end of m_zValue2AdaptorID (with reverse_iterator).

00094     {
00095         if ( ! _event->getAccepted() && ! m_adaptorID2SceneAdaptor2D[ rit->second ].m_service.expired() )
00096         {
00097             m_adaptorID2SceneAdaptor2D[ rit->second ].m_service.lock()->processInteraction( _event );
