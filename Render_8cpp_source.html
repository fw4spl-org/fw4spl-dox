90 void Render::dispatchInteraction( SPTR(::scene2D::data::Event)_event)
91 {
92  SLM_TRACE_FUNC();
93 
94  /* std::map are sorted regarding to key values. Keys of m_zValue2AdaptorID are z-values (float).
95  *
96  * The dispatching is performed to the highest z-valued adaptors first, so we start iterating
97  * from the end of m_zValue2AdaptorID (with reverse_iterator).
98  */
99  for(ZValue2AdaptorID::reverse_iterator rit = m_zValue2AdaptorID.rbegin(); rit != m_zValue2AdaptorID.rend(); ++rit )
100  {
101  if ( !_event->isAccepted() )
102  {
103  auto adaptor = m_adaptorID2SceneAdaptor2D[ rit->second ].m_service.lock();
104  if(adaptor != nullptr && adaptor->isStarted())
105  {
106  m_adaptorID2SceneAdaptor2D[ rit->second ].m_service.lock()->processInteraction( _event );
107  }
108  }
109  }

124 void Render::configuring() throw ( ::fwTools::Failed )
125 {
126  SLM_TRACE_FUNC();
127 
128  this->initialize();
129 
130  std::vector < SPTR(::fwRuntime::ConfigurationElement) > vectConfig = m_configuration->find("scene");
131  SLM_ASSERT("There is no implementation between \"scene\" tags", !vectConfig.empty());
132  m_sceneConfiguration = vectConfig.at(0);
133 
134  ::fwRuntime::ConfigurationElementContainer::Iterator iter;
135  for (iter = m_sceneConfiguration->begin(); iter != m_sceneConfiguration->end(); ++iter)
136  {
137  if ((*iter)->getName() == "axis")
138  {
139  this->configureAxis(*iter);
140  }
141  else if ((*iter)->getName() == "viewport")
142  {
143  this->configureViewport(*iter);
144  }
145  else if ((*iter)->getName() == "scene")
146  {
147  this->configureScene(*iter);
148  }
149  else if ((*iter)->getName() == "adaptor")
150  {
151  this->configureAdaptor(*iter);
152  }
153  else if((*iter)->getName() == "connect")
154  {
155  m_connect.push_back(*iter);
156  }
157  else if((*iter)->getName() == "proxy")
158  {
159  m_proxies.push_back(*iter);
160  }
161  else
162  {
163  OSLM_ASSERT("Bad scene configurationType, unknown xml node : " << (*iter)->getName(), false);
164  }
165  }

172 void Render::starting() throw ( ::fwTools::Failed )
173 {
174  SLM_TRACE_FUNC();
175 
176  this->create();
177 
178  this->startContext();
179 
180  ::fwData::Composite::sptr composite;
181  if(this->isVersion2())
182  {
183  composite = this->getInOut< ::fwData::Composite >(s_DEFAULT_OBJECT);
184  }
185  else
186  {
187  composite = this->getObject< ::fwData::Composite >();
188  }
189 
190  ObjectsID2AdaptorIDVector::iterator objectIter = m_objectsID2AdaptorIDVector.find( "self" );
191  if ( objectIter != m_objectsID2AdaptorIDVector.end() )
192  {
193  for(const AdaptorIDType& adaptorId : objectIter->second )
194  {
195  this->startAdaptor( adaptorId, composite );
196  SLM_ASSERT("Service is not started", m_adaptorID2SceneAdaptor2D[adaptorId].getService()->isStarted());
197  }
198  }
199 
200  ConstObjectMapType container;
201  if(this->isVersion2())
202  {
203  for(auto obj : this->getInputs())
204  {
205  if(obj.first != s_DEFAULT_OBJECT)
206  {
207  container[obj.first] = obj.second.lock();
208  }
209  }
210  for(auto obj : this->getInOuts())
211  {
212  if(obj.first != s_DEFAULT_OBJECT)
213  {
214  container[obj.first] = obj.second.lock();
215  }
216  }
217  this->startAdaptorsFromComposite(container);
218  }
219  else
220  {
221  auto objects = composite->getContainer();
222  std::for_each(objects.begin(), objects.end(), [&container](
223  const ::fwData::Composite::ContainerType::value_type& it)
224  {
225  container[it.first] = it.second;
226  });
227  this->startAdaptorsFromComposite(container);
228  }
229 
230  //Create connections when adaptors are started
231 
232  for(const ::fwRuntime::ConfigurationElement::sptr& connect : m_connect)
233  {
234  if(!connect->hasAttribute("waitForKey"))
235  {
236  ::fwServices::helper::Config::createConnections(connect, m_connections);
237  }
238  }
239  for(const ::fwRuntime::ConfigurationElement::sptr& proxy : m_proxies)
240  {
241  if(!proxy->hasAttribute("waitForKey"))
242  {
243  ::fwServices::helper::Config::createProxy("self", proxy, m_proxyMap);
244  }
245  }
246  this->connectAfterWait(composite->getContainer());
247 }

317 void Render::swapping(const IService::KeyType& key) throw(::fwTools::Failed)
318 {
319  if (this->isVersion2())
320  {
321  // remove connections
322  auto iter = m_objectConnections.find(key);
323  if(iter != m_objectConnections.end())
324  {
325  iter->second.disconnect();
326  m_objectConnections.erase(key);
327  }
328  ::fwServices::helper::Config::disconnectProxies(key, m_proxyMap);
329 
330  auto obj = this->getInput< ::fwData::Object>(key);
331  obj = (obj == nullptr) ? this->getInOut< ::fwData::Object>(key) : obj;
332 
333  ObjectsID2AdaptorIDVector::iterator objectIter = m_objectsID2AdaptorIDVector.find(key);
334  if (objectIter != m_objectsID2AdaptorIDVector.end())
335  {
336  for (const AdaptorIDType& adaptorId : objectIter->second)
337  {
338  SceneAdaptor2D& info = m_adaptorID2SceneAdaptor2D[adaptorId];
339 
340  if (info.getService() && info.getService()->isStarted())
341  {
342  info.getService()->stop();
343  }
344  }
345  }
346 
347  // create connections
348  if(obj)
349  {
350  ObjectsID2AdaptorIDVector::iterator objectIter = m_objectsID2AdaptorIDVector.find(key);
351  if (objectIter != m_objectsID2AdaptorIDVector.end())
352  {
353  for (const AdaptorIDType& adaptorId : objectIter->second)
354  {
355  SceneAdaptor2D& info = m_adaptorID2SceneAdaptor2D[adaptorId];
356 
357  if (info.getService())
358  {
359  if(obj != info.getService()->getObject())
360  {
361  ::fwServices::OSR::swapService(::fwData::Object::constCast(obj), info.getService());
362  }
363  info.getService()->start();
364  }
365  else
366  {
367  this->startAdaptor(adaptorId, obj);
368  }
369  }
370  }
371 
372  ::fwData::Composite::ContainerType mapConnect;
373  mapConnect[key] = ::fwData::Object::constCast(obj);
374  this->connectAfterWait(mapConnect);
375  }
376  }
377 }

399  ObjectsID2AdaptorIDVector::iterator objectIter = m_objectsID2AdaptorIDVector.find( "self" );

406  }
407 
408  ConstObjectMapType container;
409  if(this->isVersion2())
410  {
411  for(auto obj : this->getInOuts())
412  {
413  if(obj.first != s_DEFAULT_OBJECT)

415  container[obj.first] = obj.second.lock();
416  }
417  }
418  for(auto obj : this->getInputs())
419  {
420  if(obj.first != s_DEFAULT_OBJECT)
421  {
422  container[obj.first] = obj.second.lock();
423  }
424  }
425  this->stopAdaptorsFromComposite(container);
426  }
427  else
428  {
429  auto objects = composite->getContainer();
430  std::for_each(objects.begin(), objects.end(), [&container](
431  ::fwData::Composite::ContainerType::value_type it){ container[it.first] = it.second; });
432  this->stopAdaptorsFromComposite(container);

436  m_objectID2Object.clear();
437  m_objectsID2AdaptorIDVector.clear();
438 
439  this->stopContext();
440 
441  this->getContainer()->clean();
442  this->destroy();
443 }

677 void Render::startAdaptor(const AdaptorIDType& _adaptorID, const CSPTR(::fwData::Object)& _object)
678 {
679  if (!m_adaptorID2SceneAdaptor2D[_adaptorID].m_uid.empty())
680  {
681  m_adaptorID2SceneAdaptor2D[_adaptorID].m_service = ::fwServices::add< ::scene2D::adaptor::IAdaptor >(
682  _object, m_adaptorID2SceneAdaptor2D[_adaptorID].m_type, m_adaptorID2SceneAdaptor2D[_adaptorID].m_uid);
683  }
684  else
685  {
686  m_adaptorID2SceneAdaptor2D[_adaptorID].m_service = ::fwServices::add< ::scene2D::adaptor::IAdaptor >(
687  _object, m_adaptorID2SceneAdaptor2D[_adaptorID].m_type);
688  }

738  qreal x,y,w,h;
739  rec.getRect(&x,&y,&w,&h);
740 
741  if ( ratioPercent != 0 )

748  y = centerY - h/2.0;

752  m_sceneStart.setY( y );
753  m_sceneWidth.setX( w );
754  m_sceneWidth.setY( h );
