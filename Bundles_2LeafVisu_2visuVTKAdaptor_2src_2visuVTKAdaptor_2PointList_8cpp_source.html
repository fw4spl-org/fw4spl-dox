118 void PointList::createServices(WeakPointListType &wPtList)
119 {
120  for( ::fwData::Point::wptr wpt : wPtList )
121  {
122  SLM_ASSERT("Point Expired", !wpt.expired());
123 
124  ::fwData::Point::sptr pt = wpt.lock();
125  ::fwRenderVTK::IVtkAdaptorService::sptr service =
126  ::fwServices::add< ::fwRenderVTK::IVtkAdaptorService >
127  ( pt, "::visuVTKAdaptor::Point" );
128  SLM_ASSERT("service not instanced", service);
129 
130  service->setRenderService(this->getRenderService());
131  service->setRenderId( this->getRenderId() );
132  service->setPickerId( this->getPickerId() );
133  service->setAutoRender( this->getAutoRender() );
134  service->start();
135 
136  this->registerService(service);
137  }

154 PointList::WeakPointListType PointList::getNewPoints()

158 // std::set_difference (

166  isFound = false;
