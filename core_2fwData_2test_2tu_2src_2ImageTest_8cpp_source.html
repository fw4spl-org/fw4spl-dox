215 {

223  VECTORSIZE[1] = 20;

227

230 
231  // test 1 : use getPixelBuffer
232  short count = 0;

235  {
236  *iter = count++;
237  }
238 
239  for (unsigned int x = 0; x<VECTORSIZE[0]; ++x)
240  {
241  for (unsigned int y = 0; y<VECTORSIZE[1]; ++y)
242  {
243  for (unsigned int z = 0; z<VECTORSIZE[2]; ++z)
244  {
245  short val = static_cast<short>(x+y*VECTORSIZE[0]+z*VECTORSIZE[0]*VECTORSIZE[1]);
246  ::fwData::Image::IndexType index = val;
247  CPPUNIT_ASSERT_EQUAL(val, *reinterpret_cast<short*>(imgHelper.getPixelBuffer(x,y,z)));
248  CPPUNIT_ASSERT_EQUAL(val, *reinterpret_cast<short*>(imgHelper.getPixelBuffer(index)));
249  ::fwData::Image::BufferType* buffer =
250  reinterpret_cast< ::fwData::Image::BufferType* >(imgHelper.getBuffer());
251  CPPUNIT_ASSERT_EQUAL(val, *reinterpret_cast<short*>(imgHelper.getPixelBuffer(index)));
252 
253  std::stringstream ss;
254  ss << val;
255  CPPUNIT_ASSERT_EQUAL( ss.str(), imgHelper.getPixelAsString(x,y,z));
256  }
257  }
258  }
259 
260  // test 2 : use setPixelBuffer
261  for (unsigned int x = 0; x<VECTORSIZE[0]; ++x)
262  {
263  for (unsigned int y = 0; y<VECTORSIZE[1]; ++y)
264  {
265  for (unsigned int z = 0; z<VECTORSIZE[2]; ++z)
266  {
267  ::fwData::Image::IndexType index = x+y*VECTORSIZE[0]+z*VECTORSIZE[0]*VECTORSIZE[1];

274  count = 0;
