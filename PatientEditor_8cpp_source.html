00143 
00144     const bool same = srcPatient->getName() == name
00145         && srcPatient->getBirthdate() == birth
00146         && (compareSex ? (srcPatientSex == sexStr) : true)
00147         && srcPatient->getPatientId() == patient->getPatientId();
00148 
00149     ::fwData::String::sptr fieldPatient = srcPatient->getField< ::fwData::String >(s_NEW_PATIENT_FIELD_NAME);
00150     if(fieldPatient || !same)
