00042 void AppConfigParameters::parseBundleInformation()
00043 {
00044     std::vector< ::boost::shared_ptr< ::fwRuntime::Extension > >  extensions = ::fwRuntime::getAllExtensionsForPoint("::fwServices::registry::AppConfigParameters");
00045     BOOST_FOREACH( ::boost::shared_ptr< ::fwRuntime::Extension > ext, extensions )
00046     {
00047         // Get id
00048         std::string extensionId = ext->findConfigurationElement("id")->getValue();
00049 
00050         AppConfig::FieldAdaptorType parameters;
00051 
00052         // Get parmeters
00053         ::fwRuntime::ConfigurationElement::csptr parametersConfig = ext->findConfigurationElement("parameters");
00054         ::fwRuntime::ConfigurationElement::Container elements = parametersConfig->getElements();
00055         BOOST_FOREACH( ::fwRuntime::ConfigurationElement::sptr paramConfig, elements )
00056         {
00057             std::string name = paramConfig->getExistingAttributeValue("name");
00058             std::string val = paramConfig->getExistingAttributeValue("value");
00059             parameters[name] = val;
00060         }
00061         ::fwCore::mt::WriteLock lock(m_registryMutex);
00062         Registry::const_iterator iter = m_reg.find( extensionId );
00063         SLM_ASSERT("Sorry, the id " <<  extensionId
00064                    << " already exists in the application configuration parameter registry", iter == m_reg.end());
00065         m_reg[extensionId] = parameters;
00066     }
00067 }
