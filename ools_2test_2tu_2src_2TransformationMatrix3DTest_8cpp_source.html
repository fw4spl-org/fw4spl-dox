121 void TransformationMatrix3DTest::matrixTest()
122 {
123  ::fwData::TransformationMatrix3D::sptr tm1 = ::fwData::TransformationMatrix3D::New();
124  ::fwData::TransformationMatrix3D::sptr tm2 = ::fwData::TransformationMatrix3D::New();
125  ::fwData::TransformationMatrix3D::sptr tm3 = ::fwData::TransformationMatrix3D::New();
126  ::fwData::TransformationMatrix3D::sptr tm4 = ::fwData::TransformationMatrix3D::New();
127 
128  ::fwDataTools::TransformationMatrix3D::identity(tm2);
129 
130  ::fwData::TransformationMatrix3D::TMCoefArray tm1Coefs;
131  for (int i = 0; i<16; ++i)
132  {
133  tm1Coefs[i] = i+1;
134  }
135 
136  for (int i = 0; i<4; ++i)
137  {
138  for (int j = 0; j<4; ++j)
139  {
140  tm2->setCoefficient(i,j, fabs(static_cast<double>(i-j)) + 1);
141  }
142  }
143  tm1->setCoefficients(tm1Coefs);
144  tm4->setCoefficients(tm1Coefs);
145 
146  // Test matrix-matrix multiplication
147  ::fwDataTools::TransformationMatrix3D::multiply(tm1, tm2, tm3);
148  for (size_t i = 0; i<4; ++i)
149  {
150  for (size_t j = 0; j<4; ++j)
151  {
152  double val = 0;
153  for (size_t k = 0; k<4; ++k)
154  {
155  val += tm1->getCoefficient(i,k)*tm2->getCoefficient(k,j);
156  }
157  CPPUNIT_ASSERT_EQUAL(val, tm3->getCoefficient(i,j));
158  }
159  }

161  // Test invert

224 void TransformationMatrix3DTest::glmGetterSetterTest()
225 {
226  ::fwData::TransformationMatrix3D::TMCoefArray coefs = { 2, -2, .3, .12,
227  4, 8.9, 4.2, 1.2,
228  7.8, -12.1, 2.3, 1.2,
229  .3, 1.21, -3.1, 1.2};
230 
231  ::fwData::TransformationMatrix3D::sptr mat = ::fwData::TransformationMatrix3D::New();
232  mat->setCoefficients(coefs);
233 
234  // Test getter
235  ::glm::dmat4x4 glmMat;
236 
237 #ifndef FW_PROFILING_DISABLED
238  {
239  FW_PROFILE("::fwDataTools::TransformationMatrix3D::getMatrixFromTF3D");
240  for(int i = 0; i < 1000000; ++i)
241  {
242  glmMat = ::fwDataTools::TransformationMatrix3D::getMatrixFromTF3D(mat);
243  }
244  }
245 #else
246  glmMat = ::fwDataTools::TransformationMatrix3D::getMatrixFromTF3D(mat);
247 #endif
248 
249  // TransformationMatrix3D is stored row-major
250  // glm matrices are stored column-major
251  for (size_t i = 0; i < 4; ++i)
252  {
253  for (size_t j = 0; j < 4; ++j)
254  {
255  CPPUNIT_ASSERT_EQUAL(glmMat[i][j], coefs[i + j * 4]);
256  }
257  }
258 
259  // Test setter
260  double coefs2[] = { 11, -2, -.3, -.74,
261  .214, 82.9, 9.2, -5.2,
262  17.8, -2.1, 2.3, 1.2,
263  .13, 0.1, -0.1, 0.2};
264 
265  glmMat = ::glm::make_mat4<double>(coefs2);
266 #ifndef FW_PROFILING_DISABLED
267  {
268  FW_PROFILE("::fwDataTools::TransformationMatrix3D::setTF3DFromMatrix");
269  for(int i = 0; i < 1000000; ++i)
270  {
271  ::fwDataTools::TransformationMatrix3D::setTF3DFromMatrix(mat, glmMat);
272  }
273  }
274 #else
275  ::fwDataTools::TransformationMatrix3D::setTF3DFromMatrix(mat, glmMat);
276 #endif
277 
278  for (size_t i = 0; i < 4; ++i)
279  {
280  for (size_t j = 0; j < 4; ++j)
281  {
282  CPPUNIT_ASSERT_EQUAL(mat->getCoefficient(i, j), coefs2[i + j * 4]);
283  }
284  }
285 }
