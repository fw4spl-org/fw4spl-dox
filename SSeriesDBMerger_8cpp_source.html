75 void SSeriesDBMerger::updating( ) throw(::fwTools::Failed)
76 {
77  SLM_TRACE_FUNC();
78  ::fwGui::LockAction lock(this->getSptr());
79  ::fwMedData::SeriesDB::sptr seriesDB = this->getObject< ::fwMedData::SeriesDB >();
80  SLM_ASSERT("SeriesDB not instanced", seriesDB);
81 
82  // Create a new SeriesDB
83  ::fwMedData::SeriesDB::sptr localSeriesDB = ::fwMedData::SeriesDB::New();
84 
86 
87  // Get the config
88  ::fwRuntime::ConfigurationElement::csptr ioCfg;
89  ioCfg = ::fwServices::registry::ServiceConfig::getDefault()->getServiceConfig(m_ioSelectorSrvConfig,
90  "::uiIO::editor::SIOSelector");
91  SLM_ASSERT("There is no service configuration "
92  << m_ioSelectorSrvConfig
93  << " for ::uiIO::editor::SIOSelector", ioCfg);
94 
95  // Init and execute the service
96  ::fwServices::IService::sptr ioSelectorSrv;
97  ioSelectorSrv = ::fwServices::add(localSeriesDB,
98  "::gui::editor::IDialogEditor",
99  "::uiIO::editor::SIOSelector");
100 
101  ioSelectorSrv->setWorker(m_associatedWorker);
102 
103  auto jobCreatedSignal = ioSelectorSrv->signal("jobCreated");
104  if(jobCreatedSignal)
105  {
106  jobCreatedSignal->connect(m_slotForwardJob);
107  }
108 
109  ioSelectorSrv->setConfiguration( ::fwRuntime::ConfigurationElement::constCast(ioCfg) );
110  ioSelectorSrv->configure();
111  ioSelectorSrv->start();
112  ioSelectorSrv->update();
113  ioSelectorSrv->stop();
114  ::fwServices::OSR::unregisterService( ioSelectorSrv );
115 
116  ::fwComEd::helper::SeriesDB sDBhelper(seriesDB);
117  sDBhelper.merge(localSeriesDB);
118  sDBhelper.notify();
119 }
