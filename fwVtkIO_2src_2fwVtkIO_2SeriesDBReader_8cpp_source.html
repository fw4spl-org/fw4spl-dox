00338 
00339         vtkSmartPointer< vtkDataObject  > obj;

00345             if(m_lazyMode)

00347                 img = lazyRead<vtkGenericDataObjectReader>(file);

00349             if (!img)
00350             {

00353         }
00354         else if(file.extension().string() == ".vti")
00355         {
00356             if(m_lazyMode)

00358                 img = lazyRead<vtkXMLGenericDataObjectReader>(file);
00359             }
00360             if (!img)
00361             {
00362                 obj = getObj<vtkXMLGenericDataObjectReader>(file, this);
00363             }
00364         }
00365         else if(file.extension().string() == ".mhd")
00366         {
00367             obj = getObj<vtkMetaImageReader>(file, this);
00368         }
00369 
00370         if (!img)

00373             img = ::fwData::Image::dynamicCast(dataObj);

00375         }
00376         if(img)

00381             seriesDB->getContainer().push_back(imgSeries);
00382         }
00383         else if (rec)
00384         {
00385             recs.push_back(rec);
00386         }
00387         else

00389             errorFiles.push_back(file.string());

00391     }
00392

00396     }
00397

00399     if(!recs.empty())
00400     {
00401         ::fwMedData::ModelSeries::sptr modelSeries = ::fwMedData::ModelSeries::New();
00402         initSeries(modelSeries, instanceUID);
00403         modelSeries->setReconstructionDB(recs);
