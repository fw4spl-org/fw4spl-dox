361 template< typename WORLD, typename INT_INDEX >
362 void MedicalImageAdaptor::worldToSliceIndex(const WORLD world, INT_INDEX index )
363 {
364  double spacing[3];
365  this->getImageSpacing(spacing);
366  double origin[3];
367  this->getImageOrigin(origin);
368  for ( int i = 0; i<3; ++i )
369  {
370  index[i] =
371  static_cast< int >( ( (world[i] - origin[i])/spacing[i] ) +
372  ( ( (world[i] - origin[i])/spacing[i] ) >= 0 ? 0.5 : -0.5 ) );
373  }
374 }

378 template< typename WORLD, typename INT_INDEX >
379 void MedicalImageAdaptor::worldToImageSliceIndex(const WORLD world, INT_INDEX index )
380 {
381  int imageSize[3];
382  this->getImageDataSize(imageSize);
383  this->worldToSliceIndex(world, index);
384 
385  int idval;
386  for (int i = 0; i < 3; i++)
387  {
388  int max = imageSize[i]-1;
389  idval = index[i];
390  if (idval < 0)
391  {
392  index[i] = 0;
393  }
394  else if (idval > max)
395  {
396  index[i] = max;
397  }
398  }
399 }
