367 template< typename WORLD, typename INT_INDEX >
368 void MedicalImageAdaptor::worldToSliceIndex(const WORLD world, INT_INDEX index )
369 {
370  double spacing[3];
371  this->getImageSpacing(spacing);
372  double origin[3];
373  this->getImageOrigin(origin);
374  for ( int i = 0; i<3; ++i )
375  {
376  index[i] =
377  static_cast< int >( ( (world[i] - origin[i])/spacing[i] ) +
378  ( ( (world[i] - origin[i])/spacing[i] ) >= 0 ? 0.5 : -0.5 ) );
379  }
380 }

384 template< typename WORLD, typename INT_INDEX >
385 void MedicalImageAdaptor::worldToImageSliceIndex(const WORLD world, INT_INDEX index )
386 {
387  int imageSize[3];
388  this->getImageDataSize(imageSize);
389  this->worldToSliceIndex(world, index);
390 
391  int idval;
392  for (int i = 0; i < 3; i++)
393  {
394  int max = imageSize[i]-1;
395  idval = index[i];
396  if (idval < 0)
397  {
398  index[i] = 0;
399  }
400  else if (idval > max)
401  {
402  index[i] = max;
403  }
404  }
405 }
