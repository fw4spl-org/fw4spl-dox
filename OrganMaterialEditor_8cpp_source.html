138 void OrganMaterialEditor::onColorButton()
139 {
140  ::fwData::Reconstruction::sptr reconstruction = this->getObject< ::fwData::Reconstruction>();
141  SLM_ASSERT("No Reconstruction!", reconstruction);
142 
143  ::fwData::Material::sptr material = reconstruction->getMaterial();
144  int red = material->diffuse()->red()*255;
145  int green = material->diffuse()->green()*255;
146  int blue = material->diffuse()->blue()*255;
147 
148  // Create Color choice dialog.
149  ::fwGuiQt::container::QtContainer::sptr qtContainer = ::fwGuiQt::container::QtContainer::dynamicCast(
150  this->getContainer() );
151  QWidget* const container = qtContainer->getQtContainer();
152  SLM_ASSERT("container not instanced", container);
153 
154  QColor oldColor(red, green, blue);
155  QColor color = QColorDialog::getColor(oldColor, container);
156  if(color.isValid())
157  {
158  material->diffuse()->red() = color.redF();
159  material->diffuse()->green() = color.greenF();
160  material->diffuse()->blue() = color.blueF();
161  this->materialNotification();
162  refreshMaterial();
163  }

185 void OrganMaterialEditor::refreshMaterial( )
186 {
187  ::fwData::Reconstruction::sptr reconstruction = this->getObject< ::fwData::Reconstruction>();
188  SLM_ASSERT("No Reconstruction!", reconstruction);
189 
190  ::fwGuiQt::container::QtContainer::sptr qtContainer = ::fwGuiQt::container::QtContainer::dynamicCast(
191  this->getContainer() );
192  QWidget* const container = qtContainer->getQtContainer();
193  SLM_ASSERT("container not instanced", container);
194 
195  container->setEnabled(!reconstruction->getOrganName().empty());
196 
197  ::fwData::Material::sptr material = reconstruction->getMaterial();
198  QColor materialColor = QColor (
199  material->diffuse()->red()*255,
200  material->diffuse()->green()*255,
201  material->diffuse()->blue()*255,
202  material->diffuse()->alpha()*255
203  );
204 
205  int iconSize = m_colourButton->style()->pixelMetric(QStyle::PM_LargeIconSize);
206  QPixmap pix(iconSize, iconSize);
207  pix.fill(materialColor);
208 
209  m_colourButton->setIcon(QIcon(pix));
210 
211  int a = material->diffuse()->alpha()*100;
212  m_opacitySlider->setValue( a );
213  std::stringstream ss;
214  ss << a << "%";
215  m_transparencyValue->setText(QString::fromStdString(ss.str()));
216 }
