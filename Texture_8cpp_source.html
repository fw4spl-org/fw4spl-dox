118 void Texture::applyTexture( SPTR(::fwData::Material)_material )
119 {
120  if(m_materialSet.count(_material) == 0)
121  {
122  m_materialSet.insert(_material);
123  }
124 
125  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
126 
127  if(!::fwComEd::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
128  {
129  return;
130  }
131 
132  _material->setDiffuseTexture(image);
133  if(m_lighting == false)
134  {
135  _material->setShadingMode(::fwData::Material::AMBIENT);
136  }
137 
138  ::fwData::Material::FilteringType filtering = ::fwData::Material::LINEAR;
139  ::fwData::Material::WrappingType wrapping = ::fwData::Material::REPEAT;
140 
141  if(m_filtering == "nearest")
142  {
143  filtering = ::fwData::Material::NEAREST;
144  }
145  else if(m_filtering == "linear")
146  {
147  filtering = ::fwData::Material::LINEAR;
148  }
149  else
150  {
151  OSLM_WARN("Texture filtering type unknown or not supported : " << m_filtering);
152  }
153  _material->setDiffuseTextureFiltering(filtering);
154 
155  if(m_wrapping == "repeat")
156  {
157  wrapping = ::fwData::Material::REPEAT;
158  }
159  else if(m_wrapping == "clamp")
160  {
161  wrapping = ::fwData::Material::CLAMP;
162  }
163  else
164  {
165  OSLM_WARN("Texture wrapping type unknown or not supported : " << m_wrapping);
166  }
167  _material->setDiffuseTextureWrapping(wrapping);
168 
169  ::fwData::Object::ModifiedSignalType::sptr sig;
170  sig = _material->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
171  {
172  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
173  sig->asyncEmit();
174  }
175 }
