00165     _material->setDiffuseTextureFiltering(filtering);
00166 
00167     if(m_wrapping == "repeat")
00168     {
00169         wrapping = ::fwData::Material::REPEAT;
00170     }
00171     else if(m_wrapping == "clamp")
00172     {
00173         wrapping = ::fwData::Material::CLAMP;
00174     }
00175     else
00176     {

00179     _material->setDiffuseTextureWrapping(wrapping);

00182     msg->addEvent(::fwComEd::MaterialMsg::MATERIAL_IS_MODIFIED);
