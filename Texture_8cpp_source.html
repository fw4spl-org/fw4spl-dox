115 void Texture::applyTexture( SPTR(::fwData::Material)_material )
116 {
117  if(m_materialSet.count(_material) == 0)
118  {
119  m_materialSet.insert(_material);
120  }
121 
122  ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
123 
124  if(!::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(image))
125  {
126  return;
127  }
128 
129  _material->setDiffuseTexture(image);
130  if(m_lighting == false)
131  {
132  _material->setShadingMode(::fwData::Material::AMBIENT);
133  }
134 
135  ::fwData::Material::FilteringType filtering = ::fwData::Material::LINEAR;
136  ::fwData::Material::WrappingType wrapping = ::fwData::Material::REPEAT;
137 
138  if(m_filtering == "nearest")
139  {
140  filtering = ::fwData::Material::NEAREST;
141  }
142  else if(m_filtering == "linear")
143  {
144  filtering = ::fwData::Material::LINEAR;
145  }
146  else
147  {
148  OSLM_WARN("Texture filtering type unknown or not supported : " << m_filtering);
149  }
150  _material->setDiffuseTextureFiltering(filtering);
151 
152  if(m_wrapping == "repeat")
153  {
154  wrapping = ::fwData::Material::REPEAT;
155  }
156  else if(m_wrapping == "clamp")
157  {
158  wrapping = ::fwData::Material::CLAMP;
159  }
160  else
161  {
162  OSLM_WARN("Texture wrapping type unknown or not supported : " << m_wrapping);
163  }
164  _material->setDiffuseTextureWrapping(wrapping);
165 
166  ::fwData::Object::ModifiedSignalType::sptr sig;
167  sig = _material->signal< ::fwData::Object::ModifiedSignalType >(::fwData::Object::s_MODIFIED_SIG);
168  {
169  ::fwCom::Connection::Blocker block(sig->getConnection(m_slotUpdate));
170  sig->asyncEmit();
171  }
172 }
