92 void SVTKMesher::updating() throw ( ::fwTools::Failed )
93 {
94  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
95  ::fwData::Composite::iterator iterImg = composite->find(m_imageKey);
96  ::fwData::Composite::iterator iterModelSeries = composite->find(m_modelSeriesKey);
97 
98  SLM_ASSERT("Key '"+m_imageKey+"' not found in composite.", iterImg != composite->end());
99  SLM_ASSERT("Key '"+m_modelSeriesKey+"' not found in composite.", iterModelSeries != composite->end());
100 
101  ::fwData::Image::sptr image = ::fwData::Image::dynamicCast(iterImg->second);
102  SLM_ASSERT("Image '"+m_imageKey+"' is not valid.", image);
103 
104  ::fwMedData::ModelSeries::sptr modelSeries = ::fwMedData::ModelSeries::dynamicCast(iterModelSeries->second);
105  SLM_ASSERT("ModelSeries '"+m_modelSeriesKey+"' is not valid.", modelSeries);
106 
107  ::fwData::Mesh::sptr mesh = ::fwData::Mesh::New();
108 
109  // vtk img
110  vtkSmartPointer< vtkImageData > vtkImage = vtkSmartPointer< vtkImageData >::New();
111  ::fwVtkIO::toVTKImage( image, vtkImage );
112 
113  // contour filter
114  vtkSmartPointer< vtkDiscreteMarchingCubes > contourFilter = vtkSmartPointer< vtkDiscreteMarchingCubes >::New();
115  contourFilter->SetInputData(vtkImage);
116  contourFilter->SetValue(0, 255);
117  contourFilter->ComputeScalarsOn();
118  contourFilter->ComputeNormalsOn();
119  contourFilter->Update();
120 
121  // smooth filter
122  vtkSmartPointer< vtkWindowedSincPolyDataFilter > smoothFilter =
123  vtkSmartPointer< vtkWindowedSincPolyDataFilter >::New();
124  smoothFilter->SetInputConnection(contourFilter->GetOutputPort());
125  smoothFilter->SetNumberOfIterations( 50 );
126  smoothFilter->BoundarySmoothingOn();
127  smoothFilter->SetPassBand ( 0.1 );
128  smoothFilter->SetFeatureAngle(120.0);
129  smoothFilter->SetEdgeAngle(90);
130  smoothFilter->FeatureEdgeSmoothingOn();
131  smoothFilter->Update();
132 
133 
134  // Get polyData
135  vtkSmartPointer< vtkPolyData > polyData;
136 
137  // decimate filter
138  unsigned int reduction = m_reduction;
139  if( reduction > 0 )
140  {
141  vtkSmartPointer< vtkDecimatePro > decimate = vtkSmartPointer< vtkDecimatePro >::New();
142  decimate->SetInputConnection( smoothFilter->GetOutputPort() );
143  decimate->SetTargetReduction( reduction/100.0 );
144  decimate->PreserveTopologyOff();
145  decimate->SplittingOn();
146  decimate->BoundaryVertexDeletionOn();
147  decimate->SetSplitAngle( 120 );
148  decimate->Update();
149  polyData = decimate->GetOutput();
150  OSLM_TRACE("final GetNumberOfCells = " << polyData->GetNumberOfCells());
151  ::fwVtkIO::helper::Mesh::fromVTKMesh( polyData, mesh);
152  }
153  else
154  {
155  polyData = smoothFilter->GetOutput();
156  OSLM_TRACE("final GetNumberOfCells = " << polyData->GetNumberOfCells());
157  ::fwVtkIO::helper::Mesh::fromVTKMesh( polyData, mesh);
158  }
159 
160  ::fwData::Reconstruction::sptr reconstruction = ::fwData::Reconstruction::New();
161 
162  static unsigned int organNumber = 0;
163  ++organNumber;
164  reconstruction->setOrganName("OrganMesher_VTK_" + std::to_string(organNumber));
165  reconstruction->setStructureType("OrganType");
166  reconstruction->setIsVisible(true);
167  // Set Mesh
168  reconstruction->setMesh(mesh);
169 
170  ::fwMedData::ModelSeries::ReconstructionVectorType recs = modelSeries->getReconstructionDB();
171  recs.push_back(reconstruction);
172  modelSeries->setReconstructionDB(recs);
173 
174  // Notification
175  ::fwMedData::ModelSeries::ReconstructionVectorType addedRecs;
176  addedRecs.push_back(reconstruction);
177  auto sig = modelSeries->signal< ::fwMedData::ModelSeries::ReconstructionsAddedSignalType >(
178  ::fwMedData::ModelSeries::s_RECONSTRUCTIONS_ADDED_SIG);
179  sig->asyncEmit(addedRecs);
180 }
