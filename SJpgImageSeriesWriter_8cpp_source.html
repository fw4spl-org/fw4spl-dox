56 void SJpgImageSeriesWriter::configureWithIHM()
57 {
58  SLM_TRACE_FUNC();
59  static ::boost::filesystem::path _sDefaultPath;
60 
61  ::fwGui::dialog::LocationDialog dialog;
62  dialog.setTitle("Choose a directory to save image");
63  dialog.setDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
64  dialog.setOption(::fwGui::dialog::ILocationDialog::WRITE);
65  dialog.setType(::fwGui::dialog::ILocationDialog::FOLDER);
66 
67  ::fwData::location::Folder::sptr result;
68 
69  while (result = ::fwData::location::Folder::dynamicCast( dialog.show() ))
70  {
71  if( ::boost::filesystem::is_empty(result->getFolder()) )
72  {
73  break;
74  }
75  // message box
76  ::fwGui::dialog::MessageDialog messageBox;
77  messageBox.setTitle("Overwrite confirmation");
78  messageBox.setMessage("The selected directory is not empty. Write anyway ?");
79  messageBox.setIcon(::fwGui::dialog::IMessageDialog::QUESTION);
80  messageBox.addButton(::fwGui::dialog::IMessageDialog::YES);
81  messageBox.addButton(::fwGui::dialog::IMessageDialog::CANCEL);
82  if( messageBox.show() == ::fwGui::dialog::IMessageDialog::YES)
83  {
84  break;
85  }
86  }
87 
88  if (result)
89  {
90  _sDefaultPath = result->getFolder().parent_path();
91  dialog.saveDefaultLocation( ::fwData::location::Folder::New(_sDefaultPath) );
92  this->setFolder(result->getFolder());
93  }
94  else
95  {
96  this->clearLocations();
97  }
98 }

124 {
125  SLM_TRACE_FUNC();
126 
127  if( this->hasLocationDefined() )

136  JpgImageWriterService::saveImage(this->getFolder(), imageSeries->getImage());
