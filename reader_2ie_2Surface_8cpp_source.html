66 void Surface::readSurfaceSegmentationModule(::gdcm::SmartPointer< ::gdcm::Segment > segment)
67 {
68  ::fwData::Reconstruction::sptr reconstruction = ::fwData::Reconstruction::New();
69 
70  // We only handle segment containing one surface
71  if (segment->GetSurfaceCount() != 1)
72  {
73  throw ::fwGdcmIO::exception::Failed("Inappropriate surface count for a segment.");
74  }
75 
76  // Organ Name - TODO : if empty, try to get label with DictionarySegment
77  std::string organName = segment->GetSegmentDescription();
78  ::boost::algorithm::trim(organName);
79  reconstruction->setOrganName(organName);
80  SLM_TRACE("Organ Name : " + reconstruction->getOrganName());
81 
82  // Structure Type
83  std::string structureType = segment->GetSegmentLabel();
84  ::boost::algorithm::trim(structureType);
85  reconstruction->setStructureType(structureType);
86  SLM_TRACE("Structure type : " + reconstruction->getStructureType());
87 
88  // Get the associated surface of the current segmentation
89  ::gdcm::SmartPointer< ::gdcm::Surface > surface = segment->GetSurface(0);
90 
91  this->readSurfaceMeshModule(surface, reconstruction);
92 
93  // Add the Reconstruction in the ModelSeries
94  ::fwMedData::ModelSeries::ReconstructionVectorType reconstructionDB = m_object->getReconstructionDB();
95  reconstructionDB.push_back(reconstruction);
96  m_object->setReconstructionDB(reconstructionDB);
97 
98 }

102 void Surface::readSurfaceMeshModule(::gdcm::SmartPointer< ::gdcm::Surface > surface,
103  ::fwData::Reconstruction::sptr reconstruction)
104 {
105  // Create material
106  ::fwData::Material::sptr material = fwData::Material::New();
107 
108  // Convert CIE Lab to RGBA
109  const unsigned short* lab = surface->GetRecommendedDisplayCIELabValue();
110  ::gdcm::SurfaceHelper::ColorArray CIELab(3);
111  CIELab[0] = lab[0];
112  CIELab[1] = lab[1];
113  CIELab[2] = lab[2];
114  const std::vector<float> RGB = ::gdcm::SurfaceHelper::RecommendedDisplayCIELabToRGB(CIELab, 1.);
115 
116  // Adapt color to material
117  ::fwData::Color::ColorArray rgba;
118  ::boost::algorithm::clamp_range(RGB.begin(), RGB.end(), rgba.begin(), 0.f, 1.f);
119 
120  // Recommended Presentation Opacity
121  const float opacity = ::boost::algorithm::clamp(surface->GetRecommendedPresentationOpacity(), 0.f, 1.f);
122  rgba[3] = opacity; // set alpha component
123 
124  // Set reconstruction's visibility
125  reconstruction->setIsVisible(opacity > 1e-3);
126  OSLM_TRACE("Reconstruction is visible : " << reconstruction->getIsVisible());
127 
128  // Set reconstruction's color
129  ::fwData::Color::sptr color = ::fwData::Color::New();
130  color->setRGBA( rgba );
131  material->setDiffuse(color);
132  OSLM_TRACE("RGBA color : " << rgba[0]<<" "<< rgba[1]<<" "<< rgba[2]<<" "<< rgba[3]);
133 
134  // Recommended Presentation Type
135  material->setRepresentationMode(
136  ::fwGdcmIO::helper::DicomData::convertToRepresentationMode( surface->GetRecommendedPresentationType() ) );
137  OSLM_TRACE("Reconstruction's structure type : " <<
138  ::gdcm::Surface::GetVIEWTypeString( surface->GetRecommendedPresentationType() ) );
139 
140  // Manifold
141  if (surface->GetManifold() == ::gdcm::Surface::YES)
142  {
143  throw ::fwGdcmIO::exception::Failed("Manifold not handled");
144  }
145 
146  // Point Coordinates Data
147  const ::gdcm::DataElement& gdcmPointCoords = surface->GetPointCoordinatesData();
148  const ::gdcm::ByteValue* gdcmPointCoordsBV = gdcmPointCoords.GetByteValue();
149  const char* coordBuffer = gdcmPointCoordsBV->GetPointer();
150 
151  // Check that the surface contains point coordinates
152  if(gdcmPointCoordsBV == 0 || gdcmPointCoordsBV->GetPointer() == 0)
153  {
154  throw ::fwGdcmIO::exception::Failed("No point coordinates data");
155  }
156 
157  // Get size of buffer (bytes)
158  const unsigned long coordBufferSize = gdcmPointCoordsBV->GetLength();
159 
160  // Compute number of coordinates
161  const unsigned long coordSize = coordBufferSize / sizeof(float);
162 
163  // Check that the buffer has the correct size according to the number of points
164  if ( (coordSize / 3) != surface->GetNumberOfSurfacePoints())
165  {
166  throw ::fwGdcmIO::exception::Failed("Corrupted point coordinates data");
167  }
168 
169  // Mesh Primitive
170  ::gdcm::SmartPointer< ::gdcm::MeshPrimitive > gdcmMeshPrimitive = surface->GetMeshPrimitive();
171  if (gdcmMeshPrimitive->GetPrimitiveType() != ::gdcm::MeshPrimitive::TRIANGLE)
172  {
173  throw ::fwGdcmIO::exception::Failed("Primitive type not handled");
174  }
175 
176  // Surface Points Normals
177  const char* normalBuffer = NULL;
178  if (!surface->GetVectorCoordinateData().IsEmpty())
179  {
180  const ::gdcm::DataElement& gdcmNormalCoords = surface->GetVectorCoordinateData();
181  const ::gdcm::ByteValue* gdcmNormalCoordsBV = gdcmNormalCoords.GetByteValue();
182 
183  // Check that the surface contains normals
184  if(gdcmNormalCoordsBV == 0 || gdcmNormalCoordsBV->GetPointer() == 0)
185  {
186  throw ::fwGdcmIO::exception::Failed("No normal coordinates data");
187  }
188 
189  // Set normal buffer
190  normalBuffer = gdcmNormalCoordsBV->GetPointer();
191 
192  // Get size of buffer (bytes)
193  const unsigned long normalCoordBufferSize = gdcmNormalCoordsBV->GetLength();
194 
195  // Compute number of coordinates
196  const unsigned long normalCoordSize = normalCoordBufferSize / sizeof(float);
197 
198  // Check that the buffer has the correct size according to the number of normals
199  if ( (normalCoordSize / 3) != surface->GetNumberOfVectors() && normalCoordSize != coordSize)
200  {
201  throw ::fwGdcmIO::exception::Failed("Corrupted normal coordinates data");
202  }
203  }
204 
205  // Triangle Point Index List
206  const ::gdcm::DataElement& gdcmPointIndex = gdcmMeshPrimitive->GetPrimitiveData();
207  const ::gdcm::ByteValue* gdcmPointIndexBV = gdcmPointIndex.GetByteValue();
208  const char* indexBuffer = gdcmPointIndexBV->GetPointer();
209 
210  // Check that the surface contains triangle point list
211  if (indexBuffer == 0)
212  {
213  throw ::fwGdcmIO::exception::Failed("No triangle point index list");
214  }
215 
216  // Get size of buffer (bytes)
217  const unsigned long indexBufferSize = gdcmPointIndexBV->GetLength();
218 
219  // Get number of primitives
220  const unsigned long indexSize = indexBufferSize / sizeof (uint32_t);
221 
222  OSLM_TRACE("Reconstruction's number of points : " << coordSize / 3 );
223  OSLM_TRACE("Reconstruction's number of cells : " << indexSize / 3 );
224 
225  // Create a new Mesh
226  ::fwGdcmIO::container::DicomSurface surfaceContainer;
227  ::fwData::Mesh::sptr mesh = surfaceContainer.convertToData(reinterpret_cast<const float*>(coordBuffer), coordSize,
228  reinterpret_cast<const uint32_t*>(indexBuffer),
229  indexSize, reinterpret_cast<const float*>(normalBuffer));
230 
231  // Set the reconstruction
232  reconstruction->setMaterial( material );
233  reconstruction->setMesh( mesh );
234 
235 }
