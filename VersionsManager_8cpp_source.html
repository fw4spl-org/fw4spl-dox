00093 void VersionsManager::generateNewFile(const ::boost::filesystem::path& filePath,
00094                                       const std::string& context, const std::string& versionName)
00095 {
00096     FW_RAISE_EXCEPTION_IF( ::fwAtomsPatch::exceptions::BadExtension(".versions file required"),

00101     pt::ptree root;

00103     root.put("context", context);

00107     for (int i = 0; i < classCount; ++i)
00108     {
00109         const ::camp::Class& metaclass = ::camp::classByIndex(i);

00111 
00112         if (metaclass.hasTag(::fwAtomsPatch::s_OBJ_VERSION))
00113         {
00114             const ::camp::Value& value = metaclass.tag(::fwAtomsPatch::s_OBJ_VERSION);
00115             versions.put(className, value.to<std::string>());
00116         }
00117     }
00118 
00119     root.push_back( pt::ptree::value_type("versions", pt::ptree(versions)));
00120     pt::json_parser::write_json(filePath.string(), root);

00172 
00173     const std::string& patcher = root.get("patcher", "DefaultPatcher");
00174 
00175     BOOST_FOREACH(pt::ptree::value_type &child, root.get_child("links"))
00176     {
00177         BOOST_FOREACH(pt::ptree::value_type &node, (child.second).get_child(""))

00179             link.push_back(std::make_pair(node.first, node.second.data()));

00181 
00182         FW_RAISE_EXCEPTION_IF(::fwAtomsPatch::exceptions::MissingInformation(
00183                         "A link should contain an origin version and a target version."), link.size() != 2);
00184 
00185         links[link[0]] = link[1];
00186 
00187         // Clear the vector
00188         LinkType().swap(link);
00189     }
00190 
00191     file.close();
00192 
00193     return ::fwAtomsPatch::LinkDescriptor(context, originVersion, targetVersion, patcher, links);
