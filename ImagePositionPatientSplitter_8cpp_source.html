59 ImagePositionPatientSplitter::DicomSeriesContainerType ImagePositionPatientSplitter::apply(
60  const ::fwMedData::DicomSeries::sptr& series, const ::fwLog::Logger::sptr& logger)
61 const throw(::fwDicomIOFilter::exceptions::FilterFailure)
62 {
63  DicomSeriesContainerType result;
64 
65  DcmFileFormat fileFormat;
66  OFCondition status;
67  DcmDataset* dataset;
68 
69  double previousIndex = 0;
70  unsigned int instanceNumber = 0;
71  double spacingBetweenSlices = 0;
72  const double epsilon = 1e-2; // Value used to find a gap
73  ::fwMedData::DicomSeries::sptr currentSeries;
74  for(const ::fwMedData::DicomSeries::DicomPathContainerType::value_type& file : series->getLocalDicomPaths())
75  {
76  const std::string& filename = file.second.string();
77  status = fileFormat.loadFile(filename.c_str());
78  FW_RAISE_IF("Unable to read the file: \""+filename+"\"", status.bad());
79  dataset = fileFormat.getDataset();
80 
81  if(!dataset->tagExists(DCM_ImagePositionPatient) || !dataset->tagExists(DCM_ImageOrientationPatient))
82  {
83  const std::string msg =
84  "Unable to split the series using ImagePositionPatient and ImageOrientationPatient. "
85  "Tag(s) missing.";
86  throw ::fwDicomIOFilter::exceptions::FilterFailure(msg);
87  }
88 
89 
90  fwVec3d imagePosition;
91  for(unsigned int i = 0; i < 3; ++i)
92  {
93  dataset->findAndGetFloat64(DCM_ImagePositionPatient, imagePosition[i], i);
94  }
95 
96  fwVec3d imageOrientationU;
97  fwVec3d imageOrientationV;
98  for(unsigned int i = 0; i < 3; ++i)
99  {
100  dataset->findAndGetFloat64(DCM_ImageOrientationPatient, imageOrientationU[i], i);
101  dataset->findAndGetFloat64(DCM_ImageOrientationPatient, imageOrientationV[i], i+3);
102  }
103 
104  //Compute Z direction (cross product)
105  fwVec3d zVector = ::fwMath::cross(imageOrientationU, imageOrientationV);
106 
107  //Compute dot product to get the index
108  double index = ::fwMath::dot(imagePosition, zVector);
109 
110  //Compute spacing
111  double spacing = index - previousIndex;
112  if(currentSeries && spacingBetweenSlices == 0)
113  {
114  spacingBetweenSlices = spacing;
115  }
116 
117  // First frame or volume detected: We create a new Series
118  if(!currentSeries || (fabs(spacing-spacingBetweenSlices) > epsilon) )
119  {
120  if(currentSeries)
121  {
122  result.push_back(currentSeries);
123  currentSeries->setNumberOfInstances((unsigned int)(currentSeries->getLocalDicomPaths().size()));
124  }
125  instanceNumber = 0;
126  currentSeries = ::fwMedData::DicomSeries::New();
127  currentSeries->deepCopy(series);
128  ::fwMedData::DicomSeries::DicomPathContainerType dicomPathContainer;
129  currentSeries->setLocalDicomPaths(dicomPathContainer);
130  }
131 
132  currentSeries->addDicomPath(instanceNumber++, filename);
133  previousIndex = index;
134 
135  }
136 
137  // Push last series created
138  result.push_back(currentSeries);
139  currentSeries->setNumberOfInstances((unsigned int)(currentSeries->getLocalDicomPaths().size()));
140 
141  if(result.size() > 1)
142  {
143  logger->warning("Series has been split according to slice positions.");
144  }
145 
146  return result;
147 
148 }
