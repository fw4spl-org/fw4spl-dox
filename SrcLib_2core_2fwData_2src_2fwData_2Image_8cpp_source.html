00162         ("int32",  ::fwTools::makeDynamicType<std::string>("signed int")     )
00163         ("float",  ::fwTools::makeDynamicType<std::string>("float")          )
00164         ("double", ::fwTools::makeDynamicType<std::string>("double")         )
00165 
00166 //special case for dynamic type : 64bits integers was not managed by dynamic type.
00167 #if ( INT_MAX < LONG_MAX )
00168         ("uint64", ::fwTools::makeDynamicType<std::string>("unsigned long")  )
00169         ("int64",  ::fwTools::makeDynamicType<std::string>("signed long")    )
00170 #else
00171         ("uint32", ::fwTools::makeDynamicType<std::string>("unsigned long")  )
00172         ("int32",  ::fwTools::makeDynamicType<std::string>("signed long")    )
00173         ("uint64", ::fwTools::DynamicType() )
00174         ("int64",  ::fwTools::DynamicType() )
00175 #endif
00176         ;
00177 
00178     ::fwTools::DynamicType dtype = dynamicTypeMap[getType().string()] ;
