176 void MeshesBoxWidget::updateFromVtk()
177 {
178  m_vtkBoxWidget->RemoveObserver( m_boxWidgetCommand );
179 
180  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
181 
182  vtkBoxRepresentation *boxRep = vtkBoxRepresentation::SafeDownCast( m_vtkBoxWidget->GetRepresentation() );
183  vtkTransform * boxTransform = vtkTransform::New();
184  boxRep->GetTransform(boxTransform);
185 
186  for(::fwData::Composite::value_type elt : *composite)
187  {
188  ::fwData::Mesh::sptr mesh = ::fwData::Mesh::dynamicCast(elt.second);
189  ::fwData::TransformationMatrix3D::sptr fieldTransform;
190  SLM_ASSERT("Mesh must have a TransformMatrix field", mesh->getField("TransformMatrix"));
191  fieldTransform = mesh->getField< ::fwData::TransformationMatrix3D > ("TransformMatrix");
192 
193  vtkTransform * transform = vtkTransform::New();
194  vtkLinearTransform * meshTransform = m_meshMap[elt.first]->GetUserTransform();
195  transform->Concatenate(boxTransform);
196  transform->Concatenate(meshTransform);
197 
198  vtkMatrix4x4* mat = transform->GetMatrix();
199  for(int lt = 0; lt<4; lt++)
200  {
201  for(int ct = 0; ct<4; ct++)
202  {
203  fieldTransform->setCoefficient(lt,ct, mat->GetElement(lt,ct));
204  }
205  }
206 
207  auto sig = fieldTransform->signal< ::fwData::Object::ModifiedSignalType >(
208  ::fwData::Object::s_MODIFIED_SIG);
209  sig->asyncEmit();
210 
211  transform->Delete();
212  }
213  m_vtkBoxWidget->AddObserver( ::vtkCommand::InteractionEvent, m_boxWidgetCommand );
214  boxTransform->Delete();
215 }

219 void MeshesBoxWidget::updateMeshMapFromComposite(::fwData::Composite::ContainerType objects)
220 {
221  for(::fwData::Composite::value_type elt : objects)
222  {
223  ::fwData::Mesh::sptr mesh = ::fwData::Mesh::dynamicCast(elt.second);
224  vtkSmartPointer<vtkPolyData> vtkMesh = vtkSmartPointer<vtkPolyData>::New();
225  ::fwVtkIO::helper::Mesh::toVTKMesh( mesh, vtkMesh);
226 
227  ::fwData::TransformationMatrix3D::sptr fieldTransform;
228  fieldTransform = mesh->setDefaultField("TransformMatrix", ::fwData::TransformationMatrix3D::New());
229 
230  vtkSmartPointer<vtkTransform> transform = vtkSmartPointer<vtkTransform>::New();
231  transform->Identity();
232 
233  vtkSmartPointer<vtkMatrix4x4> matrix = vtkSmartPointer<vtkMatrix4x4>::New();
234 
235  for(int lt = 0; lt<4; lt++)
236  {
237  for(int ct = 0; ct<4; ct++)
238  {
239  matrix->SetElement(lt,ct, fieldTransform->getCoefficient(lt,ct));
240  }
241  }
242 
243  transform->SetMatrix(matrix);
244  vtkSmartPointer<vtkPolyDataMapper> meshMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
245  meshMapper->SetInputData(vtkMesh);
246 
247  vtkActor *meshActor = vtkActor::New();
248  meshActor->SetMapper(meshMapper);
249  meshActor->SetUserTransform(transform);
250 
251  if (m_meshMap.find(elt.first) == m_meshMap.end())
252  {
253  ::fwCom::Connection connection = fieldTransform->signal(::fwData::Object::s_MODIFIED_SIG)->
254  connect(this->slot(s_UPDATE_MATRICES_SLOT));
255  m_connections[elt.first] = connection;
256  }
257 
258  m_meshMap[elt.first] = meshActor;
259  }
260 }
