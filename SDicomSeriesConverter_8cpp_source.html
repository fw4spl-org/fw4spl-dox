77 void SDicomSeriesConverter::updating() throw(::fwTools::Failed)
78 {
79  ::fwMedData::SeriesDB::csptr dicomSeriesDB = this->getInput< ::fwMedData::SeriesDB >("source");
80  ::fwMedData::SeriesDB::sptr dummy = ::fwMedData::SeriesDB::New();
81 
82  if(dicomSeriesDB->empty())
83  {
84  ::fwGui::dialog::MessageDialog messageBox;
85  messageBox.setIcon(::fwGui::dialog::IMessageDialog::INFO);
86  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
87  messageBox.setTitle("Read DICOM series");
88  messageBox.setMessage( "There is no DICOM series that can be read." );
89  messageBox.show();
90  }
91  else
92  {
93  ::fwGdcmIO::reader::SeriesDB::sptr reader = ::fwGdcmIO::reader::SeriesDB::New();
94  reader->setObject(dummy);
95  auto job = reader->getJob();
96  m_sigJobCreated->emit(job);
97 
98  try
99  {
100  reader->readFromDicomSeriesDB(dicomSeriesDB, this->getSptr());
101 
102  ::fwLog::Logger::sptr logger = reader->getLogger();
103  logger->sort();
104 
105  std::stringstream ss;
106  ss << "The reading process is over : <b>" << dummy->size() << " series</b> "
107  << ((dummy->size() > 1) ? "have" : "has") << " been found. "
108  << "<br>Please verify the log report to be informed of the potential errors.";
109 
110  bool result = false;
111  if(!job->cancelRequested())
112  {
113  result = ::fwGui::dialog::LoggerDialog::showLoggerDialog("Reading process over", ss.str(), logger);
114  }
115 
116  // If the user cancel the reading process we delete the loaded series
117  if(!result || job->cancelRequested())
118  {
119  ::fwMedDataTools::helper::SeriesDB sDBhelper(dummy);
120  sDBhelper.clear();
121  }
122  else
123  {
124  ::fwMedDataTools::helper::SeriesDB sDBhelper(m_destinationSeriesDB);
125  sDBhelper.merge(dummy);
126  sDBhelper.notify();
127  }
128  }
129  catch (const std::exception& e)
130  {
131  std::stringstream ss;
132  ss << "Warning during loading : " << e.what();
133  ::fwGui::dialog::MessageDialog::showMessageDialog(
134  "Warning", ss.str(), ::fwGui::dialog::IMessageDialog::WARNING);
135  }
136  catch( ... )
137  {
138  ::fwGui::dialog::MessageDialog::showMessageDialog(
139  "Warning", "Warning during loading", ::fwGui::dialog::IMessageDialog::WARNING);
140  }
141  }
142 }
