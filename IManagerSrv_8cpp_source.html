00058 
00059     BOOST_FOREACH(::fwRuntime::ConfigurationElement::csptr elem,  config->getElements())
00060     {
00061         SLM_ASSERT("Bad tag '" <<elem->getName() << "', only <signal> or <slot> are allowed.",
00062                    elem->getName() == "signal" || elem->getName() == "slot");
00063 
00064         src = elem->getValue();
00065         if( ::boost::regex_match(src, match, re) )
00066         {
00067             OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
00068             uid.assign(match[1].first, match[1].second);
00069             key.assign(match[2].first, match[2].second);
00070 
00071             OSLM_ASSERT(src << " configuration is not correct for "<< elem->getName() ,
00072                         !uid.empty() && !key.empty());
00073 
00074             if (elem->getName() == "signal")
00075             {
00076                 SLM_ASSERT("There must be only one signal by connection",
00077                            signalInfo.first.empty() && signalInfo.second.empty());
00078                 signalInfo = std::make_pair(uid, key);
00079             }
00080             else if (elem->getName() == "slot")
00081             {
00082                 slotInfos.push_back( std::make_pair(uid, key) );
00083             }
00084         }
00085         else
00086         {
00087             uid = object->getID();
00088             key = src;
00089             SLM_ASSERT("Element must be a signal or must be written as <fwID/Key>", elem->getName() == "signal");
00090             SLM_ASSERT("There must be only one signal by connection",
00091                        signalInfo.first.empty() && signalInfo.second.empty());
00092             signalInfo = std::make_pair(uid, key);
00093         }
00094     }
00095

00098

00100     ObjectConnectionsMapType::iterator iter = m_objectConnections.find(objectId);
00101     if (iter != m_objectConnections.end())
00102     {
00103         connection = iter->second;
00104     }
00105     else
00106     {
00107         connection = ::fwServices::helper::SigSlotConnection::New();
00108         m_objectConnections[objectId] = connection;
00109     }
00110 
00111     BOOST_FOREACH(SlotInfoType slotInfo,  slotInfos)
00112     {
00113         ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(slotInfo.first);

00115

00117     }
