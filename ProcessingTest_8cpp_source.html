50  activeWorkers->initRegistry();

52  typedef signed short ImageType;
53  const int sizeX = 50;
54  const int sizeY = 50;
55  const int sizeZ = 50;
56  const int imageSize = sizeX * sizeY * sizeZ;
57 
58  std::string implementation = "::scene2D::processing::SComputeHistogram";
59 
60  // Configure data hirearchy
61  ::fwData::Image::sptr image = ::fwData::Image::New();
62  ::fwData::Histogram::sptr histogram = ::fwData::Histogram::New();
63 
64  // Create image.
65  image->allocate(sizeX, sizeY, sizeZ, ::fwTools::Type("int16"));
66  ::fwData::Array::sptr array = image->getDataArray();
67  ::fwComEd::helper::Array arrayHelper(array);
68 
69  ImageType* itr = arrayHelper.begin< ImageType >();
70  ImageType* itrEnd = arrayHelper.end< ImageType >();
71 
72  int count = 0;
73  for(; itr!= itrEnd; ++itr)
74  {
75  if(count < imageSize/4)
76  {
77  *itr = 10;
78  }
79  else if( count < imageSize/2)
80  {
81  *itr = 20;
82  }
83  else if( count < 3*imageSize/4)
84  {
85  *itr = 30;
86  }
87  else if( count < imageSize)
88  {
89  *itr = 40;
90  }
91  ++count;
92  }
93 
94  // Create service.
95  ::fwServices::IService::sptr srv;
96  srv = ::fwServices::registry::ServiceFactory::getDefault()->create( implementation );
97  CPPUNIT_ASSERT(srv);
98 
99  // Register service.
100  ::fwServices::OSR::registerService( image, srv );
101 
102  ::fwRuntime::EConfigurationElement::sptr srvCfg = ::fwRuntime::EConfigurationElement::New("service");
103 
104  ::fwRuntime::EConfigurationElement::sptr histogramIdCfg = ::fwRuntime::EConfigurationElement::New("histogramId");
105  histogramIdCfg->setValue(histogram->getID());
106  srvCfg->addConfigurationElement(histogramIdCfg);
107 
108  ::fwRuntime::EConfigurationElement::sptr binsWidthCfg = ::fwRuntime::EConfigurationElement::New("binsWidth");
109  binsWidthCfg->setValue("1.0");
110  srvCfg->addConfigurationElement(binsWidthCfg);
111 
112  srv->setConfiguration(srvCfg);
113  srv->configure();
114  srv->start().wait();
115  srv->stop().wait();
116  ::fwServices::OSR::unregisterService(srv);
117 
118  ::fwData::Histogram::fwHistogramValues values = histogram->getValues();

120 
121  CPPUNIT_ASSERT_EQUAL((float) 10, histogram->getMinValue());
122 
123  CPPUNIT_ASSERT_EQUAL((float) 40, histogram->getMaxValue());
124 
125  CPPUNIT_ASSERT_EQUAL((long) imageSize/4, histogram->getNbPixels(10,11));
126  CPPUNIT_ASSERT_EQUAL((long) imageSize/4, histogram->getNbPixels(20,21));
127  CPPUNIT_ASSERT_EQUAL((long) imageSize/4, histogram->getNbPixels(30,31));
128  CPPUNIT_ASSERT_EQUAL((long) imageSize/4, histogram->getNbPixels(40,41));
