00151 {
00152     SeriesVectorType vSeries;
00153     QStandardItem* item = m_model->itemFromIndex(index);
00154     int nbRow = item->rowCount();
00155     for(int row =0; row < nbRow; ++row)

00160         ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(uid);
00161         ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(obj);
