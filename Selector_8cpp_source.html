114 Selector::SeriesVectorType Selector::getSeries(const QModelIndexList& indexList)
115 {
116  SeriesVectorType vSeries;
117  for(QModelIndex index : indexList)
118  {
119  std::string uid = index.data(SelectorModel::UID).toString().toStdString();
120  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(uid);
121 
122  if (index.data(SelectorModel::ITEM_TYPE) == SelectorModel::SERIES)
123  {
124  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(obj);
125  vSeries.push_back(series);
126  }
127  }
128  return vSeries;

148 Selector::SeriesVectorType Selector::getSeriesFromStudyIndex(const QModelIndex& index )
149 {
150  SeriesVectorType vSeries;
151  QStandardItem* item = m_model->itemFromIndex(index);
152  int nbRow = item->rowCount();
153  for(int row = 0; row < nbRow; ++row)
154  {
155  QStandardItem *child = item->child(row);
156  std::string uid = child->data(SelectorModel::UID).toString().toStdString();
157  SLM_ASSERT("UID must not be empty.", !uid.empty());
158  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(uid);
159  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(obj);
160  vSeries.push_back(series);
161  }
162  return vSeries;
163 }

189 void Selector::deleteSelection()
190 {
191  QModelIndexList selection = this->selectionModel()->selectedRows(0);
192 
193  SeriesVectorType vSeries = this->getSeries(selection);
194  QModelIndexList studyIndexes = this->getStudyIndexes(selection);
195  for(QModelIndex index : studyIndexes)
196  {
197  SeriesVectorType series = getSeriesFromStudyIndex(index);
198  std::copy(series.begin(), series.end(), std::back_inserter(vSeries));
199  }
200 
201  Q_EMIT removeSeries(vSeries);
202 
203  // Remove item in Selector.
204  m_model->removeRows(selection);
205 }
