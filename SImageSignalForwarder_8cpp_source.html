99 void SImageSignalForwarder::configuring() throw ( ::fwTools::Failed )
100 {
101  SLM_TRACE_FUNC();
102 
103  ::fwRuntime::ConfigurationElement::sptr fromUidCfg = m_configuration->findConfigurationElement("fromUid");
104  if (fromUidCfg)
105  {
106  m_sourceImageUid = fromUidCfg->getValue();
107  }
108  ::fwRuntime::ConfigurationElement::sptr fromKeyCfg = m_configuration->findConfigurationElement("fromKey");
109  if (fromKeyCfg)
110  {
111  std::string src = fromKeyCfg->getValue();
112  ::boost::regex re("(.*)\\[(.*)\\]");
113  ::boost::smatch match;
114  std::string uid, key;
115 
116  if( ::boost::regex_match(src, match, re) )
117  {
118  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
119  uid.assign(match[1].first, match[1].second);
120  key.assign(match[2].first, match[2].second);
121  }
122  m_sourceCompoUid = uid;
123  m_sourceImageKey = key;
124  }
125 
126  SLM_ASSERT("only one tag 'fromUid' or 'fromKey' must be defined.",
127  (!m_sourceImageUid.empty() && m_sourceCompoUid.empty() && m_sourceImageKey.empty()) ||
128  (m_sourceImageUid.empty() && !m_sourceCompoUid.empty() && !m_sourceImageKey.empty()));
129 
130  std::vector< ::fwRuntime::ConfigurationElement::sptr > configs = m_configuration->find("forward");
131 
132  for (auto cfg : configs)
133  {
134  std::string signal = cfg->getValue();
135  SLM_ASSERT("Signal '" + signal + "' is unknown.",
136  m_availableConnection.find(signal) != m_availableConnection.end());
137  m_managedSignals.push_back(signal);
138  }
139 }

143 void SImageSignalForwarder::starting() throw ( ::fwTools::Failed )
144 {
145  m_connections = ::fwServices::helper::SigSlotConnection::New();
146  ::fwData::Image::sptr src;
147  if (!m_sourceImageUid.empty())
148  {
149  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_sourceImageUid);
150  src = ::fwData::Image::dynamicCast(obj);
151  SLM_ASSERT("Source image '" + m_sourceImageUid + "' is not found", src);
152  }
153  else
154  {
155  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_sourceCompoUid);
156  ::fwData::Composite::sptr compo = ::fwData::Composite::dynamicCast(obj);
157  src = compo->at< ::fwData::Image >(m_sourceImageKey);
158  }
159 
160  for (auto signalKey: m_managedSignals)
161  {
162  m_connections->connect(src, signalKey, this->getSptr(), m_availableConnection[signalKey]);
163  }
164 }
