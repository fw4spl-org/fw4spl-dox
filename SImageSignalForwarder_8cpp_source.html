101 void SImageSignalForwarder::configuring() throw ( ::fwTools::Failed )
102 {
103  if (!this->isVersion2())
104  {
105  ::fwRuntime::ConfigurationElement::sptr fromUidCfg = m_configuration->findConfigurationElement("fromUid");
106  if (fromUidCfg)
107  {
108  m_sourceImageUid = fromUidCfg->getValue();
109  }
110  ::fwRuntime::ConfigurationElement::sptr fromKeyCfg = m_configuration->findConfigurationElement("fromKey");
111  if (fromKeyCfg)
112  {
113  std::string src = fromKeyCfg->getValue();
114  ::boost::regex re("(.*)\\[(.*)\\]");
115  ::boost::smatch match;
116  std::string uid, key;
117 
118  if( ::boost::regex_match(src, match, re) )
119  {
120  OSLM_ASSERT("Wrong value for attribute src: "<<src, match.size() >= 3);
121  uid.assign(match[1].first, match[1].second);
122  key.assign(match[2].first, match[2].second);
123  }
124  m_sourceCompoUid = uid;
125  m_sourceImageKey = key;
126  }
127 
128  SLM_ASSERT("only one tag 'fromUid' or 'fromKey' must be defined.",
129  (!m_sourceImageUid.empty() && m_sourceCompoUid.empty() && m_sourceImageKey.empty()) ||
130  (m_sourceImageUid.empty() && !m_sourceCompoUid.empty() && !m_sourceImageKey.empty()));
131  }
132 
133  std::vector< ::fwRuntime::ConfigurationElement::sptr > configs = m_configuration->find("forward");
134 
135  for (auto cfg : configs)
136  {
137  std::string signal = cfg->getValue();
138  SLM_ASSERT("Signal '" + signal + "' is unknown.",
139  m_availableConnection.find(signal) != m_availableConnection.end());
140  m_managedSignals.push_back(signal);
141  }
142 }

146 void SImageSignalForwarder::starting() throw ( ::fwTools::Failed )
147 {
148  ::fwData::Image::csptr src;
149 
150  if (!this->isVersion2())
151  {
152  if (!m_sourceImageUid.empty())
153  {
154  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_sourceImageUid);
155  src = ::fwData::Image::dynamicCast(obj);
156  SLM_ASSERT("Source image '" + m_sourceImageUid + "' is not found", src);
157  }
158  else
159  {
160  ::fwTools::Object::sptr obj = ::fwTools::fwID::getObject(m_sourceCompoUid);
161  ::fwData::Composite::sptr compo = ::fwData::Composite::dynamicCast(obj);
162  src = compo->at< ::fwData::Image >(m_sourceImageKey);
163  }
164  }
165  else
166  {
167  src = this->getInput< ::fwData::Image >(s_SOURCE_INPUT);
168  }
169 
170  for (auto signalKey: m_managedSignals)
171  {
172  m_connections.connect(src, signalKey, this->getSptr(), m_availableConnection[signalKey]);
173  }
174 }
