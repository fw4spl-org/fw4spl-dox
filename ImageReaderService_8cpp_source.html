00150     if(ext == ".vtk")
00151     {
00152         ::fwVtkIO::ImageReader::sptr vtkReader = ::fwVtkIO::ImageReader::New();

00158     else if(ext == ".vti")

00160         ::fwVtkIO::VtiImageReader::sptr vtiReader = ::fwVtkIO::VtiImageReader::New();

00163         myReader = vtiReader;

00165     else if(ext == ".mhd")

00167         ::fwVtkIO::MetaImageReader::sptr mhdReader = ::fwVtkIO::MetaImageReader::New();

00170         myReader = mhdReader;
00171     }
00172     else
00173     {
00174         FW_RAISE_EXCEPTION(::fwTools::Failed("Only .vtk, .vti and .mhd are supported."));

00176 
00177     // Set the image (already created, but empty) that will be modified
00178     myReader->setObject(_pImg);
00179

00181     {
00182         // Launch reading process

00184     }
00185     catch (const std::exception & e)

00187         std::stringstream ss;
00188         ss << "Warning during loading : " << e.what();
00189

00193                 ::fwGui::dialog::IMessageDialog::WARNING);
00194         ok = false;
