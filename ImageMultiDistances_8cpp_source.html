00297 ::fwData::Point::sptr ImageMultiDistances::screenToWorld(int X,int Y)
00298 {
00299     double *world;
00300     double display[3];
00301     double worldTmp[4];
00302     display[0]=X;
00303     display[1]=Y;
00304     display[2]=0;
00305 
00306     if ( this->getPicker() && this->getPicker()->Pick( display , this->getRenderer() ) )
00307     {
00308         world = this->getPicker()->GetPickPosition();
00309     }
00310     else
00311     {
00312         // set temporaly the clipping around the focal point : see (1)
00313         vtkCamera *camera = this->getRenderer()->GetActiveCamera();
00314         double *clippingCamBackup = camera->GetClippingRange();
00315         camera->SetClippingRange( camera->GetDistance() - 0.1 , camera->GetDistance() + 0.1 ); // set the clipping around the focal point
00316 
00317         world=worldTmp;
00318         // (1) this function use the near clipping range to estimate the world point (by defaut 0.1 from camera view). The clipping can be modified
00319         // by insertion of new object. By setting previously the clipping to the focal point we ensure to not place new point a camera position
00320         this->getInteractor()->GetInteractorStyle()->ComputeDisplayToWorld ( this->getRenderer(), X , Y , 0 , world); // RETURN HOMOGEN COORD !!!
00321 
00322         // restore initial clipping
00323         camera->SetClippingRange( clippingCamBackup );
00324     }
00325 
00326     ::fwData::Point::sptr pt = ::fwData::Point::New();
00327     std::copy( world, world +3, pt->getRefCoord().begin() );
00328     this->setVtkPipelineModified();
00329     return pt;
00330 }
