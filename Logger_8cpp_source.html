00039 Logger::Logger()
00040 {
00041     namespace expr = ::boost::log::expressions;
00042     namespace keywords = ::boost::log::keywords;
00043 
00044     ::boost::log::add_file_log (
00045                 // file name pattern
00046                 keywords::file_name = "PATCH.log",
00047                 // rotate files every 10 MiB...
00048                 keywords::rotation_size = 10 * 1024 * 1024,
00049                 // ...or at midnight
00050                 keywords::time_based_rotation = ::boost::log::sinks::file::rotation_at_time_point(0, 0, 0),
00051                 // log record format
00052                 keywords::format = (
00053                         expr::stream
00054                         << "[" << expr::format_date_time< ::boost::posix_time::ptime >("TimeStamp", "%d.%m.%Y %H:%M:%S.%f")
00055                         << "][" << expr::format_date_time< ::boost::posix_time::ptime >("Uptime", "%H:%M:%S.%f")
00056                         << "][" << expr::attr< std::string >("Channel")
00057                         << "] " << expr::smessage
00058                 ),
00059                 // auto-flush feature of the backend
00060                 keywords::auto_flush = true,
00061                 keywords::filter = (expr::attr< std::string >("Channel") == "patch")
00062     );
00063 
00064     // Construct the sink
00065     typedef ::boost::log::sinks::synchronous_sink< ::boost::log::sinks::text_ostream_backend > text_sink;
00066     boost::shared_ptr< text_sink > pSink = boost::make_shared< text_sink >();
00067 
00068     // Add a stream to write log to
00069     pSink->locked_backend()->add_stream(s_stream);
00070 
00071     // Register the sink in the logging core
00072     ::boost::log::core::get()->add_sink(pSink);
