148 void MeshNormals::updateMeshNormals()
149 {
150  ::fwData::Mesh::sptr mesh = this->getObject < ::fwData::Mesh >();
151 
152  if(m_normalRepresentation == NONE)
153  {
154  m_actor->SetVisibility( false );
155  }
156  else
157  {
158  m_polyData = vtkSmartPointer< vtkPolyData >::New();
159  ::fwVtkIO::helper::Mesh::toVTKMesh(mesh, m_polyData);
160 
161  vtkSmartPointer<vtkPolyDataAlgorithm> algo;
162  if(m_normalRepresentation == CELL_NORMAL)
163  {
164  algo = vtkSmartPointer<vtkCellCenters>::New();
165  }
166  else if(m_normalRepresentation == POINT_NORMAL)
167  {
168  vtkSmartPointer<vtkMaskPoints> ptMask = vtkSmartPointer<vtkMaskPoints>::New();
169  ptMask->SetOnRatio(1);
170  ptMask->RandomModeOn();
171  ptMask->SetMaximumNumberOfPoints(mesh->getNumberOfPoints());
172  algo = ptMask;
173  }
174 
175  algo->SetInputData(m_polyData);
176 
177  vtkSmartPointer<vtkGlyphSource2D> arrow = vtkSmartPointer<vtkGlyphSource2D>::New();
178  arrow->SetGlyphTypeToArrow ();
179  arrow->FilledOff();
180 
181  vtkSmartPointer<vtkGlyph3D> glyph = vtkSmartPointer<vtkGlyph3D>::New();
182  glyph->SetInputConnection(algo->GetOutputPort());
183  glyph->SetSourceConnection(arrow->GetOutputPort());
184  glyph->SetVectorModeToUseNormal();
185  glyph->SetScaleModeToScaleByVector();
186  glyph->SetScaleFactor(10.0);
187 
188  vtkSmartPointer<vtkPolyDataMapper> glyphMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
189  glyphMapper->SetInputConnection(glyph->GetOutputPort());
190 
191  m_actor->SetVisibility( true );
192  this->getActor()->SetMapper(glyphMapper);
193  this->getActor()->GetProperty()->SetColor(1., 1., 1.);
194  }
195  this->setVtkPipelineModified();
196 }
