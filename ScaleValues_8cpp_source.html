39 void ScaleValues::configuring() throw ( ::fwTools::Failed )
40 {
41  SLM_TRACE_FUNC();
42 
43  SLM_ASSERT("\"config\" tag missing", m_configuration->getName() == "config");
44 
45  this->IAdaptor::configuring(); // xAxis, yAxis, zValue, opacity
46 
47  SLM_TRACE("IAdaptor configuring ok");
48 
49  // Set the min/max values
50  const std::string min = m_configuration->getAttributeValue("min");
51  const std::string max = m_configuration->getAttributeValue("max");
52  SLM_ASSERT("'min' atttribute is missing.", !min.empty());
53  SLM_ASSERT("'min' atttribute is missing.", !max.empty());
54 
55  m_min = ::boost::lexical_cast< double >( min );
56  m_max = ::boost::lexical_cast< double >( max );
57 
58 
59  // Interval configuration
60  if (!m_configuration->getAttributeValue("interval").empty())
61  {
62  m_interval = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("interval") );
63  }
64 
65 
66  // Color configuration
67  const std::string color = m_configuration->getAttributeValue("color");
68  if (!color.empty())
69  {
70  ::scene2D::data::InitQtPen::setPenColor(m_pen, color, m_opacity);
71  }
72  else
73  {
74  ::scene2D::data::InitQtPen::setPenColor(m_pen, "white", m_opacity);
75  }
76 
77 
78  // Font size configuratiion
79  if (!m_configuration->getAttributeValue("fontSize").empty())
80  {
81  m_fontSize = ::boost::lexical_cast< float >( m_configuration->getAttributeValue("fontSize") );
82  }
83 
84 
85  // Viewport configuratiion
86  SLM_ASSERT("A viewport attribute must be specified with 'viewportUID'.",
87  !m_configuration->getAttributeValue("viewportUID").empty());
88 
89  if( !m_configuration->getAttributeValue("viewportUID").empty() )
90  {
91  m_viewportID = m_configuration->getAttributeValue("viewportUID");
92  }
93 
94 
95  // Show unit

97  if(showUnit == "true" || showUnit == "false")
98  {
99  m_showUnit = (showUnit == "true");
100  }
101 
102 
103  // Unit text configuration
104  m_displayedUnit = m_configuration->getAttributeValue("unit");
105 
106 
107  // 'align' attribute configuration
108  m_align = m_configuration->getAttributeValue("align");
109 
110  SLM_ASSERT(

115  m_align == "left" || m_align == "right" || m_align == "top" || m_align == "bottom");

120 void ScaleValues::buildValues()
121 {
122  SLM_TRACE_FUNC();
123  SLM_ASSERT("m_interval can not be equal to 0", m_interval != 0);
124 
125  m_values.clear();
126 
127  double val = this->getStartVal();
128  const int range = (int) ceil(this->getEndVal() - val);
129  const int nbValues = (int)(ceil(range/ m_interval)) + 1;
130 
131  std::string format;
132 
133  if(m_align == "left")
134  {
135  format = " %1";
136  }
137  else if(m_align == "right")
138  {
139  format = "%1 ";
140  }
141  else
142  {
143  format = "%1";
144  }
145 
146  // Build the values as graphic items
147  for(int i = 0; i < nbValues; ++i, val += m_interval)
148  {
149  QGraphicsSimpleTextItem* text = new QGraphicsSimpleTextItem();
150  text->setText( QString::fromStdString(format).arg(val) );
151  text->setFont( m_font );
152  text->setBrush( m_brush );
153  text->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
154 
155  m_values.push_back( text );
156  }
157 
158  // Add the values to the item group
159  for( QGraphicsItem* item : m_values )
160  {
161  m_layer->addToGroup( item );
162  }
163 
164  // Build the unit graphic item
165  m_unit = new QGraphicsSimpleTextItem();
166  m_unit->setText( QString::fromStdString( m_displayedUnit ) );
167  m_unit->setFont( m_font );
168  m_unit->setBrush( m_brush );
169  m_unit->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
170 
171  // Add it to the items to be displayed if required
172  if( m_showUnit )
173  {
174  m_layer->addToGroup( m_unit );
175  }
176 
177  // Set the layer position (according to the related axis) and zValue
178  m_layer->setPos( m_xAxis->getOrigin(), m_yAxis->getOrigin() );
179  m_layer->setZValue( m_zValue );
180 
181  // Add the layer containing grid's lines to the scene
182  this->getScene2DRender()->getScene()->addItem( m_layer );
183 }

239 void ScaleValues::rescaleValues()
240 {
241  const double viewportX = m_viewport->getX();
242  const double viewportWidth = m_viewport->getWidth();
243  const double viewportHeight = m_viewport->getHeight();
244 
245  const double viewportSizeRatio = viewportHeight / viewportWidth;
246  const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
247 
248  Scene2DRatio ratio = this->getRatio(); // Total ratio
249  double viewportWidthRatio = this->getViewportSizeRatio().first;
250 
251  double scaleX = m_fontSize;
252  double scaleY = m_fontSize * viewportSizeRatio;
253  scaleY /= viewportWidthRatio;
254  scaleY *= viewInitialSizeRatio;
255 
256  scaleX = scaleX * ratio.first;
257  scaleY = scaleY * ratio.second;
258 
259  QTransform transform;
260  transform.scale( scaleX, scaleY );
261 
262  int step = 0;
263  double valueSize; // the size (width or height) of the current value
264  std::pair<double, double> coord; // coordinates of the current scale value item
265  std::pair<double, double> size; // size of the current scale value item
266  bool suggestResampling = false; /* scale value items resampling is suggested because of
267  a lack of sufficient width to display all of them */
268 
269  const size_t valuesSize = m_values.size();
270  float val = getStartVal();
271 
272  if(m_align == "left" || m_align == "right")
273  {
274  const double valueSizeRatio = m_interval / scaleY;
275 
276  float coeff = 0;
277 
278  double textPosX;
279 
280  if(m_align == "left")
281  {
282  textPosX = viewportX;
283  }
284  else
285  {
286  coeff = -1;
287  textPosX = viewportX + viewportWidth;
288  }
289 
290  for(int i = 0; i < valuesSize; ++i, val += m_interval)
291  {
292  valueSize = m_values[i]->boundingRect().height();
293 
294  size = this->mapAdaptorToScene(
295  std::pair<double, double>(m_values[i]->boundingRect().width(), valueSize),
296  m_xAxis, m_yAxis);
297 
298  step = (int)(valueSize / valueSizeRatio) + 1;
299 
300  if( step > m_step )
301  {
302  m_step = step;
303  suggestResampling = true;
304  }
305 
306  coord = this->mapAdaptorToScene(
307  std::pair<double, double>(textPosX, val), m_xAxis, m_yAxis);
308 
309  m_values[i]->setTransform( transform );
310 
311  m_values[i]->setPos(
312  coord.first + coeff * size.first * scaleX,
313  coord.second - (m_interval - size.second / 2) * scaleY );
314  }
315 
316  m_unit->setTransform( transform );
317 
318  val = viewportHeight * 0.8;
319 
320  coord = this->mapAdaptorToScene(
321  std::pair<double, double>(textPosX, val), m_xAxis, m_yAxis);
322 
323  coeff = (m_align == "left") ? 1 : -1.5;
324 
325  m_unit->setPos(
326  coord.first + coeff * 2 * size.first * scaleX,
327  coord.second + size.second * scaleY);
328  }
329  else // axis centered on top or bottom
330  {
331  const double valueSizeRatio = m_interval / scaleX;
332 
333  float coeff = 0.5;
334 
335  double textPosY = (m_align == "bottom")
336  ? m_viewport->getY()
337  : viewportHeight * 0.9;
338 
339  for(int i = 0; i < valuesSize; ++i, val += m_interval)
340  {
341  valueSize = m_values[i]->boundingRect().width();
342 
343  size = this->mapAdaptorToScene(
344  std::pair<double, double>(valueSize, m_values[i]->boundingRect().height()),
345  m_xAxis, m_yAxis);
346 
347  step = (int)(valueSize / valueSizeRatio) + 1;
348 
349  if( step > m_step )
350  {
351  m_step = step;
352  suggestResampling = true;
353  }
354 
355  coord = this->mapAdaptorToScene(std::pair<double, double>(val, textPosY),
356  m_xAxis, m_yAxis);
357 
358  m_values[i]->setTransform( transform );
359 
360  m_values[i]->setPos(
361  coord.first - size.first / 2 * scaleX,
362  coord.second - coeff * size.second / 2 * scaleY );
363  }
364 
365  m_unit->setTransform( transform );
366 
367  val = viewportHeight * 0.8;
368 
369  size = this->mapAdaptorToScene(
370  std::pair<double, double>(m_unit->boundingRect().width(), m_unit->boundingRect().height()),
371  m_xAxis, m_yAxis);
372 
373  coord = this->mapAdaptorToScene(
374  std::pair<double, double>(
375  viewportX + viewportWidth / 2, textPosY),
376  m_xAxis, m_yAxis);
377 
378  coeff = (m_align == "left") ? 1 : -1.5;
379 
380  m_unit->setPos(
381  coord.first - size.first * scaleX,
382  coord.second - 1.5 * size.second * scaleY);
383  }
384 
385  if( suggestResampling )
386  {
387  showHideScaleValues();
388  }
389  else if( !suggestResampling && step != m_step )
390  {
391  m_step = step;
392  showHideScaleValues();
393  }
394 }

398 void ScaleValues::showHideScaleValues()
399 {
400  double value;
401  const int size = (int)m_values.size();
402  const double startVal = this->getStartVal();
403 
404  for(int i = 0; i < size; ++i)
405  {
406  value = i * m_interval + startVal; // compute the value at index 'i'
407 
408  // Display this value or not according to the current step between two consecutive values
409  // and in keeping the displaying of the value '0':
410  m_values[i]->setVisible( fmod(value, (m_step * m_interval)) == 0.0 );
411  }
412 }
