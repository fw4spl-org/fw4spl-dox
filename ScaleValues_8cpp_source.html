00120 void ScaleValues::buildValues()
00121 {
00122     SLM_TRACE_FUNC();
00123     SLM_ASSERT("m_interval can not be equal to 0", m_interval != 0);
00124 
00125     m_values.clear();
00126 
00127     double val = this->getStartVal();
00128     const int range = (int) ceil(this->getEndVal() - val);
00129     const int nbValues = (int)(ceil(range/ m_interval)) + 1 ;
00130 
00131     std::string format;
00132 
00133     if(m_align == "left")
00134     {
00135         format = "  %1";
00136     }
00137     else if(m_align == "right")
00138     {
00139         format = "%1  ";
00140     }
00141     else
00142     {
00143         format = "%1";
00144     }
00145 
00146     // Build the values as graphic items
00147     for(int i = 0; i < nbValues; ++i, val += m_interval)
00148     {
00149         QGraphicsSimpleTextItem* text = new QGraphicsSimpleTextItem();
00150         text->setText( QString::fromStdString(format).arg(val) );
00151         text->setFont( m_font );
00152         text->setBrush( m_brush );
00153         text->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
00154 
00155         m_values.push_back( text );
00156     }
00157 
00158     // Add the values to the item group
00159     BOOST_FOREACH( QGraphicsItem* item, m_values )
00160     {
00161         m_layer->addToGroup( item );
00162     }
00163 
00164     // Build the unit graphic item
00165     m_unit = new QGraphicsSimpleTextItem();
00166     m_unit->setText( QString::fromStdString( m_displayedUnit ) );
00167     m_unit->setFont( m_font );
00168     m_unit->setBrush( m_brush );
00169     m_unit->setCacheMode( QGraphicsItem::DeviceCoordinateCache );
00170 
00171     // Add it to the items to be displayed if required
00172     if( m_showUnit )
00173     {
00174         m_layer->addToGroup( m_unit );
00175     }
00176 
00177     // Set the layer position (according to the related axis) and zValue
00178     m_layer->setPos( m_xAxis->getOrigin(), m_yAxis->getOrigin() );
00179     m_layer->setZValue( m_zValue );
00180 
00181     // Add the layer containing grid's lines to the scene
00182     this->getScene2DRender()->getScene()->addItem( m_layer );
00183 }

00238 void ScaleValues::rescaleValues()
00239 {
00240     const double viewportX = m_viewport->getX();
00241     const double viewportWidth = m_viewport->getWidth();
00242     const double viewportHeight = m_viewport->getHeight();
00243 
00244     const double viewportSizeRatio = viewportHeight / viewportWidth;
00245     const double viewInitialSizeRatio = m_viewInitialSize.first / m_viewInitialSize.second;
00246 
00247     Scene2DRatio ratio = this->getRatio();  // Total ratio
00248     double viewportWidthRatio = this->getViewportSizeRatio().first;
00249 
00250     double scaleX = m_fontSize;
00251     double scaleY = m_fontSize * viewportSizeRatio;
00252     scaleY /= viewportWidthRatio;
00253     scaleY *= viewInitialSizeRatio;
00254 
00255     scaleX = scaleX * ratio.first;
00256     scaleY = scaleY * ratio.second;
00257 
00258     QTransform transform;
00259     transform.scale( scaleX, scaleY );
00260 
00261     int step = 0;
00262     double valueSize;                   // the size (width or height) of the current value
00263     std::pair<double, double> coord;    // coordinates of the current scale value item
00264     std::pair<double, double> size;     // size of the current scale value item
00265     bool suggestResampling = false;     /* scale value items resampling is suggested because of
00266                                            a lack of sufficient width to display all of them */
00267 
00268     const int valuesSize = m_values.size();
00269     float val = getStartVal();
00270 
00271     if(m_align == "left" || m_align == "right")
00272     {
00273         const double valueSizeRatio = m_interval / scaleY;
00274 
00275         float coeff = 0;
00276 
00277         double textPosX;
00278 
00279         if(m_align == "left")
00280         {
00281             textPosX = viewportX;
00282         }
00283         else
00284         {
00285             coeff = -1;
00286             textPosX = viewportX + viewportWidth;
00287         }
00288 
00289         for(int i = 0; i < valuesSize; ++i, val += m_interval)
00290         {
00291             valueSize = m_values[i]->boundingRect().height();
00292 
00293             size = this->mapAdaptorToScene(
00294                     std::pair<double, double>(m_values[i]->boundingRect().width(), valueSize),
00295                     m_xAxis, m_yAxis);
00296 
00297             step = (int)(valueSize / valueSizeRatio) + 1;
00298 
00299             if( step > m_step )
00300             {
00301                 m_step = step;
00302                 suggestResampling = true;
00303             }
00304 
00305             coord = this->mapAdaptorToScene(
00306                 std::pair<double, double>(textPosX, val), m_xAxis, m_yAxis);
00307 
00308             m_values[i]->setTransform( transform );
00309 
00310             m_values[i]->setPos(
00311                     coord.first + coeff * size.first * scaleX,
00312                     coord.second - (m_interval - size.second / 2) * scaleY );
00313         }
00314 
00315         m_unit->setTransform( transform );
00316 
00317         val = viewportHeight * 0.8;
00318 
00319         coord = this->mapAdaptorToScene(
00320                 std::pair<double, double>(textPosX, val), m_xAxis, m_yAxis);
00321 
00322         coeff = (m_align == "left") ? 1 : -1.5;
00323 
00324         m_unit->setPos(
00325                 coord.first + coeff * 2 * size.first * scaleX,
00326                 coord.second + size.second * scaleY);
00327     }
00328     else    // axis centered on top or bottom
00329     {
00330         const double valueSizeRatio = m_interval / scaleX;
00331 
00332         float coeff = 0.5;
00333 
00334         double textPosY = (m_align == "bottom")
00335             ? m_viewport->getY()
00336             : viewportHeight * 0.9;
00337 
00338         for(int i = 0; i < valuesSize; ++i, val += m_interval)
00339         {
00340             valueSize = m_values[i]->boundingRect().width();
00341 
00342             size = this->mapAdaptorToScene(
00343                 std::pair<double, double>(valueSize, m_values[i]->boundingRect().height()),
00344                 m_xAxis, m_yAxis);
00345 
00346             step = (int)(valueSize / valueSizeRatio) + 1;
00347 
00348             if( step > m_step )
00349             {
00350                 m_step = step;
00351                 suggestResampling = true;
00352             }
00353 
00354             coord = this->mapAdaptorToScene(std::pair<double, double>(val, textPosY),
00355                     m_xAxis, m_yAxis);
00356 
00357             m_values[i]->setTransform( transform );
00358 
00359             m_values[i]->setPos(
00360                     coord.first - size.first / 2 * scaleX,
00361                     coord.second - coeff * size.second / 2 * scaleY );
00362         }
00363 
00364         m_unit->setTransform( transform );
00365 
00366         val = viewportHeight * 0.8;
00367 
00368         size = this->mapAdaptorToScene(
00369                 std::pair<double, double>(m_unit->boundingRect().width(), m_unit->boundingRect().height()),
00370                 m_xAxis, m_yAxis);
00371 
00372         coord = this->mapAdaptorToScene(
00373                 std::pair<double, double>(
00374                     viewportX + viewportWidth / 2, textPosY),
00375                 m_xAxis, m_yAxis);
00376 
00377         coeff = (m_align == "left") ? 1 : -1.5;
00378 
00379         m_unit->setPos(
00380                 coord.first - size.first * scaleX,
00381                 coord.second - 1.5 * size.second * scaleY);
00382     }
00383 
00384     if( suggestResampling )
00385     {
00386         showHideScaleValues();
00387     }
00388     else if( !suggestResampling && step != m_step )
00389     {
00390         m_step = step;
00391         showHideScaleValues();
00392     }
00393 }
