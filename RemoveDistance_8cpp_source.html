00066 ::fwData::PointList::sptr  getDistanceToRemove(::fwData::Image::sptr image, bool &removeAll)
00067 {
00068     ::fwData::PointList::sptr distToRemove;
00069     removeAll = false;
00070     ::fwData::Vector::sptr vectDist;
00071     vectDist = image->getField< ::fwData::Vector >(::fwComEd::Dictionary::m_imageDistancesId);
00072 
00073     if(vectDist)
00074     {
00075         std::vector< std::string > selections;
00076         selections.push_back("ALL");
00077         std::map< std::string , ::fwData::PointList::sptr > correspondance;
00078 
00079         BOOST_FOREACH(::fwData::Object::sptr obj, *vectDist)
00080         {
00081             ::fwData::PointList::sptr pl = ::fwData::PointList::dynamicCast(obj);
00082 
00083             if ( pl->getPoints().size()!=2 ) { continue; } // we skip no paired pointList
00084             ::fwData::Point::sptr pt1 =  pl->getPoints().front();
00085             ::fwData::Point::sptr pt2 =  pl->getPoints().back();
00086 
00087             double dist=0;
00088             double delta = pt1->getCRefCoord()[0] - pt2->getCRefCoord()[0];
00089             dist += delta*delta;
00090             delta = pt1->getCRefCoord()[1] - pt2->getCRefCoord()[1];
00091             dist += delta*delta;
00092             delta = pt1->getCRefCoord()[2] - pt2->getCRefCoord()[2];
00093             dist += delta*delta;
00094             dist = sqrt(dist);
00095 
00096             selections.push_back( distanceToStr(dist) );
00097             correspondance[ selections.back() ] = pl;
00098         }
00099 
00100         if ( !selections.empty() )
00101         {
00102             ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
00103             selector->setTitle("Select a distance to remove");
00104             selector->setSelections(selections);
00105             std::string selection = selector->show();
00106             if( ! selection.empty() )
00107             {
00108                 if (selection=="ALL")
00109                 {
00110                     removeAll=true;
00111                 }
00112                 else
00113                 {
00114                     removeAll=false;
00115                     distToRemove = correspondance[selection];
00116                 }
00117             }
00118         }
00119     }
00120     return distToRemove;
00121 }

00143 void RemoveDistance::updating( ) throw(::fwTools::Failed)
00144 {
00145     ::fwData::Image::sptr image = this->getObject< ::fwData::Image >();
00146 
00147     ::fwData::Vector::sptr vectDist;
00148     vectDist = image->getField< ::fwData::Vector >(::fwComEd::Dictionary::m_imageDistancesId);
00149

00153         bool requestAll;
00154         ::fwData::PointList::sptr distToRemove = getDistanceToRemove(image, requestAll );
00155 
00156         // perform action only available distance

00158         {
00159             SLM_ASSERT("No Field ImageDistancesId",vectDist);
00160             ::fwData::Vector::IteratorType newEnd = std::remove(vectDist->begin(), vectDist->end(), distToRemove);
00161             vectDist->getContainer().erase(newEnd, vectDist->end());
00162 
00163             this->notifyDeleteDistance(image, distToRemove);
00164         }
00165         if ( requestAll )

00167             // backup

00169 
00170             image->removeField( ::fwComEd::Dictionary::m_imageDistancesId );
