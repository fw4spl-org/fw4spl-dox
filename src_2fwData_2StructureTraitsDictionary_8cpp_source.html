00040 void StructureTraitsDictionary::addStructure(StructureTraits::sptr structureTraits)
00041 {
00042     std::string type = structureTraits->getType();
00043     StructureTraits::StructureClass structClass = structureTraits->getClass();
00044     std::string attachment = structureTraits->getAttachmentType();
00045     FW_RAISE_IF("Structure of type '" << type << "' already exist", m_structureTraitsMap.find(type) != m_structureTraitsMap.end());
00046     FW_RAISE_IF("Structure of class '" << structClass << "' can not have attachment",
00047             !(attachment.empty() || structClass == StructureTraits::LESION || structClass == StructureTraits::FUNCTIONAL) );
00048     FW_RAISE_IF("Structure attachment '" << attachment << "' not found in dictionary",
00049             !(attachment.empty() || m_structureTraitsMap.find(attachment) != m_structureTraitsMap.end() ) );
00050     FW_RAISE_IF("Structure attachment '" << attachment << "' must be of class ORGAN",
00051             !(attachment.empty() || m_structureTraitsMap[attachment]->getClass() == StructureTraits::ORGAN ) );
00052     FW_RAISE_IF("Structure must have at least one category",
00053             structureTraits->getCategories().empty());
00054     FW_RAISE_IF("Wrong structure type '" << type<< "', a type cannot contain space" , structureTraits->getType().find(" ") != std::string::npos );
00055 
00056     m_structureTraitsMap[type] = structureTraits;
00057 }
