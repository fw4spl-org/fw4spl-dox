102 void ExternalDataReaderService::updating() throw(::fwTools::Failed)
103 {
104  this->configureWithIHM();
105 
106  std::string imageName;
107  ::fwData::Composite::sptr dataComposite = this->getObject< ::fwData::Composite >();
108  ::fwComEd::helper::Composite compositeHelper(dataComposite);
109  SLM_ASSERT("dataComposite not instanced", dataComposite);
110  try
111  {
112  if (this->hasLocationDefined())
113  {
114  // reading of the file
115  std::fstream file;
116  file.open(this->getFile().string().c_str(), std::fstream::in);
117  if (!file.is_open())
118  {
119  OSLM_ERROR( "External data file loading error for " << this->getFile());
120  std::string str = "Unable to open ";
121  str += this->getFile().string();
122  throw std::ios_base::failure(str);
123  }
124  file >> imageName;
125  int readedValue = 0;
126  double value;
127  ::fwData::TransformationMatrix3D::sptr transformation1 = ::fwData::TransformationMatrix3D::New();
128  ::fwData::TransformationMatrix3D::sptr transformation2 = ::fwData::TransformationMatrix3D::New();
129  while(!file.eof())
130  {
131  readedValue = 0;
132  while ( !file.eof() && readedValue<32 )
133  {
134  file >> value;
135  if (readedValue<16)
136  {
137  transformation1->getRefCoefficients()[readedValue] = value;
138  }
139  else
140  {
141  transformation2->getRefCoefficients()[readedValue] = value;
142  }
143  readedValue++;
144  }
145  }
146  file.close();
147  // TF1 contains the first and third transformations
148  if(dataComposite->find("TF1") == dataComposite->end() )
149  {
150  compositeHelper.add("TF1", transformation1);
151  }
152  else
153  {
154  compositeHelper.swap("TF1", transformation1);
155  }
156  // TF2 contains the first and third transformations
157  if(dataComposite->find("TF2") == dataComposite->end() )
158  {
159  compositeHelper.add("TF2", transformation2);
160  }
161  else
162  {
163  compositeHelper.swap("TF2", transformation2);
164  }
165  ::fwData::String::sptr imageNameStr = ::fwData::String::New(imageName);
166  if(dataComposite->find("filename") == dataComposite->end() )
167  {
168  compositeHelper.add("filename", imageNameStr);
169  }
170  else
171  {
172  compositeHelper.swap("filename", imageNameStr);
173  }
174  SLM_ASSERT("Unable to open '"+this->getFile().string()+"'.", readedValue == 32 );
175  }
176  }
177  catch(std::ios_base::failure &exception)
178  {
179  OSLM_ERROR( "External data file loading error for " << exception.what());
180  }
181 }
