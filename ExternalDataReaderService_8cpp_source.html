00128             {
00129                 readedValue = 0;
00130                 while ( !file.eof() && readedValue<32 )
00131                 {
00132                     file >> value;

00139                         transformation2->getRefCoefficients().push_back( value );

00142                 }
00143             }
00144             file.close();
00145             // TF1 contains the first and third transformations
00146             if(dataComposite->find("TF1") == dataComposite->end() )
00147             {
00148                 compositeHelper.add("TF1", transformation1);
00149             }
00150             else

00156             {
00157                 compositeHelper.add("TF2", transformation2);

00165             {

00173         }
