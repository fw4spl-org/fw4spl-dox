54 void IFrameLayoutManager::initialize( ConfigurationType configuration)
55 {
56  OSLM_ASSERT("Bad configuration name "<<configuration->getName()<< ", must be frame",
57  configuration->getName() == "frame");
58 
59  std::vector < ConfigurationType > name = configuration->find("name");
60  std::vector < ConfigurationType > icon = configuration->find("icon");
61  std::vector < ConfigurationType > minSize = configuration->find("minSize");
62  std::vector < ConfigurationType > styles = configuration->find("style");
63 
64  if(!name.empty())
65  {
66  m_frameInfo.m_name = name.at(0)->getValue();
67  }
68 
69  if(!icon.empty())
70  {
71  m_frameInfo.m_iconPath = ::boost::filesystem::path( icon.at(0)->getValue() );
72  OSLM_ASSERT("The icon "<< m_frameInfo.m_iconPath << " doesn't exist, please ensure that the path is correct",
73  ::boost::filesystem::exists(m_frameInfo.m_iconPath));
74  }
75 
76  if(!minSize.empty())
77  {
78  if(minSize.at(0)->hasAttribute("width"))
79  {
80  m_frameInfo.m_minSize.first =
81  ::boost::lexical_cast<int >(minSize.at(0)->getExistingAttributeValue("width"));
82  }
83  if(minSize.at(0)->hasAttribute("height"))
84  {
85  m_frameInfo.m_minSize.second = ::boost::lexical_cast<int >(minSize.at(0)->getExistingAttributeValue(
86  "height"));
87  }
88  }
89 
90  if(!styles.empty())
91  {
92  ::fwRuntime::ConfigurationElement::sptr stylesCfgElt = styles.at(0);
93  SLM_FATAL_IF("<style> node must contain mode attribute", !stylesCfgElt->hasAttribute("mode") );
94  const std::string style = stylesCfgElt->getExistingAttributeValue("mode");
95 
96  if (style == "DEFAULT")
97  {
98  m_frameInfo.m_style = DEFAULT;
99  }
100  else if (style == "STAY_ON_TOP")
101  {
102  m_frameInfo.m_style = STAY_ON_TOP;
103  }
104  else if (style == "MODAL")
105  {
106  m_frameInfo.m_style = MODAL;
107  }
108  else
109  {
110  OSLM_FATAL("The style "<<style<< " is unknown, it should be DEFAULT, STAY_ON_TOP or MODAL.");
111  }
112  }
113  this->readConfig();
114 }

132 void IFrameLayoutManager::readConfig()
133 {
134  ::fwData::Composite::sptr prefUI = this->getPreferenceUI();
135  if(prefUI)
136  {
137  if ( prefUI->find( IFrameLayoutManager::FRAME_STATE_UI ) != prefUI->end() )
138  {
139  ::fwData::Integer::sptr state =
140  ::fwData::Integer::dynamicCast( (*prefUI)[ IFrameLayoutManager::FRAME_STATE_UI ] );
141  SLM_ASSERT("UI state not correct", state);
142  m_frameInfo.m_state = (FrameState) state->value();
143  }
144  if ( prefUI->find( IFrameLayoutManager::FRAME_SIZE_W_UI ) != prefUI->end() )
145  {
146  ::fwData::Integer::sptr sizew =
147  ::fwData::Integer::dynamicCast( (*prefUI)[ IFrameLayoutManager::FRAME_SIZE_W_UI ] );
148  SLM_ASSERT("UI sizeW not correct", sizew);
149  m_frameInfo.m_size.first = *sizew;
150  }
151  if ( prefUI->find( IFrameLayoutManager::FRAME_SIZE_H_UI ) != prefUI->end() )
152  {
153  ::fwData::Integer::sptr sizeh =
154  ::fwData::Integer::dynamicCast( (*prefUI)[ IFrameLayoutManager::FRAME_SIZE_H_UI ] );
155  SLM_ASSERT("UI sizeH not correct", sizeh);
156  m_frameInfo.m_size.second = *sizeh;
157  }
158  if ( prefUI->find( IFrameLayoutManager::FRAME_POSITION_X_UI ) != prefUI->end() )
159  {
160  ::fwData::Integer::sptr posx =
161  ::fwData::Integer::dynamicCast( (*prefUI)[ IFrameLayoutManager::FRAME_POSITION_X_UI ] );
162  SLM_ASSERT("UI posX not correct", posx);
163  m_frameInfo.m_position.first = *posx;
164  }
165  if ( prefUI->find( IFrameLayoutManager::FRAME_POSITION_Y_UI ) != prefUI->end() )
166  {
167  ::fwData::Integer::sptr posy =
168  ::fwData::Integer::dynamicCast( (*prefUI)[ IFrameLayoutManager::FRAME_POSITION_Y_UI ] );
169  SLM_ASSERT("UI posY not correct", posy);
170  m_frameInfo.m_position.second = *posy;
171  }
172  }
173 }

177 void IFrameLayoutManager::writeConfig()
178 {
179  ::fwData::Composite::sptr prefUI = this->getPreferenceUI();
180  if(prefUI)
181  {
182  if(m_frameInfo.m_state != ICONIZED)
183  {
184  ::fwData::Integer::sptr state = ::fwData::Integer::New(m_frameInfo.m_state);
185  (*prefUI)[ IFrameLayoutManager::FRAME_STATE_UI ] = state;
186  }
187 
188  ::fwData::Integer::sptr sizew = ::fwData::Integer::New(m_frameInfo.m_size.first);
189  (*prefUI)[ IFrameLayoutManager::FRAME_SIZE_W_UI ] = sizew;
190 
191  ::fwData::Integer::sptr sizeh = ::fwData::Integer::New(m_frameInfo.m_size.second);
192  (*prefUI)[ IFrameLayoutManager::FRAME_SIZE_H_UI ] = sizeh;
193 
194  ::fwData::Integer::sptr posx = ::fwData::Integer::New(m_frameInfo.m_position.first);
195  (*prefUI)[ IFrameLayoutManager::FRAME_POSITION_X_UI ] = posx;
196 
197  ::fwData::Integer::sptr posy = ::fwData::Integer::New(m_frameInfo.m_position.second);
198  (*prefUI)[ IFrameLayoutManager::FRAME_POSITION_Y_UI ] = posy;

204 ::fwData::Composite::sptr IFrameLayoutManager::getPreferenceUI()
205 {
206  ::fwData::Composite::sptr prefUI;
207 
208  // Get preferences
209  auto preferencesServicesList = ::fwServices::OSR::getServices("::fwPreferences::IPreferences");
210  if(!preferencesServicesList.empty())
211  {
212  ::fwServices::IService::sptr prefService = *preferencesServicesList.begin();
213  ::fwData::Composite::sptr prefs = prefService->getObject< ::fwData::Composite >();
214 
215  ::fwData::Composite::sptr framesUI;
216  // Retreives software UI pref
217  if ( prefs->find( IFrameLayoutManager::SOFTWARE_UI ) == prefs->end() )
218  {
219  framesUI = ::fwData::Composite::New();
220  (*prefs)[ IFrameLayoutManager::SOFTWARE_UI ] = framesUI;
221  }
222  else
223  {
224  framesUI = ::fwData::Composite::dynamicCast( (*prefs)[ IFrameLayoutManager::SOFTWARE_UI ]);
225  }
226  // Retreives frame UI pref
227  if ( framesUI->find( this->m_frameInfo.m_name ) != framesUI->end() )
228  {
229  prefUI = ::fwData::Composite::dynamicCast( (*framesUI)[ this->m_frameInfo.m_name ] );
230  }
231  else
232  {
233  prefUI = ::fwData::Composite::New();
234  (*framesUI)[ this->m_frameInfo.m_name ] = prefUI;
235  }
236  }
237  return prefUI;
238 }
