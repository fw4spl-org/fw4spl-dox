106 void SMeshReader::loadMesh( const ::boost::filesystem::path vtkFile, ::fwData::Mesh::sptr _pMesh )
107 {
108  SLM_TRACE_FUNC();
109 
110  ::fwVtkIO::MeshReader::sptr reader = ::fwVtkIO::MeshReader::New();
111 
112  m_sigJobCreated->emit(reader->getJob());
113 
114  reader->setObject(_pMesh);
115  reader->setFile(vtkFile);
116 
117  ::fwData::mt::ObjectWriteLock lock(_pMesh);
118 
119  try
120  {
121  reader->read();
122  }
123  catch (const std::exception & e)
124  {
125  std::stringstream ss;
126  ss << "Warning during loading : " << e.what();
127 
128  ::fwGui::dialog::MessageDialog::showMessageDialog(
129  "Warning",
130  ss.str(),
131  ::fwGui::dialog::IMessageDialog::WARNING);
132  }
133  catch( ... )
134  {
135  std::stringstream ss;
136  ss << "Warning during loading. ";
137  ::fwGui::dialog::MessageDialog::showMessageDialog(
138  "Warning",
139  "Warning during loading.",
140  ::fwGui::dialog::IMessageDialog::WARNING);
141  }
142 }
