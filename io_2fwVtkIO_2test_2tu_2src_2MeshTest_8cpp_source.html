00087 
00088     vtkSmartPointer< vtkPolyData > vtkMesh = vtkSmartPointer< vtkPolyData >::New();
00089     ::fwVtkIO::helper::Mesh::toVTKMesh( mesh1, vtkMesh);
00090     CPPUNIT_ASSERT( vtkMesh );
00091 
00092     CPPUNIT_ASSERT_EQUAL(poly_source->GetNumberOfVerts(), vtkMesh->GetNumberOfVerts());
