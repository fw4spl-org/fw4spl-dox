00078 VersionsGraph::VersionSeriesType VersionsGraph::shortestPath(const NodeType& origin, const NodeType& target)
00079 {
00080     VersionSeriesType serie;
00081 
00082     ::fwCore::mt::ReadLock nodesLock(m_nodesMutex);
00083     ::fwCore::mt::ReadLock graphLock(m_graphMutex);
00084 
00085     std::vector< NodeIDType > predecessor( ::boost::num_vertices(m_graph) );
00086     std::vector< int > distances( ::boost::num_vertices(m_graph) );
00087 
00088     const NodeIDType& vOrig = m_nodes[origin];
00089     ::boost::dijkstra_shortest_paths(m_graph, vOrig,
00090             ::boost::weight_map(get(&EdgeType::m_weight, m_graph))
00091             .predecessor_map(&predecessor[0])
00092             .distance_map(::boost::make_iterator_property_map(distances.begin(), ::boost::get(::boost::vertex_index, m_graph )))
00093             );
00094 
00095     NodeIDType current = m_nodes[target];
00096 
00097     while(current != predecessor[current])
00098     {
00099         serie.insert(serie.begin(), current);
00100         current = predecessor[current];
00101     }
00102 
00103     return serie;
00104 }
