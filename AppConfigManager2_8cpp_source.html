228 {
229  ::fwData::Object::sptr obj;
230 
231  // Look first in objects created in this appConfig

358 void AppConfigManager2::processStartItems()
359 {
360  std::vector< ::fwServices::IService::SharedFutureType > futures;
361 
362  for(const auto& elem : m_cfgElem->getElements())
363  {
364  if (elem->getName() == "start")
365  {
366  SLM_ASSERT("Missing attribute \"uid\".", elem->hasAttribute("uid"));
367  const std::string uid = elem->getAttributeValue("uid");
368  SLM_ASSERT("\"uid\" attribute is empty.", !uid.empty());
369 
370  if(!::fwTools::fwID::exist(uid))
371  {
372  if(m_deferredServices.find(uid) != m_deferredServices.end())
373  {
374  m_deferredStartSrv.push_back(uid);
375  SLM_INFO( this->msgHead() + "Start for service '" + uid + "' will be deferred since at least one "
376  "of its data is missing. With DEBUG log level, you can know which are the "
377  "missing objects.");
378  }
379  else
380  {
381  SLM_FATAL( this->msgHead() + "Start is requested for service '" + uid +
382  "', but it does not exist.");
383  }
384  }
385  else
386  {
387  const ::fwServices::IService::sptr srv = ::fwServices::get(uid);
388  SLM_FATAL_IF( this->msgHead() + "Start is requested for service '" + uid + "', though this identifier "
389  "exists, this is not a service.", !srv);
390 
391  futures.push_back(srv->start());
392 
393  m_startedSrv.push_back(srv);
394  }
395 
396  }
397  }
398  ::boost::wait_for_all(futures.begin(), futures.end());
399 }

445 void AppConfigManager2::createObjects(::fwRuntime::ConfigurationElement::csptr cfgElem)
446 {
447  for(const auto& elem : cfgElem->getElements())
448  {
449  if (elem->getName() == "object")
450  {
451  // Get attributes
452 
453  // Id
454  ConfigAttribute id("", false);
455  if (elem->hasAttribute("uid"))
456  {
457  id.first = elem->getAttributeValue("uid");
458  SLM_ASSERT(this->msgHead() + "\"uid\" attribute is empty.", !id.first.empty());
459  id.second = true;
460  }
461 
462  // Type
463  ConfigAttribute type("", false);
464  if (elem->hasAttribute("type"))
465  {
466  type.first = elem->getAttributeValue("type");
467  SLM_ASSERT(this->msgHead() + "\"type\" attribute is empty.", !type.first.empty());
468  SLM_ASSERT(this->msgHead() + "\"type\" must be a rooted namespace.", type.first.substr(0, 2) == "::");
469  type.second = true;
470  }
471 
472  // Build mode
473  ConfigAttribute buildMode("", false);
474  if (elem->hasAttribute("src"))
475  {
476  buildMode.first = elem->getAttributeValue("src");
477  SLM_ASSERT("this->msgHead() + \"src\" attribute is empty.", !buildMode.first.empty());
478 
479  SLM_ASSERT("Unhandled build mode (bad \"src\" attribute). Must be \"new\", \"deferred\" or \"ref\".",
480  buildMode.first == "ref" || buildMode.first == "src" || buildMode.first == "deferred");
481  buildMode.second = true;
482  }
483 
484  if(buildMode.first == "deferred")
485  {
486  SLM_ASSERT(this->msgHead() + "Missing attribute \"id\".", id.second);
487  auto ret = m_deferredObjects.insert( std::make_pair(id.first, DeferredObjectType()));
488  SLM_ASSERT(this->msgHead() + "Object '" + id.first + "' already exists in this config.", ret.second);
489  }
490  else
491  {
492  // Creation of a new object
493  ::fwData::Object::sptr obj;
494 
495  // Create new or get the referenced object
496  if (buildMode.second && buildMode.first == "ref")
497  {
498  SLM_ASSERT(this->msgHead() + "Missing attribute \"id\".", id.second);
499  obj = this->getRefObject(type, id.first);
500  }
501  else
502  {
503  obj = this->getNewObject(type, id);
504  }
505 
506  // Get the object parser associated with the object type
507  const auto srvFactory = ::fwServices::registry::ServiceFactory::getDefault();
508 
509  std::string srvImpl = srvFactory->getDefaultImplementationIdFromObjectAndType(
510  obj->getClassname(), "::fwServices::IXMLParser");
511 
512  ::fwServices::IService::sptr srv = srvFactory->create("::fwServices::IXMLParser", srvImpl);
513  auto objectParser = ::fwServices::IXMLParser::dynamicCast(srv);
514  objectParser->setObjectConfig(elem);
515  objectParser->createConfig(obj);
516 
517  m_createdObjects[id.first] = std::make_pair( obj, objectParser);
518  }
519  }
520  }
521 }

525 void AppConfigManager2::createServices(::fwRuntime::ConfigurationElement::csptr cfgElem)
526 {
527  for(const auto& elem : cfgElem->getElements())
528  {
529  if (elem->getName() == "service")
530  {
531  // Parse the service configuration
532  ServiceConfig srvConfig = ::fwServices::helper::Config::parseService(elem, this->msgHead());
533 
534  // Check if we can start the service now or if we must deferred its creation
535  bool createService = true;
536  std::vector<std::string> uids;
537 
538  for(const auto& objectCfg : srvConfig.m_objects)
539  {
540  // If the current service uses an object that is marked as deferred, this means
541  // we will have to manage automatically the start/stop and the connections
542  auto it = m_deferredObjects.find(objectCfg.m_uid);
543  if(it != m_deferredObjects.end())
544  {
545  it->second.m_servicesCfg.emplace_back(srvConfig);
546  uids.push_back(objectCfg.m_uid);
547  m_deferredServices.insert(srvConfig.m_uid);
548 
549  if(!objectCfg.m_optional)
550  {
551  createService = false;
552  }
553  }
554  }
555 
556  if(createService)
557  {
558  this->createService(srvConfig);
559  }
560  else
561  {
562  // Check if a service hasn't been already created with this uid
563  SLM_ASSERT(this->msgHead() + "UID " + srvConfig.m_uid + " already exists.",
564  !::fwTools::fwID::exist(srvConfig.m_uid));
565 
566  const std::string msg = AppConfigManager2::getUIDListAsString(uids);
567  SLM_INFO(this->msgHead() + "Service '" + srvConfig.m_uid +
568  "' has not been created because the object" + msg + "not available.");
569  }
570  }
571  else if (elem->getName() == "serviceList")
572  {
573  this->createServices(elem);
574  }
575  }
576 }

580 ::fwServices::IService::sptr AppConfigManager2::createService(const ::fwServices::IService::Config& srvConfig)
581 {
582  // Create and bind service
583  const ::fwServices::IService::sptr srv = this->getNewService(srvConfig.m_uid, srvConfig.m_type);
584  m_createdSrv.push_back(srv);
585 
586  if (!srvConfig.m_worker.empty())
587  {
588  ::fwServices::registry::ActiveWorkers::sptr activeWorkers = ::fwServices::registry::ActiveWorkers::getDefault();
589  ::fwThread::Worker::sptr worker;
590  worker = activeWorkers->getWorker(srvConfig.m_worker);
591  if (!worker)
592  {
593  worker = ::fwThread::Worker::New();
594  activeWorkers->addWorker(srvConfig.m_worker, worker);
595  }
596  srv->setWorker(worker);
597  }
598 
599  std::string errMsgTail = " when creating service '" + srvConfig.m_uid + "'.";
600 
601  for(const auto& objectCfg : srvConfig.m_objects)
602  {
603  srv->setObjectId(objectCfg.m_key, objectCfg.m_uid);
604 
605  ::fwData::Object::sptr obj = this->findObject(objectCfg.m_uid, errMsgTail);
606 
607  SLM_ASSERT(this->msgHead() + "Object '" + objectCfg.m_uid + "' has not been found" + errMsgTail,
608  (!objectCfg.m_optional && obj) || objectCfg.m_optional);
609  if((obj || !objectCfg.m_optional) && objectCfg.m_access != ::fwServices::IService::AccessType::OUTPUT)
610  {
611  ::fwServices::OSR::registerService(obj, objectCfg.m_key, objectCfg.m_access, srv);
612  }
613  }
614 
615  // Ok for now we assume we always need the root composite
616  // This is only true for services that will need to perform some stuff like adding new objects, etc...
617  // We will see in the future if this should be replaced or not
618  {
619  srv->setObjectId(::fwServices::IService::s_DEFAULT_OBJECT, "defaultObjectId");
620 
621  ::fwServices::OSR::registerService(m_tmpRootObject, ::fwServices::IService::s_DEFAULT_OBJECT,
622  ::fwServices::IService::AccessType::INOUT, srv);
623  }
624 
625  // Set the size of the key groups
626  srv->m_keyGroupSize = srvConfig.m_groupSize;
627 
628  // Set the proxies
629  const auto& itSrvProxy = m_servicesProxies.find(srvConfig.m_uid);
630  if(itSrvProxy != m_servicesProxies.end())
631  {
632  for(const auto& itProxy : itSrvProxy->second.m_proxyCnt)
633  {
634  srv->addProxyConnection(itProxy.second);
635  }
636  }
637 
638  // Configure
639  srv->setConfiguration(srvConfig);
640  srv->configure();
641 
642  return srv;
643 }

647 void AppConfigManager2::createConnections()
648 {
649  for(const auto& elem : m_cfgElem->getElements())
650  {
651  if (elem->getName() == "connect")
652  {
653  // Parse all connections
654  auto genIdFn = [this]()
655  {
656  return "Proxy_" + this->getID() + "_" + std::to_string(m_proxyID++);
657  };
658 
659  ProxyConnections connectionInfos = ::fwServices::helper::Config::parseConnections2(elem, this->msgHead(),
660  genIdFn);
661  // Proxy that is used for non-deferred connections
662  ProxyConnections createdObjectsProxy(connectionInfos.m_channel);
663 
664  // Register signals
665  for(const auto& signalInfo : connectionInfos.m_signals)
666  {
667  auto itDeferredObj = m_deferredObjects.find(signalInfo.first);
668  if( itDeferredObj != m_deferredObjects.end() )
669  {
670  // Deferred Object
671  ProxyConnections& proxy = itDeferredObj->second.m_proxyCnt[connectionInfos.m_channel];
672  proxy.addSignalConnection(signalInfo);
673  }
674  else
675  {
676  auto itObj = m_createdObjects.find(signalInfo.first);
677  if( itObj != m_createdObjects.end() )
678  {
679  // Regular object
680  createdObjectsProxy.addSignalConnection(signalInfo);
681  }
682  else
683  {
684  // Service
685  auto& itSrv = m_servicesProxies[signalInfo.first];

715  ProxyConnections& proxy = itSrv.m_proxyCnt[connectionInfos.m_channel];
716  proxy.addSlotConnection(slotInfo);
717  proxy.m_channel = connectionInfos.m_channel;
718  }
719  }
720  }

723  this->connectProxy(connectionInfos.m_channel, createdObjectsProxy);

791 void AppConfigManager2::addObjects(fwData::Object::sptr obj, const std::string& id)
792 {
793  FW_PROFILE("addObjects");
794 
795  // Local map used to process services only once
796  std::map< std::string, const ServiceConfig* > servicesMapCfg;
797 
798  // Local vector used to store services and keep the declare order (we could use only this one but the map is used
799  // to speedup the search
800  std::vector< const ServiceConfig* > servicesCfg;
801 
802  // Are there services that were waiting for this object ?
803  auto itDeferredObj = m_deferredObjects.find(id);
804  if(itDeferredObj != m_deferredObjects.end())
805  {
806  // For each service waiting to be started
807  for(const auto& srvCfg : itDeferredObj->second.m_servicesCfg)
808  {
809  if(servicesMapCfg.find(srvCfg.m_uid) == servicesMapCfg.end())
810  {
811  servicesMapCfg[srvCfg.m_uid] = &srvCfg;
812  servicesCfg.push_back(&srvCfg);
813  }
814  }
815 
816  // Connect signals of this deferred object
817  itDeferredObj->second.m_object = obj;
818 
819  for(const auto& connectCfg : itDeferredObj->second.m_proxyCnt)
820  {
821  this->connectProxy(connectCfg.first, connectCfg.second);
822  }
823  }
824 
825  std::unordered_map<std::string, ::fwServices::IService::sptr> newServices;
826 
827  for(const auto& itService : servicesCfg)
828  {
829  auto srvCfg = itService;
830  SLM_ASSERT("Config is null", srvCfg);
831  auto& uid = srvCfg->m_uid;
832 
833  bool createService = true;
834  bool reconnect = false;
835 
836  // Look for all objects (there could be more than the current object) and check if they are all created
837  for(const auto& objCfg : srvCfg->m_objects)
838  {
839  // Look first in objects created in this appConfig
840  if(m_createdObjects.find(objCfg.m_uid) == m_createdObjects.end())
841  {
842  // Not found, now look in the objects that were marked as "deferred"
843  const auto itDeferredObj = m_deferredObjects.find(objCfg.m_uid);
844  SLM_ASSERT( this->msgHead() + "Object '" + objCfg.m_uid + "' used by service '" + uid +
845  "' has not been declared in this AppConfig.",
846  itDeferredObj != m_deferredObjects.end());
847 
848  const auto object = itDeferredObj->second.m_object;
849  if(object == nullptr)
850  {
851  if(!objCfg.m_optional)
852  {
853  createService = false;
854 
855  SLM_INFO( this->msgHead() + "Service '" + uid + "' has not been created because the "
856  "object" + objCfg.m_uid + " is not available.");
857  }
858  }
859  else if(::fwTools::fwID::exist(uid))
860  {
861  ::fwServices::IService::sptr srv = ::fwServices::get(uid);
862  SLM_ASSERT(this->msgHead() + "No service registered with UID \"" + uid + "\".", srv);
863 
864  // We have an optional object
865  if( objCfg.m_optional && !::fwServices::OSR::isRegistered(objCfg.m_key, objCfg.m_access, srv))
866  {
867  // Register the key on the service
868  ::fwServices::OSR::registerService(object, objCfg.m_key, objCfg.m_access, srv);
869 
870  // Call the swapping callback of the service and wait for it
871  srv->swapKey(objCfg.m_key, nullptr).wait();
872  }
873 
874  createService = false;
875  reconnect = true;
876  }
877  }
878  }
879 
880  // All the data for this service is ready, create and run it
881  if(createService)
882  {
883  // Create the service
884  newServices.emplace(std::make_pair(uid, this->createService(*srvCfg)));
885 
886  // Debug message
887  SLM_INFO( this->msgHead() + "Service '" + uid + "' has been automatically created because its "
888  "objects are all available.");
889  }
890  else if(reconnect)
891  {
892  // Update auto connections
893  ::fwServices::IService::sptr srv = ::fwServices::get(uid);
894  OSLM_ASSERT(this->msgHead() + "No service registered with UID \"" << uid << "\".", srv);
895 
896  srv->autoDisconnect();
897  srv->autoConnect();
898  }
899  }
900 
901  std::vector< ::fwServices::IService::SharedFutureType > futures;
902 
903  // Start services according to the order given in the configuration
904  for(const auto& uid : m_deferredStartSrv)
905  {
906  auto itSrv = newServices.find(uid);
907  if( itSrv != newServices.end())
908  {
909  futures.push_back(itSrv->second->start());
910  m_startedSrv.push_back(itSrv->second);
911 
912  // Debug message
913  SLM_INFO( this->msgHead() + "Service '" + uid + "' has been automatically started because its "
914  "objects are all available.");
915  }
916  }
917 
918  ::boost::wait_for_all(futures.begin(), futures.end());
919  futures.clear();
920 
921  // Update services according to the order given in the configuration
922  for(const auto& uid : m_deferredUpdateSrv)
923  {
924  auto itSrv = newServices.find(uid);
925  if( itSrv != newServices.end())
926  {
927  futures.push_back(itSrv->second->update());
928 
929  // Debug message
930  SLM_INFO( this->msgHead() + "Service '" + uid + "' has been automatically update because its "
931  "objects are all available.");
932  }
933  }
934 
935  ::boost::wait_for_all(futures.begin(), futures.end());
936 }

940 void AppConfigManager2::removeObjects(fwData::Object::sptr obj, const std::string& id)
941 {
942  FW_PROFILE("removeObjects");
943 
944  // Are there services that were connected with this object ?
945  const auto itDeferredObj = m_deferredObjects.find(id);
946  if(itDeferredObj != m_deferredObjects.end())
947  {
948  for(const auto& itProxy : itDeferredObj->second.m_proxyCnt)
949  {
950  this->destroyProxy(itProxy.first, itProxy.second, id, itDeferredObj->second.m_object);
951  }
952 
953  itDeferredObj->second.m_object.reset();
954 
955  // Are there services that were using this object ?
956  for(const auto& srvCfg : itDeferredObj->second.m_servicesCfg)
957  {
958  if( ::fwTools::fwID::exist(srvCfg.m_uid) )
959  {
960  // Check all objects, to know if this object is optional
961  bool optional = true;
962 
963  for(const auto& objCfg : srvCfg.m_objects)
964  {
965  if(id == objCfg.m_uid)
966  {
967  ::fwServices::IService::sptr srv = ::fwServices::get(srvCfg.m_uid);
968  OSLM_ASSERT("No service registered with UID \"" << srvCfg.m_uid << "\".", srv);
969 
970  if(objCfg.m_optional)
971  {
972  if(::fwServices::OSR::isRegistered(objCfg.m_key, objCfg.m_access, srv))
973  {
974  ::fwServices::OSR::unregisterService(objCfg.m_key, objCfg.m_access, srv);
975 
976  srv->swapKey(objCfg.m_key, obj).wait();
977  }
978  }
979  else
980  {
981  optional = false;
982  }
983  }
984  }
985 
986  if(!optional)
987  {
988  // 1. Stop the service
989  ::fwServices::IService::sptr srv = ::fwServices::get(srvCfg.m_uid);
990  OSLM_ASSERT(this->msgHead() + "No service registered with UID \"" << srvCfg.m_uid << "\".", srv);
991 
992  OSLM_ASSERT("Service " << srv->getID() << " already stopped.", !srv->isStopped());
993  srv->stop().wait();
994 
995  for(auto it = m_startedSrv.begin(); it != m_startedSrv.end(); ++it)
996  {
997  if(it->lock() == srv)
998  {
999  m_startedSrv.erase(it);
1000  break;
1001  }
1002  }
1003 
1004  // 2. Destroy the service
1005  OSLM_ASSERT("Service " << srv->getID() << " must be stopped before destruction.",
1006  srv->isStopped());
1007  ::fwServices::OSR::unregisterService(srv);
1008 
1009  for(auto it = m_createdSrv.begin(); it != m_createdSrv.end(); ++it)
1010  {
1011  if(it->lock() == srv)
1012  {
1013  m_createdSrv.erase(it);
1014  break;
1015  }
1016  }
1017 
1018  ::fwServices::IService::wptr checkSrv = srv;
1019  srv.reset();
1020 
1021  SLM_ASSERT( this->msgHead() + "The service '" + srvCfg.m_uid +
1022  "'' should have been destroyed, but someone "
1023  "still holds a reference which prevents to destroy it properly.",
1024  checkSrv.expired());
1025 
1026  SLM_INFO( this->msgHead() + "Service '" + srvCfg.m_uid +
1027  "' has been stopped because the object " +
1028  id + " is no longer available.");
1029  }
1030  else
1031  {
1032  // Update auto connections
1033  ::fwServices::IService::sptr srv = ::fwServices::get(srvCfg.m_uid);
1034  OSLM_ASSERT(this->msgHead() + "No service registered with UID \"" << srvCfg.m_uid << "\".", srv);
1035 
1036  srv->autoDisconnect();
1037  srv->autoConnect();
1038  }
1039  }
1040  }
1041  }
