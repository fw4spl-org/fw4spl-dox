51 void SurfaceSegmentationIOD::write(::fwMedData::Series::sptr series)
52 {
53  // Retrieve model series
54  ::fwMedData::ModelSeries::sptr modelSeries = ::fwMedData::ModelSeries::dynamicCast(series);
55  SLM_ASSERT("Image series should not be null.", modelSeries);
56 
57  // Create writer
58  SPTR(::gdcm::SurfaceWriter) writer = std::make_shared< ::gdcm::SurfaceWriter >();
59 
60 
61  // Create Information Entity helpers
62  ::fwGdcmIO::writer::ie::Patient patientIE(writer, m_instance, series->getPatient());
63  ::fwGdcmIO::writer::ie::Study studyIE(writer, m_instance, series->getStudy());
64  ::fwGdcmIO::writer::ie::Series seriesIE(writer, m_instance, series);
65  // Use Image as frame of reference
66  ::fwGdcmIO::writer::ie::FrameOfReference frameOfReferenceIE(writer, m_imageInstance, series);
67  ::fwGdcmIO::writer::ie::Equipment equipmentIE(writer, m_instance, series->getEquipment());
68  ::fwGdcmIO::writer::ie::Surface surfaceIE(writer, m_instance, m_imageInstance, modelSeries);
69 
70  // Write Patient Module - PS 3.3 C.7.1.1
71  patientIE.writePatientModule();
72 
73  // Write General Study Module - PS 3.3 C.7.2.1
74  studyIE.writeGeneralStudyModule();
75 
76  // Write Patient Study Module - PS 3.3 C.7.2.2
77  studyIE.writePatientStudyModule();
78 
79  // Write General Series Module - PS 3.3 C.7.3.1
80  seriesIE.writeGeneralSeriesModule();
81 
82  // Write General Series Module - PS 3.3 C.8.20.1
83  seriesIE.writeSegmentationSeriesModule();
84 
85  // Write Frame of Reference Module - PS 3.3 C.7.4.1
86  frameOfReferenceIE.writeFrameOfReferenceModule();
87 
88  // Write General Equipment Module - PS 3.3 C.7.5.1
89  equipmentIE.writeGeneralEquipmentModule();
90 
91  // Copy dataset to avoid writing conflict with GDCM
92  const ::gdcm::DataSet datasetCopy = writer->GetFile().GetDataSet();
93 
94  // Write SOP Common Module - PS 3.3 C.12.1
95  surfaceIE.writeSOPCommonModule();
96 
97  // Skipped segmentation count
98  unsigned int skippedSegmentationCount = 0;
99 
100  // Write surface segmentations
101  for(unsigned short index = 0; index < modelSeries->getReconstructionDB().size(); ++index)
102  {
103  try
104  {
105  // Write Surface Segmentation Module - PS 3.3 C.8.23.1
106  surfaceIE.writeSurfaceSegmentationModule(index);
107 
108  // Write Surface Mesh Module - PS 3.3 C.27.1
109  surfaceIE.writeSurfaceMeshModule(index);
110 
111  }
112  catch (::fwGdcmIO::exception::Failed& e)
113  {
114  ++skippedSegmentationCount;
115  SLM_ERROR(e.what());
116  }
117  }
118 
119  if (skippedSegmentationCount == modelSeries->getReconstructionDB().size())
120  {
121  throw ::fwGdcmIO::exception::Failed("All 3D reconstructions have been rejected");
122  }
123  else if (skippedSegmentationCount > 0)
124  {
125  OSLM_WARN(skippedSegmentationCount<<" 3D reconstruction(s) have been rejected");
126  }
127 
128 
129  // Number of Surfaces Tag(0x0066,0x0001) - Type 1
130  writer->SetNumberOfSurfaces(modelSeries->getReconstructionDB().size() - skippedSegmentationCount);
131  OSLM_TRACE("Number of Surfaces : " << writer->GetNumberOfSurfaces());
132 
133  // Write the file
134  ::fwGdcmIO::helper::FileWriter::write(m_folderPath.string() + "/imSEG", writer);
135 
136 
137 }
