60  visitor::CompareObjects visitor;

65  }
66

68  CPPUNIT_ASSERT_EQUAL(size_t(3), seriesDBComp->getContainer().size());

70  CPPUNIT_ASSERT(series);
71

74 
75  patient->setName(name);
76  patient->setSex("M");
77  patient->setPatientId("42");
78 
79  visitor::CompareObjects visitor;
80  visitor.compare(seriesDBRef, seriesDBComp);

188  ::fwData::Composite::sptr compRef = ::fwData::Composite::New();
189  ::fwData::Composite::sptr compComp = ::fwData::Composite::New();

193 
194  visitor::CompareObjects visitor;

196 
197  SPTR(visitor::CompareObjects::PropsMapType) diffs = visitor.getDifferences();
198  CPPUNIT_ASSERT_EQUAL(diffs->size(), (size_t)2);
199  CPPUNIT_ASSERT(diffs->find("values.float.value") != diffs->end());
200  CPPUNIT_ASSERT(diffs->find("values.int.value") != diffs->end());

202  CPPUNIT_ASSERT_EQUAL((*diffs)["values.int.value"], visitor::CompareObjects::s_MISSING_PROPERTY);
