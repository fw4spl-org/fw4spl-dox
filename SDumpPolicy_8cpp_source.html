00070 void SDumpPolicy::starting() throw ( ::fwTools::Failed )
00071 {
00072     ::fwMemory::BufferManager::sptr manager = ::fwMemory::BufferManager::getDefault();
00073     if (manager)
00074     {
00075         ::fwMemory::IPolicy::sptr policy = ::fwMemory::policy::registry::get()->create(m_policy);
00076 
00077         if (policy)
00078         {
00079             bool success;
00080             BOOST_FOREACH(const ParametersType::value_type &param, m_policyParams)
00081             {
00082                 success = policy->setParam(param.first, param.second);
00083                 OSLM_ERROR_IF( "[" << m_policy << "] Unable to set '"
00084                                << param.first << "' parameter to " << param.second,
00085                                !success);
00086                 OSLM_INFO_IF("Set '" << param.first << "' policy parameter to " << param.second , success);
00087             }
00088             ::fwCore::mt::WriteLock lock( manager->getMutex() );
00089             manager->setDumpPolicy(policy);
00090             OSLM_INFO("Set dump policy to : " << m_policy);
00091         }
00092         OSLM_ERROR_IF("Unable to instantiate '" << m_policy << "' dump policy", !policy);
00093     }
