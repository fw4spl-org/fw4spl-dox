00104         namespace ns = ::fwComEd::fieldHelper;

00110         {
00111             ::fwGui::dialog::MessageDialog messageBox;
00112             messageBox.setTitle("Focus landmarks");
00113             messageBox.setMessage( "Sorry, it is impossible to focus image landmarks. There are not defined landmarks for this selected image." );
00114             messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
00115             messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
00116             messageBox.show();
00117         }
00118         else
00119         {
00120             // Retrieve point names
00121             std::vector< std::string > names;
00122             std::map< std::string, ::fwData::Point::sptr > name2Point;
00123 
00124             ::fwData::PointList::PointListContainer points = landmarks->getCRefPoints();
00125             BOOST_FOREACH(::fwData::Point::sptr point, points)
00126             {
00127                 std::string name =  point->getField< ::fwData::String >( ::fwComEd::Dictionary::m_labelId )->value();
00128                 OSLM_DEBUG( "Point name " << name );
00129                 names.push_back( name );
00130                 name2Point[name] = point;
00131             }
00132 
00133             // Propose to user to choose a landmark
00134             ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
00135             selector->setTitle("Select a landmark");
00136             selector->setSelections(names);
00137             std::string selection = selector->show();
00138             if( ! selection.empty() )
00139             {
00140                 ::fwData::Point::sptr selectedPoint = name2Point[ selection ];
00141                 SLM_ASSERT("selectedPoint not instanced", selectedPoint);
00142                 ::fwData::Integer::sptr paramA = ::fwData::Integer::New();
00143                 paramA->value() = static_cast<int>((selectedPoint->getRefCoord()[2] - pImage->getOrigin()[2] )/  pImage->getSpacing()[2] +0.5);
00144                 ::fwData::Integer::sptr paramF = ::fwData::Integer::New();
00145                 paramF->value() = static_cast<int>((selectedPoint->getRefCoord()[1] -  pImage->getOrigin()[1])/  pImage->getSpacing()[1] +0.5);
00146                 ::fwData::Integer::sptr paramS = ::fwData::Integer::New();
00147                 paramS->value() = static_cast<int>((selectedPoint->getRefCoord()[0] -  pImage->getOrigin()[0])/  pImage->getSpacing()[0] +0.5);
00148                 if( paramS->value() >= 0 &&
00149                         paramF->value() >= 0 &&
00150                         paramA->value() >= 0 &&
00151                         pImage->getSize()[0] > paramS->value() &&
00152                         pImage->getSize()[1] > paramF->value() &&
00153                         pImage->getSize()[2] > paramA->value() )
00154                 {
00155                     pImage->setField( ::fwComEd::Dictionary::m_axialSliceIndexId, paramA );
00156                     pImage->setField( ::fwComEd::Dictionary::m_frontalSliceIndexId, paramF );
00157                     pImage->setField( ::fwComEd::Dictionary::m_sagittalSliceIndexId, paramS );
00158 
00159                     // notify
00160                     ::fwComEd::ImageMsg::sptr msg = ::fwComEd::ImageMsg::New();

00162                     ::fwServices::IEditionService::notify(this->getSptr(), pImage, msg);
00163                 }
00164                 else
00165                 {
00166                     ::fwGui::dialog::MessageDialog::showMessageDialog("Focus landmarks",
00167                             "Sorry, it is impossible to focus image landmarks: landmark outside image.",
00168                             ::fwGui::dialog::IMessageDialog::WARNING);
00169                 }
00170             }
00171         }
00172     }
