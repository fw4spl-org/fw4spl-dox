77 void FocusLandmark::updating() throw(::fwTools::Failed)
78 {
79  SLM_TRACE_FUNC();
80 
81  ::fwData::Image::sptr pImage = this->getObject< ::fwData::Image >();
82  if (!::fwDataTools::fieldHelper::MedicalImageHelpers::checkImageValidity(pImage))
83  {
84  ::fwGui::dialog::MessageDialog messageBox;
85  messageBox.setTitle("Add landmarks");
86  messageBox.setMessage(
87  "It is impossible to add image landmarks. There is no loaded image in the software." );
88  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
89  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
90  messageBox.show();
91  return;
92  }
93 
94  else // Image is defined
95  {
96  // get landmarks
97  namespace ns = ::fwDataTools::fieldHelper;
98  ns::MedicalImageHelpers::checkLandmarks( pImage );
99  ::fwData::PointList::sptr landmarks = pImage->getField< ::fwData::PointList >(
100  ::fwDataTools::fieldHelper::Image::m_imageLandmarksId);
101  SLM_ASSERT("landmarks not instanced", landmarks);
102 
103  if( landmarks->getCRefPoints().empty() )
104  {
105  ::fwGui::dialog::MessageDialog messageBox;
106  messageBox.setTitle("Focus landmarks");
107  messageBox.setMessage(
108  "It is impossible to focus image landmarks. There are no defined landmarks for this selected image." );
109  messageBox.setIcon(::fwGui::dialog::IMessageDialog::WARNING);
110  messageBox.addButton(::fwGui::dialog::IMessageDialog::OK);
111  messageBox.show();
112  }
113  else
114  {
115  // Retrieve point names
116  std::vector< std::string > names;
117  std::map< std::string, ::fwData::Point::sptr > name2Point;
118 
119  ::fwData::PointList::PointListContainer points = landmarks->getCRefPoints();
120  for(::fwData::Point::sptr point : points)
121  {
122  std::string name =
123  point->getField< ::fwData::String >( ::fwDataTools::fieldHelper::Image::m_labelId )->value();
124  OSLM_DEBUG( "Point name " << name );
125  names.push_back( name );
126  name2Point[name] = point;
127  }
128 
129  // Propose to user to choose a landmark
130  ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
131  selector->setTitle("Select a landmark");
132  selector->setSelections(names);
133  std::string selection = selector->show();
134  if( !selection.empty() )
135  {
136  ::fwData::Point::sptr selectedPoint = name2Point[ selection ];
137  SLM_ASSERT("selectedPoint not instanced", selectedPoint);
138  ::fwData::Integer::sptr paramA = ::fwData::Integer::New();
139  paramA->value() =
140  static_cast<int>((selectedPoint->getRefCoord()[2] - pImage->getOrigin()[2] )/
141  pImage->getSpacing()[2] +0.5);
142  ::fwData::Integer::sptr paramF = ::fwData::Integer::New();
143  paramF->value() =
144  static_cast<int>((selectedPoint->getRefCoord()[1] - pImage->getOrigin()[1])/
145  pImage->getSpacing()[1] +0.5);
146  ::fwData::Integer::sptr paramS = ::fwData::Integer::New();
147  paramS->value() =
148  static_cast<int>((selectedPoint->getRefCoord()[0] - pImage->getOrigin()[0])/
149  pImage->getSpacing()[0] +0.5);
150  if( paramS->value() >= 0 &&
151  paramF->value() >= 0 &&
152  paramA->value() >= 0 &&
153  pImage->getSize()[0] > paramS->value() &&
154  pImage->getSize()[1] > paramF->value() &&
155  pImage->getSize()[2] > paramA->value() )
156  {
157  pImage->setField( ::fwDataTools::fieldHelper::Image::m_axialSliceIndexId, paramA );
158  pImage->setField( ::fwDataTools::fieldHelper::Image::m_frontalSliceIndexId, paramF );
159  pImage->setField( ::fwDataTools::fieldHelper::Image::m_sagittalSliceIndexId, paramS );
160 
161  // notify
162  auto sig = pImage->signal< ::fwData::Image::SliceIndexModifiedSignalType >(
163  ::fwData::Image::s_SLICE_INDEX_MODIFIED_SIG);
164  sig->asyncEmit(paramA->value(), paramF->value(), paramS->value());
165  }
166  else
167  {
168  ::fwGui::dialog::MessageDialog::showMessageDialog("Focus landmarks",
169  "It is impossible to focus image landmarks: "
170  "landmark is outside image.",
171  ::fwGui::dialog::IMessageDialog::WARNING);
172  }
173  }
174  }
175  }
176 }
