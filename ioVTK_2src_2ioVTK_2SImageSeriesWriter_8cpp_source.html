118 void SImageSeriesWriter::updating() throw(::fwTools::Failed)
119 {
120  SLM_TRACE_FUNC();
121 
122  if( this->hasLocationDefined() )
123  {
124  // Retrieve dataStruct associated with this service
125  ::fwMedData::ImageSeries::sptr imageSeries = this->getObject< ::fwMedData::ImageSeries >();
126  SLM_ASSERT("ImageSeries is not instanced", imageSeries);
127  SLM_ASSERT("Image from ImageSeries is not instanced", imageSeries->getImage());
128 
129  ::fwGui::Cursor cursor;
130  cursor.setCursor(::fwGui::ICursor::BUSY);
131 
132  try
133  {
134  SImageWriter::saveImage(this->getFile(), imageSeries->getImage(), m_sigJobCreated);
135  }
136  catch(::fwTools::Failed& e)
137  {
138  OSLM_TRACE("Error : " << e.what());
139  FW_RAISE_EXCEPTION(e);
140  }
141  cursor.setDefaultCursor();
142  }
143 }
