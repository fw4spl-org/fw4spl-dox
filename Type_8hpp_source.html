00237 void Type::setType()
00238 {
00239     m_name = Type::traitsToString< sizeof(T), ::boost::is_signed<T>::value, ::boost::is_integral<T>::value >();
00240 
00241     m_sizeof = sizeof(T);
00242     m_isSigned = ::boost::is_signed<T>::value;
00243     m_isFixedPrecision = ::boost::is_integral<T>::value;
00244 
00245     m_tool = SPTR(ToolBase)(new Type::Tool<T>());
00246 
00247     T min = static_cast< T >( std::numeric_limits< T >::min() );
00248     T max = static_cast< T >( std::numeric_limits< T >::max() );
00249 
00250     if(!m_isFixedPrecision)
00251     {
00252         // std::numeric_limits::min() returns the smallest positive value for floating types
00253         min = max * -1;
00254     }
00255 
00256     m_min = min;
00257     m_max = max;
00258 }
