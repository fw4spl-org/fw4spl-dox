91 void Plugin::loadStyleSheet()
92 {
93  if( this->getBundle()->hasParameter("resource") )
94  {
95  std::string resourceFile = this->getBundle()->getParameterValue("resource");
96  bool resourceLoaded = QResource::registerResource(resourceFile.c_str());
97  SLM_ASSERT("Cannot load resources '"+resourceFile+"'.", resourceLoaded);
98  }
99 
100  if( this->getBundle()->hasParameter("style") )
101  {
102  std::string style = this->getBundle()->getParameterValue("style");
103  qApp->setStyle(QStyleFactory::create(QString::fromStdString(style)));
104  }
105 
106  if( this->getBundle()->hasParameter("stylesheet") )
107  {
108  std::string stylesheetFile = this->getBundle()->getParameterValue("stylesheet");
109 
110  QFile data(QString::fromStdString(stylesheetFile));
111  QString style;
112  if(data.open(QFile::ReadOnly))
113  {
114  QTextStream styleIn(&data);
115  style = styleIn.readAll();
116  data.close();
117  qApp->setStyleSheet(style);
118  }
119  }
120 }
