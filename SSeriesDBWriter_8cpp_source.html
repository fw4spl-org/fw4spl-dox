93 void SSeriesDBWriter::updating() throw(::fwTools::Failed)
94 {
95  if( this->hasLocationDefined() )
96  {
97  const ::boost::filesystem::path& folder = this->getFolder();
98  if(!::boost::filesystem::is_empty(folder))
99  {
100  ::fwGui::dialog::MessageDialog dialog;
101  dialog.setMessage("Folder '"+folder.string()+"' isn't empty, files can be overwritten."
102  "\nDo you want to continue ?");
103  dialog.setTitle("Folder not empty.");
104  dialog.setIcon(::fwGui::dialog::MessageDialog::QUESTION);
105  dialog.addButton( ::fwGui::dialog::MessageDialog::YES_NO );
106  ::fwGui::dialog::MessageDialog::Buttons button = dialog.show();
107 
108  if(button == ::fwGui::dialog::MessageDialog::NO)
109  {
110  return;
111  }
112  }
113 
114  // Retrieve dataStruct associated with this service
115  ::fwData::Vector::sptr vector = this->getObject< ::fwData::Vector >();
116 
117  // Create SeriesDB
118  ::fwMedData::SeriesDB::sptr seriesDB = ::fwMedData::SeriesDB::New();
119  ::fwMedDataTools::helper::SeriesDB seriesDBHelper(seriesDB);
120 
121  for(::fwData::Object::sptr object: vector->getContainer())
122  {
123  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(object);
124  SLM_ASSERT("The container should only contain series.", series);
125  seriesDBHelper.add(series);
126  }
127 
128  ::fwGui::Cursor cursor;
129  cursor.setCursor(::fwGui::ICursor::BUSY);
130  this->saveSeriesDB(folder, seriesDB);
131  cursor.setDefaultCursor();
132  }
133 }

138 void SSeriesDBWriter::saveSeriesDB( const ::boost::filesystem::path folder, ::fwMedData::SeriesDB::sptr seriesDB )
139 {
140  ::fwGdcmIO::writer::SeriesDB::sptr writer = ::fwGdcmIO::writer::SeriesDB::New();

146 
147  try

151  writer->write();
152  }

155  std::stringstream ss;

159  }
160  catch( ... )
161  {
162  ::fwGui::dialog::MessageDialog::showMessageDialog(
163  "Warning", "Warning during saving", ::fwGui::dialog::IMessageDialog::WARNING);

176 bool SSeriesDBWriter::selectFiducialsExportMode()
177 {
178  // Retrieve dataStruct associated with this service
179  ::fwData::Vector::sptr vector = this->getObject< ::fwData::Vector >();
180 
181  // Create SeriesDB
182  ::fwMedData::SeriesDB::sptr seriesDB = ::fwMedData::SeriesDB::New();
183  ::fwMedDataTools::helper::SeriesDB seriesDBHelper(seriesDB);
184 
185  for(::fwData::Object::sptr object: vector->getContainer())
186  {
187  ::fwMedData::Series::sptr series = ::fwMedData::Series::dynamicCast(object);
188  SLM_ASSERT("The container should only contain series.", series);
189  seriesDBHelper.add(series);
190  }
191 
192  const bool containsLandmarks = ::fwGdcmIO::helper::Fiducial::containsLandmarks(seriesDB);
193  const bool containsDistances = ::fwGdcmIO::helper::Fiducial::containsDistances(seriesDB);
194  const bool contains3DDistances = ::fwGdcmIO::helper::Fiducial::contains3DDistances(seriesDB);
195 
196  if(containsLandmarks || containsDistances)
197  {
198  static const std::string fiducialIOD = "Spatial Fiducials";
199  static const std::string comprehensiveSRIOD = "Comprehensive SR";
200  static const std::string comprehensive3DSRIOD = "Comprehensive 3D SR";
201 
202  std::vector< std::string > exportModes;
203  if(!containsDistances)
204  {
205  exportModes.push_back(fiducialIOD);
206  }
207  if(!contains3DDistances)
208  {
209  exportModes.push_back(comprehensiveSRIOD);
210  }
211  exportModes.push_back(comprehensive3DSRIOD);
212 
213  // Create selector
214  ::fwGui::dialog::SelectorDialog::sptr selector = ::fwGui::dialog::SelectorDialog::New();
215 
216  selector->setTitle("Fiducials export mode");
217  selector->setSelections(exportModes);
218  const std::string mode = selector->show();
219  const bool modeSelectionIsCanceled = mode.empty();
220 
221  if(mode == fiducialIOD)
222  {
223  m_fiducialsExportMode = ::fwGdcmIO::writer::Series::SPATIAL_FIDUCIALS;
224  }
225  else if(mode == comprehensiveSRIOD)
226  {
227  m_fiducialsExportMode = ::fwGdcmIO::writer::Series::COMPREHENSIVE_SR;
228  }
229  else
230  {
231  m_fiducialsExportMode = ::fwGdcmIO::writer::Series::COMPREHENSIVE_3D_SR;
232  }
233 
234 
235  return !modeSelectionIsCanceled;
236  }
237 
238  return true;
239 
240 }
