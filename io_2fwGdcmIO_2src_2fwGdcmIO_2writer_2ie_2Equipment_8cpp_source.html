37 void Equipment::writeGeneralEquipmentModule()
38 {
39  // Retrieve dataset
40  ::gdcm::DataSet& dataset = m_writer->GetFile().GetDataSet();
41 
42  // Retrieve runtime profile
43  ::fwRuntime::profile::Profile::sptr profile = ::fwRuntime::profile::getCurrentProfile();
44 
45  // Institution Name - Type 3
46  const std::string& institutName = m_object->getInstitutionName();
47  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x0080 >(institutName, dataset);
48 
49  // Manufacturer - Type 2
50  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x0070 >("FW4SPL", dataset);
51 
52  // Manufacturer’s Model Name - Type 3 (Type 1 for EnhancedGeneralEquipmentModule)
53  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0008, 0x1090 >("FW4SPL_FWGDCMIO", dataset);
54 
55  // Device Serial Number - Type 3 (Type 1 for EnhancedGeneralEquipmentModule)
56  std::string deviceSerialNumber = std::string(FWGDCMIO_VER);
57  ::fwGdcmIO::helper::DicomData::setTagValue< 0x0018, 0x1000 >(deviceSerialNumber, dataset);
58 
59  // Software Versions - Type 3 (Type 1 for EnhancedGeneralEquipmentModule)
60  ::gdcm::Attribute< 0x0018, 0x1020 > attribute;
61  attribute.SetNumberOfValues(1);
62  attribute.SetValue((profile) ? (profile->getName()+" "+profile->getVersion()) : "Unknown");
63  dataset.Insert(attribute.GetAsDataElement());
64 }
