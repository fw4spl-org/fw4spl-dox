00078 void StarterActionService::updating() throw( ::fwTools::Failed )
00079 {
00080     SLM_TRACE_FUNC() ;
00081 
00082     for(size_t i = 0; i < m_uuidServices.size(); i++)
00083     {
00084         ActionType action =  m_uuidServices.at(i).second;
00085         IDSrvType uid =  m_uuidServices.at(i).first;
00086         bool srv_exists = ::fwTools::fwID::exist(uid  );
00087 
00088         // Manage special action
00089         if ( action == START_IF_EXISTS )
00090         {
00091             if ( srv_exists )
00092             {
00093                 action = START;
00094             }
00095             else
00096             {
00097                 action = DO_NOTHING;
00098             }
00099         }
00100         else if( action == STOP_IF_EXISTS )
00101         {
00102             if ( srv_exists )
00103             {
00104                 action = STOP;
00105             }
00106             else
00107             {
00108                 action = DO_NOTHING;
00109             }
00110         }
00111 
00112         if( action != DO_NOTHING)
00113         {
00114             ::fwGui::LockAction lock(this->getSptr());
00115 
00116             ::fwServices::IService::sptr service = ::fwServices::get( uid ) ;
00117             SLM_ASSERT("service not found", service);
00118             switch ( action )
00119             {
00120             case START_OR_STOP:
00121             {
00122                 if(service->isStopped())
00123                 {
00124                     service->start();
00125                     service->update();
00126                     m_idStartedSrvSet.insert(uid);
00127                 }
00128                 else
00129                 {
00130                     service->stop();
00131                     m_idStartedSrvSet.erase(uid);
00132                 }
00133                 break;
00134             }
00135             case START :
00136             {
00137                 if(service->isStopped())
00138                 {
00139                     service->start();
00140                     m_idStartedSrvSet.insert(uid);
00141                 }
00142                 else
00143                 {
00144                     OSLM_WARN("Service " << service->getID() << " is not stopped");
00145                 }
00146                 service->update();
00147                 break;
00148             }
00149             case STOP :
00150             {
00151                 if(service->isStarted())
00152                 {
00153                     service->stop();
00154                 }
00155                 else
00156                 {
00157                     OSLM_WARN("Service " << service->getID() << " is not started");
00158                 }
00159                 break;
00160             }
00161             default :
00162             {
00163                 SLM_FATAL("Sorry, this action type is not managed");
00164                 break;
00165             }
00166             }
00167         }
00168         else
00169         {
00170             ::fwGui::dialog::MessageDialog::showMessageDialog(
00171                     "Service unavailable",
00172                     "Sorry, the service is unavailable.",
00173                     ::fwGui::dialog::IMessageDialog::WARNING);
00174 
00175             OSLM_INFO("Do nothing for Service " << m_uuidServices.at(i).first);
00176         }
00177     }
00178 }

00189 void StarterActionService::configuring() throw( ::fwTools::Failed )
00190 {
00191     SLM_TRACE_FUNC() ;
00192     this->initialize();
00193 
00194     BOOST_FOREACH(ConfigurationType actionCfg, m_configuration->getElements() )
00195     {
00196         OSLM_INFO( "StarterActionService "  << actionCfg->getName());
00197 
00198         std::string actionType =  actionCfg->getName();
00199         ActionType action;
00200         if ( actionType == "start" )                { action = START; }
00201         else if ( actionType == "stop" )            { action = STOP; }
00202         else if ( actionType == "start_or_stop" )   { action = START_OR_STOP; }
00203         else if ( actionType == "start_if_exists" ) { action = START_IF_EXISTS; }
00204         else if ( actionType == "stop_if_exists" )  { action = STOP_IF_EXISTS; }
00205         else
00206         {
00207             OSLM_WARN("Sorry this type of \"actionType\":" << actionType <<" is not managed by StarterActionService");
00208             continue;
00209         }
00210         SLM_ASSERT("Attribute uid missing", actionCfg->hasAttribute("uid")) ;
00211         IDSrvType uuid = actionCfg->getExistingAttributeValue("uid") ;
00212 
00213         m_uuidServices.push_back( std::make_pair(uuid, action) );
00214     }
00215 }
