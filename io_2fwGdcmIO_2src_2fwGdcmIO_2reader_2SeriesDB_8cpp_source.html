93  m_logger->clear();

95 
96  // Finish jobs
97  m_dicomdirFileLookupJob->finish();
98  m_regularFileLookupJob->finish();
99  m_readerJob->finish();
100  m_completeDicomSeriesJob->finish();
101  m_converterJob->finish();
102  m_job->run().get();

104  }
105 
106  // Apply Default filters
107  if(!m_dicomFilterType.empty())
108  {
109  ::fwDicomIOFilter::IFilter::sptr filter = ::fwDicomIOFilter::factory::New(m_dicomFilterType);
110  ::fwDicomIOFilter::helper::Filter::applyFilter(m_dicomSeriesContainer,filter, true, m_logger);
111  }
112 
113  if(m_dicomSeriesContainer.empty())
114  {
115  m_logger->critical("Unable to retrieve series from the selected folder.");
116  m_converterJob->done();
117  m_converterJob->finish();
118  }
119  else
120  {
121  // Read series
122  this->convertDicomSeries();
123  }
124 
125  try
126  {
127  // .get() throws exception that might have occured
128  m_job->run().get();
129  }
130  catch (const std::exception& e)
131  {
132  m_logger->critical("An error has occured during the reading process : " + std::string(e.what()));
133  }
134  catch( ... )
135  {
136  m_logger->critical("An unkown error has occured during the reading process.");
137  }
138 
139  if(m_dicomSeriesContainer.empty())

141  m_logger->critical("Unable to retrieve series from the selected folder.");
142  }

210 void SeriesDB::readDicom(const bool checkIsDicom)
211 {
212  SLM_ASSERT("This reader only work on folder selection.",
213  (::fwData::location::have < ::fwData::location::Folder, ::fwDataIO::reader::IObjectReader > (this)));
214 
215  // DICOMDIR
216  auto dicomdir = ::fwGdcmIO::helper::DicomDir::findDicomDir(this->getFolder());
217  if(m_isDicomdirActivated && ::boost::filesystem::exists(dicomdir))
218  {
219  // Create Dicom Series
220  ::fwGdcmIO::helper::DicomDir::retrieveDicomSeries(dicomdir,
221  m_dicomSeriesContainer,
222  m_logger,
223  m_dicomdirFileLookupJob
224  );
225  // Fill Dicom Series
226  ::fwGdcmIO::helper::DicomSeries helper;
227  helper.complete(m_dicomSeriesContainer, m_completeDicomSeriesJob);
228 
229  }
230 
231  // Finish DICOMDIR lookup
232  m_dicomdirFileLookupJob->done();
233  m_dicomdirFileLookupJob->finish();
234 
235  // Regular read
236  if(!m_isDicomdirActivated || !::boost::filesystem::exists(dicomdir) || m_dicomSeriesContainer.empty())
237  {
238  m_readerJob->doneWork(0);
239 
240  // Recursively search for dicom files
241  std::vector< std::string > filenames;
242  ::fwGdcmIO::helper::DicomSearch::searchRecursively(
243  this->getFolder(), filenames, checkIsDicom, m_regularFileLookupJob);
244 
245  // Read Dicom Series
246  ::fwGdcmIO::helper::DicomSeries helper;
247  if(checkIsDicom)
248  {
249  m_dicomSeriesContainer = helper.read(filenames, m_readerJob, m_completeDicomSeriesJob);
250  }
251  else
252  {
253  m_dicomSeriesContainer.push_back(helper.createBlob(filenames));
254  }
255  }
256 
257  // Finish regular lookup
258  m_regularFileLookupJob->done();
259  m_regularFileLookupJob->finish();
260 
261  // Finish reading
262  m_readerJob->done();
263  m_readerJob->finish();
264 
265  // Finish completing series
266  m_completeDicomSeriesJob->done();
267  m_completeDicomSeriesJob->finish();
268

333 void SeriesDB::convertDicomSeries(::fwServices::IService::sptr notifier)
334 {
335  ::fwMedData::SeriesDB::sptr seriesDB = this->getConcreteObject();
336 
337  // Sort DicomSeries
338  std::sort(m_dicomSeriesContainer.begin(), m_dicomSeriesContainer.end(), SeriesDB::dicomSeriesComparator);
339 
340  // Create reader
341  SPTR(::fwGdcmIO::reader::Series) seriesReader = std::make_shared< ::fwGdcmIO::reader::Series >();
342  seriesReader->setBufferRotationEnabled(m_enableBufferRotation);
343  seriesReader->setLogger(m_logger);
344 
345  // needed for seriesReader
346  seriesReader->addCallback([&](unsigned int& progress)
347  {
348  });
349 
350  m_converterJob->setTotalWorkUnits(m_dicomSeriesContainer.size());
351 
352  // Read series
353  std::uint64_t progress = 0;
354  for(const ::fwMedData::DicomSeries::sptr& dicomSeries: m_dicomSeriesContainer)
355  {
356  ::fwMedData::DicomSeries::SOPClassUIDContainerType sopClassUIDContainer = dicomSeries->getSOPClassUIDs();
357  FW_RAISE_IF("The series contains several SOPClassUIDs. Try to apply a filter in order to split the series.",
358  sopClassUIDContainer.size() != 1);
359  std::string sopClassUID = sopClassUIDContainer.begin()->c_str();
360 
361  const SupportedSOPClassContainerType::iterator bIt = m_supportedSOPClassContainer.begin();
362  const SupportedSOPClassContainerType::iterator eIt = m_supportedSOPClassContainer.end();
363 
364  if(m_supportedSOPClassContainer.empty() || std::find(bIt, eIt, sopClassUID) != eIt)
365  {
366  try
367  {
368 
369  ::fwMedData::Series::sptr series = seriesReader->read(dicomSeries);
370 
371  if(series)
372  {
373  // Add the series to the DB
374  ::fwMedDataTools::helper::SeriesDB seriesDBHelper(seriesDB);
375  seriesDBHelper.add(series);
376 
377  if(notifier)
378  {
379  seriesDBHelper.notify();
380  }
381  }
382  }
383  catch (::fwGdcmIO::exception::Failed& e)
384  {
385  m_logger->critical("Unable to read series : " + dicomSeries->getInstanceUID());
386  }
387  }
388  else
389  {
390  m_logger->critical("DICOM SOP Class UID \"" + sopClassUID +"\" is not supported by the selected reader.");
391  }
392 
393  if(m_job->cancelRequested())
394  {
395  break;
396  }
397 
398  m_converterJob->doneWork(++progress);
399  }
400 
401  m_converterJob->done();
402  m_converterJob->finish();
403 
404 }

408 bool SeriesDB::dicomSeriesComparator(SPTR(::fwMedData::DicomSeries)a, SPTR(::fwMedData::DicomSeries)b)
409 {
410  ::fwMedData::DicomSeries::SOPClassUIDContainerType aSOPClassUIDContainer = a->getSOPClassUIDs();
411  std::string aSOPClassUID = *(aSOPClassUIDContainer.begin());
412  ::fwMedData::DicomSeries::SOPClassUIDContainerType bSOPClassUIDContainer = b->getSOPClassUIDs();
413  std::string bSOPClassUID = *(bSOPClassUIDContainer.begin());
414 
415  // a > b if a contains a SR and not b
416  bool aIsAnImage = (::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) == ::gdcm::MediaStorage::EnhancedSR ||
417  ::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) == ::gdcm::MediaStorage::ComprehensiveSR ||
418  aSOPClassUID == "1.2.840.10008.5.1.4.1.1.88.34" || // FIXME Replace hard coded string by "::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) == ::gdcm::MediaStorage::Comprehensive3DSR"
419  ::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) ==
420  ::gdcm::MediaStorage::SpacialFiducialsStorage ||
421  ::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) ==
422  ::gdcm::MediaStorage::SurfaceSegmentationStorage);
423 
424  bool bIsAnImage = (::gdcm::MediaStorage::GetMSType(bSOPClassUID.c_str()) == ::gdcm::MediaStorage::EnhancedSR ||
425  ::gdcm::MediaStorage::GetMSType(bSOPClassUID.c_str()) == ::gdcm::MediaStorage::ComprehensiveSR ||
426  bSOPClassUID == "1.2.840.10008.5.1.4.1.1.88.34" || // FIXME Replace hard coded string by "::gdcm::MediaStorage::GetMSType(bSOPClassUID.c_str()) == ::gdcm::MediaStorage::Comprehensive3DSR"
427  ::gdcm::MediaStorage::GetMSType(bSOPClassUID.c_str()) ==
428  ::gdcm::MediaStorage::SpacialFiducialsStorage ||
429  ::gdcm::MediaStorage::GetMSType(aSOPClassUID.c_str()) ==
430  ::gdcm::MediaStorage::SurfaceSegmentationStorage);
431 
432  return bIsAnImage && !aIsAnImage;
433 }
