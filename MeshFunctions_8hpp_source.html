00073 bool isBorderlessSurface(T* cellDataBegin, T* cellDataEnd, U* cellDataOffsetsBegin, U* cellDataOffsetsEnd, V* cellTypesBegin)
00074 {
00075     typedef std::pair< T, T >  Edge; // always Edge.first < Edge.second !!
00076     typedef boost::unordered_map< Edge, int >  EdgeHistogram;
00077     EdgeHistogram edgesHistogram;
00078     bool isBorderless = true;
00079 
00080     size_t dataLen = 0;
00081     U* iter = cellDataOffsetsBegin;
00082     U* iter2 = cellDataOffsetsBegin + 1;
00083     const U* iterEnd = cellDataOffsetsEnd - 1;
00084     V* iterTypes = cellTypesBegin;
00085 
00086     dataLen = *iter2 - *iter;
00087     for (
00088             ;
00089             iter < iterEnd || ( iter < cellDataOffsetsEnd && (dataLen = (cellDataEnd - cellDataBegin) - *iter) ) ;
00090             dataLen = *++iter2 - *++iter, ++iterTypes
00091         )
00092     {
00093         if(*iterTypes == 0)
00094         {
00095             continue;
00096         }
00097         T* iterCell = cellDataBegin + *iter;
00098         T* iterCell2 = iterCell + 1;
00099         T* beginCell = iterCell;
00100         const T* iterCellEnd = beginCell + dataLen - 1;
00101         for (
00102                 ;
00103                 iterCell < iterCellEnd || ( iterCell < (beginCell + dataLen) && (iterCell2 = beginCell) ) ;
00104                 ++iterCell, ++iterCell2
00105             )
00106         {
00107             ++edgesHistogram[makeOrderedPair(*iterCell, *(iterCell2))];
00108         }
00109     }
00110 
00111     BOOST_FOREACH(const typename EdgeHistogram::value_type &histo, edgesHistogram)
00112     {
00113         if (histo.second != 2)
00114         {
00115             isBorderless = false;
00116             break;
00117         }
00118     }
