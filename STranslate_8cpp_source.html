56 void STranslate::starting() throw ( ::fwTools::Failed )
57 {
58  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
59 
60  if (!m_sourceCompositeID.empty())
61  {
62  ::fwData::Composite::sptr source;
63  source = ::fwData::Composite::dynamicCast(::fwTools::fwID::getObject(m_sourceCompositeID));
64  SLM_ASSERT("Source composite is missing '" + m_sourceCompositeID + "'", source);
65 
66  m_connections.connect(source, this->getSptr(), this->getObjSrvConnections() );
67 
68  for( const ManagedTranslations::value_type& trans : m_managedTranslations )
69  {
70  const std::string& fromKey = trans.first;
71  const std::string& toKey = trans.second;
72 
73  ::fwData::Composite::const_iterator iter = source->find(fromKey);
74  if (iter != source->end())
75  {
76  this->updateComposite(composite, iter->second, toKey, ADD_OR_SWAP );
77  }
78  }
79  }
80 }

91 void STranslate::configuring() throw ( ::fwTools::Failed )
92 {
93  const ::fwServices::IService::ConfigType conf = this->getConfigTree().get_child("service");
94 
95  if (conf.count("source"))
96  {
97  const ::fwServices::IService::ConfigType& sourceCfg = conf.get_child("source");
98  m_sourceCompositeID = sourceCfg.get_value<std::string>();
99  SLM_ASSERT("Source composite is not defined", !m_sourceCompositeID.empty());
100  }
101 
102  SLM_ASSERT("Problem with configuration for STranslate type, missing element \"translate\"",
103  conf.count("translate") > 0);
104 
105  OSLM_DEBUG( "nb of translations = " << conf.count("translate") );
106  m_managedTranslations.clear();
107  BOOST_FOREACH( const ::fwServices::IService::ConfigType::value_type &v, conf.equal_range("translate") )
108  {
109  const ::fwServices::IService::ConfigType& translate = v.second;
110  const ::fwServices::IService::ConfigType xmlattr = translate.get_child("<xmlattr>");
111 
112  SLM_FATAL_IF( "The attribute \"fromKey\" is missing, it represents the key of the object in the source "
113  "composite that will be translated", xmlattr.count("fromKey") != 1 );
114  SLM_FATAL_IF("The attribute \"toKey\" is missing, it represents the the key of the object in the current"
115  "composite that will contain the translation", xmlattr.count("toKey") != 1);
116 
117  std::string fromKey = xmlattr.get<std::string>("fromKey");
118  std::string toKey = xmlattr.get<std::string>("toKey");
119 
120  OSLM_INFO(
121  "Manage translation from "<< fromKey << " to "<< toKey <<
122  " in my composite.");
123  m_managedTranslations[fromKey] = toKey;
124  }
125 
126 }

149 {

151  for ( ManagedTranslations::value_type elt : m_managedTranslations )
152  {
153  const std::string& fromKey = elt.first;
154  const std::string& toKey = elt.second;
155 
156  ::fwData::Composite::ContainerType::iterator iter = objects.find(fromKey);
157  if (iter != objects.end())
