56  include_patterns = common.get_option('forbidtoken-hook.' + config_name, default=tr[config_name][2]).split()

58  common.note('Checking for "' + config_name + '" tokens on ' + ', '.join(include_patterns) + ' files')
59  abort = False
60  token = tr[config_name][0]
61  line_iter = lambda x:enumerate( re.finditer(".*\n", x, re.MULTILINE), 1 )
62  line_match = lambda test, x:(n for n,m in line_iter(x) if test(m.group()))
63 
64  count = 0
65  for f in files:
66  if not any(f.fnmatch(p) for p in include_patterns):
67  continue
68  common.trace('Checking ' + str(f.path) + '...')

76  count = count + 1
77
