117 void SObjFromSlot::addOrSwap(::fwData::Object::sptr obj)
118 {
119  if(this->isVersion2())
120  {
121  this->setOutput(s_OBJECT, obj);
122  }
123  else
124  {
125  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
126  ::fwDataTools::helper::Composite helper(composite);
127  if (composite->find(m_compositeKey) == composite->end())
128  {
129  helper.add(m_compositeKey, obj);
130  }
131  else
132  {
133  helper.swap(m_compositeKey, obj);
134  }
135  helper.notify();
136  }
137 }
