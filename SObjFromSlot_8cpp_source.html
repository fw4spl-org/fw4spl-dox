107 void SObjFromSlot::addOrSwap(::fwData::Object::sptr obj)
108 {
109  ::fwData::Composite::sptr composite = this->getObject< ::fwData::Composite >();
110  ::fwComEd::helper::Composite helper(composite);
111  if (composite->find(m_compositeKey) == composite->end())
112  {
113  helper.add(m_compositeKey, obj);
114  }
115  else
116  {
117  helper.swap(m_compositeKey, obj);
118  }
119  helper.notify();
120 }
